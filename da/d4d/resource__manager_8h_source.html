71struct resource_metrics {
72 std::atomic<size_t> current_usage{0};
73 std::atomic<size_t> total_allocations{0};
74 std::atomic<size_t> peak_usage{0};
75 std::atomic<size_t> rejected_operations{0};
76 std::atomic<size_t> delayed_operations{0};
77
78 resource_metrics() = default;
79 resource_metrics(const resource_metrics& other)
80 : current_usage(other.current_usage.load())
81 , total_allocations(other.total_allocations.load())
82 , peak_usage(other.peak_usage.load())
83 , rejected_operations(other.rejected_operations.load())
84 , delayed_operations(other.delayed_operations.load()) {}
85
86 resource_metrics& operator=(const resource_metrics& other) {
87 if (this != &other) {
88 current_usage.store(other.current_usage.load());
89 total_allocations.store(other.total_allocations.load());
90 peak_usage.store(other.peak_usage.load());
91 rejected_operations.store(other.rejected_operations.load());
92 delayed_operations.store(other.delayed_operations.load());
93 }
94 return *this;
95 }
96};

122struct resource_quota {
123 resource_type type = resource_type::memory;
124 size_t max_value = 0;
125 size_t warning_threshold = 0;
126 size_t critical_threshold = 0;
127 throttling_strategy strategy = throttling_strategy::reject;
128
129 resource_quota() = default;
130 resource_quota(resource_type t, size_t max, throttling_strategy s = throttling_strategy::reject)
131 : type(t), max_value(max), warning_threshold(max * 70 / 100),
132 critical_threshold(max * 90 / 100), strategy(s) {}
133
138 bool validate() const {
139 if (max_value == 0) {
140 return false;
141 }
142 if (warning_threshold > max_value) {
143 return false;
144 }
145 if (critical_threshold > max_value) {
146 return false;
147 }
148 if (warning_threshold > critical_threshold) {
149 return false;
150 }
151 return true;
152 }
153};

158struct cpu_throttle_config {

163

189class rate_limiter {
190public:
191 virtual ~rate_limiter() = default;
192
198 virtual bool try_acquire(size_t count = 1) = 0;
199
206 template<typename Func>
207 auto execute(Func&& func) -> decltype(func()) {
208 using result_type = decltype(func());
209 using value_type = typename result_type::value_type;
210
211 if (!try_acquire(1)) {
212 return common::make_error<value_type>(static_cast<int>(monitoring_error_code::resource_exhausted),
213 "Rate limit exceeded for '" + get_name() + "'");
214 }
215 return func();
216 }
217
221 virtual const std::string& get_name() const = 0;
222};

234class token_bucket_limiter : public rate_limiter {
235public:
236 using clock = std::chrono::steady_clock;
237
238 token_bucket_limiter(const std::string& name, double rate, size_t capacity,
239 throttling_strategy /*strategy*/ = throttling_strategy::reject)
240 : name_(name)
241 , rate_(rate)
242 , capacity_(capacity)
243 , tokens_(static_cast<double>(capacity))
244 , last_refill_(clock::now()) {}
245
246 bool try_acquire(size_t count = 1) override {
247 std::lock_guard<std::mutex> lock(mutex_);
248 refill();
249
250 if (tokens_ >= static_cast<double>(count)) {
251 tokens_ -= static_cast<double>(count);
252 return true;
253 }
254 return false;
255 }
256
257 const std::string& get_name() const override {
258 return name_;
259 }
260
261private:
262 void refill() {
263 auto now = clock::now();
264 auto elapsed = std::chrono::duration<double>(now - last_refill_).count();
265 tokens_ = std::min(static_cast<double>(capacity_), tokens_ + elapsed * rate_);
266 last_refill_ = now;
267 }
268
269 std::string name_;
270 double rate_;
271 size_t capacity_;
272 double tokens_;
273 clock::time_point last_refill_;
274 mutable std::mutex mutex_;
275};

287class leaky_bucket_limiter : public rate_limiter {
288public:
289 using clock = std::chrono::steady_clock;
290
291 leaky_bucket_limiter(const std::string& name, double rate, size_t capacity)
292 : name_(name)
293 , rate_(rate)
294 , capacity_(capacity)
295 , water_(0.0)
296 , last_leak_(clock::now()) {}
297
298 bool try_acquire(size_t count = 1) override {
299 std::lock_guard<std::mutex> lock(mutex_);
300 leak();
301
302 if (water_ + static_cast<double>(count) <= static_cast<double>(capacity_)) {
303 water_ += static_cast<double>(count);
304 return true;
305 }
306 return false;
307 }
308
309 const std::string& get_name() const override {
310 return name_;
311 }
312
313private:
314 void leak() {
315 auto now = clock::now();
316 auto elapsed = std::chrono::duration<double>(now - last_leak_).count();
317 water_ = std::max(0.0, water_ - elapsed * rate_);
318 last_leak_ = now;
319 }
320
321 std::string name_;
322 double rate_;
323 size_t capacity_;
324 double water_;
325 clock::time_point last_leak_;
326 mutable std::mutex mutex_;
327};

336class memory_quota_manager {
337public:
338 memory_quota_manager(const std::string& name, const resource_quota& quota)
339 : name_(name), quota_(quota) {}
340
341 memory_quota_manager(const std::string& name, size_t max_bytes,
342 throttling_strategy strategy = throttling_strategy::reject)
343 : name_(name), quota_(resource_type::memory, max_bytes, strategy) {}
344
350 common::Result<bool> allocate(size_t bytes) {
351 std::lock_guard<std::mutex> lock(mutex_);
352
353 if (metrics_.current_usage.load() + bytes > quota_.max_value) {
354 metrics_.rejected_operations++;
355 return common::make_error<bool>(static_cast<int>(monitoring_error_code::resource_exhausted),
356 "Memory quota exceeded for '" + name_ + "'");
357 }
358
359 metrics_.current_usage += bytes;
360 metrics_.total_allocations++;
361
362 // Update peak usage
363 size_t current = metrics_.current_usage.load();
364 size_t peak = metrics_.peak_usage.load();
365 while (current > peak && !metrics_.peak_usage.compare_exchange_weak(peak, current)) {
366 // Retry until successful
367 }
368
369 return common::ok(true);
370 }
371
376 void deallocate(size_t bytes) {
377 std::lock_guard<std::mutex> lock(mutex_);
378 size_t current = metrics_.current_usage.load();
379 metrics_.current_usage.store(current >= bytes ? current - bytes : 0);
380 }
381
385 size_t current_usage() const {
386 return metrics_.current_usage.load();
387 }
388
392 bool is_over_warning_threshold() const {
393 return metrics_.current_usage.load() >= quota_.warning_threshold;
394 }
395
399 bool is_over_critical_threshold() const {
400 return metrics_.current_usage.load() >= quota_.critical_threshold;
401 }
402
406 resource_metrics get_metrics() const {
407 return metrics_;
408 }
409
413 const std::string& get_name() const {
414 return name_;
415 }
416
417private:
418 std::string name_;
419 resource_quota quota_;
420 mutable resource_metrics metrics_;
421 mutable std::mutex mutex_;
422};

478class resource_manager {
479public:
480 explicit resource_manager(const std::string& name) : name_(name) {}
481
488 common::Result<bool> add_rate_limiter(const std::string& name, const rate_limit_config& config) {
489 std::lock_guard<std::mutex> lock(mutex_);
490
491 if (rate_limiters_.find(name) != rate_limiters_.end()) {
492 return common::make_error<bool>(static_cast<int>(monitoring_error_code::already_exists),
493 "Rate limiter '" + name + "' already exists");
494 }
495
496 rate_limiters_[name] = std::make_unique<token_bucket_limiter>(
497 name, config.rate_per_second, config.burst_capacity, config.strategy);
498 return common::ok(true);
499 }
500
506 rate_limiter* get_rate_limiter(const std::string& name) {
507 std::lock_guard<std::mutex> lock(mutex_);
508 auto it = rate_limiters_.find(name);
509 return it != rate_limiters_.end() ? it->second.get() : nullptr;
510 }
511
518 common::Result<bool> add_memory_quota(const std::string& name, const resource_quota& quota) {
519 std::lock_guard<std::mutex> lock(mutex_);
520
521 if (memory_quotas_.find(name) != memory_quotas_.end()) {
522 return common::make_error<bool>(static_cast<int>(monitoring_error_code::already_exists),
523 "Memory quota '" + name + "' already exists");
524 }
525
526 memory_quotas_[name] = std::make_unique<memory_quota_manager>(name, quota);
527 return common::ok(true);
528 }
529
535 memory_quota_manager* get_memory_quota(const std::string& name) {
536 std::lock_guard<std::mutex> lock(mutex_);
537 auto it = memory_quotas_.find(name);
538 return it != memory_quotas_.end() ? it->second.get() : nullptr;
539 }
540
547 common::Result<bool> add_cpu_throttler(const std::string& name, const cpu_throttle_config& config) {
548 std::lock_guard<std::mutex> lock(mutex_);
549
550 if (cpu_throttlers_.find(name) != cpu_throttlers_.end()) {
551 return common::make_error<bool>(static_cast<int>(monitoring_error_code::already_exists),
552 "CPU throttler '" + name + "' already exists");
553 }
554
555 cpu_throttlers_[name] = std::make_unique<cpu_throttler>(name, config);
556 return common::ok(true);
557 }
558
564 cpu_throttler* get_cpu_throttler(const std::string& name) {
565 std::lock_guard<std::mutex> lock(mutex_);
566 auto it = cpu_throttlers_.find(name);
567 return it != cpu_throttlers_.end() ? it->second.get() : nullptr;
568 }
569
574 common::Result<bool> is_healthy() const {
575 std::lock_guard<std::mutex> lock(mutex_);
576
577 for (const auto& [name, manager] : memory_quotas_) {
578 if (manager->is_over_critical_threshold()) {
579 return common::ok(false);
580 }
581 }
582
583 return common::ok(true);
584 }
585
590 std::unordered_map<std::string, resource_metrics> get_all_metrics() const {
591 std::lock_guard<std::mutex> lock(mutex_);
592 std::unordered_map<std::string, resource_metrics> all_metrics;
593
594 for (const auto& [name, limiter] : rate_limiters_) {
595 all_metrics["rate_" + name] = resource_metrics{};
596 }
597
598 for (const auto& [name, manager] : memory_quotas_) {
599 all_metrics["memory_" + name] = manager->get_metrics();
600 }
601
602 for (const auto& [name, throttler] : cpu_throttlers_) {
603 all_metrics["cpu_" + name] = throttler->get_metrics();
604 }
605
606 return all_metrics;
607 }
608
609private:
610 std::string name_;
611 std::unordered_map<std::string, std::unique_ptr<rate_limiter>> rate_limiters_;
612 std::unordered_map<std::string, std::unique_ptr<memory_quota_manager>> memory_quotas_;
613 std::unordered_map<std::string, std::unique_ptr<cpu_throttler>> cpu_throttlers_;
614 mutable std::mutex mutex_;
615};
