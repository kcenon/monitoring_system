117class json_alert_formatter : public alert_formatter {
118public:
119 std::string format(const alert& a) const override {
120 std::ostringstream oss;
121 oss << "{";
122 oss << "\"name\":\"" << escape_json(a.name) << "\",";
123 oss << "\"state\":\"" << alert_state_to_string(a.state) << "\",";
124 oss << "\"severity\":\"" << alert_severity_to_string(a.severity) << "\",";
125 oss << "\"value\":" << a.value << ",";
126 oss << "\"summary\":\"" << escape_json(a.annotations.summary) << "\",";
127 oss << "\"description\":\"" << escape_json(a.annotations.description) << "\",";
128 oss << "\"fingerprint\":\"" << a.fingerprint() << "\",";
129 oss << "\"labels\":{";
130 bool first = true;
131 for (const auto& [key, value] : a.labels.labels) {
132 if (!first) oss << ",";
133 oss << "\"" << escape_json(key) << "\":\"" << escape_json(value) << "\"";
134 first = false;
135 }
136 oss << "}}";
137 return oss.str();
138 }
139
140 std::string format_group(const alert_group& group) const override {
141 std::ostringstream oss;
142 oss << "{";
143 oss << "\"group_key\":\"" << escape_json(group.group_key) << "\",";
144 oss << "\"severity\":\"" << alert_severity_to_string(group.max_severity()) << "\",";
145 oss << "\"alert_count\":" << group.size() << ",";
146 oss << "\"alerts\":[";
147 bool first = true;
148 for (const auto& a : group.alerts) {
149 if (!first) oss << ",";
150 oss << format(a);
151 first = false;
152 }
153 oss << "]}";
154 return oss.str();
155 }
156
157private:
158 static std::string escape_json(const std::string& s) {
159 std::ostringstream oss;
160 for (char c : s) {
161 switch (c) {
162 case '"': oss << "\\\""; break;
163 case '\\': oss << "\\\\"; break;
164 case '\n': oss << "\\n"; break;
165 case '\r': oss << "\\r"; break;
166 case '\t': oss << "\\t"; break;
167 default: oss << c; break;
168 }
169 }
170 return oss.str();
171 }
172};

178class text_alert_formatter : public alert_formatter {
179public:
180 std::string format(const alert& a) const override {
181 std::ostringstream oss;
182 oss << "[" << alert_state_to_string(a.state) << "] "
183 << a.name << " (" << alert_severity_to_string(a.severity) << ")\n"
184 << " Summary: " << a.annotations.summary << "\n"
185 << " Value: " << a.value << "\n"
186 << " Fingerprint: " << a.fingerprint();
187 return oss.str();
188 }
189
190 std::string format_group(const alert_group& group) const override {
191 std::ostringstream oss;
192 oss << "Alert Group: " << group.group_key << "\n"
193 << " Total alerts: " << group.size() << "\n"
194 << " Max severity: " << alert_severity_to_string(group.max_severity()) << "\n"
195 << " Alerts:\n";
196 for (const auto& a : group.alerts) {
197 oss << " - " << a.name << " (" << alert_state_to_string(a.state) << ")\n";
198 }
199 return oss.str();
200 }

221class webhook_notifier : public alert_notifier {
222public:
223 using http_sender_func = std::function<common::VoidResult(
224 const std::string& url,
225 const std::string& method,
226 const std::unordered_map<std::string, std::string>& headers,
227 const std::string& body
228 )>;
229
235 explicit webhook_notifier(const webhook_config& config,
236 std::shared_ptr<alert_formatter> formatter = nullptr)
237 : config_(config)
238 , formatter_(formatter ? formatter : std::make_shared<json_alert_formatter>()) {}
239
240 std::string name() const override {
241 return "webhook:" + config_.url;
242 }
243
244 common::VoidResult notify(const alert& a) override {
245 if (!config_.send_resolved && a.state == alert_state::resolved) {
246 return common::ok();
247 }
248
249 std::string payload = formatter_->format(a);
250 return send_with_retry(payload);
251 }
252
253 common::VoidResult notify_group(const alert_group& group) override {
254 std::string payload = formatter_->format_group(group);
255 return send_with_retry(payload);
256 }
257
258 bool is_ready() const override {
259 return config_.validate() && http_sender_ != nullptr;
260 }
261
277 void set_http_sender(http_sender_func sender) {
278 http_sender_ = std::move(sender);
279 }
280
284 const webhook_config& config() const { return config_; }
285
286private:
287 common::VoidResult send_with_retry(const std::string& payload) {
288 if (!http_sender_) {
289 return common::VoidResult::err(error_info(monitoring_error_code::operation_failed, "No HTTP sender configured").to_common_error());
290 }
291
292 auto headers = config_.headers;
293 headers["Content-Type"] = config_.content_type;
294
295 for (size_t attempt = 0; attempt <= config_.max_retries; ++attempt) {
296 auto result = http_sender_(config_.url, config_.method, headers, payload);
297 if (result.is_ok()) {
298 return result;
299 }
300
301 if (attempt < config_.max_retries) {
302 std::this_thread::sleep_for(config_.retry_delay);
303 }
304 }
305
306 return common::VoidResult::err(static_cast<int>(monitoring_error_code::retry_attempts_exhausted),
307 "Failed to send webhook after " +
308 std::to_string(config_.max_retries) + " retries");
309 }
310
311 webhook_config config_;
312 std::shared_ptr<alert_formatter> formatter_;
313 http_sender_func http_sender_;
314};

351private:
352 common::VoidResult write_to_file(const std::string& content) {
353 std::lock_guard<std::mutex> lock(mutex_);
354
355 std::ofstream file(file_path_, std::ios::app);
356 if (!file) {
357 return common::VoidResult::err(static_cast<int>(monitoring_error_code::storage_write_failed),
358 "Failed to open file: " + file_path_);
359 }
360
361 auto now = std::chrono::system_clock::now();
362 auto time_t_now = std::chrono::system_clock::to_time_t(now);
363
364 file << "=== " << std::ctime(&time_t_now);
365 file << content << "\n\n";
366
367 return common::ok();
368 }
369
370 std::string file_path_;
371 std::shared_ptr<alert_formatter> formatter_;
372 std::mutex mutex_;
373};

382class multi_notifier : public alert_notifier {
383public:
387 explicit multi_notifier(std::string notifier_name)
388 : name_(std::move(notifier_name)) {}
389
393 void add_notifier(std::shared_ptr<alert_notifier> notifier) {
394 std::lock_guard<std::mutex> lock(mutex_);
395 notifiers_.push_back(std::move(notifier));
396 }
397
398 std::string name() const override { return name_; }
399
400 common::VoidResult notify(const alert& a) override {
401 std::lock_guard<std::mutex> lock(mutex_);
402
403 std::vector<std::string> failures;
404 for (const auto& notifier : notifiers_) {
405 if (notifier && notifier->is_ready()) {
406 auto result = notifier->notify(a);
407 if (!result.is_ok()) {
408 failures.push_back(notifier->name());
409 }
410 }
411 }
412
413 if (!failures.empty()) {
414 return common::VoidResult::err(static_cast<int>(monitoring_error_code::operation_failed),
415 "Failed notifiers: " + join_strings(failures, ", "));
416 }
417 return common::ok();
418 }
419
420 common::VoidResult notify_group(const alert_group& group) override {
421 std::lock_guard<std::mutex> lock(mutex_);
422
423 std::vector<std::string> failures;
424 for (const auto& notifier : notifiers_) {
425 if (notifier && notifier->is_ready()) {
426 auto result = notifier->notify_group(group);
427 if (!result.is_ok()) {
428 failures.push_back(notifier->name());
429 }
430 }
431 }
432
433 if (!failures.empty()) {
434 return common::VoidResult::err(static_cast<int>(monitoring_error_code::operation_failed),
435 "Failed notifiers: " + join_strings(failures, ", "));
436 }
437 return common::ok();
438 }
439
440 bool is_ready() const override {
441 std::lock_guard<std::mutex> lock(mutex_);
442 return !notifiers_.empty();
443 }
444
445private:
446 static std::string join_strings(const std::vector<std::string>& strings,
447 const std::string& delimiter) {
448 std::ostringstream oss;
449 for (size_t i = 0; i < strings.size(); ++i) {
450 if (i > 0) oss << delimiter;
451 oss << strings[i];
452 }
453 return oss.str();
454 }
455
456 std::string name_;
457 mutable std::mutex mutex_;
458 std::vector<std::shared_ptr<alert_notifier>> notifiers_;
459};

468class buffered_notifier : public alert_notifier {
469public:
476 buffered_notifier(std::shared_ptr<alert_notifier> inner,
477 size_t buffer_size = 100,
478 std::chrono::milliseconds flush_interval = std::chrono::seconds(30))
479 : inner_(std::move(inner))
480 , buffer_size_(buffer_size)
481 , flush_interval_(flush_interval)
482 , last_flush_(std::chrono::steady_clock::now()) {}
483
484 std::string name() const override {
485 return "buffered:" + (inner_ ? inner_->name() : "none");
486 }
487
488 common::VoidResult notify(const alert& a) override {
489 std::lock_guard<std::mutex> lock(mutex_);
490
491 buffer_.push_back(a);
492
493 if (should_flush()) {
494 return flush_internal();
495 }
496
497 return common::ok();
498 }
499
500 common::VoidResult notify_group(const alert_group& group) override {
501 std::lock_guard<std::mutex> lock(mutex_);
502
503 for (const auto& a : group.alerts) {
504 buffer_.push_back(a);
505 }
506
507 if (should_flush()) {
508 return flush_internal();
509 }
510
511 return common::ok();
512 }
513
514 bool is_ready() const override {
515 return inner_ && inner_->is_ready();
516 }
517
521 common::VoidResult flush() {
522 std::lock_guard<std::mutex> lock(mutex_);
523 return flush_internal();
524 }
525
529 size_t pending_count() const {
530 std::lock_guard<std::mutex> lock(mutex_);
531 return buffer_.size();
532 }
533
534private:
535 bool should_flush() const {
536 if (buffer_.size() >= buffer_size_) {
537 return true;
538 }
539 auto now = std::chrono::steady_clock::now();
540 return (now - last_flush_) >= flush_interval_;
541 }
542
543 common::VoidResult flush_internal() {
544 if (buffer_.empty() || !inner_) {
545 return common::ok();
546 }
547
548 // Create a group from buffered alerts
549 alert_group group("buffered");
550 for (auto& a : buffer_) {
551 group.add_alert(std::move(a));
552 }
553 buffer_.clear();
554
555 last_flush_ = std::chrono::steady_clock::now();
556
557 return inner_->notify_group(group);
558 }
559
560 std::shared_ptr<alert_notifier> inner_;
561 size_t buffer_size_;
562 std::chrono::milliseconds flush_interval_;
563
564 mutable std::mutex mutex_;
565 std::vector<alert> buffer_;
566 std::chrono::steady_clock::time_point last_flush_;
567};

576class routing_notifier : public alert_notifier {
577public:
578 using route_condition = std::function<bool(const alert&)>;
579
583 explicit routing_notifier(std::string notifier_name)
584 : name_(std::move(notifier_name)) {}
585
591 void add_route(route_condition condition,
592 std::shared_ptr<alert_notifier> notifier) {
593 std::lock_guard<std::mutex> lock(mutex_);
594 routes_.push_back({std::move(condition), std::move(notifier)});
595 }
596
600 void set_default_route(std::shared_ptr<alert_notifier> notifier) {
601 std::lock_guard<std::mutex> lock(mutex_);
602 default_route_ = std::move(notifier);
603 }
604
605 std::string name() const override { return name_; }
606
607 common::VoidResult notify(const alert& a) override {
608 std::lock_guard<std::mutex> lock(mutex_);
609
610 for (const auto& [condition, notifier] : routes_) {
611 if (condition(a) && notifier && notifier->is_ready()) {
612 return notifier->notify(a);
613 }
614 }
615
616 if (default_route_ && default_route_->is_ready()) {
617 return default_route_->notify(a);
618 }
619
620 return common::ok();
621 }
622
623 common::VoidResult notify_group(const alert_group& group) override {
624 // Route each alert individually
625 for (const auto& a : group.alerts) {
626 auto result = notify(a);
627 if (!result.is_ok()) {
628 return result;
629 }
630 }
631 return common::ok();
632 }
633
634 bool is_ready() const override {
635 std::lock_guard<std::mutex> lock(mutex_);
636 return !routes_.empty() || default_route_ != nullptr;
637 }
638
639 // Factory methods for common routes
643 void route_by_severity(alert_severity severity,
644 std::shared_ptr<alert_notifier> notifier) {
645 add_route(
646 [severity](const alert& a) { return a.severity == severity; },
647 std::move(notifier)
648 );
649 }
650
654 void route_by_label(const std::string& key,
655 const std::string& value,
656 std::shared_ptr<alert_notifier> notifier) {
657 add_route(
658 [key, value](const alert& a) { return a.labels.get(key) == value; },
659 std::move(notifier)
660 );
661 }
662
663private:
664 struct route {
665 route_condition condition;
666 std::shared_ptr<alert_notifier> notifier;
667 };
668
669 std::string name_;
670 mutable std::mutex mutex_;
671 std::vector<route> routes_;
672 std::shared_ptr<alert_notifier> default_route_;
673};
