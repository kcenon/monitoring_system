117class json_alert_formatter : public alert_formatter {
118public:
119 std::string format(const alert& a) const override {
120 std::ostringstream oss;
121 oss << "{";
122 oss << "\"name\":\"" << escape_json(a.name) << "\",";
123 oss << "\"state\":\"" << alert_state_to_string(a.state) << "\",";
124 oss << "\"severity\":\"" << alert_severity_to_string(a.severity) << "\",";
125 oss << "\"value\":" << a.value << ",";
126 oss << "\"summary\":\"" << escape_json(a.annotations.summary) << "\",";
127 oss << "\"description\":\"" << escape_json(a.annotations.description) << "\",";
128 oss << "\"fingerprint\":\"" << a.fingerprint() << "\",";
129 oss << "\"labels\":{";
130 bool first = true;
131 for (const auto& [key, value] : a.labels.labels) {
132 if (!first) oss << ",";
133 oss << "\"" << escape_json(key) << "\":\"" << escape_json(value) << "\"";
134 first = false;
135 }
136 oss << "}}";
137 return oss.str();
138 }
139
140 std::string format_group(const alert_group& group) const override {
141 std::ostringstream oss;
142 oss << "{";
143 oss << "\"group_key\":\"" << escape_json(group.group_key) << "\",";
144 oss << "\"severity\":\"" << alert_severity_to_string(group.max_severity()) << "\",";
145 oss << "\"alert_count\":" << group.size() << ",";
146 oss << "\"alerts\":[";
147 bool first = true;
148 for (const auto& a : group.alerts) {
149 if (!first) oss << ",";
150 oss << format(a);
151 first = false;
152 }
153 oss << "]}";
154 return oss.str();
155 }
156
157private:
158 static std::string escape_json(const std::string& s) {
159 std::ostringstream oss;
160 for (char c : s) {
161 switch (c) {
162 case '"': oss << "\\\""; break;
163 case '\\': oss << "\\\\"; break;
164 case '\n': oss << "\\n"; break;
165 case '\r': oss << "\\r"; break;
166 case '\t': oss << "\\t"; break;
167 default: oss << c; break;
168 }
169 }
170 return oss.str();
171 }
172};

178class text_alert_formatter : public alert_formatter {
179public:
180 std::string format(const alert& a) const override {
181 std::ostringstream oss;
182 oss << "[" << alert_state_to_string(a.state) << "] "
183 << a.name << " (" << alert_severity_to_string(a.severity) << ")\n"
184 << " Summary: " << a.annotations.summary << "\n"
185 << " Value: " << a.value << "\n"
186 << " Fingerprint: " << a.fingerprint();
187 return oss.str();
188 }
189
190 std::string format_group(const alert_group& group) const override {
191 std::ostringstream oss;
192 oss << "Alert Group: " << group.group_key << "\n"
193 << " Total alerts: " << group.size() << "\n"
194 << " Max severity: " << alert_severity_to_string(group.max_severity()) << "\n"
195 << " Alerts:\n";
196 for (const auto& a : group.alerts) {
197 oss << " - " << a.name << " (" << alert_state_to_string(a.state) << ")\n";
198 }
199 return oss.str();
200 }

221class webhook_notifier : public alert_notifier {
222public:
223 using http_sender_func = std::function<result_void(
224 const std::string& url,
225 const std::string& method,
226 const std::unordered_map<std::string, std::string>& headers,
227 const std::string& body
228 )>;
229
235 explicit webhook_notifier(const webhook_config& config,
236 std::shared_ptr<alert_formatter> formatter = nullptr)
237 : config_(config)
238 , formatter_(formatter ? formatter : std::make_shared<json_alert_formatter>()) {}
239
240 std::string name() const override {
241 return "webhook:" + config_.url;
242 }
243
244 result_void notify(const alert& a) override {
245 if (!config_.send_resolved && a.state == alert_state::resolved) {
246 return make_void_success();
247 }
248
249 std::string payload = formatter_->format(a);
250 return send_with_retry(payload);
251 }
252
253 result_void notify_group(const alert_group& group) override {
254 std::string payload = formatter_->format_group(group);
255 return send_with_retry(payload);
256 }
257
258 bool is_ready() const override {
259 return config_.validate() && http_sender_ != nullptr;
260 }
261
277 void set_http_sender(http_sender_func sender) {
278 http_sender_ = std::move(sender);
279 }
280
284 const webhook_config& config() const { return config_; }
285
286private:
287 result_void send_with_retry(const std::string& payload) {
288 if (!http_sender_) {
289 return make_void_error(monitoring_error_code::operation_failed,
290 "No HTTP sender configured");
291 }
292
293 auto headers = config_.headers;
294 headers["Content-Type"] = config_.content_type;
295
296 for (size_t attempt = 0; attempt <= config_.max_retries; ++attempt) {
297 auto result = http_sender_(config_.url, config_.method, headers, payload);
298 if (result.is_ok()) {
299 return result;
300 }
301
302 if (attempt < config_.max_retries) {
303 std::this_thread::sleep_for(config_.retry_delay);
304 }
305 }
306
307 return make_void_error(monitoring_error_code::retry_attempts_exhausted,
308 "Failed to send webhook after " +
309 std::to_string(config_.max_retries) + " retries");
310 }
311
312 webhook_config config_;
313 std::shared_ptr<alert_formatter> formatter_;
314 http_sender_func http_sender_;
315};

353 result_void write_to_file(const std::string& content) {
354 std::lock_guard<std::mutex> lock(mutex_);
355
356 std::ofstream file(file_path_, std::ios::app);
357 if (!file) {
358 return make_void_error(monitoring_error_code::storage_write_failed,
359 "Failed to open file: " + file_path_);
360 }
361
362 auto now = std::chrono::system_clock::now();
363 auto time_t_now = std::chrono::system_clock::to_time_t(now);
364
365 file << "=== " << std::ctime(&time_t_now);
366 file << content << "\n\n";
367
368 return make_void_success();
369 }
370
371 std::string file_path_;
372 std::shared_ptr<alert_formatter> formatter_;
373 std::mutex mutex_;

383class multi_notifier : public alert_notifier {
384public:
388 explicit multi_notifier(std::string notifier_name)
389 : name_(std::move(notifier_name)) {}
390
394 void add_notifier(std::shared_ptr<alert_notifier> notifier) {
395 std::lock_guard<std::mutex> lock(mutex_);
396 notifiers_.push_back(std::move(notifier));
397 }
398
399 std::string name() const override { return name_; }
400
401 result_void notify(const alert& a) override {
402 std::lock_guard<std::mutex> lock(mutex_);
403
404 std::vector<std::string> failures;
405 for (const auto& notifier : notifiers_) {
406 if (notifier && notifier->is_ready()) {
407 auto result = notifier->notify(a);
408 if (!result.is_ok()) {
409 failures.push_back(notifier->name());
410 }
411 }
412 }
413
414 if (!failures.empty()) {
415 return make_void_error(monitoring_error_code::operation_failed,
416 "Failed notifiers: " + join_strings(failures, ", "));
417 }
418 return make_void_success();
419 }
420
421 result_void notify_group(const alert_group& group) override {
422 std::lock_guard<std::mutex> lock(mutex_);
423
424 std::vector<std::string> failures;
425 for (const auto& notifier : notifiers_) {
426 if (notifier && notifier->is_ready()) {
427 auto result = notifier->notify_group(group);
428 if (!result.is_ok()) {
429 failures.push_back(notifier->name());
430 }
431 }
432 }
433
434 if (!failures.empty()) {
435 return make_void_error(monitoring_error_code::operation_failed,
436 "Failed notifiers: " + join_strings(failures, ", "));
437 }
438 return make_void_success();
439 }
440
441 bool is_ready() const override {
442 std::lock_guard<std::mutex> lock(mutex_);
443 return !notifiers_.empty();
444 }
445
446private:
447 static std::string join_strings(const std::vector<std::string>& strings,
448 const std::string& delimiter) {
449 std::ostringstream oss;
450 for (size_t i = 0; i < strings.size(); ++i) {
451 if (i > 0) oss << delimiter;
452 oss << strings[i];
453 }
454 return oss.str();
455 }
456
457 std::string name_;
458 mutable std::mutex mutex_;
459 std::vector<std::shared_ptr<alert_notifier>> notifiers_;
460};

469class buffered_notifier : public alert_notifier {
470public:
477 buffered_notifier(std::shared_ptr<alert_notifier> inner,
478 size_t buffer_size = 100,
479 std::chrono::milliseconds flush_interval = std::chrono::seconds(30))
480 : inner_(std::move(inner))
481 , buffer_size_(buffer_size)
482 , flush_interval_(flush_interval)
483 , last_flush_(std::chrono::steady_clock::now()) {}
484
485 std::string name() const override {
486 return "buffered:" + (inner_ ? inner_->name() : "none");
487 }
488
489 result_void notify(const alert& a) override {
490 std::lock_guard<std::mutex> lock(mutex_);
491
492 buffer_.push_back(a);
493
494 if (should_flush()) {
495 return flush_internal();
496 }
497
498 return make_void_success();
499 }
500
501 result_void notify_group(const alert_group& group) override {
502 std::lock_guard<std::mutex> lock(mutex_);
503
504 for (const auto& a : group.alerts) {
505 buffer_.push_back(a);
506 }
507
508 if (should_flush()) {
509 return flush_internal();
510 }
511
512 return make_void_success();
513 }
514
515 bool is_ready() const override {
516 return inner_ && inner_->is_ready();
517 }
518
522 result_void flush() {
523 std::lock_guard<std::mutex> lock(mutex_);
524 return flush_internal();
525 }
526
530 size_t pending_count() const {
531 std::lock_guard<std::mutex> lock(mutex_);
532 return buffer_.size();
533 }
534
535private:
536 bool should_flush() const {
537 if (buffer_.size() >= buffer_size_) {
538 return true;
539 }
540 auto now = std::chrono::steady_clock::now();
541 return (now - last_flush_) >= flush_interval_;
542 }
543
544 result_void flush_internal() {
545 if (buffer_.empty() || !inner_) {
546 return make_void_success();
547 }
548
549 // Create a group from buffered alerts
550 alert_group group("buffered");
551 for (auto& a : buffer_) {
552 group.add_alert(std::move(a));
553 }
554 buffer_.clear();
555
556 last_flush_ = std::chrono::steady_clock::now();
557
558 return inner_->notify_group(group);
559 }
560
561 std::shared_ptr<alert_notifier> inner_;
562 size_t buffer_size_;
563 std::chrono::milliseconds flush_interval_;
564
565 mutable std::mutex mutex_;
566 std::vector<alert> buffer_;
567 std::chrono::steady_clock::time_point last_flush_;
568};

577class routing_notifier : public alert_notifier {
578public:
579 using route_condition = std::function<bool(const alert&)>;
580
584 explicit routing_notifier(std::string notifier_name)
585 : name_(std::move(notifier_name)) {}
586
592 void add_route(route_condition condition,
593 std::shared_ptr<alert_notifier> notifier) {
594 std::lock_guard<std::mutex> lock(mutex_);
595 routes_.push_back({std::move(condition), std::move(notifier)});
596 }
597
601 void set_default_route(std::shared_ptr<alert_notifier> notifier) {
602 std::lock_guard<std::mutex> lock(mutex_);
603 default_route_ = std::move(notifier);
604 }
605
606 std::string name() const override { return name_; }
607
608 result_void notify(const alert& a) override {
609 std::lock_guard<std::mutex> lock(mutex_);
610
611 for (const auto& [condition, notifier] : routes_) {
612 if (condition(a) && notifier && notifier->is_ready()) {
613 return notifier->notify(a);
614 }
615 }
616
617 if (default_route_ && default_route_->is_ready()) {
618 return default_route_->notify(a);
619 }
620
621 return make_void_success();
622 }
623
624 result_void notify_group(const alert_group& group) override {
625 // Route each alert individually
626 for (const auto& a : group.alerts) {
627 auto result = notify(a);
628 if (!result.is_ok()) {
629 return result;
630 }
631 }
632 return make_void_success();
633 }
634
635 bool is_ready() const override {
636 std::lock_guard<std::mutex> lock(mutex_);
637 return !routes_.empty() || default_route_ != nullptr;
638 }
639
640 // Factory methods for common routes
644 void route_by_severity(alert_severity severity,
645 std::shared_ptr<alert_notifier> notifier) {
646 add_route(
647 [severity](const alert& a) { return a.severity == severity; },
648 std::move(notifier)
649 );
650 }
651
655 void route_by_label(const std::string& key,
656 const std::string& value,
657 std::shared_ptr<alert_notifier> notifier) {
658 add_route(
659 [key, value](const alert& a) { return a.labels.get(key) == value; },
660 std::move(notifier)
661 );
662 }
663
664private:
665 struct route {
666 route_condition condition;
667 std::shared_ptr<alert_notifier> notifier;
668 };
669
670 std::string name_;
671 mutable std::mutex mutex_;
672 std::vector<route> routes_;
673 std::shared_ptr<alert_notifier> default_route_;
674};
