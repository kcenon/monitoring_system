76struct lockfree_queue_statistics {
77 std::atomic<size_t> push_attempts{0};
78 std::atomic<size_t> push_successes{0};
79 std::atomic<size_t> push_failures{0};
80 std::atomic<size_t> pop_attempts{0};
81 std::atomic<size_t> pop_successes{0};
82 std::atomic<size_t> pop_failures{0};
83
84 lockfree_queue_statistics() = default;
85 lockfree_queue_statistics(const lockfree_queue_statistics& other)
86 : push_attempts(other.push_attempts.load())
87 , push_successes(other.push_successes.load())
88 , push_failures(other.push_failures.load())
89 , pop_attempts(other.pop_attempts.load())
90 , pop_successes(other.pop_successes.load())
91 , pop_failures(other.pop_failures.load()) {}
92
97 double get_push_success_rate() const {
98 auto attempts = push_attempts.load();
99 if (attempts == 0) {
100 return 100.0;
101 }
102 return (static_cast<double>(push_successes.load()) / static_cast<double>(attempts)) * 100.0;
103 }
104
109 double get_pop_success_rate() const {
110 auto attempts = pop_attempts.load();
111 if (attempts == 0) {
112 return 100.0;
113 }
114 return (static_cast<double>(pop_successes.load()) / static_cast<double>(attempts)) * 100.0;
115 }
116
120 void reset() {
121 push_attempts.store(0);
122 push_successes.store(0);
123 push_failures.store(0);
124 pop_attempts.store(0);
125 pop_successes.store(0);
126 pop_failures.store(0);
127 }
128};

138template<typename T>
139class lockfree_queue {
140public:
144 lockfree_queue() : lockfree_queue(lockfree_queue_config{}) {}
145
150 explicit lockfree_queue(const lockfree_queue_config& config)
151 : config_(config)
152 , capacity_(config.initial_capacity)
153 , buffer_(config.initial_capacity)
154 , head_(0)
155 , tail_(0)
156 , size_(0) {
157 // Initialize each slot's sequence to its index
158 for (size_t i = 0; i < capacity_; ++i) {
159 buffer_[i].sequence.store(i, std::memory_order_relaxed);
160 }
161 }
162
163 // Disable copy
164 lockfree_queue(const lockfree_queue&) = delete;
165 lockfree_queue& operator=(const lockfree_queue&) = delete;
166
167 // Enable move
168 lockfree_queue(lockfree_queue&& other) noexcept
169 : config_(std::move(other.config_))
170 , capacity_(other.capacity_)
171 , buffer_(std::move(other.buffer_))
172 , head_(other.head_.load())
173 , tail_(other.tail_.load())
174 , size_(other.size_.load())
175 , stats_(std::move(other.stats_)) {}

177 lockfree_queue& operator=(lockfree_queue&& other) noexcept {
178 if (this != &other) {
179 config_ = std::move(other.config_);
180 capacity_ = other.capacity_;
181 buffer_ = std::move(other.buffer_);
182 head_.store(other.head_.load());
183 tail_.store(other.tail_.load());

186 }

204 common::Result<bool> push(T&& value) {

212 common::Result<T> pop() {
213 stats_.pop_attempts++;
214
215 size_t current_head = head_.load(std::memory_order_relaxed);
216
217 while (true) {
218 size_t index = current_head % capacity_;
219 auto& slot = buffer_[index];
220 size_t seq = slot.sequence.load(std::memory_order_acquire);
221 intptr_t diff = static_cast<intptr_t>(seq) - static_cast<intptr_t>(current_head + 1);
222
223 if (diff == 0) {
224 // Slot is ready to be read
225 if (head_.compare_exchange_weak(current_head, current_head + 1,
226 std::memory_order_relaxed)) {
227 T value = std::move(slot.data);
228 slot.sequence.store(current_head + capacity_, std::memory_order_release);
229 size_.fetch_sub(1, std::memory_order_relaxed);
230 stats_.pop_successes++;
231 return common::ok(std::move(value));
232 }
233 } else if (diff < 0) {
234 // Queue is empty
235 stats_.pop_failures++;
236 return common::Result<T>::err(error_info(monitoring_error_code::resource_unavailable, "Queue is empty").to_common_error());
237 } else {
238 // Another thread is modifying, retry
239 current_head = head_.load(std::memory_order_relaxed);
240 }
241 }
242 }

289 template<typename U>
290 common::Result<bool> push_impl(U&& value) {
291 stats_.push_attempts++;
292
293 size_t current_tail = tail_.load(std::memory_order_relaxed);
294
295 while (true) {
296 size_t index = current_tail % capacity_;
297 auto& slot = buffer_[index];
298 size_t seq = slot.sequence.load(std::memory_order_acquire);
299 intptr_t diff = static_cast<intptr_t>(seq) - static_cast<intptr_t>(current_tail);
300
301 if (diff == 0) {
302 // Slot is available for writing
303 if (tail_.compare_exchange_weak(current_tail, current_tail + 1,
304 std::memory_order_relaxed)) {
305 slot.data = std::forward<U>(value);
306 slot.sequence.store(current_tail + 1, std::memory_order_release);
307 size_.fetch_add(1, std::memory_order_relaxed);
308 stats_.push_successes++;
309 return common::ok(true);
310 }
311 } else if (diff < 0) {
312 // Queue is full
313 stats_.push_failures++;
314 return common::ok(false);
315 } else {
316 // Another thread is modifying, retry
317 current_tail = tail_.load(std::memory_order_relaxed);
318 }
319 }
320 }
