47 // Hash function for named_key
48 struct named_key_hash {
49 std::size_t operator()(const named_key& key) const {
50 auto h1 = std::hash<std::type_index>{}(key.first);
51 auto h2 = std::hash<std::string>{}(key.second);
52 return h1 ^ (h2 << 1);
53 }

205 result<std::any> resolve_impl(std::type_index type) override {
206 std::lock_guard<std::mutex> lock(mutex_);
207
208 // Check current container
209 auto it = type_registrations_.find(type);
210 if (it != type_registrations_.end()) {
211 return resolve_registration(it->second);
212 }
213
214 // Check parent container for scoped resolution
215 if (parent_container_) {
216 return parent_container_->resolve_impl(type);
217 }
218
219 return make_error<std::any>(
220 monitoring_error_code::collector_not_found,
221 "Service type not registered"
222 );
223 }

228 result<std::any> resolve_named_impl(

232 std::lock_guard<std::mutex> lock(mutex_);
233
234 named_key key{type, name};
235 auto it = named_registrations_.find(key);
236 if (it != named_registrations_.end()) {

239

254 bool is_registered_impl(std::type_index type) const override {

282private:
288 result<std::any> resolve_registration(service_registration& reg) {
289 switch (reg.lifetime) {
290 case service_lifetime::transient:
291 // Always create new instance
292 if (reg.factory) {
293 return make_success(reg.factory());
294 }
295 return make_error<std::any>(
296 monitoring_error_code::invalid_configuration,
297 "No factory for transient service"
298 );
299
300 case service_lifetime::scoped:
301 case service_lifetime::singleton:
302 // Return cached instance or create new one
303 if (reg.cached_instance.has_value()) {
304 return make_success(reg.cached_instance);
305 }
306 if (reg.factory) {
307 reg.cached_instance = reg.factory();
308 return make_success(reg.cached_instance);
309 }
310 return make_error<std::any>(
311 monitoring_error_code::invalid_configuration,
312 "No factory or instance for service"
313 );
314
315 default:
316 return make_error<std::any>(
317 monitoring_error_code::unknown_error,
318 "Unknown service lifetime"
319 );
320 }
321 }
