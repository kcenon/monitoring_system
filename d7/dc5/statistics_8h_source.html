123template <typename T>
124T divide(const T& value, size_t count) {
125 if (count == 0) {
126 return zero_value<T>();
127 }
128 if constexpr (is_chrono_duration_v<T>) {
129 return value / static_cast<typename T::rep>(count);
130 } else {
131 return value / static_cast<T>(count);
132 }
133}

154template <typename T>
155T percentile(const std::vector<T>& sorted_values, double percentile_value) {
156 if (sorted_values.empty()) {
157 return detail::zero_value<T>();
158 }
159
160 if (percentile_value <= 0.0) {
161 return sorted_values.front();
162 }
163
164 if (percentile_value >= 100.0) {
165 return sorted_values.back();
166 }
167
168 double rank = (percentile_value / 100.0) * (sorted_values.size() - 1);
169 size_t lower_idx = static_cast<size_t>(rank);
170 size_t upper_idx = lower_idx + 1;
171
172 if (upper_idx >= sorted_values.size()) {
173 return sorted_values[lower_idx];
174 }
175
176 // For chrono types, use simple index-based selection
177 // For numeric types, use linear interpolation
178 if constexpr (detail::is_chrono_duration_v<T>) {
179 // Use nearest-rank method for duration types
180 return sorted_values[static_cast<size_t>(std::round(rank))];
181 } else {
182 double fraction = rank - static_cast<double>(lower_idx);
183 return sorted_values[lower_idx] +
184 static_cast<T>(fraction * (sorted_values[upper_idx] - sorted_values[lower_idx]));
185 }
186}

197template <typename T>
198statistics<T> compute_sorted(const std::vector<T>& sorted_values) {
199 statistics<T> result{};
200
201 if (sorted_values.empty()) {
202 result.min = detail::zero_value<T>();
203 result.max = detail::zero_value<T>();
204 result.mean = detail::zero_value<T>();
205 result.median = detail::zero_value<T>();
206 result.p95 = detail::zero_value<T>();
207 result.p99 = detail::zero_value<T>();
208 result.total = detail::zero_value<T>();
209 result.count = 0;
210 return result;
211 }
212
213 result.count = sorted_values.size();
214 result.min = sorted_values.front();
215 result.max = sorted_values.back();
216
217 // Calculate total
218 result.total = std::accumulate(sorted_values.begin(), sorted_values.end(),
219 detail::zero_value<T>());
220
221 // Calculate mean
222 result.mean = detail::divide(result.total, result.count);
223
224 // Calculate percentiles
225 result.median = percentile(sorted_values, 50.0);
226 result.p95 = percentile(sorted_values, 95.0);
227 result.p99 = percentile(sorted_values, 99.0);
228
229 return result;
230}

252template <typename T>
253statistics<T> compute(const std::vector<T>& values) {
254 if (values.empty()) {
255 return compute_sorted<T>({});
256 }
257
258 std::vector<T> sorted = values;
259 std::sort(sorted.begin(), sorted.end());
260 return compute_sorted(sorted);
261}

272template <typename T>
273statistics<T> compute_inplace(std::vector<T>& values) {
274 if (values.empty()) {
275 return compute_sorted<T>({});
