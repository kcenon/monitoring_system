61class console_logger : public ILogger, public IMonitorable {
62private:
63 std::atomic<int> message_count_{0};
64 log_level min_level_{log_level::info};
65 std::shared_ptr<IMonitor> monitor_;
66
67public:
68 common::VoidResult log(log_level level, const std::string& message) override {
69 if (level < min_level_) {
70 return std::monostate{};
71 }
72
73 message_count_++;
74
75 auto now = std::chrono::system_clock::now();
76 auto time_t = std::chrono::system_clock::to_time_t(now);
77 auto tm = *std::localtime(&time_t);
78
79 std::cout << "[" << std::put_time(&tm, "%Y-%m-%d %H:%M:%S") << "] "
80 << "[" << to_string(level) << "] "
81 << message << std::endl;
82
83 // If monitor is injected, record logging metrics
84 if (monitor_) {
85 monitor_->record_metric("messages_logged", message_count_.load());
86 }
87
88 return std::monostate{};
89 }
90
91 common::VoidResult log(log_level level, const std::string& message,
92 const std::string& file, int line,
93 const std::string& function) override {
94 std::string detailed_message = message + " [" + file + ":" +
95 std::to_string(line) + " in " + function + "]";
96 return log(level, detailed_message);
97 }
98
99 common::VoidResult log(const log_entry& entry) override {
100 return log(entry.level, entry.message, entry.file, entry.line, entry.function);
101 }
102
103 bool is_enabled(log_level level) const override {
104 return level >= min_level_;
105 }
106
107 common::VoidResult set_level(log_level level) override {
108 min_level_ = level;
109 return std::monostate{};
110 }
111
112 log_level get_level() const override {
113 return min_level_;
114 }
115
116 common::VoidResult flush() override {
117 std::cout.flush();
118 return std::monostate{};
119 }
120
121 // IMonitorable implementation
122 common::Result<metrics_snapshot> get_monitoring_data() override {
123 metrics_snapshot snapshot;
124 snapshot.source_id = "console_logger";
125 snapshot.add_metric("total_messages", message_count_.load());
126 snapshot.add_metric("is_enabled", is_enabled(log_level::info) ? 1.0 : 0.0);
127 return snapshot;
128 }
129
130 common::Result<health_check_result> health_check() override {
131 health_check_result result;
132 result.status = health_status::healthy;
133 result.message = "Console logger operational";
134 result.metadata["message_count"] = std::to_string(message_count_.load());
135 return result;
136 }
137
138 std::string get_component_name() const override {
139 return "console_logger";
140 }
141
142 // Allow monitor injection
143 void set_monitor(std::shared_ptr<IMonitor> monitor) {
144 monitor_ = std::move(monitor);
145 }
146
147 int get_message_count() const {
148 return message_count_.load();
149 }
150};

177 std::cout << "Logger receives monitor via DI for metrics collection.\n" << std::endl;
178
179 auto monitor = std::make_shared<performance_monitor>();
180 auto logger = std::make_shared<console_logger>();
181
182 // Inject monitor into logger
183 logger->set_monitor(monitor);
184
185 // Log some messages
186 logger->log(log_level::info, "First message with monitoring");
187 logger->log(log_level::warning, "Second message with monitoring");
188 logger->log(log_level::error, "Third message with monitoring");
189
190 // Check monitor collected metrics
191 auto metrics_result = monitor->get_metrics();
192 if (std::holds_alternative<metrics_snapshot>(metrics_result)) {
193 auto& snapshot = std::get<metrics_snapshot>(metrics_result);
194 std::cout << "\n‚úì Monitor collected " << snapshot.metrics.size()
195 << " metrics from logger" << std::endl;
196
197 for (const auto& metric : snapshot.metrics) {
198 std::cout << " - " << metric.name << ": " << metric.value << std::endl;
199 }
200 }

231void demo_bidirectional_integration() {
232 std::cout << "\n\n=== Scenario 4: Bidirectional DI (No Circular Dependency!) ===" << std::endl;
233 std::cout << "Both systems integrated at RUNTIME without compile-time circular dependency.\n" << std::endl;
234
235 // Create both systems
236 auto logger = std::make_shared<console_logger>();
237 auto monitor = std::make_shared<performance_monitor>();
238
239 // Bidirectional injection
240 logger->set_monitor(monitor);
241 // Note: In real logger_system, logger would also be injected into adapter
242
243 std::cout << "\n‚úì Bidirectional dependency injection complete" << std::endl;
244 std::cout << " Logger -> uses Monitor for metrics" << std::endl;
245 std::cout << " Monitor <- logs status via Logger\n" << std::endl;
246
247 // Simulate application workload
248 std::cout << "\nSimulating application workload..." << std::endl;
249 for (int i = 0; i < 10; ++i) {
250 // Log activity
251 logger->log(log_level::info, "Processing request " + std::to_string(i));
252
253 // Record performance metrics
254 monitor->record_metric("requests_processed", i + 1);
255 monitor->record_metric("response_time_ms", 50.0 + (i * 5));
256
257 std::this_thread::sleep_for(std::chrono::milliseconds(10));
258 }
259
260 // Check both systems' health
261 std::cout << "\n=== System Health Check ===" << std::endl;
262
263 auto logger_health = logger->health_check();
264 if (std::holds_alternative<health_check_result>(logger_health)) {
265 auto& health = std::get<health_check_result>(logger_health);
266 std::cout << "Logger Status: " << to_string(health.status)
267 << " - " << health.message << std::endl;
268 std::cout << " Messages logged: " << logger->get_message_count() << std::endl;
269 }
270
271 auto monitor_health = monitor->check_health();
272 if (std::holds_alternative<health_check_result>(monitor_health)) {
273 auto& health = std::get<health_check_result>(monitor_health);
274 std::cout << "Monitor Status: " << to_string(health.status)
275 << " - " << health.message << std::endl;
276 }
277
278 // Get comprehensive metrics
279 std::cout << "\n=== Collected Metrics ===" << std::endl;
280
281 auto logger_metrics = logger->get_monitoring_data();
282 if (std::holds_alternative<metrics_snapshot>(logger_metrics)) {
283 auto& snapshot = std::get<metrics_snapshot>(logger_metrics);
284 std::cout << "Logger Metrics:" << std::endl;
285 for (const auto& metric : snapshot.metrics) {
286 std::cout << " " << metric.name << ": " << metric.value << std::endl;
287 }
288 }
289
290 auto monitor_metrics = monitor->get_metrics();
291 if (std::holds_alternative<metrics_snapshot>(monitor_metrics)) {
292 auto& snapshot = std::get<metrics_snapshot>(monitor_metrics);
293 std::cout << "\nMonitor Metrics:" << std::endl;
294 for (const auto& metric : snapshot.metrics) {
295 std::cout << " " << metric.name << ": " << metric.value << std::endl;
296 }
297 }
298
299 std::cout << "\n‚úì Both systems fully operational and integrated!" << std::endl;
300}
