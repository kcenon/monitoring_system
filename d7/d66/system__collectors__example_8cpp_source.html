66void display_system_metrics(const system_resources& resources) {
67 std::cout << "\n=== System Resource Metrics ===" << std::endl;
68
69 // CPU metrics
70 std::cout << "CPU:" << std::endl;
71 std::cout << " Usage: " << std::fixed << std::setprecision(2)
72 << resources.cpu.usage_percent << "%" << std::endl;
73 std::cout << " User: " << resources.cpu.user_percent << "%" << std::endl;
74 std::cout << " System: " << resources.cpu.system_percent << "%" << std::endl;
75 std::cout << " Idle: " << resources.cpu.idle_percent << "%" << std::endl;
76 std::cout << " Core Count: " << resources.cpu.count << std::endl;
77 std::cout << " Load Average: "
78 << resources.cpu.load.one_min << " (1m), "
79 << resources.cpu.load.five_min << " (5m), "
80 << resources.cpu.load.fifteen_min << " (15m)" << std::endl;
81
82 // Memory metrics
83 std::cout << "\nMemory:" << std::endl;
84 std::cout << " Total: " << (resources.memory.total_bytes / (1024.0 * 1024.0 * 1024.0))
85 << " GB" << std::endl;
86 std::cout << " Used: " << (resources.memory.used_bytes / (1024.0 * 1024.0 * 1024.0))
87 << " GB (" << resources.memory.usage_percent << "%)" << std::endl;
88 std::cout << " Available: " << (resources.memory.available_bytes / (1024.0 * 1024.0 * 1024.0))
89 << " GB" << std::endl;
90 std::cout << " Swap Used: " << (resources.memory.swap.used_bytes / (1024.0 * 1024.0 * 1024.0))
91 << " GB (" << resources.memory.swap.usage_percent << "%)" << std::endl;
92
93 // Disk metrics
94 std::cout << "\nDisk:" << std::endl;
95 std::cout << " Total: " << (resources.disk.total_bytes / (1024.0 * 1024.0 * 1024.0))
96 << " GB" << std::endl;
97 std::cout << " Used: " << (resources.disk.used_bytes / (1024.0 * 1024.0 * 1024.0))
98 << " GB (" << resources.disk.usage_percent << "%)" << std::endl;
99 std::cout << " I/O Read: " << (resources.disk.io.read_bytes_per_sec / (1024.0 * 1024.0))
100 << " MB/s (" << resources.disk.io.read_ops_per_sec << " ops/s)" << std::endl;
101 std::cout << " I/O Write: " << (resources.disk.io.write_bytes_per_sec / (1024.0 * 1024.0))
102 << " MB/s (" << resources.disk.io.write_ops_per_sec << " ops/s)" << std::endl;
103
104 // Network metrics
105 std::cout << "\nNetwork:" << std::endl;
106 std::cout << " RX: " << (resources.network.rx_bytes_per_sec / (1024.0 * 1024.0))
107 << " MB/s (" << resources.network.rx_packets_per_sec << " packets/s)" << std::endl;
108 std::cout << " TX: " << (resources.network.tx_bytes_per_sec / (1024.0 * 1024.0))
109 << " MB/s (" << resources.network.tx_packets_per_sec << " packets/s)" << std::endl;
110 std::cout << " Errors: RX=" << resources.network.rx_errors
111 << ", TX=" << resources.network.tx_errors << std::endl;
112 std::cout << " Dropped: RX=" << resources.network.rx_dropped
113 << ", TX=" << resources.network.tx_dropped << std::endl;
114
115 // Process metrics
116 std::cout << "\nProcess:" << std::endl;
117 std::cout << " Count: " << resources.process.count << std::endl;
118 std::cout << " Threads: " << resources.process.thread_count << std::endl;
119 std::cout << " Handles: " << resources.process.handle_count << std::endl;
120 std::cout << " Open FDs: " << resources.process.open_file_descriptors << std::endl;
121
122 // Context switches
123 std::cout << "\nContext Switches:" << std::endl;
124 std::cout << " Total: " << resources.context_switches.total << std::endl;
125 std::cout << " Per Second: " << resources.context_switches.per_sec << std::endl;
126 std::cout << " Voluntary: " << resources.context_switches.voluntary << std::endl;
127 std::cout << " Non-voluntary: " << resources.context_switches.nonvoluntary << std::endl;
128}

133void display_network_collector_metrics(const std::vector<metric>& metrics) {
134 std::cout << "\n=== Network Collector Metrics ===" << std::endl;
135
136 for (const auto& m : metrics) {
137 std::cout << " " << m.name << ": ";
138 std::visit([](const auto& val) { std::cout << val; }, m.value);
139 auto unit_it = m.tags.find("unit");
140 if (unit_it != m.tags.end() && !unit_it->second.empty()) {
141 std::cout << " " << unit_it->second;
142 }
143 std::cout << std::endl;
144 }
145}

165 std::cout << "=== System Collectors Example ===" << std::endl;

167 try {
168 // Step 1: Create and configure system_resource_collector
169 std::cout << "\n1. Creating system_resource_collector..." << std::endl;
170
171 system_metrics_config sys_config;
172 sys_config.collect_cpu = true;
173 sys_config.collect_memory = true;
174 sys_config.collect_disk = true;
175 sys_config.collect_network = true;
176 sys_config.collect_process = true;
177 sys_config.enable_load_history = true;
178 sys_config.load_history_max_samples = 100;
179 sys_config.interval = std::chrono::milliseconds(1000);
180
181 system_resource_collector sys_collector(sys_config);
182
183 // Initialize the collector
184 std::unordered_map<std::string, std::string> init_config;
185 if (!sys_collector.initialize(init_config)) {
186 std::cerr << "Failed to initialize system_resource_collector" << std::endl;
187 return 1;
188 }
189
190 std::cout << " Initialized: " << sys_collector.get_name() << std::endl;
191 std::cout << " Health: " << (sys_collector.is_healthy() ? "OK" : "UNHEALTHY") << std::endl;
192
193 // Step 2: Create network_metrics_collector
194 std::cout << "\n2. Creating network_metrics_collector..." << std::endl;
195
196 network_metrics_collector net_collector;
197 if (!net_collector.initialize(init_config)) {
198 std::cerr << "Failed to initialize network_metrics_collector" << std::endl;
199 return 1;
200 }
201
202 std::cout << " Initialized: " << net_collector.get_name() << std::endl;
203 std::cout << " Health: " << (net_collector.is_healthy() ? "OK" : "UNHEALTHY") << std::endl;
204
205 // Step 3: Create process_metrics_collector
206 std::cout << "\n3. Creating process_metrics_collector..." << std::endl;
207
208 process_metrics_collector proc_collector;
209 if (!proc_collector.initialize(init_config)) {
210 std::cerr << "Failed to initialize process_metrics_collector" << std::endl;
211 return 1;
212 }
213
214 std::cout << " Initialized: " << proc_collector.get_name() << std::endl;
215 std::cout << " Health: " << (proc_collector.is_healthy() ? "OK" : "UNHEALTHY") << std::endl;
216
217#ifdef THREAD_SYSTEM_AVAILABLE
218 // Step 4: Create thread_system_collector (if available)
219 std::cout << "\n4. Creating thread_system_collector..." << std::endl;
220
221 thread_system_collector thread_collector;
222 if (!thread_collector.initialize(init_config)) {
223 std::cerr << "Failed to initialize thread_system_collector" << std::endl;
224 return 1;
225 }
226
227 std::cout << " Initialized: " << thread_collector.get_name() << std::endl;
228 std::cout << " Health: " << (thread_collector.is_healthy() ? "OK" : "UNHEALTHY") << std::endl;
229#else
230 std::cout << "\n4. thread_system_collector not available (THREAD_SYSTEM_AVAILABLE not defined)" << std::endl;
231#endif
232
233#ifdef LOGGER_SYSTEM_AVAILABLE
234 // Step 5: Create logger_system_collector (if available)
235 std::cout << "\n5. Creating logger_system_collector..." << std::endl;
236
237 logger_system_collector logger_collector;
238 if (!logger_collector.initialize(init_config)) {
239 std::cerr << "Failed to initialize logger_system_collector" << std::endl;
240 return 1;
241 }
242
243 std::cout << " Initialized: " << logger_collector.get_name() << std::endl;
244 std::cout << " Health: " << (logger_collector.is_healthy() ? "OK" : "UNHEALTHY") << std::endl;
245#else
246 std::cout << "\n5. logger_system_collector not available (LOGGER_SYSTEM_AVAILABLE not defined)" << std::endl;
247#endif
248
249 // Step 6: Collector lifecycle demonstration
250 std::cout << "\n6. Demonstrating collector lifecycle (3 iterations)..." << std::endl;
251
252 for (int i = 0; i < 3; ++i) {
253 std::cout << "\n--- Iteration " << (i + 1) << "/3 ---" << std::endl;
254
255 // Collect system resources
256 auto sys_metrics = sys_collector.collect();
257 std::cout << "System metrics collected: " << sys_metrics.size() << std::endl;
258
259 // Get last collected resources for detailed display
260 auto resources = sys_collector.get_last_resources();
261 display_system_metrics(resources);
262
263 // Collect network metrics
264 auto net_metrics = net_collector.collect();
265 std::cout << "Network metrics collected: " << net_metrics.size() << std::endl;
266 display_network_collector_metrics(net_metrics);
267
268 // Collect process metrics
269 auto proc_metrics = proc_collector.collect();
270 std::cout << "Process metrics collected: " << proc_metrics.size() << std::endl;
271 display_process_collector_metrics(proc_metrics);
272
273#ifdef THREAD_SYSTEM_AVAILABLE
274 // Collect thread system metrics
275 auto thread_metrics = thread_collector.collect();
276 std::cout << "Thread system metrics collected: " << thread_metrics.size() << std::endl;
277#endif
278
279#ifdef LOGGER_SYSTEM_AVAILABLE
280 // Collect logger system metrics
281 auto logger_metrics = logger_collector.collect();
282 std::cout << "Logger system metrics collected: " << logger_metrics.size() << std::endl;
283#endif
284
285 // Wait before next collection
286 if (i < 2) {
287 std::cout << "\nWaiting 2 seconds before next collection..." << std::endl;
288 std::this_thread::sleep_for(2s);
289 }
290 }
291
292 // Step 7: Display collector statistics
293 std::cout << "\n7. Collector Statistics:" << std::endl;
294
295 auto sys_stats = sys_collector.get_statistics();
296 std::cout << "\nSystem Resource Collector:" << std::endl;
297 for (const auto& [key, value] : sys_stats) {
298 std::cout << " " << key << ": " << value << std::endl;
299 }
300
301 auto net_stats = net_collector.get_statistics();
302 std::cout << "\nNetwork Metrics Collector:" << std::endl;
303 for (const auto& [key, value] : net_stats) {
304 std::cout << " " << key << ": " << value << std::endl;
305 }
306
307 auto proc_stats = proc_collector.get_statistics();
308 std::cout << "\nProcess Metrics Collector:" << std::endl;
309 for (const auto& [key, value] : proc_stats) {
310 std::cout << " " << key << ": " << value << std::endl;
311 }
312
313 // Step 8: Load history demonstration (if enabled)
314 if (sys_config.enable_load_history) {
315 std::cout << "\n8. Load Average History:" << std::endl;
316
317 auto load_history = sys_collector.get_all_load_history();
318 std::cout << " Total samples: " << load_history.size() << std::endl;
319
320 if (!load_history.empty()) {
321 auto load_stats = sys_collector.get_all_load_statistics();
322 std::cout << " 1-min avg: " << load_stats.load_1m_stats.avg << std::endl;
323 std::cout << " 5-min avg: " << load_stats.load_5m_stats.avg << std::endl;
324 std::cout << " 15-min avg: " << load_stats.load_15m_stats.avg << std::endl;
325 }
326 }
327
328 std::cout << "\n=== Example completed successfully ===" << std::endl;
329
330 } catch (const std::exception& e) {
331 std::cerr << "Exception: " << e.what() << std::endl;
332 return 1;
333 }
334
335 return 0;
336}
