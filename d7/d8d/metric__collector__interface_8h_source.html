147class metric_filter {
148public:
149 enum class filter_type {
150 include_all,
151 include_specific,
152 exclude_specific
153 };
154
155 metric_filter(filter_type type = filter_type::include_all)
156 : type_(type) {}
157
158 void add_metric_type(const std::string& type) {
159 metric_types_.push_back(type);
160 }
161
162 bool should_collect(const std::string& metric_type) const {
163 switch (type_) {
164 case filter_type::include_all:
165 return true;
166 case filter_type::include_specific:
167 return std::find(metric_types_.begin(), metric_types_.end(), metric_type) != metric_types_.end();
168 case filter_type::exclude_specific:
169 return std::find(metric_types_.begin(), metric_types_.end(), metric_type) == metric_types_.end();
170 }
171 return true;
172 }
173
174private:
175 filter_type type_;
176 std::vector<std::string> metric_types_;
177};

206struct collection_config {
207 std::chrono::milliseconds interval{std::chrono::seconds(1)};
208 metric_filter filter;
209 bool batch_collection{false};
210 size_t batch_size{100};
211 bool async_collection{true};
212
221 [[nodiscard]] common::VoidResult validate() const {
222 if (interval.count() <= 0) {
223 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration, "Collection interval must be positive").to_common_error());
224 }
225 if (batch_collection && batch_size == 0) {
226 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration, "Batch size must be positive when batch collection is enabled").to_common_error());
227 }
228 return common::ok();
229 }
230};
