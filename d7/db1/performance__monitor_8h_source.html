87struct tagged_metric {
88 std::string name;
89 double value;
90 recorded_metric_type type;
91 tag_map tags;
92 std::chrono::system_clock::time_point timestamp;
93
94 tagged_metric(const std::string& n, double v, recorded_metric_type t,
95 const tag_map& tgs = {})
96 : name(n)
97 , value(v)
98 , type(t)
99 , tags(tgs)
100 , timestamp(std::chrono::system_clock::now()) {}
101
105 std::string key() const {
106 std::string k = name;
107 // Sort tags for consistent key generation
108 std::vector<std::pair<std::string, std::string>> sorted_tags(
109 tags.begin(), tags.end());
110 std::sort(sorted_tags.begin(), sorted_tags.end());
111 for (const auto& [tag_key, tag_value] : sorted_tags) {
112 k += ";" + tag_key + "=" + tag_value;
113 }
114 return k;
115 }
116};

121struct performance_metrics {
122 std::string operation_name;
123 std::chrono::nanoseconds min_duration{std::chrono::nanoseconds::max()};
124 std::chrono::nanoseconds max_duration{std::chrono::nanoseconds::zero()};
125 std::chrono::nanoseconds total_duration{std::chrono::nanoseconds::zero()};
126 std::chrono::nanoseconds mean_duration{std::chrono::nanoseconds::zero()};
127 std::chrono::nanoseconds median_duration{std::chrono::nanoseconds::zero()};
128 std::chrono::nanoseconds p95_duration{std::chrono::nanoseconds::zero()};
129 std::chrono::nanoseconds p99_duration{std::chrono::nanoseconds::zero()};
130 std::uint64_t call_count{0};
131 std::uint64_t error_count{0};
132 double throughput{0.0}; // Operations per second
133
137 static std::chrono::nanoseconds calculate_percentile(
138 const std::vector<std::chrono::nanoseconds>& sorted_durations,
139 double percentile) {
140
141 if (sorted_durations.empty()) {
142 return std::chrono::nanoseconds::zero();
143 }
144
145 size_t index = static_cast<size_t>(
146 (percentile / 100.0) * (sorted_durations.size() - 1)
147 );
148 return sorted_durations[index];
149 }
150
154 void update_statistics(const std::vector<std::chrono::nanoseconds>& durations) {
155 if (durations.empty()) return;
156
157 auto sorted = durations;
158 std::sort(sorted.begin(), sorted.end());
159
160 min_duration = sorted.front();
161 max_duration = sorted.back();
162 median_duration = calculate_percentile(sorted, 50.0);
163 p95_duration = calculate_percentile(sorted, 95.0);
164 p99_duration = calculate_percentile(sorted, 99.0);
165
166 total_duration = std::accumulate(
167 sorted.begin(), sorted.end(),
168 std::chrono::nanoseconds::zero()
169 );
170
171 mean_duration = total_duration / sorted.size();
172 }
173};

297class scoped_timer {
298private:
299 performance_profiler* profiler_;
300 std::string operation_name_;
301 std::chrono::high_resolution_clock::time_point start_time_;
302 bool success_{true};
303 bool completed_{false};
304
305public:
306 scoped_timer(performance_profiler* profiler, const std::string& operation_name)
307 : profiler_(profiler)
308 , operation_name_(operation_name)
309 , start_time_(std::chrono::high_resolution_clock::now()) {}
310
311 ~scoped_timer() {
312 if (!completed_ && profiler_) {
313 complete();
314 }
315 }
316
320 void mark_failed() { success_ = false; }
321
325 void complete() {
326 if (completed_) return;
327
328 auto end_time = std::chrono::high_resolution_clock::now();
329 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
330 end_time - start_time_
331 );
332
333 if (profiler_) {
334 profiler_->record_sample(operation_name_, duration, success_);
335 }
336
337 completed_ = true;
338 }
339
343 std::chrono::nanoseconds elapsed() const {
344 auto now = std::chrono::high_resolution_clock::now();
345 return std::chrono::duration_cast<std::chrono::nanoseconds>(
346 now - start_time_
347 );
348 }

688class performance_benchmark {
689private:
690 performance_profiler profiler_;
691 std::string name_;
692 std::uint32_t iterations_{1000};
693 std::uint32_t warmup_iterations_{100};
694
695public:
696 explicit performance_benchmark(const std::string& name)
697 : name_(name) {}
698
702 void set_iterations(std::uint32_t iterations) {
703 iterations_ = iterations;
704 }
705
709 void set_warmup_iterations(std::uint32_t warmup) {
710 warmup_iterations_ = warmup;
711 }
712
716 template<typename Func>
717 kcenon::monitoring::result<performance_metrics> run(
718 const std::string& operation_name,
719 Func&& func
720 ) {
721 // Warmup
722 for (std::uint32_t i = 0; i < warmup_iterations_; ++i) {
723 func();
724 }
725
726 // Actual benchmark
727 for (std::uint32_t i = 0; i < iterations_; ++i) {
728 auto start = std::chrono::high_resolution_clock::now();
729
730 try {
731 func();
732 } catch (...) {
733 // Record error
734 auto end = std::chrono::high_resolution_clock::now();
735 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
736 end - start
737 );
738 profiler_.record_sample(operation_name, duration, false);
739 continue;
740 }
741
742 auto end = std::chrono::high_resolution_clock::now();
743 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
744 end - start
745 );
746 profiler_.record_sample(operation_name, duration, true);
747 }
748
749 return profiler_.get_metrics(operation_name);
750 }
751
755 template<typename Func1, typename Func2>
756 kcenon::monitoring::result<std::pair<performance_metrics, performance_metrics>> compare(
757 const std::string& operation1_name,
758 Func1&& func1,
759 const std::string& operation2_name,
760 Func2&& func2
761 ) {
762 auto result1 = run(operation1_name, std::forward<Func1>(func1));
763 if (result1.is_err()) {
764 return result<std::pair<performance_metrics, performance_metrics>>::err(result1.error());
765 }
766
767 auto result2 = run(operation2_name, std::forward<Func2>(func2));
768 if (result2.is_err()) {
769 return result<std::pair<performance_metrics, performance_metrics>>::err(result2.error());
770 }
771
772 return std::make_pair(result1.value(), result2.value());
773 }
