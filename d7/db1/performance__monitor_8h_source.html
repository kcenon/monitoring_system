88struct tagged_metric {
89 std::string name;
90 double value;
91 recorded_metric_type type;
92 tag_map tags;
93 std::chrono::system_clock::time_point timestamp;
94
95 tagged_metric(const std::string& n, double v, recorded_metric_type t,
96 const tag_map& tgs = {})
97 : name(n)
98 , value(v)
99 , type(t)
100 , tags(tgs)
101 , timestamp(std::chrono::system_clock::now()) {}
102
106 std::string key() const {
107 std::string k = name;
108 // Sort tags for consistent key generation
109 std::vector<std::pair<std::string, std::string>> sorted_tags(
110 tags.begin(), tags.end());
111 std::sort(sorted_tags.begin(), sorted_tags.end());
112 for (const auto& [tag_key, tag_value] : sorted_tags) {
113 k += ";" + tag_key + "=" + tag_value;
114 }
115 return k;
116 }
117};

285class scoped_timer {
286private:
287 performance_profiler* profiler_;
288 std::string operation_name_;
289 std::chrono::high_resolution_clock::time_point start_time_;
290 bool success_{true};
291 bool completed_{false};
292
293public:
294 scoped_timer(performance_profiler* profiler, const std::string& operation_name)
295 : profiler_(profiler)
296 , operation_name_(operation_name)
297 , start_time_(std::chrono::high_resolution_clock::now()) {}
298
299 ~scoped_timer() {
300 if (!completed_ && profiler_) {
301 complete();
302 }
303 }
304
308 void mark_failed() { success_ = false; }
309
313 void complete() {
314 if (completed_) return;
315
316 auto end_time = std::chrono::high_resolution_clock::now();
317 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
318 end_time - start_time_
319 );
320
321 if (profiler_) {
322 profiler_->record_sample(operation_name_, duration, success_);
323 }
324
325 completed_ = true;
326 }
327
331 std::chrono::nanoseconds elapsed() const {
332 auto now = std::chrono::high_resolution_clock::now();
333 return std::chrono::duration_cast<std::chrono::nanoseconds>(
334 now - start_time_
335 );
336 }

653class performance_benchmark {
654private:
655 performance_profiler profiler_;
656 std::string name_;
657 std::uint32_t iterations_{1000};
658 std::uint32_t warmup_iterations_{100};
659
660public:
661 explicit performance_benchmark(const std::string& name)
662 : name_(name) {}
663
667 void set_iterations(std::uint32_t iterations) {
668 iterations_ = iterations;
669 }
670
674 void set_warmup_iterations(std::uint32_t warmup) {
675 warmup_iterations_ = warmup;
676 }
677
681 template<typename Func>
682 common::Result<performance_metrics> run(
683 const std::string& operation_name,
684 Func&& func
685 ) {
686 // Warmup
687 for (std::uint32_t i = 0; i < warmup_iterations_; ++i) {
688 func();
689 }
690
691 // Actual benchmark
692 for (std::uint32_t i = 0; i < iterations_; ++i) {
693 auto start = std::chrono::high_resolution_clock::now();
694
695 try {
696 func();
697 } catch (...) {
698 // Record error
699 auto end = std::chrono::high_resolution_clock::now();
700 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
701 end - start
702 );
703 profiler_.record_sample(operation_name, duration, false);
704 continue;
705 }
706
707 auto end = std::chrono::high_resolution_clock::now();
708 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
709 end - start
710 );
711 profiler_.record_sample(operation_name, duration, true);
712 }
713
714 return profiler_.get_metrics(operation_name);
715 }
716
720 template<typename Func1, typename Func2>
721 common::Result<std::pair<performance_metrics, performance_metrics>> compare(
722 const std::string& operation1_name,
723 Func1&& func1,
724 const std::string& operation2_name,
725 Func2&& func2
726 ) {
727 auto result1 = run(operation1_name, std::forward<Func1>(func1));
728 if (result1.is_err()) {
729 return common::Result<std::pair<performance_metrics, performance_metrics>>::err(result1.error());
730 }
731
732 auto result2 = run(operation2_name, std::forward<Func2>(func2));
733 if (result2.is_err()) {
734 return common::Result<std::pair<performance_metrics, performance_metrics>>::err(result2.error());
735 }
736
737 return common::ok(std::make_pair(result1.value(), result2.value()));
738 }
