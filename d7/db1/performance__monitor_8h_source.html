87struct tagged_metric {
88 std::string name;
89 double value;
90 recorded_metric_type type;
91 tag_map tags;
92 std::chrono::system_clock::time_point timestamp;
93
94 tagged_metric(const std::string& n, double v, recorded_metric_type t,
95 const tag_map& tgs = {})
96 : name(n)
97 , value(v)
98 , type(t)
99 , tags(tgs)
100 , timestamp(std::chrono::system_clock::now()) {}
101
105 std::string key() const {
106 std::string k = name;
107 // Sort tags for consistent key generation
108 std::vector<std::pair<std::string, std::string>> sorted_tags(
109 tags.begin(), tags.end());
110 std::sort(sorted_tags.begin(), sorted_tags.end());
111 for (const auto& [tag_key, tag_value] : sorted_tags) {
112 k += ";" + tag_key + "=" + tag_value;
113 }
114 return k;
115 }
116};

258class scoped_timer {
259private:
260 performance_profiler* profiler_;
261 std::string operation_name_;
262 std::chrono::high_resolution_clock::time_point start_time_;
263 bool success_{true};
264 bool completed_{false};
265
266public:
267 scoped_timer(performance_profiler* profiler, const std::string& operation_name)
268 : profiler_(profiler)
269 , operation_name_(operation_name)
270 , start_time_(std::chrono::high_resolution_clock::now()) {}
271
272 ~scoped_timer() {
273 if (!completed_ && profiler_) {
274 complete();
275 }
276 }
277
281 void mark_failed() { success_ = false; }
282
286 void complete() {
287 if (completed_) return;
288
289 auto end_time = std::chrono::high_resolution_clock::now();
290 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
291 end_time - start_time_
292 );
293
294 if (profiler_) {
295 profiler_->record_sample(operation_name_, duration, success_);
296 }
297
298 completed_ = true;
299 }
300
304 std::chrono::nanoseconds elapsed() const {
305 auto now = std::chrono::high_resolution_clock::now();
306 return std::chrono::duration_cast<std::chrono::nanoseconds>(
307 now - start_time_
308 );
309 }

626class performance_benchmark {
627private:
628 performance_profiler profiler_;
629 std::string name_;
630 std::uint32_t iterations_{1000};
631 std::uint32_t warmup_iterations_{100};
632
633public:
634 explicit performance_benchmark(const std::string& name)
635 : name_(name) {}
636
640 void set_iterations(std::uint32_t iterations) {
641 iterations_ = iterations;
642 }
643
647 void set_warmup_iterations(std::uint32_t warmup) {
648 warmup_iterations_ = warmup;
649 }
650
654 template<typename Func>
655 common::Result<performance_metrics> run(
656 const std::string& operation_name,
657 Func&& func
658 ) {
659 // Warmup
660 for (std::uint32_t i = 0; i < warmup_iterations_; ++i) {
661 func();
662 }
663
664 // Actual benchmark
665 for (std::uint32_t i = 0; i < iterations_; ++i) {
666 auto start = std::chrono::high_resolution_clock::now();
667
668 try {
669 func();
670 } catch (...) {
671 // Record error
672 auto end = std::chrono::high_resolution_clock::now();
673 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
674 end - start
675 );
676 profiler_.record_sample(operation_name, duration, false);
677 continue;
678 }
679
680 auto end = std::chrono::high_resolution_clock::now();
681 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
682 end - start
683 );
684 profiler_.record_sample(operation_name, duration, true);
685 }
686
687 return profiler_.get_metrics(operation_name);
688 }
689
693 template<typename Func1, typename Func2>
694 common::Result<std::pair<performance_metrics, performance_metrics>> compare(
695 const std::string& operation1_name,
696 Func1&& func1,
697 const std::string& operation2_name,
698 Func2&& func2
699 ) {
700 auto result1 = run(operation1_name, std::forward<Func1>(func1));
701 if (result1.is_err()) {
702 return common::Result<std::pair<performance_metrics, performance_metrics>>::err(result1.error());
703 }
704
705 auto result2 = run(operation2_name, std::forward<Func2>(func2));
706 if (result2.is_err()) {
707 return common::Result<std::pair<performance_metrics, performance_metrics>>::err(result2.error());
708 }
709
710 return common::ok(std::make_pair(result1.value(), result2.value()));
711 }
