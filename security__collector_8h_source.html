59inline std::string security_event_type_to_string(security_event_type type) {
60 switch (type) {
61 case security_event_type::login_success: return "LOGIN_SUCCESS";
62 case security_event_type::login_failure: return "LOGIN_FAILURE";
63 case security_event_type::logout: return "LOGOUT";
64 case security_event_type::sudo_usage: return "SUDO_USAGE";
65 case security_event_type::permission_change: return "PERMISSION_CHANGE";
66 case security_event_type::account_created: return "ACCOUNT_CREATED";
67 case security_event_type::account_deleted: return "ACCOUNT_DELETED";
68 case security_event_type::account_modified: return "ACCOUNT_MODIFIED";
69 case security_event_type::session_start: return "SESSION_START";
70 case security_event_type::session_end: return "SESSION_END";
71 default: return "UNKNOWN";
72 }

108 uint64_t get_count(security_event_type type) const {
109 switch (type) {
110 case security_event_type::login_success: return login_success;
111 case security_event_type::login_failure: return login_failure;
112 case security_event_type::logout: return logout;
113 case security_event_type::sudo_usage: return sudo_usage;
114 case security_event_type::permission_change: return permission_change;
115 case security_event_type::account_created: return account_created;
116 case security_event_type::account_deleted: return account_deleted;
117 case security_event_type::account_modified: return account_modified;
118 default: return unknown;
119 }
120 }
121
126 void increment(security_event_type type) {
127 switch (type) {
128 case security_event_type::login_success: ++login_success; break;
129 case security_event_type::login_failure: ++login_failure; break;
130 case security_event_type::logout: ++logout; break;
131 case security_event_type::sudo_usage: ++sudo_usage; break;
132 case security_event_type::permission_change: ++permission_change; break;
133 case security_event_type::account_created: ++account_created; break;
134 case security_event_type::account_deleted: ++account_deleted; break;
135 case security_event_type::account_modified: ++account_modified; break;
136 default: ++unknown; break;
137 }
138 }
139
144 uint64_t total() const {
145 return login_success + login_failure + logout + sudo_usage +
146 permission_change + account_created + account_deleted +
147 account_modified + unknown;
148 }
