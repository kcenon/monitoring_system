58class example_plugin : public collector_plugin {
59public:
60 example_plugin() : generator_(std::random_device{}()) {}
61
62 auto name() const -> std::string_view override {
63 return "example_plugin";
64 }
65
66 auto initialize(const config_map& config) -> bool override {
67 (void)config; // Unused
68 initialized_ = true;
69 return true;
70 }
71
72 auto shutdown() -> void override {
73 initialized_ = false;
74 }
75
76 auto collect() -> std::vector<metric_data> override {
77 std::vector<metric_data> metrics;
78
79 if (!initialized_) {
80 return metrics;
81 }
82
83 // Generate some dummy metrics
84 metric_data cpu_metric;
85 cpu_metric.name = "example.cpu_usage";
86 cpu_metric.value = generate_random_value(0.0, 100.0);
87 cpu_metric.unit = "%";
88 cpu_metric.timestamp = std::chrono::system_clock::now();
89 cpu_metric.labels["plugin"] = "example";
90 cpu_metric.labels["type"] = "cpu";
91 metrics.push_back(cpu_metric);
92
93 metric_data memory_metric;
94 memory_metric.name = "example.memory_usage";
95 memory_metric.value = generate_random_value(0.0, 1024.0);
96 memory_metric.unit = "MB";
97 memory_metric.timestamp = std::chrono::system_clock::now();
98 memory_metric.labels["plugin"] = "example";
99 memory_metric.labels["type"] = "memory";
100 metrics.push_back(memory_metric);
101
102 metric_data counter_metric;
103 counter_metric.name = "example.request_count";
104 counter_metric.value = static_cast<double>(++request_counter_);
105 counter_metric.unit = "requests";
106 counter_metric.timestamp = std::chrono::system_clock::now();
107 counter_metric.labels["plugin"] = "example";
108 counter_metric.labels["type"] = "counter";
109 metrics.push_back(counter_metric);
110
111 return metrics;
112 }
113
114 auto is_available() const -> bool override {
115 // This plugin is available on all platforms
116 return true;
117 }
118
119 auto get_metadata() const -> plugin_metadata_t override {
120 return plugin_metadata_t{
121 plugin_category::custom,
122 plugin_type::collector,
123 "Example Plugin",
124 "1.0.0",
125 "Demonstrates dynamic plugin loading"
126 };
127 }
128
129private:
130 auto generate_random_value(double min, double max) -> double {
131 std::uniform_real_distribution<double> dist(min, max);
132 return dist(generator_);
133 }
134
135 bool initialized_{false};
136 uint64_t request_counter_{0};
137 std::mt19937 generator_;
138};
