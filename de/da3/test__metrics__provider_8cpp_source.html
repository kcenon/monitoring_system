261// Test that multiple calls return consistent results

265 }
266
267 // Get uptime twice - should be close
268 auto uptime1 = provider_->get_uptime();
269 auto uptime2 = provider_->get_uptime();
270
271 if (uptime1.available && uptime2.available) {
272 // Uptime should not decrease
