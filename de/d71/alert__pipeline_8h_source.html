104class alert_aggregator {
105public:
109 explicit alert_aggregator(const alert_aggregator_config& config)
110 : config_(config) {}
111
117 std::string add_alert(const alert& a) {
118 std::lock_guard<std::mutex> lock(mutex_);
119
120 std::string group_key = compute_group_key(a);
121 auto now = std::chrono::steady_clock::now();
122
123 auto it = groups_.find(group_key);
124 if (it == groups_.end()) {
125 alert_group new_group(group_key);
126 new_group.common_labels = extract_common_labels(a);
127 new_group.add_alert(a);
128 groups_[group_key] = std::move(new_group);
129 first_seen_[group_key] = now;
130 } else {
131 // Check for duplicate
132 bool is_duplicate = false;
133 for (auto& existing : it->second.alerts) {
134 if (existing.fingerprint() == a.fingerprint()) {
135 existing = a; // Update existing alert
136 is_duplicate = true;
137 break;
138 }
139 }
140 if (!is_duplicate) {
141 it->second.add_alert(a);
142 }
143 }
144
145 return group_key;
146 }
147
152 std::vector<alert_group> get_ready_groups() {
153 std::lock_guard<std::mutex> lock(mutex_);
154
155 std::vector<alert_group> ready;
156 auto now = std::chrono::steady_clock::now();
157
158 for (auto& [key, group] : groups_) {
159 if (group.empty()) {
160 continue;
161 }
162
163 // Check if first wait has elapsed
164 auto first_seen = first_seen_[key];
165 if (now - first_seen < config_.group_wait) {
166 continue;
167 }
168
169 // Check if enough time since last send
170 auto last_sent_it = last_sent_.find(key);
171 if (last_sent_it != last_sent_.end()) {
172 if (now - last_sent_it->second < config_.group_interval) {
173 continue;
174 }
175 }
176
177 ready.push_back(group);
178 }
179
180 return ready;
181 }
182
187 void mark_sent(const std::string& group_key) {
188 std::lock_guard<std::mutex> lock(mutex_);
189 last_sent_[group_key] = std::chrono::steady_clock::now();
190 }
191
195 void cleanup() {
196 std::lock_guard<std::mutex> lock(mutex_);
197
198 auto now = std::chrono::steady_clock::now();
199
200 for (auto it = groups_.begin(); it != groups_.end(); ) {
201 auto& group = it->second;
202
203 // Remove resolved alerts that have timed out
204 auto alert_it = group.alerts.begin();
205 while (alert_it != group.alerts.end()) {
206 if (alert_it->state == alert_state::resolved) {
207 auto resolved_time = alert_it->resolved_at.value_or(now);
208 if (now - resolved_time > config_.resolve_timeout) {
209 alert_it = group.alerts.erase(alert_it);
210 continue;
211 }
212 }
213 ++alert_it;
214 }
215
216 // Remove empty groups
217 if (group.empty()) {
218 first_seen_.erase(it->first);
219 last_sent_.erase(it->first);
220 it = groups_.erase(it);
221 } else {
222 ++it;
223 }
224 }
225 }
226
230 size_t group_count() const {
231 std::lock_guard<std::mutex> lock(mutex_);
232 return groups_.size();
233 }
234
238 size_t total_alert_count() const {
239 std::lock_guard<std::mutex> lock(mutex_);
240 size_t count = 0;
241 for (const auto& [key, group] : groups_) {
242 count += group.size();
243 }
244 return count;
245 }
246
247private:
248 std::string compute_group_key(const alert& a) const {
249 if (config_.group_by_labels.empty()) {
250 return a.rule_name;
251 }
252
253 std::string key;
254 for (const auto& label : config_.group_by_labels) {
255 key += a.labels.get(label) + ":";
256 }
257 return key;
258 }
259
260 alert_labels extract_common_labels(const alert& a) const {
261 alert_labels common;
262 for (const auto& label : config_.group_by_labels) {
263 std::string val = a.labels.get(label);
264 if (!val.empty()) {
265 common.set(label, val);
266 }
267 }
268 return common;
269 }
270
271 alert_aggregator_config config_;
272 mutable std::mutex mutex_;
273
274 std::unordered_map<std::string, alert_group> groups_;
275 std::unordered_map<std::string, std::chrono::steady_clock::time_point> first_seen_;
276 std::unordered_map<std::string, std::chrono::steady_clock::time_point> last_sent_;
277};

286struct inhibition_rule {
287 std::string name;
288 alert_labels source_match;
289 alert_labels target_match;
290 std::vector<std::string> equal;
291
295 bool matches_source(const alert& a) const {
296 for (const auto& [key, value] : source_match.labels) {
297 if (a.labels.get(key) != value) {
298 return false;
299 }
300 }
301 return true;
302 }
303
307 bool should_inhibit(const alert& source, const alert& target) const {
308 // Source must match
309 if (!matches_source(source)) {
310 return false;
311 }
312
313 // Target must match target_match labels
314 for (const auto& [key, value] : target_match.labels) {
315 if (target.labels.get(key) != value) {
316 return false;
317 }
318 }
319
320 // Equal labels must match between source and target
321 for (const auto& label : equal) {
322 if (source.labels.get(label) != target.labels.get(label)) {
323 return false;
324 }
325 }
326
327 return true;
328 }
329};

355class alert_inhibitor {
356public:
360 void add_rule(const inhibition_rule& rule) {
361 std::lock_guard<std::mutex> lock(mutex_);
362 rules_.push_back(rule);
363 }
364
368 void remove_rule(const std::string& name) {
369 std::lock_guard<std::mutex> lock(mutex_);
370 rules_.erase(
371 std::remove_if(rules_.begin(), rules_.end(),
372 [&name](const inhibition_rule& r) { return r.name == name; }),
373 rules_.end()
374 );
375 }
376
383 bool is_inhibited(const alert& target,
384 const std::vector<alert>& active_alerts) const {
385 std::lock_guard<std::mutex> lock(mutex_);
386
387 for (const auto& rule : rules_) {
388 for (const auto& source : active_alerts) {
389 // Source must be firing
390 if (source.state != alert_state::firing) {
391 continue;
392 }
393
394 // Don't inhibit self
395 if (source.fingerprint() == target.fingerprint()) {
396 continue;
397 }
398
399 if (rule.should_inhibit(source, target)) {
400 return true;
401 }
402 }
403 }
404
405 return false;
406 }
407
411 std::vector<inhibition_rule> get_rules() const {
412 std::lock_guard<std::mutex> lock(mutex_);
413 return rules_;
414 }
415
416private:
417 mutable std::mutex mutex_;
418 std::vector<inhibition_rule> rules_;
419};

428class cooldown_tracker {
429public:
433 explicit cooldown_tracker(std::chrono::milliseconds default_cooldown)
434 : default_cooldown_(default_cooldown) {}
435
441 bool is_in_cooldown(const std::string& fingerprint) const {
442 std::lock_guard<std::mutex> lock(mutex_);
443
444 auto it = last_notification_.find(fingerprint);
445 if (it == last_notification_.end()) {
446 return false;
447 }
448
449 auto now = std::chrono::steady_clock::now();
450 auto cooldown = get_cooldown_for(fingerprint);
451 return (now - it->second) < cooldown;
452 }
453
458 void record_notification(const std::string& fingerprint) {
459 std::lock_guard<std::mutex> lock(mutex_);
460 last_notification_[fingerprint] = std::chrono::steady_clock::now();
461 }
462
468 void set_cooldown(const std::string& fingerprint,
469 std::chrono::milliseconds cooldown) {
470 std::lock_guard<std::mutex> lock(mutex_);
471 custom_cooldowns_[fingerprint] = cooldown;
472 }
473
479 std::chrono::milliseconds remaining_cooldown(const std::string& fingerprint) const {
480 std::lock_guard<std::mutex> lock(mutex_);
481
482 auto it = last_notification_.find(fingerprint);
483 if (it == last_notification_.end()) {
484 return std::chrono::milliseconds::zero();
485 }
486
487 auto now = std::chrono::steady_clock::now();
488 auto cooldown = get_cooldown_for(fingerprint);
489 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
490 now - it->second);
491
492 if (elapsed >= cooldown) {
493 return std::chrono::milliseconds::zero();
494 }
495 return cooldown - elapsed;
496 }
497
501 void clear_cooldown(const std::string& fingerprint) {
502 std::lock_guard<std::mutex> lock(mutex_);
503 last_notification_.erase(fingerprint);
504 }
505
509 void reset() {
510 std::lock_guard<std::mutex> lock(mutex_);
511 last_notification_.clear();
512 }
513
514private:
515 std::chrono::milliseconds get_cooldown_for(const std::string& fingerprint) const {
516 auto it = custom_cooldowns_.find(fingerprint);
517 if (it != custom_cooldowns_.end()) {
518 return it->second;
519 }
520 return default_cooldown_;
521 }
522
523 std::chrono::milliseconds default_cooldown_;
524 mutable std::mutex mutex_;
525 std::unordered_map<std::string, std::chrono::steady_clock::time_point> last_notification_;
526 std::unordered_map<std::string, std::chrono::milliseconds> custom_cooldowns_;
527};

536class alert_deduplicator {
537public:
542 explicit alert_deduplicator(std::chrono::milliseconds cache_duration)
543 : cache_duration_(cache_duration) {}
544
550 bool is_duplicate(const alert& a) {
551 std::lock_guard<std::mutex> lock(mutex_);
552
553 cleanup_expired();
554
555 auto fingerprint = a.fingerprint();
556 auto it = seen_.find(fingerprint);
557
558 if (it == seen_.end()) {
559 seen_[fingerprint] = std::chrono::steady_clock::now();
560 return false;
561 }
562
563 // Same state is duplicate
564 auto state_it = last_state_.find(fingerprint);
565 if (state_it != last_state_.end() && state_it->second == a.state) {
566 return true;
567 }
568
569 // State changed - not duplicate
570 last_state_[fingerprint] = a.state;
571 return false;
572 }
573
577 void mark_seen(const alert& a) {
578 std::lock_guard<std::mutex> lock(mutex_);
579 auto fingerprint = a.fingerprint();
580 seen_[fingerprint] = std::chrono::steady_clock::now();
581 last_state_[fingerprint] = a.state;
582 }
583
587 void reset() {
588 std::lock_guard<std::mutex> lock(mutex_);
589 seen_.clear();
590 last_state_.clear();
591 }
592
593private:
594 void cleanup_expired() {
595 auto now = std::chrono::steady_clock::now();
596 for (auto it = seen_.begin(); it != seen_.end(); ) {
597 if (now - it->second > cache_duration_) {
598 last_state_.erase(it->first);
599 it = seen_.erase(it);
600 } else {
601 ++it;
602 }
603 }
604 }
605
606 std::chrono::milliseconds cache_duration_;
607 mutable std::mutex mutex_;
608 std::unordered_map<std::string, std::chrono::steady_clock::time_point> seen_;
609 std::unordered_map<std::string, alert_state> last_state_;
610};

646 stages_.push_back(std::move(stage));
647 }
648
