54inline void* aligned_alloc_impl(size_t alignment, size_t size) {
55#ifdef _MSC_VER
56 // MSVC doesn't support std::aligned_alloc
57 // Size must be a multiple of alignment for _aligned_malloc
58 size_t aligned_size = (size + alignment - 1) & ~(alignment - 1);
59 return _aligned_malloc(aligned_size, alignment);
60#else
61 // POSIX/C11 aligned_alloc requires size to be multiple of alignment
62 size_t aligned_size = (size + alignment - 1) & ~(alignment - 1);
63 return std::aligned_alloc(alignment, aligned_size);
64#endif
65}

71inline void aligned_free_impl(void* ptr) {
72#ifdef _MSC_VER
73 _aligned_free(ptr);
74#else
75 std::free(ptr);
76#endif
77}

86struct memory_pool_config {
87 size_t initial_blocks = 256;
88 size_t max_blocks = 4096;
89 size_t block_size = 64;
90 size_t alignment = 8;
91 bool use_thread_local_cache = false;
92
97 bool validate() const {
98 if (initial_blocks == 0) {
99 return false;
100 }
101 if (max_blocks != 0 && max_blocks < initial_blocks) {
102 return false;
103 }
104 if (block_size == 0) {
105 return false;
106 }
107 // Block size must be 8-byte aligned
108 if (block_size % 8 != 0) {
109 return false;
110 }
111 // Alignment must be power of 2
112 if (alignment == 0 || (alignment & (alignment - 1)) != 0) {
113 return false;
114 }
115 return true;
116 }
117};

165 double get_allocation_success_rate() const {
166 auto total = total_allocations.load() + allocation_failures.load();
167 if (total == 0) {
168 return 100.0;
169 }
170 return (static_cast<double>(total_allocations.load()) / static_cast<double>(total)) * 100.0;
171 }

249 common::Result<void*> allocate() {
250 std::lock_guard<std::mutex> lock(mutex_);
251
252 if (free_blocks_.empty()) {
253 // Try to grow the pool
254 if (!grow_pool()) {
255 stats_.allocation_failures++;
256 return common::Result<void*>::err(error_info(monitoring_error_code::resource_unavailable, "Memory pool exhausted").to_common_error());
257 }
258 }
259
260 void* block = free_blocks_.back();
261 free_blocks_.pop_back();
262
263 stats_.total_allocations++;
264 update_peak_usage();
265
266 return common::ok(block);
267 }
268
274 common::VoidResult deallocate(void* ptr) {
275 if (ptr == nullptr) {
276 return common::VoidResult::err(error_info(monitoring_error_code::invalid_argument, "Cannot deallocate null pointer").to_common_error());
277 }
278
279 std::lock_guard<std::mutex> lock(mutex_);
280
281 // Verify the pointer belongs to this pool
282 if (!is_owned_block(ptr)) {
283 return common::VoidResult::err(error_info(monitoring_error_code::invalid_argument, "Pointer does not belong to this pool").to_common_error());
284 }
285
286 free_blocks_.push_back(ptr);
287 stats_.total_deallocations++;
288
289 return common::ok();
290 }

299 template<typename T, typename... Args>
300 common::Result<T*> allocate_object(Args&&... args) {
301 if (sizeof(T) > block_size_) {
302 return common::Result<T*>::err(error_info(monitoring_error_code::invalid_argument, "Object size exceeds block size").to_common_error());
303 }
304
305 auto result = allocate();
306 if (result.is_err()) {
307 return common::Result<T*>::err(error_info(monitoring_error_code::resource_unavailable, "Failed to allocate memory for object").to_common_error());
308 }
309
310 void* ptr = result.value();
311 T* obj = new (ptr) T(std::forward<Args>(args)...);
312
313 return common::ok(obj);
314 }

324 if (obj == nullptr) {
325 return common::VoidResult::err(error_info(monitoring_error_code::invalid_argument, "Cannot deallocate null object").to_common_error());

372private:
373 void initialize_pool() {
374 size_t chunk_size = total_blocks_ * block_size_;
375 void* chunk = detail::aligned_alloc_impl(config_.alignment, chunk_size);
376
377 if (chunk == nullptr) {
378 throw std::bad_alloc();
379 }
380
381 memory_chunks_.push_back(chunk);
382
383 // Initialize free block list
384 free_blocks_.reserve(total_blocks_);
385 char* ptr = static_cast<char*>(chunk);
386 for (size_t i = 0; i < total_blocks_; ++i) {
387 free_blocks_.push_back(ptr + i * block_size_);
388 }
389 }
390
391 bool grow_pool() {
392 if (config_.max_blocks > 0 && total_blocks_ >= config_.max_blocks) {
393 return false;
394 }
395
396 size_t new_blocks = std::min(total_blocks_, config_.max_blocks - total_blocks_);
397 if (new_blocks == 0) {
398 new_blocks = total_blocks_; // Double the size
399 }
400
401 size_t chunk_size = new_blocks * block_size_;
402 void* chunk = detail::aligned_alloc_impl(config_.alignment, chunk_size);
403
404 if (chunk == nullptr) {
405 return false;
406 }
407
408 memory_chunks_.push_back(chunk);
409
410 char* ptr = static_cast<char*>(chunk);
411 for (size_t i = 0; i < new_blocks; ++i) {
412 free_blocks_.push_back(ptr + i * block_size_);
413 }
414
415 total_blocks_ += new_blocks;
416 return true;
417 }
418
419 bool is_owned_block(void* ptr) const {
420 for (size_t i = 0; i < memory_chunks_.size(); ++i) {
421 char* chunk_start = static_cast<char*>(memory_chunks_[i]);
422 size_t chunk_blocks = (i == 0) ? config_.initial_blocks :
423 ((total_blocks_ - config_.initial_blocks) / (memory_chunks_.size() - 1));
424 char* chunk_end = chunk_start + chunk_blocks * block_size_;
425
426 if (ptr >= chunk_start && ptr < chunk_end) {
427 // Verify alignment
428 size_t offset = static_cast<char*>(ptr) - chunk_start;
429 if (offset % block_size_ == 0) {
430 return true;
431 }
432 }
433 }
434 return false;
435 }
436
437 void update_peak_usage() {
438 size_t current_usage = total_blocks_ - free_blocks_.size();
439 size_t peak = stats_.peak_usage.load();
440 while (current_usage > peak) {
441 if (stats_.peak_usage.compare_exchange_weak(peak, current_usage)) {
442 break;
443 }
444 }
445 }

453 mutable memory_pool_statistics stats_;
