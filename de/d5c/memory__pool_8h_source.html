46struct memory_pool_config {
47 size_t initial_blocks = 256;
48 size_t max_blocks = 4096;
49 size_t block_size = 64;
50 size_t alignment = 8;
51 bool use_thread_local_cache = false;
52
57 bool validate() const {
58 if (initial_blocks == 0) {
59 return false;
60 }
61 if (max_blocks != 0 && max_blocks < initial_blocks) {
62 return false;
63 }
64 if (block_size == 0) {
65 return false;
66 }
67 // Block size must be 8-byte aligned
68 if (block_size % 8 != 0) {
69 return false;
70 }
71 // Alignment must be power of 2
72 if (alignment == 0 || (alignment & (alignment - 1)) != 0) {
73 return false;
74 }
75 return true;
76 }
77};

125 double get_allocation_success_rate() const {
126 auto total = total_allocations.load() + allocation_failures.load();
127 if (total == 0) {
128 return 100.0;
129 }
130 return (static_cast<double>(total_allocations.load()) / static_cast<double>(total)) * 100.0;
131 }

209 result<void*> allocate() {
210 std::lock_guard<std::mutex> lock(mutex_);
211
212 if (free_blocks_.empty()) {
213 // Try to grow the pool
214 if (!grow_pool()) {
215 stats_.allocation_failures++;
216 return make_error<void*>(monitoring_error_code::resource_unavailable,
217 "Memory pool exhausted");
218 }
219 }
220
221 void* block = free_blocks_.back();
222 free_blocks_.pop_back();
223
224 stats_.total_allocations++;
225 update_peak_usage();
226
227 return common::ok(block);
228 }
229
235 result_void deallocate(void* ptr) {
236 if (ptr == nullptr) {
237 return make_void_error(monitoring_error_code::invalid_argument,
238 "Cannot deallocate null pointer");
239 }
240
241 std::lock_guard<std::mutex> lock(mutex_);
242
243 // Verify the pointer belongs to this pool
244 if (!is_owned_block(ptr)) {
245 return make_void_error(monitoring_error_code::invalid_argument,
246 "Pointer does not belong to this pool");
247 }
248
249 free_blocks_.push_back(ptr);
250 stats_.total_deallocations++;
251
252 return make_void_success();
253 }

262 template<typename T, typename... Args>
263 result<T*> allocate_object(Args&&... args) {
264 if (sizeof(T) > block_size_) {
265 return make_error<T*>(monitoring_error_code::invalid_argument,
266 "Object size exceeds block size");
267 }
268
269 auto result = allocate();
270 if (result.is_err()) {
271 return make_error<T*>(monitoring_error_code::resource_unavailable,
272 "Failed to allocate memory for object");
273 }
274
275 void* ptr = result.value();
276 T* obj = new (ptr) T(std::forward<Args>(args)...);
277
278 return common::ok(obj);
279 }

289 if (obj == nullptr) {
290 return make_void_error(monitoring_error_code::invalid_argument,
291 "Cannot deallocate null object");
292 }

338private:
339 void initialize_pool() {
340 size_t chunk_size = total_blocks_ * block_size_;
341 void* chunk = std::aligned_alloc(config_.alignment, chunk_size);
342
343 if (chunk == nullptr) {
344 throw std::bad_alloc();
345 }
346
347 memory_chunks_.push_back(chunk);
348
349 // Initialize free block list
350 free_blocks_.reserve(total_blocks_);
351 char* ptr = static_cast<char*>(chunk);
352 for (size_t i = 0; i < total_blocks_; ++i) {
353 free_blocks_.push_back(ptr + i * block_size_);
354 }
355 }
356
357 bool grow_pool() {
358 if (config_.max_blocks > 0 && total_blocks_ >= config_.max_blocks) {
359 return false;
360 }
361
362 size_t new_blocks = std::min(total_blocks_, config_.max_blocks - total_blocks_);
363 if (new_blocks == 0) {
364 new_blocks = total_blocks_; // Double the size
365 }
366
367 size_t chunk_size = new_blocks * block_size_;
368 void* chunk = std::aligned_alloc(config_.alignment, chunk_size);
369
370 if (chunk == nullptr) {
371 return false;
372 }
373
374 memory_chunks_.push_back(chunk);
375
376 char* ptr = static_cast<char*>(chunk);
377 for (size_t i = 0; i < new_blocks; ++i) {
378 free_blocks_.push_back(ptr + i * block_size_);
379 }
380
381 total_blocks_ += new_blocks;
382 return true;
383 }
384
385 bool is_owned_block(void* ptr) const {
386 for (size_t i = 0; i < memory_chunks_.size(); ++i) {
387 char* chunk_start = static_cast<char*>(memory_chunks_[i]);
388 size_t chunk_blocks = (i == 0) ? config_.initial_blocks :
389 ((total_blocks_ - config_.initial_blocks) / (memory_chunks_.size() - 1));
390 char* chunk_end = chunk_start + chunk_blocks * block_size_;
391
392 if (ptr >= chunk_start && ptr < chunk_end) {
393 // Verify alignment
394 size_t offset = static_cast<char*>(ptr) - chunk_start;
395 if (offset % block_size_ == 0) {
396 return true;
397 }
398 }
399 }
400 return false;
401 }
402
403 void update_peak_usage() {
404 size_t current_usage = total_blocks_ - free_blocks_.size();
405 size_t peak = stats_.peak_usage.load();
406 while (current_usage > peak) {
407 if (stats_.peak_usage.compare_exchange_weak(peak, current_usage)) {
408 break;
409 }
410 }
411 }

417 std::vector<void*> memory_chunks_;

419 mutable memory_pool_statistics stats_;
