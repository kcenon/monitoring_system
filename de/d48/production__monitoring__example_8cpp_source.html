151 std::cout << " - Grouping: enabled" << std::endl;

156 health_config.check_interval = 5000ms;
157 health_config.enable_auto_recovery = true;

162 std::cout << std::endl;
163
164 // =====================================================================
165 // Section 2: Initialize Monitoring Components
166 // =====================================================================

247 health_mon.start();
248 std::cout << " [OK] Health monitor started" << std::endl;
249
250 alert_mgr.start();
251 std::cout << " [OK] Alert manager started" << std::endl;
252 std::cout << std::endl;
253
254 // Install signal handlers
255 std::signal(SIGINT, signal_handler);
256 std::signal(SIGTERM, signal_handler);
257
258 std::cout << "7. Monitoring Active (Ctrl+C to shutdown)" << std::endl;
259 std::cout << " ========================================" << std::endl;
260 std::cout << std::endl;
261
262 // =====================================================================
263 // Section 7: Workload Simulation
264 // =====================================================================
265 int iteration = 0;
266 while (!shutdown_requested && iteration < 10) {
267 std::cout << " Iteration " << (iteration + 1) << "/10" << std::endl;
268
269 // Collect performance metrics
270 {
271 auto timer = perf_monitor.time_operation("iteration_" + std::to_string(iteration));
272 std::this_thread::sleep_for(200ms);
273 }
274
275 // Check health
276 auto health_result = health_mon.check_health();
277 if (health_result.status == health_status::healthy) {
278 std::cout << " Health: Healthy" << std::endl;
279 }
280
281 // Get system metrics
282 auto system_metrics = perf_monitor.get_system_monitor().get_current_metrics();
283 if (system_metrics.is_ok()) {
284 const auto& metrics = system_metrics.value();
285 std::cout << " CPU: " << metrics.cpu_usage_percent
286 << "%, Memory: " << (metrics.memory_usage_bytes / (1024.0 * 1024.0))
287 << " MB" << std::endl;
288
289 // Process metrics for alerting
290 alert_mgr.process_metric("cpu_usage", metrics.cpu_usage_percent);
291 }
292
293 std::cout << std::endl;
294 std::this_thread::sleep_for(2s);
295 iteration++;
296 }
297
298 // =====================================================================
299 // Section 8: Graceful Shutdown
300 // =====================================================================
301 std::cout << "8. Graceful Shutdown" << std::endl;
302 std::cout << " ==================" << std::endl;
303 std::cout << std::endl;
304
305 alert_mgr.stop();
306 std::cout << " [OK] Alert manager stopped" << std::endl;
307
308 health_mon.stop();
309 std::cout << " [OK] Health monitor stopped" << std::endl;
310
311 perf_monitor.cleanup();
312 std::cout << " [OK] Performance monitor cleaned up" << std::endl;
313
314 storage->flush();
315 std::cout << " [OK] Storage flushed" << std::endl;
316 std::cout << std::endl;
317
318 std::cout << "=== Production Monitoring Completed Successfully ===" << std::endl;
319
320 } catch (const std::exception& e) {
321 std::cerr << "Exception: " << e.what() << std::endl;
322 return 1;
323 }
324
325 return 0;
326}
