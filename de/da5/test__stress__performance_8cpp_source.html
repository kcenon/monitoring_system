94 const int NUM_THREADS = 100;
95 const int OPERATIONS_PER_THREAD = 10000;
96 const auto TEST_DURATION = 30s;
97
98 // Setup components
99 auto tracer = std::make_unique<distributed_tracer>();

102 // Metrics collection
103 std::atomic<int64_t> total_operations{0};
104 std::atomic<int64_t> failed_operations{0};
105 std::atomic<int64_t> total_latency_us{0};

107 std::mutex latency_mutex;
108
109 // Start time
110 auto start_time = std::chrono::steady_clock::now();
111
112 // Launch worker threads
113 std::vector<std::thread> workers;
114 for (int t = 0; t < NUM_THREADS; ++t) {
115 workers.emplace_back([&, t]() {
116 std::random_device rd;
117 std::mt19937 gen(rd());
118 std::uniform_int_distribution<> dis(1, 100);
119
120 for (int i = 0; i < OPERATIONS_PER_THREAD; ++i) {
121 auto op_start = std::chrono::high_resolution_clock::now();
122

133 span_result.value()->tags["operation_id"] = std::to_string(i);

136 } else {

138 }
139

200TEST_F(StressPerformanceTest, MemoryLeakDetectionTest) {
201 const int ITERATIONS = 1000;
202 const int OBJECTS_PER_ITERATION = 100;
203
204 // Track memory usage
205 std::vector<size_t> memory_samples;
206
207 for (int iter = 0; iter < ITERATIONS; ++iter) {
208 // Create and destroy many objects
209 std::vector<std::unique_ptr<distributed_tracer>> tracers;
210 std::vector<std::unique_ptr<circuit_breaker<bool>>> breakers;
211
212 circuit_breaker_config cb_config;
213 cb_config.failure_threshold = 3;
214 cb_config.reset_timeout = 100ms;
215
216 for (int i = 0; i < OBJECTS_PER_ITERATION; ++i) {
217 tracers.push_back(std::make_unique<distributed_tracer>());
218 breakers.push_back(std::make_unique<circuit_breaker<bool>>("breaker_" + std::to_string(i), cb_config));
219
220 // Create spans
221 auto span = tracers.back()->start_span("test_span_" + std::to_string(i));
222 if (span.is_ok()) {
223 span.value()->tags["iteration"] = std::to_string(iter);
224 }
225 }
226
227 // Clear objects (should free memory)
228 tracers.clear();
229 breakers.clear();
230
231 // Sample memory usage periodically
232 if (iter % 100 == 0) {
233 memory_samples.push_back(GetCurrentMemoryUsage());
234 }
235 }
236
237 // Analyze memory trend
238 if (memory_samples.size() > 2) {
239 // Check if memory is growing linearly (indicates leak)
240 double correlation = 0;
241 double mean_x = memory_samples.size() / 2.0;
242 double mean_y = std::accumulate(memory_samples.begin(), memory_samples.end(), 0.0) / memory_samples.size();
243
244 double sum_xy = 0, sum_xx = 0;
245 for (size_t i = 0; i < memory_samples.size(); ++i) {
246 sum_xy += (i - mean_x) * (memory_samples[i] - mean_y);
247 sum_xx += (i - mean_x) * (i - mean_x);
248 }
249
250 if (sum_xx > 0) {
251 correlation = sum_xy / sqrt(sum_xx);
252 }
253
254 // High positive correlation indicates memory leak
255 EXPECT_LT(correlation, 0.8) << "Potential memory leak detected";
256 }

273 std::atomic<int> counter{0};
274 std::atomic<bool> race_detected{false};
275
276 // Synchronization for thread start
277 std::mutex start_mutex;
278 std::condition_variable start_cv;
279 std::atomic<int> ready_threads{0};

281
282 // Launch concurrent threads

285 threads.emplace_back([&, t]() {

289 ready_threads++;
290 if (ready_threads == NUM_THREADS) {
291 start_flag = true;
292 start_cv.notify_all();
293 } else {
294 start_cv.wait(lock, [&] { return start_flag; });
295 }
296 }
297
298 for (int i = 0; i < OPERATIONS; ++i) {
299 // Concurrent writes
300 metrics_snapshot snapshot;
301 snapshot.add_metric("thread_" + std::to_string(t), i);
302
303 auto result = storage->store(snapshot);
304 counter++;
305
306 // Check storage operation succeeded
307 if (!result.is_ok()) {
308 race_detected = true;
309 }
310 }
311 });
312 }
313
314 // Wait for completion
315 for (auto& thread : threads) {
316 thread.join();
317 }
318
319 // Verify results
320 EXPECT_FALSE(race_detected) << "Race condition detected";
321 EXPECT_EQ(counter, NUM_THREADS * OPERATIONS);
322 EXPECT_LE(storage->size(), config.max_capacity);

474 EXPECT_LT(max_latency, 10000); // Max under 10 seconds

481TEST_F(StressPerformanceTest, DeadlockDetectionTest) {
482 const int NUM_THREADS = 10;
483 const int ITERATIONS = 100;
484
485 // Shared resources with potential for deadlock
486 std::timed_mutex mutex1, mutex2;
487 std::atomic<int> deadlock_timeouts{0};
488
489 std::vector<std::thread> threads;
490 for (int t = 0; t < NUM_THREADS; ++t) {
491 threads.emplace_back([&, t]() {
492 for (int i = 0; i < ITERATIONS; ++i) {
493 // Alternate lock order to create potential deadlock
494 if (t % 2 == 0) {
495 // Even threads: lock mutex1 then mutex2
496 if (mutex1.try_lock_for(100ms)) {
497 std::lock_guard<std::timed_mutex> lock1(mutex1, std::adopt_lock);
498 if (mutex2.try_lock_for(100ms)) {
499 std::lock_guard<std::timed_mutex> lock2(mutex2, std::adopt_lock);
500 // Critical section
501 std::this_thread::sleep_for(1ms);
502 } else {
503 deadlock_timeouts++;
504 }
505 } else {
506 deadlock_timeouts++;
507 }
508 } else {
509 // Odd threads: lock mutex2 then mutex1
510 if (mutex2.try_lock_for(100ms)) {
511 std::lock_guard<std::timed_mutex> lock2(mutex2, std::adopt_lock);
512 if (mutex1.try_lock_for(100ms)) {
513 std::lock_guard<std::timed_mutex> lock1(mutex1, std::adopt_lock);
514 // Critical section
515 std::this_thread::sleep_for(1ms);
516 } else {
517 deadlock_timeouts++;
518 }
519 } else {
520 deadlock_timeouts++;
521 }
522 }
523 }
524 });
525 }
526
527 // Set timeout for test
528 auto start = std::chrono::steady_clock::now();
529 bool all_finished = true;
530
531 for (auto& thread : threads) {
532 if (thread.joinable()) {
533 thread.join();
534 }
535
536 // Check if test is taking too long (potential deadlock)
537 if (std::chrono::steady_clock::now() - start > 30s) {
538 all_finished = false;

540 }
541 }

543 EXPECT_TRUE(all_finished) << "Potential deadlock detected - test timed out";
544 std::cout << "Deadlock timeouts encountered: " << deadlock_timeouts << std::endl;
