338class health_check_event : public event_base {
339public:
340 enum class health_status {
341 healthy,
342 degraded,
343 unhealthy,
344 unknown
345 };
346
347 struct health_check_result {
348 std::string check_name;
349 health_status status;
350 std::string message;
351 std::chrono::milliseconds response_time;
352 std::optional<std::unordered_map<std::string, std::string>> metadata;
353 };
354
355 health_check_event(const std::string& component,
356 std::vector<health_check_result> results)
357 : component_(component), results_(std::move(results)) {}
358
359 std::string get_type_name() const override {
360 return "health_check_event";
361 }
362
363 const std::string& get_component() const { return component_; }
364 const std::vector<health_check_result>& get_results() const { return results_; }
365
366 health_status get_overall_status() const {
367 health_status overall = health_status::healthy;
368 for (const auto& result : results_) {
369 if (result.status == health_status::unhealthy) {
370 return health_status::unhealthy;
371 }
372 if (result.status == health_status::degraded) {
373 overall = health_status::degraded;
374 }
375 if (result.status == health_status::unknown && overall == health_status::healthy) {
376 overall = health_status::unknown;
377 }
378 }
379 return overall;
380 }
381
382private:
383 std::string component_;
384 std::vector<health_check_result> results_;
385};
