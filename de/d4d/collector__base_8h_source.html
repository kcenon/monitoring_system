124 bool initialize(const config_map& config) {
125 // Parse common configuration
126 if (auto it = config.find("enabled"); it != config.end()) {
127 enabled_ = (it->second == "true" || it->second == "1");
128 }
129
130 // Delegate to derived class for specific initialization
131 return derived().do_initialize(config);
132 }
133
138 std::vector<metric> collect() {
139 if (!enabled_) {
140 return {};
141 }
142
143 try {
144 auto metrics = derived().do_collect();
145 ++collection_count_;
146 return metrics;
147 } catch (...) {
148 ++collection_errors_;
149 return {};
150 }
151 }

157 std::string get_name() const {
158 return Derived::collector_name;
159 }
160
165 std::vector<std::string> get_metric_types() const {
166 return derived().do_get_metric_types();
167 }
168
173 bool is_healthy() const {
174 if (!enabled_) {
175 return true; // Disabled collectors are considered healthy
176 }
177 return derived().is_available();
178 }
179
184 stats_map get_statistics() const {
185 std::lock_guard<std::mutex> lock(stats_mutex_);
186 stats_map stats;
187
188 // Common statistics
189 stats["enabled"] = enabled_ ? 1.0 : 0.0;
190 stats["collection_count"] = static_cast<double>(collection_count_.load());
191 stats["collection_errors"] = static_cast<double>(collection_errors_.load());
192
193 // Let derived class add specific statistics
194 derived().do_add_statistics(stats);
195
196 return stats;
197 }
198
203 bool is_enabled() const { return enabled_; }

226 metric create_base_metric(const std::string& name, double value,
227 const std::unordered_map<std::string, std::string>& tags = {},
228 const std::string& /* unit */ = "") const {
229 metric m;
230 m.name = name;
231 m.value = value;
232 m.timestamp = std::chrono::system_clock::now();
233 m.tags = tags;
234 m.tags["collector"] = Derived::collector_name;
235 return m;
236 }

246 private:
