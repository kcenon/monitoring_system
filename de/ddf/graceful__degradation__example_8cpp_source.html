57// Simulated unreliable service
58class unreliable_service {
59private:
60 std::mt19937 rng_;
61 std::uniform_real_distribution<> dist_;
62 double failure_rate_;
63 std::atomic<int> call_count_{0};
64
65public:
66 explicit unreliable_service(double failure_rate)
67 : rng_(std::random_device{}())
68 , dist_(0.0, 1.0)
69 , failure_rate_(failure_rate) {}
70
71 kcenon::common::Result<std::string> call() {
72 call_count_++;
73 std::this_thread::sleep_for(50ms);
74
75 if (dist_(rng_) < failure_rate_) {
76 return kcenon::common::Result<std::string>::err(
77 error_info{monitoring_error_code::service_unavailable,
78 "Service temporarily unavailable"}.to_common_error()
79 );
80 }
81
82 return kcenon::common::ok(std::string("Service response: SUCCESS"));
83 }
84
85 void set_failure_rate(double rate) {
86 failure_rate_ = rate;
87 }
88};
