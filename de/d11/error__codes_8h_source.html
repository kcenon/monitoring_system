213inline std::string get_error_details(monitoring_error_code code) {
214 switch (code) {
215 case monitoring_error_code::collector_not_found:
216 return "The specified collector was not found. Check collector name and ensure it's registered.";
217 case monitoring_error_code::storage_full:
218 return "Storage capacity exceeded. Consider increasing buffer size or reducing collection frequency.";
219 case monitoring_error_code::invalid_configuration:
220 return "Configuration validation failed. Review configuration parameters and constraints.";
221 case monitoring_error_code::thread_system_not_available:
222 return "Thread system integration not available. Ensure thread_system is properly linked.";
223 default:
224 return error_code_to_string(code);
225 }
