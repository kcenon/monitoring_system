41// Example function that may fail
42kcenon::common::Result<double> divide(double a, double b) {
43 if (b == 0) {
44 return kcenon::common::Result<double>::err(error_info(monitoring_error_code::invalid_configuration, "Division by zero").to_common_error());
45 }
46 return kcenon::common::ok(a / b);
47}

49// Example function using kcenon::common::VoidResult
50kcenon::common::VoidResult validate_range(double value, double min, double max) {
51 if (value < min || value > max) {
52 error_info err(monitoring_error_code::invalid_configuration,
53 "Value out of range [" + std::to_string(min) + ", " + std::to_string(max) + "]");
54 return kcenon::common::VoidResult::err(err.to_common_error());
55 }
56 return kcenon::common::ok();
57}

59// Example using monadic operations
60kcenon::common::Result<std::string> process_metric(double value) {
61 // Chain operations using map and and_then
62 return divide(100.0, value)
63 .map([](double x) { return x * 2; })
64 .and_then([](double x) {
65 if (x > 50) {
66 return kcenon::common::ok("High value: " + std::to_string(x));
67 }
68 return kcenon::common::ok("Normal value: " + std::to_string(x));
69 });
70}
