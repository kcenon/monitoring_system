195 common::Result<otel_span_data> convert_span(const trace_span& span) {
196 otel_span_data otel_span;
197
198 // Convert basic span information
199 otel_span.name = span.operation_name;
200 otel_span.context = otel_span_context(span.trace_id, span.span_id);
201
202 if (!span.parent_span_id.empty()) {
203 otel_span.parent_context = otel_span_context(span.trace_id, span.parent_span_id);
204 }
205
206 otel_span.start_time = span.start_time;
207 otel_span.end_time = span.end_time;
208 otel_span.resource = resource_;
209
210 // First determine span kind from tags
211 auto kind_it = span.tags.find("span.kind");
212 if (kind_it != span.tags.end()) {
213 otel_span.kind = parse_span_kind(kind_it->second);
214 }
215
216 // Set status from tags
217 auto error_it = span.tags.find("error");
218 if (error_it != span.tags.end() && error_it->second == "true") {
219 otel_span.status_code = otel_status_code::error;
220 auto error_msg_it = span.tags.find("error.message");
221 if (error_msg_it != span.tags.end()) {
222 otel_span.status_message = error_msg_it->second;
223 }
224 } else {
225 otel_span.status_code = otel_status_code::ok;
226 }
227
228 // Convert tags to attributes, excluding special OpenTelemetry fields
229 for (const auto& [key, value] : span.tags) {
230 // Skip special fields that are handled above
231 if (key != "span.kind" && key != "error" && key != "error.message") {
232 otel_span.add_attribute(key, value);
233 }
234 }
235
236 return common::Result<otel_span_data>::ok(std::move(otel_span));
237 }
238
242 common::Result<std::vector<otel_span_data>> convert_spans(const std::vector<trace_span>& spans) {
243 std::vector<otel_span_data> otel_spans;
244 otel_spans.reserve(spans.size());
245
246 for (const auto& span : spans) {
247 auto convert_result = convert_span(span);
248 if (convert_result.is_err()) {
249 return common::Result<std::vector<otel_span_data>>::err(error_info(monitoring_error_code::processing_failed,
250 "Failed to convert span: " + convert_result.error().message).to_common_error());
251 }
252 otel_spans.push_back(convert_result.value());
253 }
254
255 return common::Result<std::vector<otel_span_data>>::ok(std::move(otel_spans));
256 }
257
261 common::Result<otel_span_context> create_context(const trace_context& context) {
262 return common::Result<otel_span_context>::ok(otel_span_context(context.trace_id, context.span_id));
263 }
264
265private:
266 otel_span_kind parse_span_kind(const std::string& kind_str) {
267 if (kind_str == "server") return otel_span_kind::server;
268 if (kind_str == "client") return otel_span_kind::client;
269 if (kind_str == "producer") return otel_span_kind::producer;
270 if (kind_str == "consumer") return otel_span_kind::consumer;
271 if (kind_str == "internal") return otel_span_kind::internal;
272 return otel_span_kind::unspecified;
273 }

290 common::Result<std::vector<otel_metric_data>> convert_metrics(const metrics_snapshot& snapshot) {
291 std::vector<otel_metric_data> otel_metrics;
292
293 for (const auto& metric_value : snapshot.metrics) {
294 otel_metric_data metric;
295 metric.name = metric_value.name;
296 metric.value = metric_value.value;
297 metric.timestamp = metric_value.timestamp;
298 metric.resource = resource_;
299
300 // Add tags as attributes
301 for (const auto& [tag_name, tag_value] : metric_value.tags) {
302 metric.add_attribute(tag_name, tag_value);
303 }
304
305 // Add common resource attributes
306 metric.add_attribute("service.name", resource_.get_attribute("service.name").value_or("unknown"));
307 metric.add_attribute("service.version", resource_.get_attribute("service.version").value_or("unknown"));
308
309 otel_metrics.push_back(std::move(metric));
310 }
311
312 return common::Result<std::vector<otel_metric_data>>::ok(std::move(otel_metrics));
313 }
314
318 common::Result<std::vector<otel_metric_data>> convert_monitoring_data(const monitoring_data& data) {
319 std::vector<otel_metric_data> otel_metrics;
320
321 for (const auto& [name, value] : data.get_metrics()) {
322 otel_metric_data metric;
323 metric.name = name;
324 metric.value = value;
325 metric.timestamp = data.get_timestamp();
326 metric.resource = resource_;
327
328 // Add tags as attributes
329 for (const auto& [tag_name, tag_value] : data.get_tags()) {
330 metric.add_attribute(tag_name, tag_value);
331 }
332
333 otel_metrics.push_back(std::move(metric));
334 }
335
336 return common::Result<std::vector<otel_metric_data>>::ok(std::move(otel_metrics));
337 }

347struct opentelemetry_exporter_config {
348 std::string endpoint{"http://localhost:4317"};
349 std::string protocol{"grpc"}; // grpc, http/protobuf, http/json
350 std::chrono::milliseconds timeout{30000};
351 std::chrono::milliseconds export_interval{5000};
352 std::size_t max_batch_size{512};
353 std::unordered_map<std::string, std::string> headers;
354 bool compression_enabled{true};
355 std::string compression_type{"gzip"};
356
357 common::VoidResult validate() const {
358 if (endpoint.empty()) {
359 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration,
360 "Exporter endpoint cannot be empty", "monitoring_system").to_common_error());
361 }
362 if (protocol != "grpc" && protocol != "http/protobuf" && protocol != "http/json") {
363 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration,
364 "Invalid protocol: " + protocol, "monitoring_system").to_common_error());
365 }
366 if (timeout <= std::chrono::milliseconds(0)) {
367 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration,
368 "Timeout must be positive", "monitoring_system").to_common_error());
369 }
370 if (max_batch_size == 0) {
371 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration,
372 "Batch size must be positive", "monitoring_system").to_common_error());
373 }
374 return common::ok();
375 }

382class opentelemetry_compatibility_layer {
383public:
384 explicit opentelemetry_compatibility_layer(const otel_resource& resource)
385 : resource_(resource)
386 , tracer_adapter_(resource)
387 , metrics_adapter_(resource) {}
388
392 common::VoidResult initialize() {
393 std::lock_guard<std::mutex> lock(mutex_);
394 if (initialized_) {
395 return common::VoidResult::err(error_info(monitoring_error_code::already_exists,
396 "Compatibility layer already initialized", "monitoring_system").to_common_error());
397 }
398
399 initialized_ = true;
400 return common::ok();
401 }
402
406 common::VoidResult shutdown() {
407 std::lock_guard<std::mutex> lock(mutex_);
408 if (!initialized_) {
409 return common::ok();
410 }
411
412 // Flush any pending data without re-locking
413 // In a real implementation, this would send data to OpenTelemetry collectors
414 pending_spans_.clear();
415 pending_metrics_.clear();
416
417 initialized_ = false;
418 return common::ok();
419 }
420
424 common::VoidResult export_spans(const std::vector<trace_span>& spans) {
425 if (!initialized_) {
426 return common::VoidResult::err(error_info(monitoring_error_code::invalid_state,
427 "Compatibility layer not initialized", "monitoring_system").to_common_error());
428 }
429
430 auto convert_result = tracer_adapter_.convert_spans(spans);
431 if (convert_result.is_err()) {
432 return common::VoidResult::err(error_info(monitoring_error_code::processing_failed,
433 "Failed to convert spans: " + convert_result.error().message).to_common_error());
434 }
435
436 // Store converted spans for batching
437 std::lock_guard<std::mutex> lock(mutex_);
438 const auto& otel_spans = convert_result.value();
439 pending_spans_.insert(pending_spans_.end(), otel_spans.begin(), otel_spans.end());
440
441 return common::ok();
442 }
443
447 common::VoidResult export_metrics(const monitoring_data& data) {
448 if (!initialized_) {
449 return common::VoidResult::err(error_info(monitoring_error_code::invalid_state,
450 "Compatibility layer not initialized", "monitoring_system").to_common_error());
451 }
452
453 auto convert_result = metrics_adapter_.convert_monitoring_data(data);
454 if (convert_result.is_err()) {
455 return common::VoidResult::err(error_info(monitoring_error_code::processing_failed,
456 "Failed to convert metrics: " + convert_result.error().message).to_common_error());
457 }
458
459 // Store converted metrics for batching
460 std::lock_guard<std::mutex> lock(mutex_);
461 const auto& otel_metrics = convert_result.value();
462 pending_metrics_.insert(pending_metrics_.end(), otel_metrics.begin(), otel_metrics.end());
463
464 return common::ok();
465 }
466
470 common::VoidResult flush() {
471 std::lock_guard<std::mutex> lock(mutex_);
472
473 // In a real implementation, this would send data to OpenTelemetry collectors
474 // For now, we'll just clear the pending data
475 pending_spans_.clear();
476 pending_metrics_.clear();
477
478 return common::ok();
479 }
