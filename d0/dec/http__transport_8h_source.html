192 url_parts parse_url(const std::string& url) {
193 url_parts parts;
194
195 // Find scheme
196 auto scheme_end = url.find("://");
197 if (scheme_end == std::string::npos) {
198 return parts;
199 }
200 parts.scheme = url.substr(0, scheme_end);
201
202 // Find host and port
203 auto host_start = scheme_end + 3;
204 auto path_start = url.find('/', host_start);
205 if (path_start == std::string::npos) {
206 path_start = url.length();
207 }
208
209 auto host_port = url.substr(host_start, path_start - host_start);
210 auto colon_pos = host_port.find(':');
211 if (colon_pos != std::string::npos) {
212 parts.host = host_port.substr(0, colon_pos);
213 try {
214 parts.port = std::stoi(host_port.substr(colon_pos + 1));
215 } catch (...) {
216 return parts;
217 }
218 } else {
219 parts.host = host_port;
220 parts.port = (parts.scheme == "https") ? 443 : 80;
221 }
222
223 // Extract path
224 if (path_start < url.length()) {
225 parts.path = url.substr(path_start);
226 } else {
227 parts.path = "/";
228 }
229
230 parts.valid = !parts.host.empty();
231 return parts;
232 }

245class network_http_transport : public http_transport {
246private:
247 std::shared_ptr<network_system::core::http_client> client_;
248
249public:
250 explicit network_http_transport(std::chrono::milliseconds timeout = std::chrono::milliseconds(30000))
251 : client_(std::make_shared<network_system::core::http_client>(timeout)) {}
252
253 result<http_response> send(const http_request& request) override {
254 // Convert headers from unordered_map to map
255 std::map<std::string, std::string> headers(request.headers.begin(), request.headers.end());
256
257 // Make the actual HTTP request using network_system
258 network_system::core::Result<network_system::core::internal::http_response> net_result;
259
260 if (request.method == "GET") {
261 net_result = client_->get(request.url, {}, headers);
262 } else if (request.method == "POST") {
263 net_result = client_->post(request.url, request.body, headers);
264 } else if (request.method == "PUT") {
265 std::string body_str(request.body.begin(), request.body.end());
266 net_result = client_->put(request.url, body_str, headers);
267 } else if (request.method == "DELETE") {
268 net_result = client_->del(request.url, headers);
269 } else if (request.method == "HEAD") {
270 net_result = client_->head(request.url, headers);
271 } else if (request.method == "PATCH") {
272 std::string body_str(request.body.begin(), request.body.end());
273 net_result = client_->patch(request.url, body_str, headers);
274 } else {
275 return make_error<http_response>(monitoring_error_code::invalid_configuration,
276 "Unsupported HTTP method: " + request.method);
277 }
278
279 if (!net_result) {
280 return make_error<http_response>(monitoring_error_code::operation_failed,
281 "HTTP request failed: " + net_result.error().message);
282 }
283
284 // Convert network_system response to monitoring response
285 http_response response;
286 response.status_code = net_result->status_code;
287 response.status_message = net_result->status_text;
288 response.body = net_result->body;
289
290 // Convert headers
291 for (const auto& [key, value] : net_result->headers) {
292 response.headers[key] = value;
293 }
294
295 return make_success(response);
296 }
297
298 bool is_available() const override {
299 return true;
300 }
301
302 std::string name() const override {
303 return "network_system";
304 }
305};
