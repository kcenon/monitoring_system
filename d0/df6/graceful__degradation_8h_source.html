59struct graceful_degradation_metrics {
60 std::atomic<size_t> total_degradations{0};
61 std::atomic<size_t> successful_degradations{0};
62 std::atomic<size_t> failed_degradations{0};
63 std::atomic<size_t> recovery_attempts{0};
64 std::atomic<size_t> successful_recoveries{0};
65
66 graceful_degradation_metrics() = default;
67
68 graceful_degradation_metrics(const graceful_degradation_metrics& other)
69 : total_degradations(other.total_degradations.load())
70 , successful_degradations(other.successful_degradations.load())
71 , failed_degradations(other.failed_degradations.load())
72 , recovery_attempts(other.recovery_attempts.load())
73 , successful_recoveries(other.successful_recoveries.load()) {}
74
75 graceful_degradation_metrics& operator=(const graceful_degradation_metrics& other) {
76 if (this != &other) {
77 total_degradations = other.total_degradations.load();
78 successful_degradations = other.successful_degradations.load();
79 failed_degradations = other.failed_degradations.load();
80 recovery_attempts = other.recovery_attempts.load();
81 successful_recoveries = other.successful_recoveries.load();
82 }
83 return *this;
84 }

137class graceful_degradation_manager {
138public:
139 graceful_degradation_manager() : name_("default") {}
140
141 explicit graceful_degradation_manager(const std::string& name) : name_(name) {}
142
146 common::VoidResult register_service(const service_config& config) {
147 if (!config.validate()) {
148 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration, "Invalid service configuration").to_common_error());
149 }
150
151 std::lock_guard<std::mutex> lock(mutex_);
152 if (services_.find(config.name) != services_.end()) {
153 return common::VoidResult::err(static_cast<int>(monitoring_error_code::already_exists),
154 "Service already registered: " + config.name);
155 }
156
157 service_state state;
158 state.config = config;
159 state.current_level = degradation_level::normal;
160 state.last_state_change = std::chrono::steady_clock::now();
161 services_[config.name] = state;
162
163 return common::ok();
164 }
165
169 common::VoidResult unregister_service(const std::string& name) {
170 std::lock_guard<std::mutex> lock(mutex_);
171 auto it = services_.find(name);
172 if (it == services_.end()) {
173 return common::VoidResult::err(static_cast<int>(monitoring_error_code::not_found),
174 "Service not found: " + name);
175 }
176 services_.erase(it);
177 return common::ok();
178 }
179
183 common::VoidResult degrade_service(const std::string& name, degradation_level level,
184 const std::string& reason) {
185 std::lock_guard<std::mutex> lock(mutex_);
186 auto it = services_.find(name);
187 if (it == services_.end()) {
188 metrics_.failed_degradations++;
189 return common::VoidResult::err(static_cast<int>(monitoring_error_code::not_found),
190 "Service not found: " + name);
191 }
192
193 metrics_.total_degradations++;
194 it->second.current_level = level;
195 it->second.last_degradation_reason = reason;
196 it->second.last_state_change = std::chrono::steady_clock::now();
197 metrics_.successful_degradations++;
198
199 return common::ok();
200 }
201
205 common::VoidResult recover_service(const std::string& name) {
206 std::lock_guard<std::mutex> lock(mutex_);
207 auto it = services_.find(name);
208 if (it == services_.end()) {
209 return common::VoidResult::err(static_cast<int>(monitoring_error_code::not_found),
210 "Service not found: " + name);
211 }
212
213 metrics_.recovery_attempts++;
214 it->second.current_level = degradation_level::normal;
215 it->second.last_degradation_reason.clear();
216 it->second.last_state_change = std::chrono::steady_clock::now();
217 metrics_.successful_recoveries++;
218
219 return common::ok();
220 }
221
225 common::VoidResult recover_all_services() {
226 std::lock_guard<std::mutex> lock(mutex_);
227 for (auto& [name, state] : services_) {
228 metrics_.recovery_attempts++;
229 state.current_level = degradation_level::normal;
230 state.last_degradation_reason.clear();
231 state.last_state_change = std::chrono::steady_clock::now();
232 metrics_.successful_recoveries++;
233 }
234 return common::ok();
235 }
236
240 degradation_level get_service_level(const std::string& name) const {
241 std::lock_guard<std::mutex> lock(mutex_);
242 auto it = services_.find(name);
243 if (it == services_.end()) {
244 return degradation_level::normal;
245 }
246 return it->second.current_level;
247 }
248
252 common::VoidResult add_degradation_plan(const degradation_plan& plan) {
253 if (!plan.validate()) {
254 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration, "Invalid degradation plan").to_common_error());
255 }
256
257 std::lock_guard<std::mutex> lock(mutex_);
258 plans_[plan.name] = plan;
259 return common::ok();
260 }
261
265 common::VoidResult execute_plan(const std::string& plan_name, const std::string& reason) {
266 std::lock_guard<std::mutex> lock(mutex_);
267 auto it = plans_.find(plan_name);
268 if (it == plans_.end()) {
269 return common::VoidResult::err(static_cast<int>(monitoring_error_code::not_found),
270 "Plan not found: " + plan_name);
271 }
272
273 const auto& plan = it->second;
274
275 // Degrade services to maintain to target level
276 for (const auto& service_name : plan.services_to_maintain) {
277 auto service_it = services_.find(service_name);
278 if (service_it != services_.end()) {
279 metrics_.total_degradations++;
280 service_it->second.current_level = plan.target_level;
281 service_it->second.last_degradation_reason = reason;
282 service_it->second.last_state_change = std::chrono::steady_clock::now();
283 metrics_.successful_degradations++;
284 }
285 }
286
287 // Disable (set to emergency) services to disable
288 for (const auto& service_name : plan.services_to_disable) {
289 auto service_it = services_.find(service_name);
290 if (service_it != services_.end()) {
291 metrics_.total_degradations++;
292 service_it->second.current_level = degradation_level::emergency;
293 service_it->second.last_degradation_reason = reason;
294 service_it->second.last_state_change = std::chrono::steady_clock::now();
295 metrics_.successful_degradations++;
296 }
297 }
298
299 return common::ok();
300 }
301
305 common::Result<bool> is_healthy() const {
306 std::lock_guard<std::mutex> lock(mutex_);
307 if (services_.empty()) {
308 return common::ok(true);
309 }
310
311 size_t normal_count = 0;
312 for (const auto& [name, state] : services_) {
313 if (state.current_level == degradation_level::normal) {
314 normal_count++;
315 }
316 }
317
318 double healthy_ratio = static_cast<double>(normal_count) / static_cast<double>(services_.size());
319 return common::ok(healthy_ratio > 0.5);
320 }
321
325 graceful_degradation_metrics get_metrics() const {
326 return metrics_;
327 }
328
332 const std::string& get_name() const {
333 return name_;
334 }
335
339 std::vector<std::string> get_service_names() const {
340 std::lock_guard<std::mutex> lock(mutex_);
341 std::vector<std::string> names;
342 names.reserve(services_.size());
343 for (const auto& [name, state] : services_) {
344 names.push_back(name);
345 }
346 return names;
347 }

349private:
350 std::string name_;
351 mutable std::mutex mutex_;
352 std::unordered_map<std::string, service_state> services_;
353 std::unordered_map<std::string, degradation_plan> plans_;
354 mutable graceful_degradation_metrics metrics_;
355};

362template<typename T>
363class degradable_service {
364public:
365 using normal_operation = std::function<common::Result<T>()>;
366 using degraded_operation = std::function<common::Result<T>(degradation_level)>;
367
368 degradable_service(const std::string& name,
369 std::shared_ptr<graceful_degradation_manager> manager,
370 normal_operation normal_op,
371 degraded_operation degraded_op)
372 : name_(name)
373 , manager_(std::move(manager))
374 , normal_op_(std::move(normal_op))
375 , degraded_op_(std::move(degraded_op)) {}
376
380 common::Result<T> execute() {
381 if (!manager_) {
382 return normal_op_();
383 }
384
385 auto level = manager_->get_service_level(name_);
386 if (level == degradation_level::normal) {
387 return normal_op_();
388 }
389
390 if (degraded_op_) {
391 return degraded_op_(level);
392 }
393
394 return common::Result<T>::err(error_info(monitoring_error_code::service_degraded, "Service is degraded and no fallback available").to_common_error());
395 }
396
400 const std::string& get_name() const {
401 return name_;
402 }
403
404private:
405 std::string name_;
406 std::shared_ptr<graceful_degradation_manager> manager_;
407 normal_operation normal_op_;
408 degraded_operation degraded_op_;
409};

414class error_boundary_registry {
415public:
416 template<typename T>
417 void register_boundary(const std::string& name, std::shared_ptr<error_boundary<T>> boundary) {
418 std::lock_guard<std::mutex> lock(mutex_);
419 registry_[name] = std::move(boundary);
420 }
421
422 template<typename T>
423 std::shared_ptr<error_boundary<T>> get_boundary(const std::string& name) {
424 std::lock_guard<std::mutex> lock(mutex_);
425 auto it = registry_.find(name);
426 if (it != registry_.end()) {
427 try {
428 return std::any_cast<std::shared_ptr<error_boundary<T>>>(it->second);
429 } catch (const std::bad_any_cast&) {
430 return nullptr;
431 }
432 }
433 return nullptr;
434 }
435
436 void remove_boundary(const std::string& name) {
437 std::lock_guard<std::mutex> lock(mutex_);
438 registry_.erase(name);
439 }
440
441 std::vector<std::string> get_all_names() const {
442 std::lock_guard<std::mutex> lock(mutex_);
443 std::vector<std::string> names;
444 names.reserve(registry_.size());
445 for (const auto& [name, boundary] : registry_) {
446 names.push_back(name);
447 }
448 return names;
449 }
450
451 void clear() {
452 std::lock_guard<std::mutex> lock(mutex_);
453 registry_.clear();
454 }
455
456private:
457 mutable std::mutex mutex_;
458 std::unordered_map<std::string, std::any> registry_;
459};
