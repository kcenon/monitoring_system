60class imonitor_interface {
61public:
62 virtual ~imonitor_interface() = default;
63 virtual void record_metric(const std::string& name, double value) = 0;
64 virtual void get_health() = 0;

67// Current implementation with multiple inheritance
68class performance_monitor_old
69 : public metrics_collector_interface // Interface 1
70 , public imonitor_interface { // Interface 2
71public:
72 explicit performance_monitor_old(const std::string& name) : name_(name) {}
73
74 // Implementing metrics_collector_interface
75 std::string get_name() const override { return name_; }
76
77 void initialize() override {
78 std::cout << "[Old] Initializing monitor: " << name_ << std::endl;
79 }
80
81 void collect_metrics() override {
82 std::cout << "[Old] Collecting metrics..." << std::endl;
83 metric_count_++;
84 }
85
86 // Implementing imonitor_interface
87 void record_metric(const std::string& name, double value) override {
88 std::cout << "[Old] Recording metric: " << name << " = " << value << std::endl;
89 metrics_[name] = value;
90 }
91
92 void get_health() override {
93 std::cout << "[Old] Health check: OK" << std::endl;
94 }
95
96 // Problems with this approach:
97 // 1. Method name conflicts possible (both interfaces might have same method)
98 // 2. Single class must satisfy multiple interface contracts
99 // 3. Hard to test each interface independently
100 // 4. Violates Single Responsibility Principle
101
102private:
103 std::string name_;
104 int metric_count_ = 0;
105 std::unordered_map<std::string, double> metrics_;
106};

112// Step 1: Core implementation (no interfaces)
113class performance_monitor_impl {

122
123 void collect_metrics_internal() {

125 metric_count_++;
126 }
127
128 void record_metric_internal(const std::string& name, double value) {

130 metrics_[name] = value;
131 }
132
133 void check_health_internal() const {
134 std::cout << "[Core] Health check: " << (initialized_ ? "OK" : "NOT_INITIALIZED") << std::endl;
135 }

138 int get_metric_count() const { return metric_count_; }
139
140private:
141 std::string name_;

191// Step 4: Facade for unified access
192class performance_monitor_facade {
193public:
194 explicit performance_monitor_facade(const std::string& name) {
195 // Create core implementation
196 impl_ = std::make_shared<performance_monitor_impl>(name);
197
198 // Create adapters
199 metrics_adapter_ = std::make_shared<metrics_collector_adapter>(impl_);
200 imonitor_adapter_ = std::make_shared<imonitor_adapter>(impl_);
201 }
202
203 // Explicit interface access
204 metrics_collector_interface& as_metrics_collector() {
205 return *metrics_adapter_;
206 }
207
208 imonitor_interface& as_imonitor() {
209 return *imonitor_adapter_;
210 }
211
212 // Direct access to implementation for advanced use
213 performance_monitor_impl& impl() {
214 return *impl_;
215 }
216
217private:
218 std::shared_ptr<performance_monitor_impl> impl_;
219 std::shared_ptr<metrics_collector_adapter> metrics_adapter_;
220 std::shared_ptr<imonitor_adapter> imonitor_adapter_;
221};

313
314 // Old approach (multiple inheritance)
315 {
316 performance_monitor_old old_monitor("bench_old");
317 auto start = std::chrono::high_resolution_clock::now();
318
319 for (int i = 0; i < iterations; ++i) {
320 old_monitor.record_metric("test", 1.0);
321 }
322
323 auto end = std::chrono::high_resolution_clock::now();
324 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start);
325 std::cout << "Old (multiple inheritance): "
326 << duration.count() / iterations << " ns/call" << std::endl;
327 }
328
329 // New approach (facade + adapters)
330 {
331 performance_monitor_facade new_monitor("bench_new");
332 auto& imonitor = new_monitor.as_imonitor();
333 auto start = std::chrono::high_resolution_clock::now();
334
335 for (int i = 0; i < iterations; ++i) {
336 imonitor.record_metric("test", 1.0);
337 }
338
339 auto end = std::chrono::high_resolution_clock::now();
340 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start);
341 std::cout << "New (facade + adapters): "

343 }
