120struct trace_context {
121 std::string trace_id;
122 std::string span_id;
123 std::string trace_flags;
124 std::string trace_state;
125 std::unordered_map<std::string, std::string> baggage;
126
130 std::string to_w3c_traceparent() const {
131 // Version-TraceId-SpanId-TraceFlags
132 return "00-" + trace_id + "-" + span_id + "-" + trace_flags;
133 }
134
138 static common::Result<trace_context> from_w3c_traceparent(const std::string& header) {
139 // Parse format: version-traceid-spanid-traceflags
140 if (header.length() < 55) { // Minimum valid length
141 error_info err(kcenon::monitoring::monitoring_error_code::invalid_argument,
142 "Invalid traceparent header length");
143 return common::Result<trace_context>::err(err.to_common_error());
144 }
145
146 trace_context ctx;
147 // Simple parsing - production code would be more robust
148 size_t pos = 0;
149 size_t dash1 = header.find('-', pos);
150 size_t dash2 = header.find('-', dash1 + 1);
151 size_t dash3 = header.find('-', dash2 + 1);
152
153 if (dash1 == std::string::npos || dash2 == std::string::npos || dash3 == std::string::npos) {
154 error_info err(kcenon::monitoring::monitoring_error_code::invalid_argument,
155 "Invalid traceparent header format");
156 return common::Result<trace_context>::err(err.to_common_error());
157 }
158
159 ctx.trace_id = header.substr(dash1 + 1, dash2 - dash1 - 1);
160 ctx.span_id = header.substr(dash2 + 1, dash3 - dash2 - 1);
161 ctx.trace_flags = header.substr(dash3 + 1);
162
163 return common::ok(ctx);
164 }
165};

284 template<typename Carrier>
285 void inject_context(const trace_context& context, Carrier& carrier) {
286 carrier["traceparent"] = context.to_w3c_traceparent();
287 if (!context.trace_state.empty()) {
288 carrier["tracestate"] = context.trace_state;
289 }
290 // Inject baggage
291 for (const auto& [key, value] : context.baggage) {
292 carrier["baggage-" + key] = value;
293 }
294 }
295
299 template<typename Carrier>
300 common::Result<trace_context> extract_context_from_carrier(const Carrier& carrier) {
301 auto traceparent_it = carrier.find("traceparent");
302 if (traceparent_it == carrier.end()) {
303 return common::Result<trace_context>::err(error_info(kcenon::monitoring::monitoring_error_code::not_found,
304 "Traceparent not found in carrier").to_common_error());
305 }
306
307 auto ctx_result = trace_context::from_w3c_traceparent(traceparent_it->second);
308 if (ctx_result.is_err()) {
309 return common::Result<trace_context>::err(ctx_result.error());
310 }
311
312 auto ctx = ctx_result.value();
313
314 // Extract tracestate if present
315 auto tracestate_it = carrier.find("tracestate");
316 if (tracestate_it != carrier.end()) {
317 ctx.trace_state = tracestate_it->second;
318 }
319
320 // Extract baggage
321 for (const auto& [key, value] : carrier) {
322 if (key.find("baggage-") == 0) { // C++17 compatible
323 ctx.baggage[key.substr(8)] = value;
324 }
325 }
326
327 return ctx;
328 }

380class scoped_span {
381private:
382 std::shared_ptr<trace_span> span_;
383 distributed_tracer* tracer_;
384
385public:
386 scoped_span(std::shared_ptr<trace_span> span, distributed_tracer* tracer)
387 : span_(span), tracer_(tracer) {
388 if (tracer_) {
389 tracer_->set_current_span(span_);
390 }
391 }
392
393 ~scoped_span() {
394 if (span_ && tracer_) {
395 tracer_->finish_span(span_);
396 }
397 }
398
399 // Disable copy
400 scoped_span(const scoped_span&) = delete;
401 scoped_span& operator=(const scoped_span&) = delete;
402
403 // Enable move
404 scoped_span(scoped_span&& other) noexcept
405 : span_(std::move(other.span_)), tracer_(other.tracer_) {
406 other.tracer_ = nullptr;
407 }
408
409 scoped_span& operator=(scoped_span&& other) noexcept {
410 if (this != &other) {
411 if (span_ && tracer_) {
412 tracer_->finish_span(span_);
413 }
414 span_ = std::move(other.span_);
415 tracer_ = other.tracer_;
416 other.tracer_ = nullptr;
417 }
418 return *this;
419 }
420
421 trace_span* operator->() { return span_.get(); }
422 const trace_span* operator->() const { return span_.get(); }
423 trace_span& operator*() { return *span_; }
424 const trace_span& operator*() const { return *span_; }
425};
