114 common::Result<common::interfaces::metrics_snapshot> get_metrics() override {
115 try {
116 // Get all recorded metrics from the performance_monitor
117 common::interfaces::metrics_snapshot snapshot;
118 snapshot.source_id = monitor_->get_name();
119 snapshot.capture_time = std::chrono::system_clock::now();
120
121 // Get performance metrics from the monitor's profiler
122 const auto& perf_metrics = monitor_->get_profiler().get_all_metrics();
123
124 // Convert each performance_metrics (operation timing) into multiple metric_value entries
125 for (const auto& perf : perf_metrics) {
126 // Add various duration metrics as gauges
127 snapshot.add_metric(perf.operation_name + "_min_ns",
128 static_cast<double>(perf.min_duration.count()));
129 snapshot.add_metric(perf.operation_name + "_max_ns",
130 static_cast<double>(perf.max_duration.count()));
131 snapshot.add_metric(perf.operation_name + "_mean_ns",
132 static_cast<double>(perf.mean_duration.count()));
133 snapshot.add_metric(perf.operation_name + "_median_ns",
134 static_cast<double>(perf.median_duration.count()));
135 snapshot.add_metric(perf.operation_name + "_p95_ns",
136 static_cast<double>(perf.p95_duration.count()));
137 snapshot.add_metric(perf.operation_name + "_p99_ns",
138 static_cast<double>(perf.p99_duration.count()));
139
140 // Add count metrics as counters (conceptually)
141 snapshot.add_metric(perf.operation_name + "_call_count",
142 static_cast<double>(perf.call_count));
143 snapshot.add_metric(perf.operation_name + "_error_count",
144 static_cast<double>(perf.error_count));
145 }
146
147 // Add tagged metrics (counters, gauges, histograms with tags)
148 const auto& tagged_metrics = monitor_->get_all_tagged_metrics();
149 for (const auto& metric : tagged_metrics) {
150 // Create metric_value with tags directly
151 common::interfaces::metric_value mv(metric.name, metric.value);
152 mv.tags = std::unordered_map<std::string, std::string>(
153 metric.tags.begin(), metric.tags.end());
154 mv.timestamp = metric.timestamp;
155 snapshot.metrics.push_back(std::move(mv));
156 }
157
158 return common::ok(std::move(snapshot));
159 } catch (const std::exception& e) {
160 return common::Result<common::interfaces::metrics_snapshot>(
161 common::error_info{3, std::string("Exception in get_metrics: ") + e.what(), "performance_monitor_adapter"}
162 );
163 }
164 }

186 common::interfaces::health_check_result result;
187 result.status = common::interfaces::health_status::unhealthy;
188 result.message = std::string("Health check failed: ") + e.what();
189 result.timestamp = std::chrono::system_clock::now();
190 return common::ok(std::move(result));
