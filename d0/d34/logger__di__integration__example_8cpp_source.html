28class simple_console_logger : public ILogger {
29private:
30 log_level min_level_ = log_level::debug;
31 std::atomic<size_t> log_count_{0};
32
33public:
34 explicit simple_console_logger(log_level min = log_level::debug)
35 : min_level_(min) {}
36
37 common::VoidResult log(log_level level, const std::string& message) override {
38 if (!is_enabled(level)) {
39 return common::VoidResult::success();
40 }
41
42 auto now = std::chrono::system_clock::now();
43 auto time = std::chrono::system_clock::to_time_t(now);
44
45 std::cout << "[" << std::put_time(std::localtime(&time), "%H:%M:%S")
46 << "] [" << to_string(level) << "] "
47 << message << std::endl;
48
49 log_count_++;
50 return common::VoidResult::success();
51 }
52
53 bool is_enabled(log_level level) const override {
54 return static_cast<int>(level) >= static_cast<int>(min_level_);
55 }
56
57 size_t get_log_count() const { return log_count_.load(); }
58};
