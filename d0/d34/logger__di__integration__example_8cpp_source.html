25class simple_console_logger : public ILogger {
26private:
27 log_level min_level_ = log_level::debug;

30public:

34 common::VoidResult log(log_level level, const std::string& message) override {
35 if (!is_enabled(level)) {
36 return common::ok();
37 }
38
39 auto now = std::chrono::system_clock::now();
40 auto time = std::chrono::system_clock::to_time_t(now);
41
42 // Thread-safe time conversion
43 std::tm tm_buf;
44#ifdef _WIN32
45 localtime_s(&tm_buf, &time);
46#else
47 localtime_r(&time, &tm_buf);
48#endif
49
50 std::cout << "[" << std::put_time(&tm_buf, "%H:%M:%S")
51 << "] [" << to_string(level) << "] "
52 << message << std::endl;
53
54 log_count_++;
55 return common::ok();
56 }
57
58 common::VoidResult log(log_level level, const std::string& message,
59 const std::string& file, int line, const std::string& function) override {
60 return log(level, message + " [" + file + ":" + std::to_string(line) + " " + function + "]");
61 }

64 return log(entry.level, entry.message, entry.file, entry.line, entry.function);
