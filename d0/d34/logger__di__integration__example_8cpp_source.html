54class simple_console_logger : public ILogger {
55private:
56 log_level min_level_ = log_level::debug;
57 std::atomic<size_t> log_count_{0};
58
59public:
60 explicit simple_console_logger(log_level min = log_level::debug)
61 : min_level_(min) {}
62
63 kcenon::common::VoidResult log(log_level level, const std::string& message) override {
64 if (!is_enabled(level)) {
65 return kcenon::common::ok();
66 }
67
68 auto now = std::chrono::system_clock::now();
69 auto time = std::chrono::system_clock::to_time_t(now);
70
71 // Thread-safe time conversion
72 std::tm tm_buf;
73#ifdef _WIN32
74 localtime_s(&tm_buf, &time);
75#else
76 localtime_r(&time, &tm_buf);
77#endif
78
79 std::cout << "[" << std::put_time(&tm_buf, "%H:%M:%S")
80 << "] [" << to_string(level) << "] "
81 << message << std::endl;
82
83 log_count_++;
84 return kcenon::common::ok();
85 }
86
87 kcenon::common::VoidResult log(log_level level, const std::string& message,
88 const std::string& file, int line, const std::string& function) override {
89 return log(level, message + " [" + file + ":" + std::to_string(line) + " " + function + "]");
90 }
91
92 kcenon::common::VoidResult log(const log_entry& entry) override {
93 return log(entry.level, entry.message, entry.file, entry.line, entry.function);
94 }
95
96 bool is_enabled(log_level level) const override {
97 return static_cast<int>(level) >= static_cast<int>(min_level_);
98 }
99
100 kcenon::common::VoidResult set_level(log_level level) override {
101 min_level_ = level;
102 return kcenon::common::ok();
103 }
104
105 log_level get_level() const override {
106 return min_level_;
107 }
108
109 kcenon::common::VoidResult flush() override {
110 std::cout << std::flush;
111 return kcenon::common::ok();
112 }
113
114 size_t get_log_count() const { return log_count_.load(); }
115};

184 std::cout << " Duration: " << health.check_duration.count() << "ms" << std::endl;
185
186 if (!health.metadata.empty()) {
187 std::cout << " Metadata:" << std::endl;
188 for (const auto& [key, value] : health.metadata) {
189 std::cout << " " << key << ": " << value << std::endl;

247void example_6_monitoring_workflow() {
248 std::cout << "\n=== Example 6: Monitoring Workflow ===" << std::endl;
249
250 auto monitor = std::make_shared<performance_monitor>();
251 auto logger = std::make_shared<simple_console_logger>();
252
253 std::cout << "\nSimulating application workload..." << std::endl;
254
255 // Simulate application metrics
256 for (int i = 0; i < 5; ++i) {
257 auto result = monitor->record_metric("requests", static_cast<double>(i * 10));
258 if (kcenon::common::is_ok(result)) {
259 logger->log(log_level::info, "Recorded metric: requests = " + std::to_string(i * 10));
260 }
261 std::this_thread::sleep_for(std::chrono::milliseconds(100));
262 }
263
264 // Check health and log results
265 auto health = monitor->check_health();
266 if (kcenon::common::is_ok(health)) {
267 auto health_status = kcenon::common::get_value(health);
268 logger->log(log_level::info, "Monitor health: " + to_string(health_status.status));
269 }
270
271 // Get metrics and log summary
272 auto metrics = monitor->get_metrics();
273 if (kcenon::common::is_ok(metrics)) {
274 auto snapshot = kcenon::common::get_value(metrics);
275 logger->log(log_level::info, "Collected " + std::to_string(snapshot.metrics.size()) + " metrics");
276 }
277
278 std::cout << "\n‚úì Workflow completed successfully" << std::endl;
279 std::cout << " Logger events: " << logger->get_log_count() << std::endl;
280}
