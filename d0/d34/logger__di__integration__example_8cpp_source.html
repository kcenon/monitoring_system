54class simple_console_logger : public ILogger {
55private:
56 log_level min_level_ = log_level::debug;
57 std::atomic<size_t> log_count_{0};
58
59public:
60 explicit simple_console_logger(log_level min = log_level::debug)
61 : min_level_(min) {}
62
63 kcenon::common::VoidResult log(log_level level, const std::string& message) override {
64 if (!is_enabled(level)) {
65 return kcenon::common::ok();
66 }
67
68 auto now = std::chrono::system_clock::now();
69 auto time = std::chrono::system_clock::to_time_t(now);
70
71 // Thread-safe time conversion
72 std::tm tm_buf;
73#ifdef _WIN32
74 localtime_s(&tm_buf, &time);
75#else
76 localtime_r(&time, &tm_buf);
77#endif
78
79 std::cout << "[" << std::put_time(&tm_buf, "%H:%M:%S")
80 << "] [" << to_string(level) << "] "
81 << message << std::endl;
82
83 log_count_++;
84 return kcenon::common::ok();
85 }
86
87 kcenon::common::VoidResult log(const log_entry& entry) override {
88 std::string msg = entry.message;
89 if (!entry.file.empty()) {
90 msg += " [" + entry.file + ":" + std::to_string(entry.line) + " " + entry.function + "]";
91 }
92 return log(entry.level, msg);
93 }
94
95 bool is_enabled(log_level level) const override {
96 return static_cast<int>(level) >= static_cast<int>(min_level_);
97 }
98
99 kcenon::common::VoidResult set_level(log_level level) override {
100 min_level_ = level;
101 return kcenon::common::ok();
102 }
103
104 log_level get_level() const override {
105 return min_level_;
106 }
107
108 kcenon::common::VoidResult flush() override {
109 std::cout << std::flush;
110 return kcenon::common::ok();
111 }
112
113 size_t get_log_count() const { return log_count_.load(); }
114};

169
170 auto monitor = std::make_shared<performance_monitor>();
171
172 std::cout << "\nConfiguring thresholds..." << std::endl;
173
174 // Set thresholds
175 monitor->set_cpu_threshold(80.0);
176 monitor->set_memory_threshold(90.0);
177 monitor->set_latency_threshold(std::chrono::milliseconds(1000));
178
179 // Check thresholds
180 auto threshold_result = monitor->check_thresholds();
181 if (threshold_result.is_ok()) {
182 bool exceeded = threshold_result.value();
183 std::cout << " Thresholds exceeded: " << (exceeded ? "Yes" : "No") << std::endl;
184
185 auto thresholds = monitor->get_thresholds();
186 std::cout << " CPU threshold: " << thresholds.cpu_threshold << "%" << std::endl;
187 std::cout << " Memory threshold: " << thresholds.memory_threshold << "%" << std::endl;
188 std::cout << " Latency threshold: " << thresholds.latency_threshold.count() << "ms" << std::endl;
189 }

248void example_6_monitoring_workflow() {
249 std::cout << "\n=== Example 6: Monitoring Workflow ===" << std::endl;
250
251 auto monitor = std::make_shared<performance_monitor>();
252 auto logger = std::make_shared<simple_console_logger>();
253
254 std::cout << "\nSimulating application workload..." << std::endl;
255
256 // Simulate application metrics (updated API: record_gauge)
257 for (int i = 0; i < 5; ++i) {
258 auto result = monitor->record_gauge("requests", static_cast<double>(i * 10));
259 if (result.is_ok()) {
260 logger->log(log_level::info, "Recorded metric: requests = " + std::to_string(i * 10));
261 }
262 std::this_thread::sleep_for(std::chrono::milliseconds(100));
263 }
264
265 // Check thresholds and log results
266 auto threshold_result = monitor->check_thresholds();
267 if (threshold_result.is_ok()) {
268 bool exceeded = threshold_result.value();
269 logger->log(log_level::info, "Thresholds exceeded: " + std::string(exceeded ? "Yes" : "No"));
270 }
271
272 // Get metrics and log summary
273 auto metrics = monitor->collect();
274 if (metrics.is_ok()) {
275 const auto& snapshot = metrics.value();
276 logger->log(log_level::info, "Collected " + std::to_string(snapshot.metrics.size()) + " metrics");
277 }
278
279 std::cout << "\n Workflow completed successfully" << std::endl;
280 std::cout << " Logger events: " << logger->get_log_count() << std::endl;
281}
