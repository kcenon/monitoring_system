80inline std::string power_source_type_to_string(power_source_type type) {
81 switch (type) {
82 case power_source_type::battery:
83 return "battery";
84 case power_source_type::ac:
85 return "ac";
86 case power_source_type::usb:
87 return "usb";
88 case power_source_type::wireless:
89 return "wireless";
90 case power_source_type::cpu:
91 return "cpu";
92 case power_source_type::gpu:
93 return "gpu";
94 case power_source_type::memory:
95 return "memory";
96 case power_source_type::package:
97 return "package";
98 case power_source_type::platform:
99 return "platform";
100 case power_source_type::other:
101 return "other";
102 default:
103 return "unknown";
104 }
