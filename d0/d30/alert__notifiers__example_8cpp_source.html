63// Helper function to create sample alerts
64alert create_sample_alert(const std::string& name,
65 alert_severity severity,
66 alert_state state,
67 double value,
68 const std::string& team = "ops") {
69 alert a;
70 a.name = name;
71 a.severity = severity;
72 a.state = state;
73 a.value = value;
74 a.labels.set("team", team);
75 a.labels.set("environment", "production");

82// Custom notifier implementation: Console notifier with color
83class console_color_notifier : public alert_notifier {
84public:
85 explicit console_color_notifier(std::string notifier_name)
86 : name_(std::move(notifier_name)) {}
87
88 std::string name() const override { return name_; }
89
90 common::VoidResult notify(const alert& a) override {
91 std::string color = get_severity_color(a.severity);
92 std::string reset = "\033[0m";
93
94 std::cout << color << "[" << name_ << "] "
95 << alert_severity_to_string(a.severity) << ": "
96 << a.name << " (" << alert_state_to_string(a.state) << ")"
97 << reset << std::endl;
98 std::cout << " Summary: " << a.annotations.summary << std::endl;
99 std::cout << " Value: " << a.value << std::endl;
100
101 return common::ok();
102 }
103
104 common::VoidResult notify_group(const alert_group& group) override {
105 std::cout << "[" << name_ << "] Alert Group: " << group.group_key
106 << " (" << group.size() << " alerts)" << std::endl;
107
108 for (const auto& alert_item : group.alerts) {
109 auto result = notify(alert_item);
110 if (!result.is_ok()) {
111 return result;
112 }
113 }
114 return common::ok();
115 }
116
117 bool is_ready() const override { return true; }
118
119private:
120 static std::string get_severity_color(alert_severity sev) {
121 switch (sev) {
122 case alert_severity::emergency: return "\033[41m\033[37m"; // White on red
123 case alert_severity::critical: return "\033[31m"; // Red
124 case alert_severity::warning: return "\033[33m"; // Yellow
125 case alert_severity::info: return "\033[32m"; // Green
126 default: return "\033[0m"; // Reset
127 }
128 }
129
130 std::string name_;
131};

133// Custom notifier: Statistics collector
134class statistics_notifier : public alert_notifier {
135public:
136 explicit statistics_notifier(std::string notifier_name)
137 : name_(std::move(notifier_name)) {}
138
139 std::string name() const override { return name_; }
140
141 common::VoidResult notify(const alert& a) override {
142 std::lock_guard<std::mutex> lock(mutex_);
143 total_alerts_++;
144 severity_counts_[a.severity]++;
145 state_counts_[a.state]++;
146 return common::ok();
147 }
148
149 common::VoidResult notify_group(const alert_group& group) override {
150 for (const auto& alert_item : group.alerts) {
151 auto result = notify(alert_item);
152 if (!result.is_ok()) {
153 return result;
154 }
155 }
156 return common::ok();
157 }
158
159 bool is_ready() const override { return true; }
160
161 void print_statistics() const {
162 std::lock_guard<std::mutex> lock(mutex_);
163
164 std::cout << "Statistics from " << name_ << ":" << std::endl;
165 std::cout << " Total alerts: " << total_alerts_ << std::endl;
166 std::cout << " By severity:" << std::endl;
167 for (const auto& [sev, count] : severity_counts_) {
168 std::cout << " " << alert_severity_to_string(sev) << ": " << count << std::endl;
169 }
170 std::cout << " By state:" << std::endl;
171 for (const auto& [state, count] : state_counts_) {
172 std::cout << " " << alert_state_to_string(state) << ": " << count << std::endl;
173 }
174 }
175
176 void reset() {
177 std::lock_guard<std::mutex> lock(mutex_);
178 total_alerts_ = 0;
179 severity_counts_.clear();
180 state_counts_.clear();
181 }
182
183private:
184 std::string name_;
185 mutable std::mutex mutex_;
186 size_t total_alerts_ = 0;
187 std::map<alert_severity, size_t> severity_counts_;
188 std::map<alert_state, size_t> state_counts_;
189};
