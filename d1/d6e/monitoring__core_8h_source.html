181struct monitoring_config {
182 std::size_t history_size = 1000;
183 std::chrono::milliseconds collection_interval{1000};
184 bool enable_compression = false;
185 bool enable_persistence = false;
186 std::size_t max_collectors = 100;
187 std::size_t buffer_size = 10000;
188
193 result_void validate() const {
194 if (history_size == 0) {
195 return make_void_error(monitoring_error_code::invalid_capacity,
196 "History size must be greater than 0");
197 }
198 if (collection_interval.count() < 10) {
199 return make_void_error(monitoring_error_code::invalid_interval,
200 "Collection interval must be at least 10ms");
201 }
202 if (buffer_size < history_size) {
203 return make_void_error(monitoring_error_code::invalid_capacity,
204 "Buffer size must be at least as large as history size");
205 }
206 return make_void_success();
207 }
208};
