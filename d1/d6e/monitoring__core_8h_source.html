181struct monitoring_config {
182 std::size_t history_size = 1000;
183 std::chrono::milliseconds collection_interval{1000};
184 bool enable_compression = false;
185 bool enable_persistence = false;
186 std::size_t max_collectors = 100;
187 std::size_t buffer_size = 10000;
188
193 common::VoidResult validate() const {
194 if (history_size == 0) {
195 error_info err(monitoring_error_code::invalid_capacity,
196 "History size must be greater than 0");
197 return common::VoidResult::err(err.to_common_error());
198 }
199 if (collection_interval.count() < 10) {
200 error_info err(monitoring_error_code::invalid_interval,
201 "Collection interval must be at least 10ms");
202 return common::VoidResult::err(err.to_common_error());
203 }
204 if (buffer_size < history_size) {
205 error_info err(monitoring_error_code::invalid_capacity,
206 "Buffer size must be at least as large as history size");
207 return common::VoidResult::err(err.to_common_error());
208 }
209 return common::ok();
210 }
211};
