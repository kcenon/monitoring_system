294 // 1. Setup resilience components with fault_tolerance_manager
295 fault_tolerance_config ft_config;
296 ft_config.enable_circuit_breaker = true;
297 ft_config.enable_retry = true;
298 ft_config.circuit_config.failure_threshold = 3;
299 ft_config.circuit_config.reset_timeout = std::chrono::milliseconds(100);
300 ft_config.retry_cfg.max_attempts = 5;
301 ft_config.retry_cfg.initial_delay = std::chrono::milliseconds(10);
302
303 fault_tolerance_manager<bool> ft_manager("test_manager", ft_config);
304
305 // 2. Simulate component with intermittent failures
306 std::atomic<int> call_count{0};
307 std::atomic<bool> should_fail{true};
308
309 auto unreliable_operation = [&call_count, &should_fail]() -> kcenon::common::Result<bool> {
310 call_count++;
311
312 // Fail first 3 calls, then succeed
313 if (call_count <= 3 && should_fail) {
314 return kcenon::common::make_error<bool>(static_cast<int>(monitoring_error_code::operation_failed),
315 "Simulated failure");
316 }
317
318 return kcenon::common::ok(true);
319 };
320
321 // 3. Test fault tolerance execution
322 auto ft_result = ft_manager.execute(unreliable_operation);
323 EXPECT_TRUE(ft_result.is_ok());
324
325 // 4. Reset and test standalone circuit breaker
326 call_count = 0;
327 should_fail = true;
328
329 circuit_breaker_config cb_config;
330 cb_config.failure_threshold = 3;
331 cb_config.reset_timeout = std::chrono::milliseconds(100);
332 circuit_breaker<bool> breaker("test_breaker", cb_config);
333
334 // Trigger circuit breaker with failures
335 for (int i = 0; i < 3; ++i) {
336 auto cb_result = breaker.execute(unreliable_operation);
337 EXPECT_FALSE(cb_result.is_ok());
338 }
339
340 // Circuit should be open
341 EXPECT_EQ(breaker.get_state(), circuit_state::open);
342
343 // Further calls should fail fast (circuit open)
344 auto open_result = breaker.execute(unreliable_operation);
345 EXPECT_FALSE(open_result.is_ok());
346
347 // 5. Wait for circuit recovery
348 std::this_thread::sleep_for(std::chrono::milliseconds(150));
349
350 // Allow success for recovery
351 should_fail = false;
352 call_count = 0;
353
354 // Circuit should transition to half-open and then closed
355 auto recovery_result = breaker.execute(unreliable_operation);
356 EXPECT_TRUE(recovery_result.is_ok());
357
358 // After several successes, circuit should be closed
359 for (int i = 0; i < 5; ++i) {
360 auto stable_result = breaker.execute(unreliable_operation);
361 EXPECT_TRUE(stable_result.is_ok());
362 }
363
364 EXPECT_EQ(breaker.get_state(), circuit_state::closed);
365}

426TEST_F(IntegrationE2ETest, FullSystemLoadTest) {
427 // 1. Setup components
428 distributed_tracer tracer;
429 auto perf_monitor = std::make_shared<performance_monitor>("load_perf");
430 auto& health_mon = global_health_monitor();
431
432 // 2. Configure for high load
433 const int num_operations = 1000;
434 const int num_threads = 10;
435
436 // 3. Generate load
437 auto start_time = std::chrono::steady_clock::now();
438 std::vector<std::thread> load_generators;
439 std::atomic<int> total_operations{0};
440
441 for (int t = 0; t < num_threads; ++t) {
442 load_generators.emplace_back([&tracer, &total_operations, t]() {
443 std::random_device rd;
444 std::mt19937 gen(rd());
445 std::uniform_real_distribution<> dis(0.0, 100.0);
446
447 for (int i = 0; i < 100; ++i) {
448 // Create span
449 auto span_result = tracer.start_span("load_test_" + std::to_string(t), "load_service");
450 if (span_result.is_ok()) {
451 auto span = span_result.value();
452 span->tags["thread"] = std::to_string(t);
453 span->tags["value"] = std::to_string(dis(gen));
454 tracer.finish_span(span);
455 total_operations++;
456 }
457
458 // Small delay to prevent overwhelming
459 if (i % 10 == 0) {
460 std::this_thread::sleep_for(std::chrono::microseconds(100));
461 }
462 }
463 });
464 }
465
466 // 4. Monitor while load is running
467 std::thread monitor_thread([&health_mon, &start_time]() {
468 while (std::chrono::steady_clock::now() - start_time < std::chrono::seconds(5)) {
469 auto health = health_mon.check_health();
470 // System should remain operational under load
471 EXPECT_TRUE(health.is_operational());
472 std::this_thread::sleep_for(std::chrono::milliseconds(100));
473 }
474 });
475
476 // 5. Wait for completion
477 for (auto& t : load_generators) {
478 t.join();
479 }
480 monitor_thread.join();
481
482 auto end_time = std::chrono::steady_clock::now();
483 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end_time - start_time);
484
485 // 6. Verify performance
486 EXPECT_EQ(total_operations, num_operations);
487
488 // Calculate throughput
489 double throughput = (total_operations * 1000.0) / duration.count();
490 std::cout << "Load test throughput: " << throughput << " ops/sec" << std::endl;
491
492 // Should achieve reasonable throughput
493 EXPECT_GT(throughput, 100.0); // At least 100 ops/sec
494}
