299 static bool parse_bool(const std::string& str) {
300 if (str.empty()) {
301 return false;
302 }
303 // Case-insensitive comparison
304 std::string lower = str;
305 for (auto& c : lower) {
306 c = static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
307 }
308 return lower == "true" || lower == "1" || lower == "yes" || lower == "on";
309 }

314 template <typename T>
315 static T parse_integral(const std::string& str) {
316 static_assert(std::is_integral_v<T>, "parse_integral requires integral type");
317 if constexpr (std::is_signed_v<T>) {
318 long long value = std::stoll(str);
319 return static_cast<T>(value);
320 } else {
321 unsigned long long value = std::stoull(str);
322 return static_cast<T>(value);
323 }
324 }

329 template <typename T>
330 static T parse_floating(const std::string& str) {
331 static_assert(std::is_floating_point_v<T>, "parse_floating requires floating point type");
332 if constexpr (std::is_same_v<T, float>) {
333 return std::stof(str);
334 } else if constexpr (std::is_same_v<T, double>) {
335 return std::stod(str);
336 } else {
337 return static_cast<T>(std::stold(str));

345 template <typename Duration>
346 static Duration parse_duration(const std::string& str, const Duration& default_value) {
347 try {
348 if (str.empty()) {
349 return default_value;
350 }
351
352 // Find where the number ends and suffix begins
353 size_t suffix_start = str.find_first_not_of("0123456789.-");
354
355 if (suffix_start == std::string::npos) {
356 // Plain number, interpret as Duration's unit
357 long long value = std::stoll(str);
358 return Duration(value);
359 }
360
361 // Parse the numeric part
362 long long value = std::stoll(str.substr(0, suffix_start));
363 std::string suffix = str.substr(suffix_start);
364
365 // Trim whitespace from suffix
366 while (!suffix.empty() && std::isspace(static_cast<unsigned char>(suffix.front()))) {
367 suffix.erase(0, 1);
368 }
369
370 // Convert to lowercase
371 for (auto& c : suffix) {
372 c = static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
373 }
374
375 // Parse suffix and convert to Duration
376 if (suffix == "ms" || suffix == "millisecond" || suffix == "milliseconds") {
377 return std::chrono::duration_cast<Duration>(std::chrono::milliseconds(value));
378 } else if (suffix == "s" || suffix == "sec" || suffix == "second" || suffix == "seconds") {
379 return std::chrono::duration_cast<Duration>(std::chrono::seconds(value));
380 } else if (suffix == "m" || suffix == "min" || suffix == "minute" || suffix == "minutes") {
381 return std::chrono::duration_cast<Duration>(std::chrono::minutes(value));
382 } else if (suffix == "h" || suffix == "hr" || suffix == "hour" || suffix == "hours") {
383 return std::chrono::duration_cast<Duration>(std::chrono::hours(value));
384 } else {
385 // Unknown suffix, treat as plain number
386 return Duration(value);
387 }
388 } catch (...) {
389 return default_value;
390 }
391 }

396 template <typename T>
397 static std::vector<T> parse_list(const std::string& str, const std::vector<T>& default_values) {
398 try {
399 if (str.empty()) {
400 return default_values;
401 }
402
403 std::vector<T> result;
404 std::string current;
405
406 for (char c : str) {
407 if (c == ',') {
408 // Trim whitespace
409 size_t start = current.find_first_not_of(" \t");
410 size_t end = current.find_last_not_of(" \t");
411 if (start != std::string::npos) {
412 std::string trimmed = current.substr(start, end - start + 1);
413 auto parsed = parse_value_optional<T>(trimmed);
414 if (parsed) {
415 result.push_back(*parsed);
416 }
417 }
418 current.clear();
419 } else {
420 current += c;
421 }
422 }
423
424 // Handle last element
425 if (!current.empty()) {
426 size_t start = current.find_first_not_of(" \t");
427 size_t end = current.find_last_not_of(" \t");
428 if (start != std::string::npos) {
429 std::string trimmed = current.substr(start, end - start + 1);
430 auto parsed = parse_value_optional<T>(trimmed);
431 if (parsed) {
432 result.push_back(*parsed);
433 }
434 }
435 }
436
437 return result.empty() ? default_values : result;
438 } catch (...) {
439 return default_values;
440 }
441 }
