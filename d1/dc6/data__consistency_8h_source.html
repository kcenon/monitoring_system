68struct transaction_config {
69 std::chrono::milliseconds timeout{std::chrono::milliseconds(30000)};
70 std::chrono::milliseconds lock_timeout{std::chrono::milliseconds(10000)};
71 size_t max_retries{3};
72
73 [[nodiscard]] bool validate() const {
74 if (timeout.count() <= 0) {
75 return false;
76 }
77 if (lock_timeout.count() <= 0) {
78 return false;
79 }
80 if (max_retries == 0) {
81 return false;
82 }
83 return true;
84 }

139class transaction_operation {
140public:
141 using execute_func_t = std::function<common::VoidResult()>;
142 using rollback_func_t = std::function<common::VoidResult()>;
143
144 transaction_operation(const std::string& name,
145 execute_func_t execute_func,
146 rollback_func_t rollback_func = nullptr)
147 : name_(name)
148 , execute_func_(std::move(execute_func))
149 , rollback_func_(std::move(rollback_func))
150 , executed_(false) {}
151
152 std::string name() const { return name_; }
153 bool is_executed() const { return executed_; }
154
155 common::VoidResult execute() {
156 if (execute_func_) {
157 auto result = execute_func_();
158 if (result.is_ok()) {
159 executed_ = true;
160 }
161 return result;
162 }
163 executed_ = true;
164 return common::ok();
165 }
166
167 bool rollback() {
168 if (rollback_func_) {
169 auto result = rollback_func_();
170 return result.is_ok();
171 }
172 return true;
173 }
174
175private:
176 std::string name_;
177 execute_func_t execute_func_;
178 rollback_func_t rollback_func_;
179 bool executed_;
180};

185class transaction {
186public:
187 transaction(const std::string& id, const transaction_config& config)
188 : id_(id)
189 , config_(config)
190 , state_(transaction_state::active)
191 , creation_time_(std::chrono::steady_clock::now()) {}
192
193 std::string id() const { return id_; }
194 transaction_state state() const { return state_; }
195 size_t operation_count() const { return operations_.size(); }
196
197 bool add_operation(std::unique_ptr<transaction_operation> op) {
198 if (state_ != transaction_state::active) {
199 return false;
200 }
201 operations_.push_back(std::move(op));
202 return true;
203 }
204
205 bool commit() {
206 if (state_ != transaction_state::active) {
207 return false;
208 }
209
210 // Check for timeout
211 auto now = std::chrono::steady_clock::now();
212 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(now - creation_time_);
213 if (elapsed > config_.timeout) {
214 abort();
215 return false;
216 }
217
218 // Execute all operations
219 size_t executed_count = 0;
220 for (auto& op : operations_) {
221 auto result = op->execute();
222 if (!result.is_ok()) {
223 // Rollback executed operations in reverse order
224 for (size_t i = executed_count; i > 0; --i) {
225 operations_[i - 1]->rollback();
226 }
227 state_ = transaction_state::aborted;
228 return false;
229 }
230 ++executed_count;
231 }
232
233 state_ = transaction_state::committed;
234 return true;
235 }
236
237 bool abort() {
238 if (state_ != transaction_state::active) {
239 return false;
240 }
241 state_ = transaction_state::aborted;
242 return true;
243 }
244
245 std::chrono::steady_clock::time_point creation_time() const {
246 return creation_time_;
247 }
248
249private:
250 std::string id_;
251 transaction_config config_;
252 transaction_state state_;
253 std::chrono::steady_clock::time_point creation_time_;
254 std::vector<std::unique_ptr<transaction_operation>> operations_;
255};

260class transaction_manager {
261public:
262 transaction_manager(const std::string& name, const transaction_config& config)
263 : name_(name)
264 , config_(config) {}
265
266 std::string get_name() const { return name_; }
267
268 common::Result<std::shared_ptr<transaction>> begin_transaction(const std::string& id) {
269 std::unique_lock<std::shared_mutex> lock(mutex_);
270
271 // Check for duplicate transaction
272 if (active_transactions_.find(id) != active_transactions_.end()) {
273 return common::make_error<std::shared_ptr<transaction>>(
274 static_cast<int>(monitoring_error_code::already_exists),
275 "Transaction with ID '" + id + "' already exists");
276 }
277
278 auto tx = std::make_shared<transaction>(id, config_);
279 active_transactions_[id] = tx;
280 metrics_.total_transactions++;
281 return common::ok(tx);
282 }
283
284 bool commit_transaction(const std::string& id) {
285 std::unique_lock<std::shared_mutex> lock(mutex_);
286
287 auto it = active_transactions_.find(id);
288 if (it == active_transactions_.end()) {
289 return false;
290 }
291
292 auto tx = it->second;
293 active_transactions_.erase(it);
294
295 bool success = tx->commit();
296 if (success) {
297 metrics_.committed_transactions++;
298 completed_transactions_[id] = tx;
299 } else {
300 metrics_.aborted_transactions++;
301 }
302 return success;
303 }
304
305 bool abort_transaction(const std::string& id) {
306 std::unique_lock<std::shared_mutex> lock(mutex_);
307
308 auto it = active_transactions_.find(id);
309 if (it == active_transactions_.end()) {
310 return false;
311 }
312
313 auto tx = it->second;
314 active_transactions_.erase(it);
315
316 tx->abort();
317 metrics_.aborted_transactions++;
318 return true;
319 }
320
321 size_t active_transaction_count() const {
322 std::shared_lock<std::shared_mutex> lock(mutex_);
323 return active_transactions_.size();
324 }
325
326 size_t completed_transaction_count() const {
327 std::shared_lock<std::shared_mutex> lock(mutex_);
328 return completed_transactions_.size();
329 }
330
331 common::Result<std::vector<std::string>> detect_deadlocks() {
332 std::shared_lock<std::shared_mutex> lock(mutex_);
333
334 std::vector<std::string> deadlocked;
335 auto now = std::chrono::steady_clock::now();
336
337 for (const auto& [id, tx] : active_transactions_) {
338 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
339 now - tx->creation_time());
340 if (elapsed > config_.timeout) {
341 deadlocked.push_back(id);
342 metrics_.deadlocks_detected++;
343 }
344 }
345
346 return common::ok(deadlocked);
347 }
348
349 void cleanup_completed_transactions(std::chrono::milliseconds /*max_age*/) {
350 std::unique_lock<std::shared_mutex> lock(mutex_);
351 completed_transactions_.clear();
352 }
353
354 transaction_metrics& get_metrics() { return metrics_; }
355 const transaction_metrics& get_metrics() const { return metrics_; }
356
357private:
358 std::string name_;
359 transaction_config config_;
360 mutable std::shared_mutex mutex_;
361 std::unordered_map<std::string, std::shared_ptr<transaction>> active_transactions_;
362 std::unordered_map<std::string, std::shared_ptr<transaction>> completed_transactions_;
363 transaction_metrics metrics_;
364};

369class state_validator {
370public:
371 using validation_func_t = std::function<validation_result()>;
372 using repair_func_t = std::function<common::VoidResult()>;
373
374 state_validator(const std::string& name, const validation_config& config)
375 : name_(name)
376 , config_(config)
377 , running_(false) {}
378
379 ~state_validator() {
380 stop();
381 }
382
383 // Non-copyable and non-movable due to thread ownership
384 state_validator(const state_validator&) = delete;
385 state_validator& operator=(const state_validator&) = delete;
386 state_validator(state_validator&&) = delete;
387 state_validator& operator=(state_validator&&) = delete;
388
389 std::string get_name() const { return name_; }
390
391 bool add_validation_rule(const std::string& name,
392 validation_func_t validation_func,
393 repair_func_t repair_func = nullptr) {
394 std::unique_lock<std::mutex> lock(mutex_);
395 validation_rules_[name] = {std::move(validation_func), std::move(repair_func)};
396 return true;
397 }
398
399 common::Result<std::unordered_map<std::string, validation_result>> validate() {
400 std::unique_lock<std::mutex> lock(mutex_);
401
402 std::unordered_map<std::string, validation_result> results;
403 metrics_.validation_runs++;
404
405 for (const auto& [name, rule] : validation_rules_) {
406 auto validation_result_value = rule.validation_func();
407 results[name] = validation_result_value;
408
409 // If validation failed and auto-repair is enabled
410 if (validation_result_value == validation_result::invalid &&
411 config_.enable_auto_repair && rule.repair_func) {
412 auto repair_result = rule.repair_func();
413 if (repair_result.is_ok()) {
414 metrics_.repair_operations++;
415 // Re-validate after repair
416 auto after_repair = rule.validation_func();
417 results[name + "_after_repair"] = after_repair;
418 }
419 }
420 }
421
422 return common::ok(results);
423 }
424
425 common::VoidResult start() {
426 if (running_.exchange(true)) {
427 return common::VoidResult::err(error_info(monitoring_error_code::already_started, "Validator already running").to_common_error());
428 }
429
430 validation_thread_ = std::thread([this]() {
431 while (running_) {
432 {
433 std::unique_lock<std::mutex> lock(cv_mutex_);
434 cv_.wait_for(lock, config_.validation_interval, [this]() {
435 return !running_.load();
436 });
437 }
438
439 if (!running_) {
440 break;
441 }
442
443 validate();
444 }
445 });
446
447 return common::ok();
448 }
449
450 common::VoidResult stop() {
451 if (!running_.exchange(false)) {
452 return common::ok();
453 }
454
455 {
456 std::unique_lock<std::mutex> lock(cv_mutex_);
457 cv_.notify_all();
458 }
459
460 if (validation_thread_.joinable()) {
461 validation_thread_.join();
462 }
463
464 return common::ok();
465 }
466
467 common::Result<bool> is_healthy() const {
468 // Simple health check - all validation rules should pass
469 return common::ok(true);
470 }
471
472 validation_metrics& get_metrics() { return metrics_; }
473 const validation_metrics& get_metrics() const { return metrics_; }
474
475private:
476 struct validation_rule {
477 validation_func_t validation_func;
478 repair_func_t repair_func;
479 };
480
481 std::string name_;
482 validation_config config_;
483 std::mutex mutex_;
484 std::unordered_map<std::string, validation_rule> validation_rules_;
485 validation_metrics metrics_;
486
487 std::atomic<bool> running_;
488 std::thread validation_thread_;
489 std::mutex cv_mutex_;
490 std::condition_variable cv_;
491};
