163class event_handler {
164public:
165 using handler_func = std::function<void(const EventType&)>;
166
167 event_handler(handler_func handler, event_priority priority = event_priority::normal)
168 : handler_(handler), priority_(priority), id_(generate_id()) {}
169
170 void operator()(const EventType& event) const {
171 if (handler_) {
172 handler_(event);
173 }
174 }
175
176 event_priority get_priority() const { return priority_; }
177 uint64_t get_id() const { return id_; }
178
179private:
180 static uint64_t generate_id() {

182 return counter.fetch_add(1, std::memory_order_relaxed);
183 }
184
185 handler_func handler_;
186 event_priority priority_;
187 uint64_t id_;
188};
