224TEST_F(MonitorServiceRegistrationTest, ThreadSafeResolution) {
225 monitoring_di::monitor_registration_config config;
226 config.lifetime = common_di::service_lifetime::singleton;
227
228 auto reg_result = monitoring_di::register_monitor_services(*container_, config);
229 ASSERT_TRUE(reg_result.is_ok());
230
231 std::vector<std::thread> threads;
232 std::vector<std::shared_ptr<interfaces::IMonitor>> results(10);
233 std::atomic<int> success_count{0};
234
235 for (size_t i = 0; i < 10; ++i) {
236 threads.emplace_back([this, &results, &success_count, i]() {
237 auto result = container_->resolve<interfaces::IMonitor>();
238 if (result.is_ok()) {
239 results[i] = result.value();
240 ++success_count;
241 }
242 });
243 }
244
245 for (auto& t : threads) {
246 t.join();
247 }
248
249 EXPECT_EQ(success_count.load(), 10);
250
251 // All threads should get the same singleton instance
252 auto first = results[0];
253 for (const auto& result : results) {
254 EXPECT_EQ(result, first) << "All threads should get same singleton";
255 }
256}
