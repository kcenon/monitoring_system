229TEST_F(MonitorServiceRegistrationTest, ThreadSafeResolution) {
230 auto config = default_test_config();
231 config.lifetime = common_di::service_lifetime::singleton;
232
233 auto reg_result = monitoring_di::register_monitor_services(container_, config);
234 ASSERT_TRUE(reg_result.is_ok());
235
236 std::vector<std::thread> threads;
237 std::vector<std::shared_ptr<interfaces::IMonitor>> results(10);
238 std::atomic<int> success_count{0};
239
240 for (size_t i = 0; i < 10; ++i) {
241 threads.emplace_back([this, &results, &success_count, i]() {
242 auto result = container_.resolve<interfaces::IMonitor>();
243 if (result.is_ok()) {
244 results[i] = result.value();
245 ++success_count;
246 }
247 });
248 }
249
250 for (auto& t : threads) {
251 t.join();
252 }
253
254 EXPECT_EQ(success_count.load(), 10);
255
256 // All threads should get the same singleton instance
257 auto first = results[0];
258 for (const auto& result : results) {
259 EXPECT_EQ(result, first) << "All threads should get same singleton";
260 }
261}
