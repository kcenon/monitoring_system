230 create_result create(const std::string& name, const config_map& config = {}) {
231 create_result result;
232
233 // Find factory
234 collector_factory_fn factory;
235 {
236 std::lock_guard<std::mutex> lock(mutex_);
237 auto it = factories_.find(name);
238 if (it == factories_.end()) {
239 result.error_message = "Unknown collector: " + name;
240 return result;
241 }
242 factory = it->second;
243 }
244
245 // Create collector
246 try {
247 result.collector = factory();
248 if (!result.collector) {
249 result.error_message = "Factory returned null for: " + name;
250 return result;
251 }
252 } catch (const std::exception& e) {
253 result.error_message = "Failed to create collector '" + name + "': " + e.what();
254 return result;
255 } catch (...) {
256 result.error_message = "Unknown error creating collector: " + name;
257 return result;
258 }
259
260 // Initialize collector
261 try {
262 if (!result.collector->initialize(config)) {
263 result.error_message = "Initialization failed for: " + name;
264 result.collector.reset();
265 return result;
266 }
267 } catch (const std::exception& e) {
268 result.error_message =
269 "Exception during initialization of '" + name + "': " + e.what();
270 result.collector.reset();
271 return result;
272 } catch (...) {
273 result.error_message = "Unknown error initializing collector: " + name;
274 result.collector.reset();
275 return result;
276 }
277
278 result.success = true;
279 return result;
280 }

299 std::vector<std::unique_ptr<collector_interface>> create_multiple(
300 const std::unordered_map<std::string, config_map>& configs) {
301 std::vector<std::unique_ptr<collector_interface>> collectors;
302 collectors.reserve(configs.size());
303
304 for (const auto& [name, config] : configs) {
305 auto result = create(name, config);
