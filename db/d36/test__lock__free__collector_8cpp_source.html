52TEST_F(LockFreeCollectorTest, BasicSampleRecording) {
53 thread_local_buffer buffer(256, collector);
54
55 // Record some samples
56 metric_sample sample1{"operation1", std::chrono::nanoseconds(1000), true};
57 metric_sample sample2{"operation1", std::chrono::nanoseconds(2000), true};
58 metric_sample sample3{"operation1", std::chrono::nanoseconds(1500), false};
59
60 EXPECT_TRUE(buffer.record(sample1));
61 EXPECT_TRUE(buffer.record(sample2));
62 EXPECT_TRUE(buffer.record(sample3));
63

65

69 EXPECT_EQ(buffer.size(), 0);
70

75 EXPECT_EQ(stats.operation_count, 1);
76
77 // Get profile

79 ASSERT_TRUE(profile_result.is_ok());
80
81 auto profile = profile_result.value();
82 EXPECT_EQ(profile.total_calls, 3);
83 EXPECT_EQ(profile.error_count, 1);
84 EXPECT_EQ(profile.min_duration_ns, 1000);
85 EXPECT_EQ(profile.max_duration_ns, 2000);

88TEST_F(LockFreeCollectorTest, BufferAutoFlush) {
89 thread_local_buffer buffer(10, collector); // Small buffer
90
91 // Fill buffer
92 for (int i = 0; i < 10; ++i) {
93 metric_sample sample{"op", std::chrono::nanoseconds(100), true};
94 EXPECT_TRUE(buffer.record(sample));
95 }
96
97 EXPECT_TRUE(buffer.is_full());
98
99 // Next record should fail
100 metric_sample sample{"op", std::chrono::nanoseconds(100), true};
101 EXPECT_FALSE(buffer.record(sample));
102
103 // But auto-flush should work
104 EXPECT_TRUE(buffer.record_auto_flush(sample));
105 EXPECT_EQ(buffer.size(), 1); // Buffer now has 1 sample after flush+record
106
107 // Check collector received the batch
108 auto stats = collector->get_stats();
109 EXPECT_EQ(stats.total_samples, 10); // First 10 samples flushed
110}

112TEST_F(LockFreeCollectorTest, MultiThreadedCollection) {
113 constexpr int NUM_THREADS = 4;
114 constexpr int SAMPLES_PER_THREAD = 1000;
115
116 std::vector<std::thread> threads;
117
118 for (int t = 0; t < NUM_THREADS; ++t) {
119 threads.emplace_back([this, t]() {
120 thread_local_buffer buffer(256, collector);
121 std::string op_name = "thread_op_" + std::to_string(t);
122
123 for (int i = 0; i < SAMPLES_PER_THREAD; ++i) {
124 metric_sample sample{
125 op_name,
126 std::chrono::nanoseconds(100 + i),
127 (i % 10) != 0 // 10% error rate
128 };
129 buffer.record_auto_flush(sample);
130 }
131
132 // Final flush
133 buffer.flush();
134 });
135 }
136
137 // Wait for all threads
138 for (auto& t : threads) {
139 t.join();
140 }
141
142 // Verify all samples were collected
143 auto stats = collector->get_stats();
144 EXPECT_EQ(stats.total_samples, NUM_THREADS * SAMPLES_PER_THREAD);
145 EXPECT_EQ(stats.operation_count, NUM_THREADS);
146
147 // Verify each thread's profile
148 for (int t = 0; t < NUM_THREADS; ++t) {
149 std::string op_name = "thread_op_" + std::to_string(t);
150 auto profile_result = collector->get_profile(op_name);
151 ASSERT_TRUE(profile_result.is_ok());
152
153 auto profile = profile_result.value();
154 EXPECT_EQ(profile.total_calls, SAMPLES_PER_THREAD);
155 EXPECT_EQ(profile.error_count, SAMPLES_PER_THREAD / 10);
156 }
157}
