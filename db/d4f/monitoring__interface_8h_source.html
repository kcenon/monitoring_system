122struct monitoring_config {
123 std::size_t history_size = 1000;
124 std::chrono::milliseconds collection_interval{1000};
125 bool enable_compression = false;
126 bool enable_persistence = false;
127 std::size_t max_collectors = 100;
128 std::size_t buffer_size = 10000;
129
134 result_void validate() const {
135 if (history_size == 0) {
136 return result_void(monitoring_error_code::invalid_capacity,
137 "History size must be greater than 0");
138 }
139 if (collection_interval.count() < 10) {
140 return result_void(monitoring_error_code::invalid_interval,
141 "Collection interval must be at least 10ms");
142 }
143 if (buffer_size < history_size) {
144 return result_void(monitoring_error_code::invalid_capacity,
145 "Buffer size must be at least as large as history size");
146 }
147 return result_void::success();
148 }
149};
