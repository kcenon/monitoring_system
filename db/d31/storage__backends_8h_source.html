75struct storage_config {
76 storage_backend_type type{storage_backend_type::memory_buffer};
77 std::string path;
78 std::string data_directory;
79 compression_algorithm compression{compression_algorithm::none};
80 size_t max_size_mb{100};
81 bool auto_flush{true};
82 std::chrono::milliseconds flush_interval{std::chrono::milliseconds(5000)};
83
84 // Extended configuration for Phase 2
85 size_t max_capacity{1000};
86 size_t batch_size{100};
87 std::string table_name;
88 std::string host;
89 uint16_t port{0};
90 std::string database_name;
91 std::string username;
92 std::string password;
93
98 result<bool> validate() const {
99 // Memory buffer doesn't require path
100 if (type != storage_backend_type::memory_buffer) {
101 if (path.empty() && host.empty()) {
102 return make_error<bool>(monitoring_error_code::invalid_configuration,
103 "Path or host required for non-memory storage");
104 }
105 }
106
107 if (max_capacity == 0) {
108 return make_error<bool>(monitoring_error_code::invalid_capacity,
109 "Capacity must be greater than 0");
110 }
111
112 if (batch_size == 0) {
113 return make_error<bool>(monitoring_error_code::invalid_configuration,
114 "Batch size must be greater than 0");
115 }
116
117 if (batch_size > max_capacity) {
118 return make_error<bool>(monitoring_error_code::invalid_configuration,
119 "Batch size cannot exceed capacity");
120 }
121
122 return make_success(true);
123 }
124};

146class file_storage_backend : public snapshot_storage_backend {
147public:
148 file_storage_backend() : config_() {}
149
150 explicit file_storage_backend(const storage_config& config)
151 : config_(config) {}
152
153 result<bool> store(const metrics_snapshot& snapshot) override {
154 std::lock_guard<std::mutex> lock(mutex_);
155
156 // Remove oldest if at capacity
157 if (snapshots_.size() >= config_.max_capacity) {
158 snapshots_.pop_front();
159 }
160
161 snapshots_.push_back(snapshot);
162 return make_success(true);
163 }
164
165 result<metrics_snapshot> retrieve(size_t index) override {
166 std::lock_guard<std::mutex> lock(mutex_);
167
168 if (index >= snapshots_.size()) {
169 return make_error<metrics_snapshot>(monitoring_error_code::not_found,
170 "Snapshot index out of range");
171 }
172
173 return make_success(snapshots_[index]);
174 }
175
176 result<std::vector<metrics_snapshot>> retrieve_range(size_t start, size_t count) override {
177 std::lock_guard<std::mutex> lock(mutex_);
178
179 std::vector<metrics_snapshot> result;
180 size_t end = std::min(start + count, snapshots_.size());
181
182 for (size_t i = start; i < end; ++i) {
183 result.push_back(snapshots_[i]);
184 }
185
186 return make_success(std::move(result));
187 }
188
189 size_t size() const override {
190 std::lock_guard<std::mutex> lock(mutex_);
191 return snapshots_.size();
192 }
193
194 size_t capacity() const override {
195 return config_.max_capacity;
196 }
197
198 result<bool> flush() override {
199 // Stub implementation - actual file I/O would go here
200 return make_success(true);
201 }
202
203 result<bool> clear() override {
204 std::lock_guard<std::mutex> lock(mutex_);
205 snapshots_.clear();
206 return make_success(true);
207 }
208
209 std::unordered_map<std::string, size_t> get_stats() const override {
210 std::lock_guard<std::mutex> lock(mutex_);
211 return {
212 {"total_snapshots", snapshots_.size()},
213 {"capacity", config_.max_capacity}
214 };
215 }
216
217private:
218 storage_config config_;
219 std::deque<metrics_snapshot> snapshots_;
220 mutable std::mutex mutex_;
221};

384class storage_backend_factory {
385public:
391 static std::unique_ptr<snapshot_storage_backend> create_backend(const storage_config& config) {
392 switch (config.type) {
393 case storage_backend_type::memory_buffer:
394 case storage_backend_type::file_json:
395 case storage_backend_type::file_binary:
396 case storage_backend_type::file_csv:
397 return std::make_unique<file_storage_backend>(config);
398
399 case storage_backend_type::database_sqlite:
400 case storage_backend_type::database_postgresql:
401 case storage_backend_type::database_mysql:
402 return std::make_unique<database_storage_backend>(config);
403
404 case storage_backend_type::cloud_s3:
405 case storage_backend_type::cloud_gcs:
406 case storage_backend_type::cloud_azure_blob:
407 return std::make_unique<cloud_storage_backend>(config);
408
409 default:
410 return nullptr;
411 }
412 }

512class memory_storage_backend : public kv_storage_backend {
513public:
