75struct storage_config {
76 storage_backend_type type{storage_backend_type::memory_buffer};
77 std::string path;
78 std::string data_directory;
79 compression_algorithm compression{compression_algorithm::none};
80 size_t max_size_mb{100};
81 bool auto_flush{true};
82 std::chrono::milliseconds flush_interval{std::chrono::milliseconds(5000)};
83
84 // Extended configuration for Phase 2
85 size_t max_capacity{1000};
86 size_t batch_size{100};
87 std::string table_name;
88 std::string host;
89 uint16_t port{0};
90 std::string database_name;
91 std::string username;
92 std::string password;
93
98 common::Result<bool> validate() const {
99 // Memory buffer doesn't require path
100 if (type != storage_backend_type::memory_buffer) {
101 if (path.empty() && host.empty()) {
102 return common::Result<bool>::err(error_info(monitoring_error_code::invalid_configuration, "Path or host required for non-memory storage").to_common_error());
103 }
104 }
105
106 if (max_capacity == 0) {
107 return common::Result<bool>::err(error_info(monitoring_error_code::invalid_capacity, "Capacity must be greater than 0").to_common_error());
108 }
109
110 if (batch_size == 0) {
111 return common::Result<bool>::err(error_info(monitoring_error_code::invalid_configuration, "Batch size must be greater than 0").to_common_error());
112 }
113
114 if (batch_size > max_capacity) {
115 return common::Result<bool>::err(error_info(monitoring_error_code::invalid_configuration, "Batch size cannot exceed capacity").to_common_error());
116 }
117
118 return common::ok(true);
119 }
120};

142class file_storage_backend : public snapshot_storage_backend {
143public:
144 file_storage_backend() : config_() {}
145
146 explicit file_storage_backend(const storage_config& config)
147 : config_(config) {}
148
149 common::Result<bool> store(const metrics_snapshot& snapshot) override {
150 std::lock_guard<std::mutex> lock(mutex_);
151
152 // Remove oldest if at capacity
153 if (snapshots_.size() >= config_.max_capacity) {
154 snapshots_.pop_front();
155 }
156
157 snapshots_.push_back(snapshot);
158 return common::ok(true);
159 }
160
161 common::Result<metrics_snapshot> retrieve(size_t index) override {
162 std::lock_guard<std::mutex> lock(mutex_);
163
164 if (index >= snapshots_.size()) {
165 return common::Result<metrics_snapshot>::err(error_info(monitoring_error_code::not_found, "Snapshot index out of range").to_common_error());
166 }
167
168 return common::ok(snapshots_[index]);
169 }
170
171 common::Result<std::vector<metrics_snapshot>> retrieve_range(size_t start, size_t count) override {
172 std::lock_guard<std::mutex> lock(mutex_);
173
174 std::vector<metrics_snapshot> result;
175 size_t end = std::min(start + count, snapshots_.size());
176
177 for (size_t i = start; i < end; ++i) {
178 result.push_back(snapshots_[i]);
179 }
180
181 return common::ok(std::move(result));
182 }
183
184 size_t size() const override {
185 std::lock_guard<std::mutex> lock(mutex_);
186 return snapshots_.size();
187 }
188
189 size_t capacity() const override {
190 return config_.max_capacity;
191 }
192
193 common::Result<bool> flush() override {
194 // Stub implementation - actual file I/O would go here
195 return common::ok(true);
196 }
197
198 common::Result<bool> clear() override {
199 std::lock_guard<std::mutex> lock(mutex_);
200 snapshots_.clear();
201 return common::ok(true);
202 }
203
204 std::unordered_map<std::string, size_t> get_stats() const override {
205 std::lock_guard<std::mutex> lock(mutex_);
206 return {
207 {"total_snapshots", snapshots_.size()},
208 {"capacity", config_.max_capacity}
209 };
210 }
211
212private:
213 storage_config config_;
214 std::deque<metrics_snapshot> snapshots_;
215 mutable std::mutex mutex_;

454class storage_backend_factory {
455public:
461 static std::unique_ptr<snapshot_storage_backend> create_backend(const storage_config& config) {
462 switch (config.type) {
463 case storage_backend_type::memory_buffer:
464 return std::make_unique<memory_storage_backend>(config);
465
466 case storage_backend_type::file_json:
467 case storage_backend_type::file_binary:
468 case storage_backend_type::file_csv:
469 return std::make_unique<file_storage_backend>(config);
470
471 case storage_backend_type::database_sqlite:
472 case storage_backend_type::database_postgresql:
473 case storage_backend_type::database_mysql:
474 return std::make_unique<database_storage_backend>(config);
475
476 case storage_backend_type::cloud_s3:
477 case storage_backend_type::cloud_gcs:
478 case storage_backend_type::cloud_azure_blob:
479 return std::make_unique<cloud_storage_backend>(config);
480
481 default:
482 return nullptr;
483 }
484 }
485
490 static std::vector<storage_backend_type> get_supported_backends() {
491 return {
