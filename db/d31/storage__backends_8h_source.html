93class memory_storage_backend : public kv_storage_backend {
94public:
95 bool store(const std::string& key, const std::string& value) override {
96 data_[key] = value;
97 return true;
98 }
99
100 std::string retrieve(const std::string& key) override {
101 auto it = data_.find(key);
102 return it != data_.end() ? it->second : "";
103 }
104
105 bool remove(const std::string& key) override {
106 return data_.erase(key) > 0;
107 }
108
109private:
110 std::unordered_map<std::string, std::string> data_;
111};
