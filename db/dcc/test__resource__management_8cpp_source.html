411// Concurrency Tests
412TEST_F(ResourceManagementTest, RateLimiterConcurrency) {
413 auto limiter = create_token_bucket_limiter("concurrent_test", 1000, 100,
414 throttling_strategy::reject);
415
416 const int num_threads = 10;
417 const int operations_per_thread = 10;
418 std::atomic<int> successful_operations{0};
419
420 std::vector<std::thread> threads;
421 for (int i = 0; i < num_threads; ++i) {
422 threads.emplace_back([&]() {
423 for (int j = 0; j < operations_per_thread; ++j) {
424 if (limiter->try_acquire(1)) {
425 successful_operations++;
426 }
427 }
428 });
429 }
430
431 for (auto& thread : threads) {
432 thread.join();
433 }
434
435 // Should have some successful operations
436 EXPECT_GT(successful_operations.load(), 0);
437 EXPECT_LE(successful_operations.load(), 100); // Limited by burst capacity
438}
