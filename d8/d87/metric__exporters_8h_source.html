124struct prometheus_metric_data {
125 std::string name;
126 metric_type type;
127 double value;
128 std::chrono::system_clock::time_point timestamp;
129 std::unordered_map<std::string, std::string> labels;
130 std::string help_text;
131
135 std::string to_prometheus_text() const {
136 std::ostringstream ss;
137
138 // Add HELP line
139 if (!help_text.empty()) {
140 ss << "# HELP " << name << " " << help_text << "\n";
141 }
142
143 // Add TYPE line
144 std::string type_str;
145 switch (type) {
146 case metric_type::counter: type_str = "counter"; break;
147 case metric_type::gauge: type_str = "gauge"; break;
148 case metric_type::histogram: type_str = "histogram"; break;
149 case metric_type::summary: type_str = "summary"; break;
150 case metric_type::timer: type_str = "gauge"; break; // Timer as gauge in Prometheus
151 }
152 ss << "# TYPE " << name << " " << type_str << "\n";
153
154 // Add metric line
155 ss << name;
156
157 if (!labels.empty()) {
158 ss << "{";
159 bool first = true;
160 for (const auto& [key, label_value] : labels) {
161 if (!first) ss << ",";
162 ss << key << "=\"" << escape_label_value(label_value) << "\"";
163 first = false;
164 }
165 ss << "}";
166 }
167
168 ss << " " << value;
169
170 // Add timestamp if available
171 if (timestamp != std::chrono::system_clock::time_point{}) {
172 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
173 timestamp.time_since_epoch()).count();
174 ss << " " << ms;
175 }
176
177 ss << "\n";
178 return ss.str();
179 }
180
181private:
182 std::string escape_label_value(const std::string& label_value) const {
183 std::string escaped = label_value;
184 // Escape backslashes, quotes, and newlines
185 escaped = std::regex_replace(escaped, std::regex("\\\\"), "\\\\");
186 escaped = std::regex_replace(escaped, std::regex("\""), "\\\"");
187 escaped = std::regex_replace(escaped, std::regex("\n"), "\\n");
188 return escaped;
189 }

196struct statsd_metric_data {
197 std::string name;
198 metric_type type;
199 double value;
200 double sample_rate = 1.0;
201 std::unordered_map<std::string, std::string> tags;
202
206 std::string to_statsd_format(bool datadog_format = false) const {
207 std::ostringstream ss;
208
209 ss << name << ":" << value << "|";
210
211 // Add type indicator
212 switch (type) {
213 case metric_type::counter: ss << "c"; break;
214 case metric_type::gauge: ss << "g"; break;
215 case metric_type::timer: ss << "ms"; break;
216 case metric_type::histogram: ss << "h"; break;
217 case metric_type::summary: ss << "s"; break;
218 }
219
220 // Add sample rate if not 1.0
221 if (sample_rate != 1.0) {
222 ss << "|@" << sample_rate;
223 }
224
225 // Add tags (DataDog format)
226 if (datadog_format && !tags.empty()) {
227 ss << "|#";
228 bool first = true;
229 for (const auto& [key, tag_value] : tags) {
230 if (!first) ss << ",";
231 ss << key << ":" << tag_value;
232 first = false;
233 }
234 }
235
236 return ss.str();
237 }

480 // Simple heuristics for metric type inference

733 common::VoidResult send_udp_batch(const std::vector<std::string>& lines) {
734 if (!transport_) {
735 return common::VoidResult::err(error_info(
736 monitoring_error_code::dependency_missing,
737 "UDP transport not available",
738 "statsd_exporter"
739 ).to_common_error());
740 }
741
742 // Auto-connect if not already connected
743 if (!transport_->is_connected()) {
744 auto connect_result = transport_->connect(config_.endpoint, config_.port);
745 if (connect_result.is_err()) {
746 return connect_result;
747 }
748 }
749
750 // Combine lines into a single packet (newline-separated)
751 // StatsD typically accepts multiple metrics per packet
752 std::string batch;
753 for (const auto& line : lines) {
754 if (!batch.empty()) {
755 batch += '\n';
756 }
757 batch += line;
758 }
759
760 // Send the batch
761 return transport_->send(batch);
762 }

974 common::VoidResult send_via_http(const std::vector<otel_metric_data>& metrics) {
975 if (!http_transport_) {
976 return common::VoidResult::err(error_info(
977 monitoring_error_code::dependency_missing,
978 "HTTP transport not available",
979 "otlp_metrics_exporter"
980 ).to_common_error());
981 }
982
983 // Build OTLP HTTP endpoint
984 std::string endpoint = config_.endpoint;
985 if (config_.port != 0) {
986 endpoint += ":" + std::to_string(config_.port);
987 }
988 endpoint += "/v1/metrics";
989
990 // Serialize metrics to JSON or Protobuf
991 std::vector<uint8_t> body = serialize_metrics(metrics);
992
993 http_request request;
994 request.url = endpoint;
995 request.method = "POST";
996 request.headers["Content-Type"] = get_content_type();
997 request.body = std::move(body);
998 request.timeout = config_.timeout;
999
1000 // Add custom headers
1001 for (const auto& [key, value] : config_.headers) {
1002 request.headers[key] = value;
1003 }
1004
1005 auto result = http_transport_->send(request);
1006 if (result.is_err()) {
1007 return common::VoidResult::err(error_info(
1008 monitoring_error_code::network_error,
1009 "HTTP send failed: " + result.error().message,
1010 "otlp_metrics_exporter"
1011 ).to_common_error());
1012 }
1013
1014 const auto& response = result.value();
1015 if (response.status_code < 200 || response.status_code >= 300) {
1016 return common::VoidResult::err(error_info(
1017 monitoring_error_code::operation_failed,
1018 "OTLP HTTP request failed with status " + std::to_string(response.status_code),
1019 "otlp_metrics_exporter"
1020 ).to_common_error());
1021 }
1022
1023 return common::ok();
1024 }

1026 common::VoidResult send_via_grpc(const std::vector<otel_metric_data>& metrics) {
1027 if (!grpc_transport_) {
1028 return common::VoidResult::err(error_info(
1029 monitoring_error_code::dependency_missing,
1030 "gRPC transport not available",
1031 "otlp_metrics_exporter"
1032 ).to_common_error());
1033 }
1034
1035 // Connect if not already connected
1036 if (!grpc_transport_->is_connected()) {
1037 auto connect_result = grpc_transport_->connect(config_.endpoint, config_.port);
1038 if (connect_result.is_err()) {
1039 return connect_result;
1040 }
1041 }
1042
1043 // Serialize metrics to protobuf
1044 std::vector<uint8_t> body = serialize_metrics(metrics);
1045
1046 grpc_request request;
1047 request.service = "opentelemetry.proto.collector.metrics.v1.MetricsService";
1048 request.method = "Export";
1049 request.body = std::move(body);
1050 request.timeout = config_.timeout;
1051
1052 auto result = grpc_transport_->send(request);
1053 if (result.is_err()) {
1054 return common::VoidResult::err(error_info(
1055 monitoring_error_code::network_error,
1056 "gRPC send failed: " + result.error().message,
1057 "otlp_metrics_exporter"
1058 ).to_common_error());
1059 }
1060
1061 const auto& response = result.value();
1062 if (response.status_code != 0) { // gRPC OK is 0
1063 return common::VoidResult::err(error_info(
1064 monitoring_error_code::operation_failed,
1065 "OTLP gRPC request failed: " + response.status_message,
1066 "otlp_metrics_exporter"
1067 ).to_common_error());
1068 }
1069
1070 return common::ok();
1071 }

1073 std::vector<uint8_t> serialize_metrics(const std::vector<otel_metric_data>& metrics) const {
1074 // Serialize metrics based on format
1075 // For JSON format, convert to JSON string
1076 // For protobuf format, serialize to protobuf bytes
1077 // This is a simplified implementation
1078 std::string json = "{\"resourceMetrics\":[";
1079
1080 bool first = true;
1081 for (const auto& metric : metrics) {
1082 if (!first) json += ",";
1083 first = false;
1084
1085 json += "{\"resource\":{},\"scopeMetrics\":[{\"metrics\":[{";
1086 json += "\"name\":\"" + metric.name + "\",";
1087 json += "\"gauge\":{\"dataPoints\":[{\"asDouble\":" +
1088 std::to_string(metric.value) + "}]}";
1089 json += "}]}]}";
1090 }
1091
1092 json += "]}";
1093
1094 return std::vector<uint8_t>(json.begin(), json.end());
1095 }
