154 void add_check(std::shared_ptr<health_check> check) {
155 std::lock_guard<std::mutex> lock(mutex_);
156 checks_.push_back(std::move(check));
157 }
158
159 health_check_result check() override {
160 std::lock_guard<std::mutex> lock(mutex_);
161
162 if (checks_.empty()) {
163 return health_check_result::healthy("No checks configured");
164 }
165
166 std::vector<health_check_result> results;
167 results.reserve(checks_.size());
168
169 for (const auto& chk : checks_) {
170 results.push_back(chk->check());
171 }
172
173 if (all_required_) {
174 return check_all_required(results);
175 } else {
176 return check_any_required(results);
177 }
178 }
179
180private:
181 health_check_result check_all_required(const std::vector<health_check_result>& results) {
182 bool has_unhealthy = false;
183 bool has_degraded = false;
184 std::string message;
185
186 for (const auto& result : results) {
187 if (result.status == health_status::unhealthy) {
188 has_unhealthy = true;
189 message += result.message + "; ";
190 } else if (result.status == health_status::degraded) {
191 has_degraded = true;
192 message += result.message + "; ";
193 }
194 }
195
196 if (has_unhealthy) {
197 return health_check_result::unhealthy(message.empty() ? "One or more checks failed" : message);
198 }
199 if (has_degraded) {
200 return health_check_result::degraded(message.empty() ? "One or more checks degraded" : message);
201 }
202 return health_check_result::healthy("All checks passed");
203 }
204
205 health_check_result check_any_required(const std::vector<health_check_result>& results) {
206 bool any_healthy = false;
207 std::string message;
208
209 for (const auto& result : results) {
210 if (result.status == health_status::healthy) {
211 any_healthy = true;
212 break;
213 }
214 message += result.message + "; ";
215 }
216
217 if (any_healthy) {
218 return health_check_result::healthy("At least one check passed");
219 }
220 return health_check_result::unhealthy(message.empty() ? "All checks failed" : message);
221 }
222
223 std::string name_;
224 health_check_type type_;
225 bool all_required_;
226 std::vector<std::shared_ptr<health_check>> checks_;
227 mutable std::mutex mutex_;
228};

248 common::Result<bool> add_dependency(const std::string& dependent, const std::string& dependency) {
249 std::lock_guard<std::shared_mutex> lock(mutex_);
250
251 if (nodes_.find(dependent) == nodes_.end()) {
252 return common::Result<bool>::err(error_info(monitoring_error_code::not_found, "Dependent '" + dependent + "' not found").to_common_error());
253 }
254 if (nodes_.find(dependency) == nodes_.end()) {
255 return common::Result<bool>::err(error_info(monitoring_error_code::not_found, "Dependency '" + dependency + "' not found").to_common_error());
256 }
257
258 if (would_create_cycle_internal(dependent, dependency)) {
259 return common::Result<bool>::err(error_info(monitoring_error_code::invalid_state, "Adding dependency would create a cycle").to_common_error());
260 }
261
262 dependencies_[dependent].push_back(dependency);
263 dependents_[dependency].push_back(dependent);
264 return common::ok(true);
265 }

292 std::vector<std::string> topological_sort() const {
293 std::shared_lock<std::shared_mutex> lock(mutex_);
294
295 std::unordered_map<std::string, int> in_degree;
296 for (const auto& [name, _] : nodes_) {
297 in_degree[name] = 0;
298 }
299
300 for (const auto& [name, deps] : dependencies_) {
301 in_degree[name] = static_cast<int>(deps.size());
302 }
303
304 std::queue<std::string> queue;
305 for (const auto& [name, degree] : in_degree) {
306 if (degree == 0) {
307 queue.push(name);
308 }
309 }
310
311 std::vector<std::string> result;
312 result.reserve(nodes_.size());
313
314 while (!queue.empty()) {
315 std::string current = queue.front();
316 queue.pop();
317 result.push_back(current);
318
319 auto it = dependents_.find(current);
320 if (it != dependents_.end()) {
321 for (const auto& dep : it->second) {
322 if (--in_degree[dep] == 0) {
323 queue.push(dep);
324 }
325 }
326 }
327 }
328
329 return result;
330 }

332 health_check_result check_with_dependencies(const std::string& name) {
333 std::shared_lock<std::shared_mutex> lock(mutex_);
334
335 auto it = nodes_.find(name);
336 if (it == nodes_.end()) {
337 return health_check_result::unhealthy("Node '" + name + "' not found");
338 }
339
340 auto deps_it = dependencies_.find(name);
341 if (deps_it != dependencies_.end()) {
342 for (const auto& dep_name : deps_it->second) {
343 auto dep_it = nodes_.find(dep_name);
344 if (dep_it != nodes_.end()) {
345 auto dep_result = dep_it->second->check();
346 if (dep_result.status == health_status::unhealthy) {
347 return health_check_result::unhealthy(
348 "Dependency '" + dep_name + "' is unhealthy: " + dep_result.message);
349 }
350 if (dep_result.status == health_status::degraded) {
351 return health_check_result::degraded(
352 "Dependency '" + dep_name + "' is degraded: " + dep_result.message);
353 }
354 }
355 }
356 }
357
358 return it->second->check();
359 }

361 std::vector<std::string> get_failure_impact(const std::string& name) const {
362 std::shared_lock<std::shared_mutex> lock(mutex_);
363
364 std::vector<std::string> impacted;
365 std::unordered_set<std::string> visited;
366 std::queue<std::string> to_visit;
367
368 auto it = dependents_.find(name);
369 if (it != dependents_.end()) {
370 for (const auto& dep : it->second) {
371 to_visit.push(dep);
372 }
373 }
374
375 while (!to_visit.empty()) {
376 std::string current = to_visit.front();
377 to_visit.pop();
378
379 if (visited.find(current) != visited.end()) {
380 continue;
381 }
382 visited.insert(current);
383 impacted.push_back(current);
384
385 auto dep_it = dependents_.find(current);
386 if (dep_it != dependents_.end()) {
387 for (const auto& dep : dep_it->second) {
388 if (visited.find(dep) == visited.end()) {
389 to_visit.push(dep);
390 }
391 }
392 }
393 }
394
395 return impacted;
396 }

398private:
399 bool would_create_cycle_internal(const std::string& from, const std::string& to) const {
400 if (from == to) {
401 return true;
402 }
403
404 std::unordered_set<std::string> visited;
405 std::queue<std::string> to_visit;
406 to_visit.push(to);
407
408 while (!to_visit.empty()) {
409 std::string current = to_visit.front();
410 to_visit.pop();
411
412 if (current == from) {
413 return true;
414 }
415
416 if (visited.find(current) != visited.end()) {
417 continue;
418 }
419 visited.insert(current);
420
421 auto it = dependencies_.find(current);
422 if (it != dependencies_.end()) {
423 for (const auto& dep : it->second) {
424 if (visited.find(dep) == visited.end()) {
425 to_visit.push(dep);
426 }
427 }
428 }
429 }
430
431 return false;
432 }

492 common::Result<bool> register_check(const std::string& name, std::shared_ptr<health_check> check) {
493 std::lock_guard<std::shared_mutex> lock(mutex_);
494
495 if (checks_.find(name) != checks_.end()) {
496 return common::Result<bool>::err(error_info(monitoring_error_code::already_exists, "Check '" + name + "' already registered").to_common_error());
497 }
498
499 checks_[name] = std::move(check);
500 auto graph_result = dependency_graph_.add_node(name, checks_[name]);
501 if (graph_result.is_err()) {
502 checks_.erase(name);
503 return common::Result<bool>::err(graph_result.error());
504 }
505 return common::ok(true);
506 }
507
508 common::Result<bool> unregister_check(const std::string& name) {
509 std::lock_guard<std::shared_mutex> lock(mutex_);
510
511 if (checks_.find(name) == checks_.end()) {
512 error_info err(monitoring_error_code::not_found,
513 "Check '" + name + "' not found");
514 return common::Result<bool>::err(err.to_common_error());
515 }
516
517 checks_.erase(name);
518 recovery_handlers_.erase(name);
519 return common::ok(true);
520 }
521
522 common::Result<health_check_result> check(const std::string& name) {
523 std::lock_guard<std::shared_mutex> lock(mutex_);
524
525 auto it = checks_.find(name);
526 if (it == checks_.end()) {
527 error_info err(monitoring_error_code::not_found,
528 "Check '" + name + "' not found");
529 return common::Result<health_check_result>::err(err.to_common_error());
530 }
531
532 auto result = dependency_graph_.check_with_dependencies(name);
533 update_stats(result);
534 cached_results_[name] = result;
535 return common::ok(result);
536 }
537
538 std::unordered_map<std::string, health_check_result> check_all() {
539 std::lock_guard<std::shared_mutex> lock(mutex_);
540
541 std::unordered_map<std::string, health_check_result> results;
542 for (const auto& [name, check] : checks_) {
543 auto result = check->check();
544 results[name] = result;
545 cached_results_[name] = result;
546 update_stats(result);
547 }
548 return results;
549 }

589 void refresh() {
590 std::lock_guard<std::shared_mutex> lock(mutex_);
591
592 for (const auto& [name, check] : checks_) {
593 auto result = check->check();
594 cached_results_[name] = result;
595 update_stats(result);
596
597 if (result.status == health_status::unhealthy) {
598 auto it = recovery_handlers_.find(name);
599 if (it != recovery_handlers_.end() && config_.enable_auto_recovery) {
600 stats_.recovery_attempts++;
601 if (it->second()) {
602 stats_.successful_recoveries++;
603 }
604 }
605 }
606 }
607
608 stats_.last_check_time = std::chrono::system_clock::now();
609 }

617 health_status get_overall_status() {
618 std::shared_lock<std::shared_mutex> lock(mutex_);
619
620 if (checks_.empty()) {
621 return cached_results_.empty() ? health_status::healthy : health_status::unknown;
622 }
623
624 bool has_unhealthy = false;
625 bool has_degraded = false;
626
627 for (const auto& [name, result] : cached_results_) {
628 if (result.status == health_status::unhealthy) {

631 has_degraded = true;
632 }
633 }
634

636 if (has_degraded) return health_status::degraded;
637 return health_status::healthy;

686 void run_monitoring_loop() {
687 while (running_.load()) {
688 refresh();
689
690 std::unique_lock<std::mutex> lock(cv_mutex_);

692 return !running_.load();

697 void update_stats(const health_check_result& result) {
698 stats_.total_checks++;
699 switch (result.status) {
700 case health_status::healthy:
701 stats_.healthy_checks++;
702 break;
703 case health_status::unhealthy:
704 stats_.unhealthy_checks++;
705 break;
706 case health_status::degraded:
707 stats_.degraded_checks++;
708 break;
709 default:
710 break;
711 }
712 }
