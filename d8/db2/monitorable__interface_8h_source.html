486 return comp->get_monitoring_id() == id;

500 common::Result<monitoring_data> collect_all() const {
501 monitoring_data aggregated(aggregator_id_);
502
503 for (const auto& component : components_) {
504 if (!component->is_monitoring_enabled()) {
505 continue;
506 }
507
508 auto result = component->get_monitoring_data();
509 if (result.is_ok()) {
510 aggregated.merge(result.value(), component->get_monitoring_id());
511 } else {
512 // Log error but continue with other components
513 aggregated.add_tag(
514 component->get_monitoring_id() + ".error",
515 result.error().message
516 );
517 }
518 }
519
520 // Add aggregator metadata
521 aggregated.add_metric("aggregator.component_count",
522 static_cast<double>(components_.size()));
523 aggregated.add_metric("aggregator.total_metrics",
524 static_cast<double>(aggregated.metric_count()));
525
526 return common::ok(std::move(aggregated));
527 }
528
534 std::shared_ptr<monitorable_interface> get_component(const std::string& id) const {
535 auto it = std::find_if(components_.begin(), components_.end(),
536 [&id](const auto& comp) {
537 return comp->get_monitoring_id() == id;
538 });
539
540 return (it != components_.end()) ? *it : nullptr;
541 }
542
547 std::vector<std::string> get_component_ids() const {
