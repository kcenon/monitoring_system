52// Test that all builtin collectors are registered with the registry
53TEST_F(CollectorRegistryIntegrationTest, BuiltinCollectorsRegisteredWithRegistry) {
54 auto& registry = collector_registry::instance();
55
56 // Register all builtin collectors
57 bool success = register_builtin_collectors();
58 EXPECT_TRUE(success);
59
60 // Check that we have 9 plugin-based collectors registered
61 // (battery, uptime, interrupt, network_metrics, platform_metrics,
62 // process_metrics, security, smart, vm)
63 auto total_plugins = registry.plugin_count();
64 EXPECT_EQ(total_plugins, 9);
65}

67// Test that get_plugins returns all registered collectors
68TEST_F(CollectorRegistryIntegrationTest, GetPluginsReturnsAllCollectors) {
69 auto& registry = collector_registry::instance();
70
71 // Register builtin collectors
72 register_builtin_collectors();
73
74 // Try to get a specific plugin first (safer)
75 auto* battery = registry.get_plugin("battery_collector");
76
77 // If battery collector is available on this platform

79 // Check that it has a valid name (the actual implementation uses "battery")
80 EXPECT_FALSE(std::string(battery->name()).empty());
