70struct time_point_data {
71 std::chrono::system_clock::time_point timestamp;
72 double value;
73 uint32_t sample_count; // Number of samples aggregated into this point
74
75 time_point_data() noexcept : value(0.0), sample_count(0) {}
76
77 time_point_data(std::chrono::system_clock::time_point ts, double val, uint32_t count = 1) noexcept
78 : timestamp(ts), value(val), sample_count(count) {}
79
83 void merge(const time_point_data& other) {
84 if (sample_count == 0) {
85 *this = other;
86 } else if (other.sample_count > 0) {
87 double total_weight = sample_count + other.sample_count;
88 value = (value * sample_count + other.value * other.sample_count) / total_weight;
89 sample_count += other.sample_count;
90
91 // Use the later timestamp
92 if (other.timestamp > timestamp) {
93 timestamp = other.timestamp;
94 }
95 }
96 }
97
101 bool is_valid(std::chrono::system_clock::time_point cutoff) const noexcept {
102 return timestamp >= cutoff;
103 }

148struct aggregation_result {
149 std::vector<time_point_data> points;
150 std::chrono::system_clock::time_point query_start;
151 std::chrono::system_clock::time_point query_end;
152 size_t total_samples;
153
154 aggregation_result() : total_samples(0) {}
155
159 summary_data get_summary() const {
160 summary_data summary;
161 for (const auto& point : points) {
162 summary.add_sample(point.value);
163 }
164 return summary;
165 }
166
170 double get_average() const {
171 if (points.empty()) return 0.0;
172
173 double sum = 0.0;
174 uint64_t total_weight = 0;
175
176 for (const auto& point : points) {
177 sum += point.value * point.sample_count;
178 total_weight += point.sample_count;
179 }
180
181 return total_weight > 0 ? sum / total_weight : 0.0;
182 }
183
187 double get_rate() const {
188 if (points.size() < 2) return 0.0;
189
190 const auto& first = points.front();
191 const auto& last = points.back();
192
193 auto duration = std::chrono::duration_cast<std::chrono::seconds>(
194 last.timestamp - first.timestamp);
195
196 if (duration.count() <= 0) return 0.0;
197
198 return (last.value - first.value) / duration.count();
199 }

206class time_series {
207private:
208 mutable std::mutex mutex_;
209 std::vector<time_point_data> data_;
210 time_series_config config_;
211 std::string series_name_;
212 size_t insertion_count_ = 0; // Track insertions for periodic maintenance
213
217 void cleanup_old_data() {
218 auto cutoff = std::chrono::system_clock::now() - config_.retention_period;
219
220 auto it = std::remove_if(data_.begin(), data_.end(),
221 [cutoff](const time_point_data& point) {
222 return !point.is_valid(cutoff);
223 });
224
225 data_.erase(it, data_.end());
226 }
227
231 void compress_data() {
232 if (!config_.enable_compression || data_.size() < 3) {
233 return;
234 }
235
236 // Simple compression: remove points that don't add significant information
237 std::vector<time_point_data> compressed;
238 compressed.reserve(data_.size());
239
240 if (!data_.empty()) {
241 compressed.push_back(data_[0]);
242
243 for (size_t i = 1; i < data_.size() - 1; ++i) {
244 const auto& prev = data_[i - 1];
245 const auto& curr = data_[i];
246 const auto& next = data_[i + 1];
247
248 // Check if current point is significantly different from linear interpolation
249 double expected = prev.value + (next.value - prev.value) *
250 (std::chrono::duration<double>(curr.timestamp - prev.timestamp).count() /
251 std::chrono::duration<double>(next.timestamp - prev.timestamp).count());
252
253 if (std::abs(curr.value - expected) > config_.compression_threshold) {
254 compressed.push_back(curr);
255 }
256 }
257
258 if (data_.size() > 1) {
259 compressed.push_back(data_.back());
260 }
261 }
262
263 data_ = std::move(compressed);
264 }
265
269 void enforce_size_limit() {
270 if (data_.size() > config_.max_points) {
271 size_t remove_count = data_.size() - config_.max_points;
272 data_.erase(data_.begin(), data_.begin() + remove_count);
273 }
274 }

305 result_void add_point(double value,
306 std::chrono::system_clock::time_point timestamp =
307 std::chrono::system_clock::now()) {
308 std::lock_guard<std::mutex> lock(mutex_);
309
310 time_point_data point(timestamp, value);
311
312 // Optimize: if timestamp is newest, append directly (O(1) vs O(n))
313 if (data_.empty() || point.timestamp >= data_.back().timestamp) {
314 data_.push_back(point);
315 } else {
316 // Insert in chronological order only when necessary
317 auto it = std::upper_bound(data_.begin(), data_.end(), point,
318 [](const time_point_data& a, const time_point_data& b) {
319 return a.timestamp < b.timestamp;
320 });
321
322 data_.insert(it, point);
323 }
324
325 // Perform maintenance periodically (every 100 insertions) instead of every time
326 ++insertion_count_;
327 if (insertion_count_ % 100 == 0) {
328 cleanup_old_data();
329 compress_data();
330 enforce_size_limit();
331 }
332
333 return make_void_success();
334 }

339 result_void add_points(const std::vector<time_point_data>& points) {
340 std::lock_guard<std::mutex> lock(mutex_);
341
342 for (const auto& point : points) {
343 // Optimize: if timestamp is newest, append directly
344 if (data_.empty() || point.timestamp >= data_.back().timestamp) {
345 data_.push_back(point);
346 } else {
347 auto it = std::upper_bound(data_.begin(), data_.end(), point,
348 [](const time_point_data& a, const time_point_data& b) {
349 return a.timestamp < b.timestamp;
350 });
351
352 data_.insert(it, point);
353 }
354 }
355
356 // Perform maintenance after batch insert
357 cleanup_old_data();
358 compress_data();
359 enforce_size_limit();
360
361 return make_void_success();
362 }

367 result<aggregation_result> query(const time_series_query& query) const {
368 auto validation = query.validate();
369 if (validation.is_err()) {
370 return result<aggregation_result>::err(error_info(monitoring_error_code::invalid_argument,
371 validation.error().message, "monitoring_system").to_common_error());
372 }
373
374 std::lock_guard<std::mutex> lock(mutex_);
375
376 aggregation_result result;
377 result.query_start = query.start_time;
378 result.query_end = query.end_time;
379
380 // Find data points in the time range
381 auto start_it = std::lower_bound(data_.begin(), data_.end(), query.start_time,
382 [](const time_point_data& point, std::chrono::system_clock::time_point time) {
383 return point.timestamp < time;
384 });
385
386 auto end_it = std::upper_bound(data_.begin(), data_.end(), query.end_time,
387 [](std::chrono::system_clock::time_point time, const time_point_data& point) {
388 return time < point.timestamp;
389 });
390
391 if (start_it == end_it) {
392 return make_success(std::move(result)); // No data in range
393 }
394
395 // Aggregate data by step size
396 auto current_step_start = query.start_time;
397
398 while (current_step_start < query.end_time) {
399 auto current_step_end = current_step_start + query.step;
400 if (current_step_end > query.end_time) {
401 current_step_end = query.end_time;
402 }
403
404 // Find points in this step
405 std::vector<time_point_data> step_points;
406 for (auto it = start_it; it != end_it; ++it) {
407 if (it->timestamp >= current_step_start && it->timestamp < current_step_end) {
408 step_points.push_back(*it);
409 result.total_samples += it->sample_count;
410 }
411 }
412
413 // Aggregate points in this step
414 if (!step_points.empty()) {
415 time_point_data aggregated_point;
416 aggregated_point.timestamp = current_step_start + query.step / 2; // Middle of step
417
418 for (const auto& point : step_points) {
419 aggregated_point.merge(point);
420 }
421
422 result.points.push_back(aggregated_point);
423 }
424
425 current_step_start = current_step_end;
426 }
427
428 return make_success(std::move(result));
429 }
