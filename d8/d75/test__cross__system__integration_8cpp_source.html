57class simple_mock_logger : public ILogger, public IMonitorable {
58private:
59 std::shared_ptr<IMonitor> monitor_;
60 int log_count_{0};
61
62public:
63 VoidResult log(log_level level, const std::string& message) override {
64 log_count_++;
65 if (monitor_) {
66 monitor_->record_metric("logs_written", log_count_);
67 }
68 return std::monostate{};
69 }
70
71 VoidResult log(log_level level, const std::string& message,
72 const std::string& file, int line,
73 const std::string& function) override {
74 return log(level, message);
75 }
76
77 VoidResult log(const log_entry& entry) override {
78 return log(entry.level, entry.message);
79 }
80
81 bool is_enabled(log_level level) const override { return true; }
82 VoidResult set_level(log_level level) override { return std::monostate{}; }
83 log_level get_level() const override { return log_level::info; }
84 VoidResult flush() override { return std::monostate{}; }
85
86 // IMonitorable implementation
87 Result<metrics_snapshot> get_monitoring_data() override {
88 metrics_snapshot snapshot;
89 snapshot.source_id = "simple_mock_logger";
90 snapshot.add_metric("total_logs", log_count_);
91 return snapshot;
92 }
93
94 Result<health_check_result> health_check() override {
95 health_check_result result;
96 result.status = health_status::healthy;
97 result.message = "Mock logger operational";
98 return result;
99 }
100
101 std::string get_component_name() const override {
102 return "simple_mock_logger";
103 }
104
105 // For bidirectional DI
106 void set_monitor(std::shared_ptr<IMonitor> monitor) {
107 monitor_ = monitor;
108 }
109
110 int get_log_count() const { return log_count_; }
111};

186TEST(CrossSystemIntegrationTest, BidirectionalDependencyInjection) {
187 // Create both systems
188 auto logger = std::make_shared<simple_mock_logger>();
189 auto monitor = std::make_shared<performance_monitor>();
190
191 // Bidirectional injection
192 logger->set_monitor(monitor);
193 // Note: monitor doesn't need to set logger in this example,
194 // but adapters can use ILogger interface
195
196 // Use both systems
197 for (int i = 0; i < 10; ++i) {
198 logger->log(log_level::info, "Request " + std::to_string(i));
199 monitor->record_metric("requests_processed", i + 1);
200 }
201
202 // Verify logger health
203 auto logger_health = logger->health_check();
204 ASSERT_TRUE(std::holds_alternative<health_check_result>(logger_health));
205 EXPECT_TRUE(std::get<health_check_result>(logger_health).is_healthy());
206
207 // Verify monitor health
208 auto monitor_health = monitor->check_health();
209 ASSERT_TRUE(std::holds_alternative<health_check_result>(monitor_health));
210 EXPECT_TRUE(std::get<health_check_result>(monitor_health).is_healthy());
211
212 // Verify metrics
213 auto monitor_metrics = monitor->get_metrics();
214 ASSERT_TRUE(std::holds_alternative<metrics_snapshot>(monitor_metrics));
215
216 auto& snapshot = std::get<metrics_snapshot>(monitor_metrics);
217 EXPECT_FALSE(snapshot.metrics.empty());
218
219 // Logger should have logged
220 EXPECT_EQ(10, logger->get_log_count());
221}

273TEST(CrossSystemIntegrationTest, IntegrationPerformanceOverhead) {
274 auto logger = std::make_shared<simple_mock_logger>();
275 auto monitor = std::make_shared<performance_monitor>();
276
277 logger->set_monitor(monitor);
278
279 // Measure time for integrated operations
280 auto start = std::chrono::high_resolution_clock::now();
281
282 for (int i = 0; i < 1000; ++i) {
283 logger->log(log_level::info, "Performance test");
284 monitor->record_metric("perf_test", i);
285 }
286
287 auto end = std::chrono::high_resolution_clock::now();
288 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);

290 // Integration should be fast (< 100ms for 1000 operations)
