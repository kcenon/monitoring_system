120class mock_monitor : public common_if::IMonitor {
121private:
122 std::vector<common_if::metric_value> metrics_;
123 mutable std::mutex mutex_;
124
125public:
126 VoidResult record_metric(const std::string& name, double value) override {
127 std::lock_guard<std::mutex> lock(mutex_);
128 metrics_.emplace_back(name, value);
129 return kcenon::common::ok();
130 }
131
132 VoidResult record_metric(
133 const std::string& name,
134 double value,
135 const std::unordered_map<std::string, std::string>& tags) override {
136 std::lock_guard<std::mutex> lock(mutex_);
137 common_if::metric_value mv(name, value);
138 mv.tags = tags;
139 metrics_.push_back(mv);
140 return kcenon::common::ok();
141 }
142
143 Result<common_if::metrics_snapshot> get_metrics() override {
144 std::lock_guard<std::mutex> lock(mutex_);
145 common_if::metrics_snapshot snapshot;
146 snapshot.metrics = metrics_;
147 snapshot.source_id = "mock_monitor";
148 return snapshot;
149 }
150
151 Result<common_if::health_check_result> check_health() override {
152 common_if::health_check_result result;
153 result.status = common_if::health_status::healthy;
154 result.message = "Mock monitor operational";
155 return result;
156 }
157
158 VoidResult reset() override {
159 std::lock_guard<std::mutex> lock(mutex_);
160 metrics_.clear();
161 return kcenon::common::ok();
162 }
163
164 size_t get_metric_count() const {
165 std::lock_guard<std::mutex> lock(mutex_);
166 return metrics_.size();
167 }
168};
