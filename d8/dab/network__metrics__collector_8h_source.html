89inline std::string tcp_state_to_string(tcp_state state) {

129 uint64_t get_count(tcp_state state) const {
130 switch (state) {
131 case tcp_state::ESTABLISHED: return established;
132 case tcp_state::SYN_SENT: return syn_sent;
133 case tcp_state::SYN_RECV: return syn_recv;
134 case tcp_state::FIN_WAIT1: return fin_wait1;
135 case tcp_state::FIN_WAIT2: return fin_wait2;
136 case tcp_state::TIME_WAIT: return time_wait;
137 case tcp_state::CLOSE: return close;
138 case tcp_state::CLOSE_WAIT: return close_wait;
139 case tcp_state::LAST_ACK: return last_ack;
140 case tcp_state::LISTEN: return listen;
141 case tcp_state::CLOSING: return closing;
142 default: return unknown;
143 }
144 }
145
150 void increment(tcp_state state) {
151 switch (state) {
152 case tcp_state::ESTABLISHED: ++established; break;
153 case tcp_state::SYN_SENT: ++syn_sent; break;
154 case tcp_state::SYN_RECV: ++syn_recv; break;
155 case tcp_state::FIN_WAIT1: ++fin_wait1; break;
156 case tcp_state::FIN_WAIT2: ++fin_wait2; break;
157 case tcp_state::TIME_WAIT: ++time_wait; break;
158 case tcp_state::CLOSE: ++close; break;
159 case tcp_state::CLOSE_WAIT: ++close_wait; break;
160 case tcp_state::LAST_ACK: ++last_ack; break;
161 case tcp_state::LISTEN: ++listen; break;
162 case tcp_state::CLOSING: ++closing; break;
163 default: ++unknown; break;
164 }
165 }
166
171 uint64_t total() const {
172 return established + syn_sent + syn_recv + fin_wait1 + fin_wait2 +
173 time_wait + close + close_wait + last_ack + listen + closing + unknown;
174 }
