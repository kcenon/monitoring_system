57otlp_grpc_config create_otlp_config() {
58 otlp_grpc_config config;
59
60 // Endpoint configuration
61 config.endpoint = "localhost:4317"; // Default OTLP gRPC port
62
63 // Timeout settings
64 config.timeout = 10s;
65 config.batch_timeout = 5s;
66
67 // Batch configuration for optimization
68 config.max_batch_size = 512; // Export up to 512 spans at once
69 config.max_queue_size = 2048; // Queue up to 2048 spans before dropping
70
71 // Retry configuration with exponential backoff
72 config.max_retry_attempts = 3;
73 config.initial_backoff = 100ms;
74 config.max_backoff = 10s;
75
76 // TLS configuration (optional)
77 config.use_tls = false;

83
84 // Custom headers (e.g., for authentication)
85 config.headers["x-api-key"] = "example-key";
86 config.headers["x-environment"] = "development";
87
88 // Resource attributes (describe the service)
89 config.resource_attributes["service.namespace"] = "examples";
90 config.resource_attributes["service.instance.id"] = "instance-001";
91 config.resource_attributes["deployment.environment"] = "dev";
92 config.resource_attributes["host.name"] = "example-host";

100std::vector<trace_span> create_sample_spans() {
101 std::vector<trace_span> spans;
102
103 auto now = std::chrono::system_clock::now();
104 std::string trace_id = "0123456789abcdef0123456789abcdef";
105
106 // Root span
107 trace_span root;
108 root.trace_id = trace_id;
109 root.span_id = "0123456789abcdef";
110 root.operation_name = "http_request";
111 root.service_name = "api_gateway";
112 root.start_time = now;
113 root.end_time = now + 150ms;
114 root.calculate_duration();
115 root.status = trace_span::status_code::ok;
116 root.tags["http.method"] = "GET";
117 root.tags["http.url"] = "/api/users";
118 root.tags["http.status_code"] = "200";
119 spans.push_back(root);
120
121 // Child span 1: database query
122 trace_span db_span;
123 db_span.trace_id = trace_id;
124 db_span.span_id = "fedcba9876543210";
125 db_span.parent_span_id = root.span_id;
126 db_span.operation_name = "db_query";
127 db_span.service_name = "user_service";
128 db_span.start_time = now + 10ms;
129 db_span.end_time = now + 100ms;
130 db_span.calculate_duration();
131 db_span.status = trace_span::status_code::ok;
132 db_span.tags["db.system"] = "postgresql";
133 db_span.tags["db.statement"] = "SELECT * FROM users";
134 db_span.tags["db.name"] = "user_db";
135 spans.push_back(db_span);
136
137 // Child span 2: cache lookup
138 trace_span cache_span;
139 cache_span.trace_id = trace_id;
140 cache_span.span_id = "1234567890abcdef";
141 cache_span.parent_span_id = root.span_id;
142 cache_span.operation_name = "cache_get";
143 cache_span.service_name = "cache_service";
144 cache_span.start_time = now + 5ms;
145 cache_span.end_time = now + 8ms;
146 cache_span.calculate_duration();
147 cache_span.status = trace_span::status_code::ok;
148 cache_span.tags["cache.key"] = "user:123";
149 cache_span.tags["cache.hit"] = "false";
150 spans.push_back(cache_span);
151
152 return spans;
153}

158void demonstrate_otlp_export() {
159 std::cout << "=== OTLP Export Example ===" << std::endl;
160
161 try {
162 // Step 1: Create and validate configuration

175 std::cout << " Endpoint: " << config.endpoint << std::endl;

180
181 // Step 2: Create exporter
182 std::cout << "\n2. Creating OTLP exporter..." << std::endl;
183
184 auto exporter = create_otlp_grpc_exporter(config);
185 std::cout << " ‚úì Exporter created" << std::endl;
186
187 // Step 3: Start exporter (connects to OTLP receiver)
188 std::cout << "\n3. Starting exporter..." << std::endl;

253void demonstrate_batch_export() {
254 std::cout << "\n=== Batch Export Optimization ===" << std::endl;
255
256 try {
257 // Configure for batch export
258 otlp_grpc_config config = create_otlp_config();
259 config.max_batch_size = 10; // Small batch for demonstration
260
261 auto exporter = create_otlp_grpc_exporter(config);
262 auto start_result = exporter->start();
263
264 if (start_result.is_err()) {
265 std::cerr << " Skipping batch demo (no OTLP receiver available)" << std::endl;
266 return;
267 }
268
269 std::cout << "\n1. Creating large batch of spans..." << std::endl;
270
271 // Create 25 spans (will be split into 3 batches of 10, 10, 5)
272 std::vector<trace_span> large_batch;
273 auto now = std::chrono::system_clock::now();
274
275 for (int i = 0; i < 25; ++i) {
276 trace_span span;
277 span.trace_id = "batch00000000000000000000000000" + std::to_string(i);
278 span.span_id = "span000000000000" + std::to_string(i);
279 span.operation_name = "batch_operation_" + std::to_string(i);
280 span.service_name = "batch_service";
281 span.start_time = now + std::chrono::milliseconds(i);
282 span.end_time = span.start_time + 10ms;
283 span.calculate_duration();
284 span.status = trace_span::status_code::ok;
285 large_batch.push_back(span);
286 }
287
288 std::cout << " Created " << large_batch.size() << " spans" << std::endl;
289 std::cout << " Batch size: " << config.max_batch_size << std::endl;
290
291 // Export in batch
292 std::cout << "\n2. Exporting batch..." << std::endl;
293 auto export_result = exporter->export_spans(large_batch);
294
295 if (export_result.is_ok()) {
296 std::cout << " ‚úì Batch export succeeded" << std::endl;
297 } else {
298 std::cerr << " ‚úó Batch export failed: "
299 << export_result.error().message << std::endl;
300 }
301
302 // Show final statistics
303 auto stats = exporter->get_detailed_stats();
304 std::cout << "\n3. Final statistics:" << std::endl;
305 std::cout << " Total exported: " << stats.spans_exported << std::endl;
306 std::cout << " Batches sent: " << stats.batches_sent << std::endl;
307
308 exporter->shutdown();
309
310 } catch (const std::exception& e) {
311 std::cerr << "Exception: " << e.what() << std::endl;
312 }
313}
