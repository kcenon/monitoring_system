32class ServiceA : public IService {
33private:
34 static std::atomic<int> instance_count_;
35 int id_;
36
37public:
38 ServiceA() : id_(++instance_count_) {}
39 ~ServiceA() { --instance_count_; }
40
41 std::string get_name() const override {
42 return "ServiceA_" + std::to_string(id_);
43 }
44
45 int get_id() const { return id_; }
46
47 static int get_instance_count() {
48 return instance_count_.load();
49 }
50
51 static void reset_count() {
52 instance_count_ = 0;
53 }
54};
