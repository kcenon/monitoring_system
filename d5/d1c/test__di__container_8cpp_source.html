115 template<typename TInterface>
116 result<std::shared_ptr<TInterface>> resolve() const {
117 std::lock_guard<std::recursive_mutex> lock(mutex_);
118
119 std::string key = get_type_key(typeid(TInterface));
120
121 // Check singletons first
122 auto singleton_it = singletons_.find(key);
123 if (singleton_it != singletons_.end()) {
124 return make_success(std::static_pointer_cast<TInterface>(singleton_it->second));
125 }
126
127 // Check factories
128 auto factory_it = factories_.find(key);
129 if (factory_it != factories_.end()) {
130 auto lifetime_it = lifetimes_.find(key);
131 if (lifetime_it != lifetimes_.end() &&
132 (lifetime_it->second == service_lifetime::singleton ||
133 lifetime_it->second == service_lifetime::scoped)) {
134 // Create singleton/scoped instance
135 auto instance = factory_it->second();
136 singletons_[key] = instance;
137 return make_success(std::static_pointer_cast<TInterface>(instance));
138 } else {
139 // Create transient instance
140 auto instance = factory_it->second();
141 return make_success(std::static_pointer_cast<TInterface>(instance));
142 }
143 }
144
145 // Check parent container
146 if (parent_ != nullptr) {
147 return parent_->resolve<TInterface>();
148 }
149
150 return make_error<std::shared_ptr<TInterface>>(
151 monitoring_error_code::collector_not_found, "Service not found");
152 }
