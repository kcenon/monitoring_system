124class ServiceA : public IService {
125private:
126 static std::atomic<int> instance_count_;
127 int id_;
128
129public:
130 ServiceA() : id_(++instance_count_) {}
131 ~ServiceA() { --instance_count_; }
132
133 std::string get_name() const override {
134 return "ServiceA_" + std::to_string(id_);

136
137 int get_id() const { return id_; }
138
139 static int get_instance_count() {

142

145 }
