66struct error_boundary_metrics {
67 std::atomic<size_t> total_operations{0};
68 std::atomic<size_t> successful_operations{0};
69 std::atomic<size_t> failed_operations{0};
70 std::atomic<size_t> recovered_operations{0};
71 std::atomic<size_t> recovery_attempts{0};
72
73 error_boundary_metrics() = default;
74
75 error_boundary_metrics(const error_boundary_metrics& other)
76 : total_operations(other.total_operations.load())
77 , successful_operations(other.successful_operations.load())
78 , failed_operations(other.failed_operations.load())
79 , recovered_operations(other.recovered_operations.load())
80 , recovery_attempts(other.recovery_attempts.load()) {}
81
82 error_boundary_metrics& operator=(const error_boundary_metrics& other) {
83 if (this != &other) {
84 total_operations = other.total_operations.load();
85 successful_operations = other.successful_operations.load();
86 failed_operations = other.failed_operations.load();
87 recovered_operations = other.recovered_operations.load();
88 recovery_attempts = other.recovery_attempts.load();
89 }
90 return *this;
91 }
92
93 double get_success_rate() const {
94 size_t total = total_operations.load();
95 if (total == 0) {
96 return 1.0;
97 }
98 return static_cast<double>(successful_operations.load()) / static_cast<double>(total);
99 }
100};

155template<typename T>
156class cached_value_strategy : public fallback_strategy_interface<T> {
157public:
158 explicit cached_value_strategy(std::chrono::seconds ttl = std::chrono::seconds(60))
159 : ttl_(ttl), has_value_(false) {}
160
161 void update_cache(const T& value) {
162 std::lock_guard<std::mutex> lock(mutex_);
163 cached_value_ = value;
164 cache_time_ = std::chrono::steady_clock::now();
165 has_value_ = true;
166 }
167
168 result<T> get_fallback(const error_info& /*err*/, degradation_level /*level*/) override {
169 std::lock_guard<std::mutex> lock(mutex_);
170 if (!has_value_) {
171 return make_error<T>(monitoring_error_code::operation_failed, "No cached value available");
172 }
173
174 auto now = std::chrono::steady_clock::now();
175 auto age = std::chrono::duration_cast<std::chrono::seconds>(now - cache_time_);
176 if (age > ttl_) {
177 return make_error<T>(monitoring_error_code::operation_failed, "Cached value expired");
178 }
179
180 return make_success(cached_value_);
181 }
182
183private:
184 std::chrono::seconds ttl_;
185 T cached_value_;
186 std::chrono::steady_clock::time_point cache_time_;
187 bool has_value_;
188 mutable std::mutex mutex_;
189};

234 template<typename Func>
235 auto execute(Func&& func) -> result<T> {
236 metrics_.total_operations++;
237
238 try {
239 auto op_result = func();
240
241 if (op_result.is_ok()) {
242 metrics_.successful_operations++;
243 handle_success();
244 return op_result;
245 }
246
247 metrics_.failed_operations++;
248 return handle_failure(op_result.error());
249
250 } catch (const std::exception& e) {
251 metrics_.failed_operations++;
252 error_info err(monitoring_error_code::operation_failed, e.what());
253 return handle_failure(err.to_common_error());
254 } catch (...) {
255 metrics_.failed_operations++;
256 error_info err(monitoring_error_code::operation_failed, "Unknown exception");
257 return handle_failure(err.to_common_error());
258 }
259 }
260
264 template<typename Func, typename FallbackFunc>
265 auto execute(Func&& func, FallbackFunc&& fallback) -> result<T> {
266 metrics_.total_operations++;
267
268 try {
269 auto op_result = func();
270
271 if (op_result.is_ok()) {
272 metrics_.successful_operations++;
273 handle_success();
274 return op_result;
275 }
276
277 metrics_.failed_operations++;
278 error_info err = error_info::from_common_error(op_result.error());
279 return fallback(err, current_degradation_level_);
280
281 } catch (const std::exception& e) {
282 metrics_.failed_operations++;
283 error_info err(monitoring_error_code::operation_failed, e.what());
284 return fallback(err, current_degradation_level_);
285 }
286 }

340private:
341 result<T> handle_failure(const common::error_info& err) {
342 consecutive_failures_++;
343
344 // Check if we should degrade
345 if (config_.policy == error_boundary_policy::degrade &&
346 consecutive_failures_ >= config_.error_threshold) {
347 upgrade_degradation();
348 }
349
350 // Call error handler if set
351 if (error_handler_) {
352 error_info monitoring_err = error_info::from_common_error(err);
353 error_handler_(monitoring_err, current_degradation_level_);
354 }
355
356 // Apply policy
357 switch (config_.policy) {
358 case error_boundary_policy::fail_fast:
359 return result<T>::err(err);
360
361 case error_boundary_policy::isolate:
362 return make_error<T>(monitoring_error_code::service_degraded,
363 "Service isolated due to error");
364
365 case error_boundary_policy::fallback:
366 if (fallback_strategy_) {
367 error_info monitoring_err = error_info::from_common_error(err);
368 return fallback_strategy_->get_fallback(monitoring_err, current_degradation_level_);
369 }
370 return result<T>::err(err);
371
372 case error_boundary_policy::degrade:
373 default:
374 return result<T>::err(err);
375 }
376 }
377
378 void handle_success() {
379 consecutive_failures_ = 0;
380
381 // Check for recovery
382 if (config_.enable_automatic_recovery &&
383 current_degradation_level_ != degradation_level::normal) {
384 metrics_.recovery_attempts++;
385 downgrade_degradation();
386 }
387 }
388
389 void upgrade_degradation() {
390 auto current = static_cast<int>(current_degradation_level_);
391 auto max_level = static_cast<int>(config_.max_degradation);
392 if (current < max_level) {
393 current_degradation_level_ = static_cast<degradation_level>(current + 1);
394 }
395 }
396
397 void downgrade_degradation() {
398 auto current = static_cast<int>(current_degradation_level_);
399 if (current > 0) {
400 current_degradation_level_ = static_cast<degradation_level>(current - 1);
401 }
402 }
403
404 std::string name_;
405 config config_;
406 std::function<void(const error_info&, degradation_level)> error_handler_;
407 std::shared_ptr<fallback_strategy_interface<T>> fallback_strategy_;
408 mutable error_boundary_metrics metrics_;
409 degradation_level current_degradation_level_ = degradation_level::normal;
410 size_t consecutive_failures_ = 0;
411};
