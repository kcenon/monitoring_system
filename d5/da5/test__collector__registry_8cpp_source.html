38// Mock collector plugin for testing
39class mock_collector_plugin : public collector_plugin {
40public:
41 explicit mock_collector_plugin(std::string_view name,
42 plugin_category category = plugin_category::custom,
43 bool available = true)
44 : name_(name), category_(category), available_(available) {}
45
46 auto name() const -> std::string_view override { return name_; }
47
48 auto collect() -> std::vector<metric> override {
49 ++collect_count_;
50 return {};
51 }
52
53 auto interval() const -> std::chrono::milliseconds override {
54 return std::chrono::seconds(1);
55 }
56
57 auto is_available() const -> bool override { return available_; }
58
59 auto get_metadata() const -> plugin_metadata override {
60 return plugin_metadata{
61 .name = name_,
62 .description = "Mock plugin for testing",
63 .category = category_,
64 .version = "1.0.0",
65 .dependencies = {},
66 .requires_platform_support = false
67 };
68 }
69
70 auto initialize(const config_map& /* config */) -> bool override {
71 initialized_ = true;
72 return true;
73 }
74
75 void shutdown() override { shutdown_called_ = true; }
76
77 auto get_metric_types() const -> std::vector<std::string> override {
78 return {"test_metric"};
79 }
80
81 // Test helpers
82 bool is_initialized() const { return initialized_; }
83 bool is_shutdown_called() const { return shutdown_called_; }
84 size_t get_collect_count() const { return collect_count_; }
85
86private:
87 std::string name_;
88 plugin_category category_;
89 bool available_;
90 bool initialized_{false};
91 bool shutdown_called_{false};
92 mutable size_t collect_count_{0};
93};
