382 explicit timer_data(size_t reservoir_size)
383 : max_samples(reservoir_size) {
384 samples.reserve(max_samples);

390 void record(double duration_ms) {
391 total_count++;
392 sum += duration_ms;
393 min_value = (std::min)(min_value, duration_ms);
394 max_value = (std::max)(max_value, duration_ms);
395 sorted = false;
396
397 if (samples.size() < max_samples) {
398 samples.push_back(duration_ms);
399 } else {
400 // Reservoir sampling for memory efficiency
401 size_t idx = static_cast<size_t>(rand()) % total_count;
402 if (idx < max_samples) {
403 samples[idx] = duration_ms;
404 }
405 }
406 }

411 template<typename Rep, typename Period>
412 void record(std::chrono::duration<Rep, Period> duration) {
413 auto ms = std::chrono::duration_cast<std::chrono::microseconds>(duration).count() / 1000.0;
414 record(ms);
415 }

422 double get_percentile(double percentile) const {
423 if (samples.empty()) return 0.0;
424 if (percentile <= 0) return min_value;
425 if (percentile >= 100) return max_value;
426
427 ensure_sorted();
428
429 double rank = (percentile / 100.0) * (samples.size() - 1);
430 size_t lower_idx = static_cast<size_t>(rank);
431 size_t upper_idx = lower_idx + 1;
432 double fraction = rank - lower_idx;
433
434 if (upper_idx >= samples.size()) {
435 return samples[lower_idx];
436 }
437
438 // Linear interpolation between adjacent values
439 return samples[lower_idx] + fraction * (samples[upper_idx] - samples[lower_idx]);
440 }

508 double stddev() const {
509 if (samples.size() < 2) return 0.0;
510
511 double avg = mean();
512 double variance = 0.0;
513 for (double sample : samples) {
514 double diff = sample - avg;
515 variance += diff * diff;
516 }
517 variance /= samples.size();
518 return std::sqrt(variance);
519 }

605inline uint32_t hash_metric_name(const std::string& name) noexcept {
606 // Simple FNV-1a hash for fast metric name hashing
607 uint32_t hash = 2166136261U;
608 for (char c : name) {
609 hash ^= static_cast<uint32_t>(c);
610 hash *= 16777619U;
611 }
612 return hash;
