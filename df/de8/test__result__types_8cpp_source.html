151TEST_F(ResultTypesTest, MonitoringConfigValidation) {
152 monitoring_config config;
153
154 // Valid configuration
155 config.history_size = 100;
156 config.collection_interval = std::chrono::milliseconds(100);
157 config.buffer_size = 1000;
158 auto result = config.validate();
159 EXPECT_TRUE(result.is_ok());
160
161 // Invalid history size
162 config.history_size = 0;
163 result = config.validate();
164 EXPECT_FALSE(result.is_ok());
165 EXPECT_EQ(static_cast<monitoring_error_code>(result.error().code), monitoring_error_code::invalid_capacity);
166
167 // Invalid interval
168 config.history_size = 100;
169 config.collection_interval = std::chrono::milliseconds(5);
170 result = config.validate();
171 EXPECT_FALSE(result.is_ok());
172 EXPECT_EQ(static_cast<monitoring_error_code>(result.error().code), monitoring_error_code::invalid_interval);
173
174 // Invalid buffer size
175 config.collection_interval = std::chrono::milliseconds(100);
176 config.buffer_size = 50; // Less than history_size
177 result = config.validate();
178 EXPECT_FALSE(result.is_ok());
179 EXPECT_EQ(static_cast<monitoring_error_code>(result.error().code), monitoring_error_code::invalid_capacity);
180}
