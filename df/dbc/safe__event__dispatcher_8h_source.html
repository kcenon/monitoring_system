75 template<typename EventType, typename Handler>
76 uint64_t subscribe_safe(Handler&& handler, event_priority priority = event_priority::normal) {
77 // Wrap handler with exception handling
78 auto safe_handler = [this, h = std::forward<Handler>(handler), id = next_id_++]
79 (const EventType& event) {
80 try {
81 h(event);
82 } catch (const std::exception& e) {
83 handle_exception(id, typeid(EventType), e.what());
84 } catch (...) {
85 handle_exception(id, typeid(EventType), "Unknown exception");
86 }
87 };
88
89 return bus_->subscribe<EventType>(std::move(safe_handler));
90 }

98 template<typename EventType>
99 bool publish_safe(const EventType& event, event_priority priority = event_priority::normal) {
100 try {
101 bus_->publish(event, priority);
102 return true;
103 } catch (const std::exception& e) {
104 // Log error
105 if (error_callback_) {
106 handler_error_info info{
107 0, // No specific handler
108 std::string("Publish failed: ") + e.what(),
109 typeid(EventType),
110 std::chrono::steady_clock::now()
111 };
112 error_callback_(info);
113 }
114
115 // Add to dead letter queue
116 add_to_dead_letter_queue(typeid(EventType), e.what());
117
118 return false;
119 } catch (...) {
120 if (error_callback_) {
121 handler_error_info info{
122 0,
123 "Publish failed: Unknown exception",
124 typeid(EventType),
125 std::chrono::steady_clock::now()
126 };
127 error_callback_(info);
128 }
129
130 add_to_dead_letter_queue(typeid(EventType), "Unknown exception");
131 return false;
132 }
133 }

216 void handle_exception(uint64_t handler_id,
217 std::type_index event_type,
218 const std::string& error_msg) {
219 total_exceptions_.fetch_add(1, std::memory_order_relaxed);
220
221 std::lock_guard<std::mutex> lock(error_mutex_);
222
223 // Record error
224 handler_error_info info{
225 handler_id,
226 error_msg,
227 event_type,
228 std::chrono::steady_clock::now()
229 };
230
231 recent_errors_.push_back(info);
232
233 // Keep only last 100 errors
234 if (recent_errors_.size() > 100) {
235 recent_errors_.erase(recent_errors_.begin());
236 }
237
238 // Track failure count
239 auto& count = failed_handler_counts_[handler_id];
240 count++;
241
242 // Circuit breaker check
243 if (count >= circuit_breaker_threshold_) {
244 // Handler has failed too many times
245 // In production, we would disable this handler
246 if (error_callback_) {
247 error_callback_(info);
248 }
249 }
250
251 // Notify error callback
252 if (error_callback_) {
253 error_callback_(info);
254 }
255 }

257 void add_to_dead_letter_queue(std::type_index event_type,
258 const std::string& error_msg) {
259 std::lock_guard<std::mutex> lock(dlq_mutex_);
260
261 dead_letter_entry entry{
262 event_type,
263 error_msg,
264 std::chrono::steady_clock::now()
265 };
266
267 dead_letter_queue_.push_back(entry);
268
269 // Limit dead letter queue size
270 if (dead_letter_queue_.size() > 1000) {
271 dead_letter_queue_.erase(dead_letter_queue_.begin());
272 }
273 }
