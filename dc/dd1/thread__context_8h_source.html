43struct context_metadata {
44 std::string request_id;
45 std::string correlation_id;
46 std::string user_id;
47 std::unordered_map<std::string, std::string> tags;
48
49 explicit context_metadata(std::string req_id = "")
50 : request_id(std::move(req_id)) {}
51
52 bool empty() const {
53 return request_id.empty() && correlation_id.empty() && user_id.empty() && tags.empty();
54 }
55
56 void set_tag(const std::string& key, const std::string& value) {
57 tags[key] = value;
58 }
59
60 std::string get_tag(const std::string& key) const {
61 auto it = tags.find(key);
62 return it != tags.end() ? it->second : "";
63 }
64};
