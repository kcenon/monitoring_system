161
167 static context_metadata& create(const std::string& request_id = "") {
168 auto metadata = std::make_unique<context_metadata>(request_id);
169 if (request_id.empty()) {
170 // Generate a unique request ID if not provided
171 metadata->request_id = generate_request_id();
172 }
173 current_context_ = std::move(metadata);
174 return *current_context_;
175 }
176
180 static void clear() {
181 current_context_.reset();
182 }
183
188 static bool has_context() {
189 return current_context_ != nullptr;
190 }
191
196 static std::string generate_request_id() {
197 auto tid = std::this_thread::get_id();
198 auto counter = context_counter_.fetch_add(1);
199 auto timestamp = std::chrono::system_clock::now().time_since_epoch().count();
200
201 // Create a unique ID from thread ID, counter, and timestamp
202 std::hash<std::thread::id> hasher;
203 auto tid_hash = hasher(tid);
204
205 char buffer[64];
206 snprintf(buffer, sizeof(buffer), "%zx-%llu-%llx",
207 tid_hash, counter, static_cast<unsigned long long>(timestamp));
208 return std::string(buffer);
209 }
210
215 static std::string generate_correlation_id() {
216 return "corr-" + generate_request_id();
217 }
218
224 static result_void copy_from(const context_metadata& source) {
225 try {
226 current_context_ = std::make_unique<context_metadata>(source);
227 return result_void::success();
228 } catch (const std::exception& e) {
229 return result_void(monitoring_error_code::out_of_memory,
230 std::string("Failed to copy context: ") + e.what());
231 }
232 }

250public:

257 bool preserve_previous = false)

259
260 if (should_restore_ && thread_context::has_context()) {
261 // Save current context
262 previous_context_ = std::make_unique<context_metadata>(*thread_context::current());
263 }
264
265 thread_context::set_current(std::move(metadata));

267
272 explicit context_scope(const std::string& request_id)
273 : should_restore_(false) {
274 thread_context::create(request_id);
275 }
276
280 ~context_scope() {
281 if (should_restore_ && previous_context_) {
282 thread_context::set_current(std::move(previous_context_));
283 } else {
284 thread_context::clear();
285 }
286 }

381class context_aware_monitoring {
382public:
383 virtual ~context_aware_monitoring() = default;
384
391 virtual result_void enrich_with_context(
392 monitoring_data& data,
393 const context_metadata* context = nullptr) const {
394
395 const context_metadata* ctx = context;
396 if (!ctx) {
397 ctx = thread_context::current();
398 }
399
400 if (!ctx) {
401 // No context available, not an error
402 return result_void::success();
403 }
404
405 // Add context metadata as tags
406 if (!ctx->request_id.empty()) {
407 data.add_tag("request_id", ctx->request_id);
408 }
409 if (!ctx->correlation_id.empty()) {
410 data.add_tag("correlation_id", ctx->correlation_id);
411 }
412 if (!ctx->user_id.empty()) {
413 data.add_tag("user_id", ctx->user_id);
414 }
415 if (!ctx->session_id.empty()) {
416 data.add_tag("session_id", ctx->session_id);
417 }
418 if (!ctx->trace_id.empty()) {
419 data.add_tag("trace_id", ctx->trace_id);
420 }
421 if (!ctx->span_id.empty()) {
422 data.add_tag("span_id", ctx->span_id);
423 }
424
425 // Add custom tags with prefix
426 for (const auto& [key, value] : ctx->custom_tags) {
427 data.add_tag("ctx." + key, value);
428 }
429
430 return result_void::success();
431 }
432
437 virtual bool is_context_aware_enabled() const {
438 return true;
439 }
