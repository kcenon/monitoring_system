93 // Convert to compact representation
94 compact_metric_value to_compact() const {
95 // Simple hash function for name
96 uint32_t hash = 0;
97 for (char c : name) {
98 hash = hash * 31 + static_cast<uint32_t>(c);
99 }
100
101 metric_metadata meta(hash, type, static_cast<uint8_t>(tags.size()));
102
103 if (std::holds_alternative<double>(value)) {
104 return compact_metric_value(meta, std::get<double>(value));
105 } else if (std::holds_alternative<int64_t>(value)) {
106 return compact_metric_value(meta, std::get<int64_t>(value));
107 } else {
108 return compact_metric_value(meta, std::get<std::string>(value));
109 }
110 }
111};
