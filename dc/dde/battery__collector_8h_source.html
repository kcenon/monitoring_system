73inline std::string battery_status_to_string(battery_status status) {
74 switch (status) {
75 case battery_status::charging:
76 return "charging";
77 case battery_status::discharging:
78 return "discharging";
79 case battery_status::not_charging:
80 return "not_charging";
81 case battery_status::full:
82 return "full";
83 default:
84 return "unknown";
85 }
86}
