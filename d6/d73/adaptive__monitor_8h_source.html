181class adaptive_collector {
182private:
183 std::shared_ptr<kcenon::monitoring::metrics_collector> collector_;
184 adaptive_config config_;
185 mutable std::mutex config_mutex_; // Protects config_
186 adaptation_stats stats_;
187 std::atomic<bool> enabled_{true};
188 std::atomic<double> current_sampling_rate_{1.0};
189 mutable std::mutex stats_mutex_;
190
191public:
192 adaptive_collector(
193 std::shared_ptr<kcenon::monitoring::metrics_collector> collector,
194 const adaptive_config& config = {}
195 ) : collector_(collector), config_(config) {
196 stats_.current_interval = config_.moderate_interval;
197 stats_.last_adaptation = std::chrono::system_clock::now();

203 common::Result<kcenon::monitoring::metrics_snapshot> collect() {
204 if (!should_sample()) {
205 stats_.samples_dropped++;
206 return common::Result<kcenon::monitoring::metrics_snapshot>::err(error_info(kcenon::monitoring::monitoring_error_code::operation_cancelled, "Sample dropped due to adaptive sampling").to_common_error());
207 }
208
209 stats_.samples_collected++;
210 return collector_->collect();
211 }

217 void adapt(const kcenon::monitoring::system_metrics& sys_metrics) {
218 // Copy config under lock to avoid holding both locks
219 adaptive_config local_config;
220 {
221 std::lock_guard<std::mutex> config_lock(config_mutex_);
222 local_config = config_;
223 }
224
225 std::lock_guard<std::mutex> lock(stats_mutex_);
226
227 bool is_first_adaptation = (stats_.total_adaptations == 0);
228
229 // Initialize averages on first adaptation
230 if (is_first_adaptation) {
231 stats_.average_cpu_usage = sys_metrics.cpu_usage_percent;
232 stats_.average_memory_usage = sys_metrics.memory_usage_percent;
233 } else {
234 // Update average metrics using exponential smoothing
235 stats_.average_cpu_usage =
236 local_config.smoothing_factor * sys_metrics.cpu_usage_percent +
237 (1.0 - local_config.smoothing_factor) * stats_.average_cpu_usage;
238
239 stats_.average_memory_usage =
240 local_config.smoothing_factor * sys_metrics.memory_usage_percent +
241 (1.0 - local_config.smoothing_factor) * stats_.average_memory_usage;
242 }
243
244 // Determine load level with hysteresis support
245 auto new_level = calculate_load_level_with_hysteresis(
246 stats_.average_cpu_usage,
247 stats_.average_memory_usage,
248 stats_.current_load_level,
249 local_config
250 );
251
252 // Adapt if load level changed
253 if (new_level != stats_.current_load_level) {
254 auto now = std::chrono::system_clock::now();
255
256 // Check cooldown period (ARC-005)
257 // Skip cooldown check for first adaptation to allow initial level setting
258 if (local_config.enable_cooldown && !is_first_adaptation) {
259 auto time_since_last_change = std::chrono::duration_cast<std::chrono::milliseconds>(
260 now - stats_.last_level_change
261 );
262 if (time_since_last_change < local_config.cooldown_period) {
263 stats_.cooldown_prevented_changes++;
264 return; // Skip this adaptation due to cooldown
265 }
266 }
267
268 if (new_level > stats_.current_load_level) {
269 stats_.downscale_count++;
270 } else {
271 stats_.upscale_count++;
272 }
273
274 stats_.current_load_level = new_level;
275 stats_.current_interval = local_config.get_interval_for_load(new_level);
276 current_sampling_rate_ = local_config.get_sampling_rate_for_load(new_level);
277 stats_.current_sampling_rate = current_sampling_rate_;
278 stats_.total_adaptations++;
279 stats_.last_adaptation = now;
280 stats_.last_level_change = now;
281 }
282 }

336 bool should_sample() const {
337 if (!enabled_) return true;
338
339 // Use random sampling based on current rate
340 static thread_local std::mt19937 gen(std::random_device{}());
341 std::uniform_real_distribution<> dis(0.0, 1.0);
342 return dis(gen) < current_sampling_rate_.load();
343 }

349 static load_level calculate_load_level_with_config(
350 double cpu_usage,
351 double memory_usage,
352 const adaptive_config& cfg
353 ) {
354 // Consider memory pressure in load calculation
355 double effective_load = cpu_usage;
356
357 // Memory pressure should escalate load level
358 if (memory_usage > cfg.memory_critical_threshold) {
359 // Critical memory -> at least high load
360 effective_load = std::max(effective_load, cfg.high_threshold + 1.0);
361 } else if (memory_usage > cfg.memory_warning_threshold) {
362 // Warning memory -> at least moderate load
363 effective_load = std::max(effective_load, cfg.moderate_threshold + 1.0);
364 }
365
366 // Apply strategy-specific adjustments BEFORE determining level
367 switch (cfg.strategy) {
368 case adaptation_strategy::conservative:
369 effective_load *= 0.8; // Be more conservative
370 break;
371 case adaptation_strategy::aggressive:
372 effective_load *= 1.2; // Be more aggressive
373 break;
374 case adaptation_strategy::balanced:
375 default:
376 break;
377 }
378
379 // Determine load level
380 if (effective_load >= cfg.high_threshold) {
381 return load_level::critical;
382 } else if (effective_load >= cfg.moderate_threshold) {
383 return load_level::high;
384 } else if (effective_load >= cfg.low_threshold) {
385 return load_level::moderate;
386 } else if (effective_load >= cfg.idle_threshold) {
387 return load_level::low;
388 } else {
389 return load_level::idle;
390 }
391 }

405 static load_level calculate_load_level_with_hysteresis(
406 double cpu_usage,
407 double memory_usage,
408 load_level current_level,
409 const adaptive_config& cfg
410 ) {
411 // First calculate what the level would be without hysteresis
412 load_level raw_level = calculate_load_level_with_config(cpu_usage, memory_usage, cfg);
413
414 // If hysteresis is disabled, return raw level
415 if (!cfg.enable_hysteresis) {
416 return raw_level;
417 }
418
419 // If no change, return current level
420 if (raw_level == current_level) {
421 return current_level;
422 }
423
424 // Calculate effective load (same logic as calculate_load_level_with_config)
425 double effective_load = cpu_usage;
426 if (memory_usage > cfg.memory_critical_threshold) {
427 effective_load = std::max(effective_load, cfg.high_threshold + 1.0);
428 } else if (memory_usage > cfg.memory_warning_threshold) {
429 effective_load = std::max(effective_load, cfg.moderate_threshold + 1.0);
430 }
431
432 switch (cfg.strategy) {
433 case adaptation_strategy::conservative:
434 effective_load *= 0.8;
435 break;
436 case adaptation_strategy::aggressive:
437 effective_load *= 1.2;
438 break;
439 case adaptation_strategy::balanced:
440 default:
441 break;
442 }
443
444 // Get threshold for current level
445 double current_threshold = get_threshold_for_level(current_level, cfg);
446 double margin = cfg.hysteresis_margin;
447
448 // For upward transitions (higher load), require crossing threshold + margin
449 // For downward transitions (lower load), require crossing threshold - margin
450 if (raw_level > current_level) {
451 // Moving to higher load level - need to exceed threshold by margin
452 double next_threshold = get_threshold_for_level(
453 static_cast<load_level>(static_cast<int>(current_level) + 1), cfg);
454 if (effective_load < next_threshold + margin) {
455 return current_level; // Stay at current level (hysteresis)
456 }
457 } else {
458 // Moving to lower load level - need to drop below threshold by margin
459 if (effective_load > current_threshold - margin) {
460 return current_level; // Stay at current level (hysteresis)
461 }
462 }
463
464 return raw_level;
465 }
466
470 static double get_threshold_for_level(load_level level, const adaptive_config& cfg) {
471 switch (level) {
472 case load_level::idle: return 0.0;
473 case load_level::low: return cfg.idle_threshold;
474 case load_level::moderate: return cfg.low_threshold;
475 case load_level::high: return cfg.moderate_threshold;
476 case load_level::critical: return cfg.high_threshold;
477 }
478 return cfg.moderate_threshold;
479 }

575class adaptive_scope {

595 }
596
597 // Disable copy
598 adaptive_scope(const adaptive_scope&) = delete;
599 adaptive_scope& operator=(const adaptive_scope&) = delete;
600
601 // Enable move
602 adaptive_scope(adaptive_scope&& other) noexcept
603 : monitor_(other.monitor_)
604 , collector_name_(std::move(other.collector_name_))
605 , registered_(other.registered_) {
606 other.monitor_ = nullptr;
607 other.registered_ = false;
608 }
609
610 adaptive_scope& operator=(adaptive_scope&& other) noexcept {
611 if (this != &other) {
612 if (registered_ && monitor_) {
613 monitor_->unregister_collector(collector_name_);
614 }
615 monitor_ = other.monitor_;
616 collector_name_ = std::move(other.collector_name_);
617 registered_ = other.registered_;
618 other.monitor_ = nullptr;
619 other.registered_ = false;
620 }
621 return *this;
622 }
623
624 bool is_registered() const { return registered_; }
625};
