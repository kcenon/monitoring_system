61struct alert_rule_config {

72 bool validate() const {

281 alert create_alert(double value) const {
282 alert a(name_, labels_);
283 a.annotations = annotations_;
284 a.severity = severity_;
285 a.value = value;

295 common::VoidResult validate() const {
296 if (name_.empty()) {
297 error_info err(monitoring_error_code::invalid_argument,
298 "Rule name cannot be empty");
299 return common::VoidResult::err(err.to_common_error());
300 }
301 if (!config_.validate()) {
302 error_info err(monitoring_error_code::invalid_configuration,
303 "Rule configuration is invalid");
304 return common::VoidResult::err(err.to_common_error());
305 }
306 if (!trigger_) {
307 error_info err(monitoring_error_code::invalid_argument,
308 "Rule must have a trigger");
309 return common::VoidResult::err(err.to_common_error());
310 }
311 return common::ok();
312 }
