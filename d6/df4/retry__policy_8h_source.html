153class retry_executor {
154public:
155 using config = retry_config;
156
157 retry_executor() : name_("default"), config_() {}
158
159 explicit retry_executor(const std::string& name)
160 : name_(name), config_() {}

162 explicit retry_executor(const std::string& name, const config& cfg)
163 : name_(name), config_(cfg) {}

172 template<typename Func>
173 common::Result<T> execute(Func&& func) {
174 metrics_.total_executions++;
175
176 common::Result<T> last_result = common::Result<T>::err(error_info(monitoring_error_code::operation_failed, "No attempts made").to_common_error());
177
178 for (size_t attempt = 0; attempt < config_.max_attempts; ++attempt) {
179 if (attempt > 0) {
180 metrics_.total_retries++;
181 auto delay = calculate_delay(attempt);
182 std::this_thread::sleep_for(delay);
183 }
184
185 last_result = func();
186
187 if (last_result.is_ok()) {
188 metrics_.successful_executions++;
189 return last_result;
190 }
191
192 if (config_.should_retry) {
193 auto err_info = error_info::from_common_error(last_result.error());
194 if (!config_.should_retry(err_info)) {
195 metrics_.failed_executions++;
196 return last_result;
197 }
198 }
199 }
200
201 metrics_.failed_executions++;
202 return last_result;
203 }
204
208 retry_metrics get_metrics() const {
209 return metrics_;
210 }
211
215 void reset_metrics() {
216 metrics_.reset();
217 }
218
222 const std::string& get_name() const {
223 return name_;
224 }
225
226private:
232 std::chrono::milliseconds calculate_delay(size_t attempt) const {
233 std::chrono::milliseconds delay;
234
235 switch (config_.strategy) {
236 case retry_strategy::fixed_delay:
237 delay = config_.initial_delay;
238 break;
239
240 case retry_strategy::exponential_backoff: {
241 auto multiplier = std::pow(config_.backoff_multiplier, static_cast<double>(attempt - 1));
242 auto delay_ms = static_cast<long long>(config_.initial_delay.count() * multiplier);
243 delay = std::chrono::milliseconds(delay_ms);
244 break;
245 }
246
247 case retry_strategy::linear_backoff:
248 delay = config_.initial_delay * attempt;
249 break;
250
251 case retry_strategy::fibonacci_backoff: {
252 delay = config_.initial_delay * fibonacci(attempt);
253 break;
254 }
255
256 default:
257 delay = config_.initial_delay;
258 break;
259 }
260
261 return std::min(delay, config_.max_delay);
262 }
263
269 static size_t fibonacci(size_t n) {
270 if (n <= 1) return 1;
271
272 size_t prev = 1;
273 size_t curr = 1;
274
275 for (size_t i = 2; i <= n; ++i) {
276 size_t next = prev + curr;
277 prev = curr;
278 curr = next;
279 }
280
281 return curr;
282 }

286 mutable retry_metrics metrics_;
