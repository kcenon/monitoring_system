498class callback_notifier : public alert_notifier {
499public:
500 using callback_func = std::function<void(const alert&)>;
501 using group_callback_func = std::function<void(const alert_group&)>;
502
509 callback_notifier(std::string notifier_name,
510 callback_func callback,
511 group_callback_func group_callback = nullptr)
512 : name_(std::move(notifier_name))
513 , callback_(std::move(callback))
514 , group_callback_(std::move(group_callback)) {}
515
516 std::string name() const override { return name_; }
517
518 common::VoidResult notify(const alert& a) override {
519 if (callback_) {
520 callback_(a);
521 return common::ok();
522 }
523 return common::VoidResult::err(error_info(monitoring_error_code::operation_failed, "No callback configured").to_common_error());
524 }
525
526 common::VoidResult notify_group(const alert_group& group) override {
527 if (group_callback_) {
528 group_callback_(group);
529 return common::ok();
530 }
531 // Fall back to individual notifications
532 for (const auto& a : group.alerts) {
533 auto result = notify(a);
534 if (!result.is_ok()) {
535 return result;
536 }
537 }
538 return common::ok();
539 }
540
541 bool is_ready() const override { return callback_ != nullptr; }
542
543private:
544 std::string name_;
545 callback_func callback_;
546 group_callback_func group_callback_;
547};
