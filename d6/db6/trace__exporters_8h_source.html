63struct trace_export_config {
64 std::string endpoint;
65 trace_export_format format = trace_export_format::otlp_grpc;
66 std::chrono::milliseconds timeout{30000};
67 std::chrono::milliseconds batch_timeout{5000};
68 std::size_t max_batch_size = 512;
69 std::size_t max_queue_size = 2048;
70 bool enable_compression = true;
71 std::unordered_map<std::string, std::string> headers;
72 std::optional<std::string> service_name;
73
77 common::VoidResult validate() const {
78 if (endpoint.empty()) {
79 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration,
80 "Export endpoint cannot be empty", "monitoring_system").to_common_error());
81 }
82
83 if (timeout.count() <= 0) {
84 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration,
85 "Timeout must be positive", "monitoring_system").to_common_error());
86 }
87
88 if (max_batch_size == 0) {
89 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration,
90 "Batch size must be greater than 0", "monitoring_system").to_common_error());
91 }
92
93 if (max_queue_size < max_batch_size) {
94 return common::VoidResult::err(error_info(monitoring_error_code::invalid_configuration,
95 "Queue size must be at least batch size", "monitoring_system").to_common_error());
96 }
97
98 return common::ok();
99 }

290 jaeger_span_data convert_span(const trace_span& span) const {

370 common::VoidResult send_thrift_batch(const std::vector<jaeger_span_data>& spans) {
371 // Build JSON payload for Thrift over HTTP
372 std::ostringstream payload;
373 payload << "{\"data\":[{\"spans\":[";
374 bool first = true;
375 for (const auto& span : spans) {
376 if (!first) payload << ",";
377 payload << span.to_thrift_json();
378 first = false;
379 }
380 payload << "]}]}";
381
382 std::string body = payload.str();
383
384 // Build HTTP request
385 http_request request;
386 request.url = config_.endpoint + "/api/traces";
387 request.method = "POST";
388 request.headers["Content-Type"] = "application/x-thrift";
389 request.headers["Accept"] = "application/json";
390 for (const auto& [key, value] : config_.headers) {
391 request.headers[key] = value;
392 }
393 request.body = std::vector<uint8_t>(body.begin(), body.end());
394 request.timeout = config_.timeout;
395
396 // Send with retry
397 return send_with_retry(request);
398 }

400 common::VoidResult send_grpc_batch(const std::vector<jaeger_span_data>& spans) {
401 // gRPC would require a different transport mechanism
402 // For now, fall back to HTTP POST with protobuf
403 std::vector<uint8_t> payload;
404 for (const auto& span : spans) {
405 auto proto = span.to_protobuf();
406 payload.insert(payload.end(), proto.begin(), proto.end());
407 }
408
409 http_request request;
410 request.url = config_.endpoint;
411 request.method = "POST";
412 request.headers["Content-Type"] = "application/grpc+proto";
413 for (const auto& [key, value] : config_.headers) {
414 request.headers[key] = value;
415 }
416 request.body = payload;
417 request.timeout = config_.timeout;
418
419 return send_with_retry(request);
420 }

422 common::VoidResult send_with_retry(const http_request& request) {
423 std::size_t attempt = 0;
424 std::chrono::milliseconds delay = base_retry_delay_;
425
426 while (attempt < max_retries_) {
427 auto result = transport_->send(request);
428 if (result.is_ok()) {
429 const auto& response = result.value();
430 if (response.status_code >= 200 && response.status_code < 300) {
431 return common::ok();
432 }
433 // Retry on 5xx errors
434 if (response.status_code >= 500) {
435 attempt++;
436 if (attempt < max_retries_) {
437 std::this_thread::sleep_for(delay);
438 delay *= 2; // Exponential backoff
439 }
440 continue;
441 }
442 // Non-retryable error
443 return common::VoidResult::err(error_info(monitoring_error_code::operation_failed,
444 "Jaeger export failed with status: " + std::to_string(response.status_code),
445 "monitoring_system").to_common_error());
446 }
447 attempt++;
448 if (attempt < max_retries_) {
449 std::this_thread::sleep_for(delay);
450 delay *= 2;
451 }
452 }
453 return common::VoidResult::err(error_info(monitoring_error_code::operation_failed,
454 "Jaeger export failed after " + std::to_string(max_retries_) + " retries",
455 "monitoring_system").to_common_error());
456 }
