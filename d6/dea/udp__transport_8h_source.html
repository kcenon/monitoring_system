241class common_udp_transport : public udp_transport {
242private:
243 std::shared_ptr<::kcenon::common::interfaces::IUdpClient> client_;
244 mutable std::atomic<std::size_t> packets_sent_{0};
245 mutable std::atomic<std::size_t> bytes_sent_{0};
246 mutable std::atomic<std::size_t> send_failures_{0};
247
248public:
249 explicit common_udp_transport(std::shared_ptr<::kcenon::common::interfaces::IUdpClient> client)
250 : client_(std::move(client)) {}
251
252 result_void connect(const std::string& host, uint16_t port) override {
253 if (!client_) {
254 return result_void::err(error_info(
255 monitoring_error_code::dependency_missing,
256 "UDP client not available",
257 "common_udp_transport"
258 ).to_common_error());
259 }
260
261 auto result = client_->connect(host, port);
262 if (result.is_err()) {
263 return result_void::err(error_info(
264 monitoring_error_code::network_error,
265 "Connection failed: " + result.error().message,
266 "common_udp_transport"
267 ).to_common_error());
268 }
269 return common::ok();
270 }
271
272 result_void send(std::span<const uint8_t> data) override {
273 if (!client_) {
274 send_failures_.fetch_add(1, std::memory_order_relaxed);
275 return result_void::err(error_info(
276 monitoring_error_code::dependency_missing,
277 "UDP client not available",
278 "common_udp_transport"
279 ).to_common_error());
280 }
281
282 if (!client_->is_connected()) {
283 send_failures_.fetch_add(1, std::memory_order_relaxed);
284 return result_void::err(error_info(
285 monitoring_error_code::network_error,
286 "Not connected",
287 "common_udp_transport"
288 ).to_common_error());
289 }
290
291 auto result = client_->send(data);
292 if (result.is_err()) {
293 send_failures_.fetch_add(1, std::memory_order_relaxed);
294 return result_void::err(error_info(
295 monitoring_error_code::network_error,
296 "Send failed: " + result.error().message,
297 "common_udp_transport"
298 ).to_common_error());
299 }
300
301 packets_sent_.fetch_add(1, std::memory_order_relaxed);
302 bytes_sent_.fetch_add(data.size(), std::memory_order_relaxed);
303 return common::ok();
304 }
305
306 bool is_connected() const override {
307 return client_ && client_->is_connected();
308 }
309
310 void disconnect() override {
311 if (client_) {
312 client_->disconnect();
313 }
314 }
315
316 bool is_available() const override {
317 return client_ != nullptr;
318 }
319
320 std::string name() const override {
321 if (client_) {
322 return "common:" + client_->get_implementation_name();
323 }
324 return "common:unavailable";
325 }
326
327 udp_statistics get_statistics() const override {
328 return {
329 packets_sent_.load(std::memory_order_relaxed),

358class network_udp_transport : public udp_transport {
359private:
360 std::unique_ptr<network_system::udp::udp_client> client_;
361 std::string host_;
362 uint16_t port_{0};

371 result_void connect(const std::string& host, uint16_t port) override {
372 try {
373 client_ = std::make_unique<network_system::udp::udp_client>(host, port);
374 host_ = host;
375 port_ = port;
376 connected_ = true;
377 return common::ok();
378 } catch (const std::exception& e) {

388 if (!connected_ || !client_) {
389 send_failures_.fetch_add(1, std::memory_order_relaxed);
390 return result_void::err(error_info(
391 monitoring_error_code::network_error,
392 "Not connected",
393 "network_udp_transport"
394 ).to_common_error());
395 }
396
397 try {
398 auto result = client_->send(data);
399 if (!result) {
400 send_failures_.fetch_add(1, std::memory_order_relaxed);
401 return result_void::err(error_info(
402 monitoring_error_code::network_error,
403 "Send failed: " + result.error().message,
404 "network_udp_transport"
405 ).to_common_error());
406 }
407
408 packets_sent_.fetch_add(1, std::memory_order_relaxed);
409 bytes_sent_.fetch_add(data.size(), std::memory_order_relaxed);
410 return common::ok();
411 } catch (const std::exception& e) {
412 send_failures_.fetch_add(1, std::memory_order_relaxed);
413 return result_void::err(error_info(
414 monitoring_error_code::network_error,
415 std::string("Send failed: ") + e.what(),
416 "network_udp_transport"
417 ).to_common_error());
418 }
419 }

422 return connected_ && client_ != nullptr;
423 }

429 port_ = 0;
