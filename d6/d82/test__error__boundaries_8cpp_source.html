39class ErrorBoundariesTest : public ::testing::Test {
40protected:
41 void SetUp() override {
42 call_count = 0;
43 success_after_attempts = 0;

51 std::atomic<int> call_count{0};
52 std::atomic<int> success_after_attempts{0};
53
54 // Helper function that fails for the first N attempts
55 result<int> failing_operation() {
56 int current_call = ++call_count;
57 if (success_after_attempts > 0 && current_call <= success_after_attempts) {
58 return make_error<int>(monitoring_error_code::operation_failed,
59 "Simulated failure on attempt " + std::to_string(current_call));
60 }
61 return make_success(42);
62 }
63
64 // Helper function that always fails
65 result<int> always_failing_operation() {
66 ++call_count;
67 return make_error<int>(monitoring_error_code::operation_failed, "Always fails");
68 }
69
70 // Helper function that always succeeds
71 result<int> always_succeeding_operation() {
72 ++call_count;
73 return make_success(100);
74 }
75
76 // Throwing operation for exception handling
77 result<int> throwing_operation() {
78 ++call_count;
79 throw std::runtime_error("Simulated exception");
80 }
81};

183 error_boundary_config config;
184 config.policy = error_boundary_policy::degrade;
185 config.error_threshold = 1;
186 config.enable_automatic_recovery = true;
187 config.recovery_timeout = std::chrono::milliseconds(100);
188
189 error_boundary<int> boundary("test_boundary", config);
190
191 // Trigger degradation
192 boundary.execute([this]() { return always_failing_operation(); });
193 EXPECT_GT(boundary.get_degradation_level(), degradation_level::normal);
194
195 // Wait for recovery timeout
196 std::this_thread::sleep_for(std::chrono::milliseconds(150));
197
198 // Execute successful operation to trigger recovery
199 auto result = boundary.execute([this]() { return always_succeeding_operation(); });
200 EXPECT_TRUE(result.is_ok());
201
202 // Check if recovery occurred (might need multiple successful operations)
203 for (int i = 0; i < 5; ++i) {
204 boundary.execute([this]() { return always_succeeding_operation(); });
205 }
206
207 auto metrics = boundary.get_metrics();
208 EXPECT_GT(metrics.recovery_attempts.load(), 0);
209}
