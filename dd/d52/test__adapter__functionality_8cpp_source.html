194
195 EXPECT_TRUE(adapter->is_logger_system_available());
196
197 // Collect metrics should retrieve from IMonitorable

199 ASSERT_TRUE(metrics_result.is_ok());
200

210 found_metric = true;
211 break;
212 }
213 }
214 }
