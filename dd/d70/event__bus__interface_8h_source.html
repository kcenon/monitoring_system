188class event_handler {
189public:
190 using handler_func = std::function<void(const EventType&)>;
191
192 event_handler(handler_func handler, event_priority priority = event_priority::normal)
193 : handler_(handler), priority_(priority), id_(generate_id()) {}
194
195 void operator()(const EventType& event) const {
196 if (handler_) {
197 handler_(event);
198 }
199 }
200
201 event_priority get_priority() const { return priority_; }
202 uint64_t get_id() const { return id_; }
203
204private:
205 static uint64_t generate_id() {
206 static std::atomic<uint64_t> counter{0};
207 return counter.fetch_add(1, std::memory_order_relaxed);
208 }
209
210 handler_func handler_;
211 event_priority priority_;
212 uint64_t id_;
213};
