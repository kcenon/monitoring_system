49struct fault_tolerance_metrics {
50 size_t total_operations = 0;
51 size_t successful_operations = 0;
52 size_t failed_operations = 0;
53 size_t circuit_breaker_rejections = 0;
54 size_t timeouts = 0;
55
56 double get_overall_success_rate() const {
57 if (total_operations == 0) {
58 return 1.0;
59 }
60 return static_cast<double>(successful_operations) / static_cast<double>(total_operations);
61 }
62};

67struct fault_tolerance_config {
68 bool enable_circuit_breaker = true;
69 bool enable_retry = true;
70 bool circuit_breaker_first = true;
71#ifdef _MSC_VER
72#pragma warning(push)
73#pragma warning(disable: 4996) // Disable deprecation warnings for internal use
74#endif
75 circuit_breaker_config circuit_config;
76#ifdef _MSC_VER
77#pragma warning(pop)
78#endif
79 retry_config retry_cfg;
80
81 bool validate() const {
82 if (!enable_circuit_breaker && !enable_retry) {
83 return false;
84 }
85#ifdef _MSC_VER
86#pragma warning(push)
87#pragma warning(disable: 4996) // Disable deprecation warnings for internal use
88#endif
89 if (enable_circuit_breaker && !circuit_config.validate()) {
90 return false;
91 }
92#ifdef _MSC_VER
93#pragma warning(pop)
94#endif
95 if (enable_retry && !retry_cfg.validate()) {
96 return false;
97 }
98 return true;
99 }
100};

133 template<typename Func>
134 common::Result<T> execute(Func&& func) {
135 metrics_.total_operations++;
136
137 common::Result<T> op_result = common::Result<T>::err(error_info(monitoring_error_code::operation_failed, "Not executed").to_common_error());
138
139 if (config_.circuit_breaker_first) {
140 op_result = execute_circuit_breaker_first(std::forward<Func>(func));
141 } else {
142 op_result = execute_retry_first(std::forward<Func>(func));
143 }
144
145 if (op_result.is_ok()) {
146 metrics_.successful_operations++;
147 } else {
148 metrics_.failed_operations++;
149 }
150
151 return op_result;
152 }
153
162 template<typename Func>
163 common::Result<T> execute_with_timeout(Func&& func, std::chrono::milliseconds timeout) {
164 metrics_.total_operations++;
165
166 auto future_result = std::async(std::launch::async, [this, func = std::forward<Func>(func)]() mutable {
167 return this->execute_internal(std::move(func));
168 });
169
170 if (future_result.wait_for(timeout) == std::future_status::timeout) {
171 metrics_.timeouts++;
172 metrics_.failed_operations++;
173 return common::make_error<T>(static_cast<int>(monitoring_error_code::operation_timeout),
174 "Operation timed out after " + std::to_string(timeout.count()) + "ms");
175 }
176
177 auto op_result = future_result.get();
178 if (op_result.is_ok()) {
179 metrics_.successful_operations++;
180 } else {
181 metrics_.failed_operations++;
182 }
183
184 return op_result;
185 }
186
191 common::Result<bool> is_healthy() {
192 if (config_.enable_circuit_breaker && circuit_breaker_) {
193 auto state = circuit_breaker_->get_state();
194 if (state == circuit_state::OPEN) {
195 return common::ok(false);
196 }
197 }
198 return common::ok(true);
199 }
200
204 fault_tolerance_metrics get_metrics() const {
205 return metrics_;
206 }
207
211 const std::string& get_name() const {
212 return name_;
213 }
214
215private:
216 void initialize() {
217 if (config_.enable_circuit_breaker) {
218 circuit_breaker_ = std::make_unique<circuit_breaker<T>>(name_ + "_cb", config_.circuit_config);
219 }
220 if (config_.enable_retry) {
221 retry_executor_ = std::make_unique<retry_executor<T>>(name_ + "_retry", config_.retry_cfg);
222 }
223 }

225 template<typename Func>
226 common::Result<T> execute_internal(Func&& func) {
227 if (config_.circuit_breaker_first) {
228 return execute_circuit_breaker_first(std::forward<Func>(func));
229 }
230 return execute_retry_first(std::forward<Func>(func));
231 }
232
233 template<typename Func>
234 common::Result<T> execute_circuit_breaker_first(Func&& func) {
235 if (config_.enable_circuit_breaker && circuit_breaker_) {
236 if (config_.enable_retry && retry_executor_) {
237 return circuit_breaker_->execute([this, &func]() {
238 return retry_executor_->execute(func);
239 });
240 }
241 return circuit_breaker_->execute(std::forward<Func>(func));
242 }
243 if (config_.enable_retry && retry_executor_) {
244 return retry_executor_->execute(std::forward<Func>(func));
245 }
246 return func();
247 }
248
249 template<typename Func>
250 common::Result<T> execute_retry_first(Func&& func) {
251 if (config_.enable_retry && retry_executor_) {
252 if (config_.enable_circuit_breaker && circuit_breaker_) {
253 return retry_executor_->execute([this, &func]() {
254 return circuit_breaker_->execute(func);
255 });
256 }
257 return retry_executor_->execute(std::forward<Func>(func));
258 }
259 if (config_.enable_circuit_breaker && circuit_breaker_) {
260 return circuit_breaker_->execute(std::forward<Func>(func));
261 }
262 return func();
263 }
264
265 std::string name_;
266 fault_tolerance_config config_;
267#ifdef _MSC_VER
268#pragma warning(push)
269#pragma warning(disable: 4996) // Disable deprecation warnings for internal use
270#endif
271 std::unique_ptr<circuit_breaker<T>> circuit_breaker_;
272#ifdef _MSC_VER
273#pragma warning(pop)
274#endif
275 std::unique_ptr<retry_executor<T>> retry_executor_;
276 mutable fault_tolerance_metrics metrics_;
277};

290 std::lock_guard<std::mutex> lock(mutex_);
291 registry_[name] = std::move(breaker);
292 }
293
294 template<typename T>
295 std::shared_ptr<circuit_breaker<T>> get_circuit_breaker(const std::string& name) {
296 std::lock_guard<std::mutex> lock(mutex_);
297 auto it = registry_.find(name);
298 if (it != registry_.end()) {
299 return std::any_cast<std::shared_ptr<circuit_breaker<T>>>(it->second);
300 }
301 return nullptr;
302 }
303#ifdef _MSC_VER
304#pragma warning(pop)
305#endif
306
307 void remove_circuit_breaker(const std::string& name) {
308 std::lock_guard<std::mutex> lock(mutex_);
309 registry_.erase(name);
310 }

312 std::vector<std::string> get_all_names() const {
313 std::lock_guard<std::mutex> lock(mutex_);
314 std::vector<std::string> names;
315 names.reserve(registry_.size());
316 for (const auto& pair : registry_) {
317 names.push_back(pair.first);
318 }
319 return names;
320 }
321
322 void clear() {
323 std::lock_guard<std::mutex> lock(mutex_);
324 registry_.clear();
325 }
326
327private:
328 mutable std::mutex mutex_;
329 std::unordered_map<std::string, std::any> registry_;
