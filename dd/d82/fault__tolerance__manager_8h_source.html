49struct fault_tolerance_metrics {
50 size_t total_operations = 0;
51 size_t successful_operations = 0;
52 size_t failed_operations = 0;
53 size_t circuit_breaker_rejections = 0;
54 size_t timeouts = 0;
55
56 double get_overall_success_rate() const {
57 if (total_operations == 0) {
58 return 1.0;
59 }
60 return static_cast<double>(successful_operations) / static_cast<double>(total_operations);
61 }
62};

67struct fault_tolerance_config {
68 bool enable_circuit_breaker = true;
69 bool enable_retry = true;
70 bool circuit_breaker_first = true;
71 circuit_breaker_config circuit_config;
72 retry_config retry_cfg;
73
74 bool validate() const {
75 if (!enable_circuit_breaker && !enable_retry) {
76 return false;
77 }
78 if (enable_circuit_breaker && !circuit_config.validate()) {
79 return false;
80 }
81 if (enable_retry && !retry_cfg.validate()) {
82 return false;
83 }
84 return true;
85 }
86};

95template<typename T>
96class fault_tolerance_manager {
97public:
98 fault_tolerance_manager() : name_("default"), config_() {
99 initialize();
100 }
101
102 explicit fault_tolerance_manager(const std::string& name)
103 : name_(name), config_() {
104 initialize();
105 }
106
107 explicit fault_tolerance_manager(const std::string& name, const fault_tolerance_config& cfg)
108 : name_(name), config_(cfg) {
109 initialize();
110 }
111
119 template<typename Func>
120 result<T> execute(Func&& func) {
121 metrics_.total_operations++;
122
123 result<T> op_result = make_error<T>(monitoring_error_code::operation_failed, "Not executed");
124
125 if (config_.circuit_breaker_first) {
126 op_result = execute_circuit_breaker_first(std::forward<Func>(func));
127 } else {
128 op_result = execute_retry_first(std::forward<Func>(func));
129 }
130
131 if (op_result.is_ok()) {
132 metrics_.successful_operations++;
133 } else {
134 metrics_.failed_operations++;
135 }
136
137 return op_result;
138 }
139
148 template<typename Func>
149 result<T> execute_with_timeout(Func&& func, std::chrono::milliseconds timeout) {
150 metrics_.total_operations++;
151
152 auto future_result = std::async(std::launch::async, [this, func = std::forward<Func>(func)]() mutable {
153 return this->execute_internal(std::move(func));
154 });
155
156 if (future_result.wait_for(timeout) == std::future_status::timeout) {
157 metrics_.timeouts++;
158 metrics_.failed_operations++;
159 return make_error<T>(monitoring_error_code::operation_timeout,
160 "Operation timed out after " + std::to_string(timeout.count()) + "ms");
161 }
162
163 auto op_result = future_result.get();
164 if (op_result.is_ok()) {
165 metrics_.successful_operations++;
166 } else {
167 metrics_.failed_operations++;
168 }
169
170 return op_result;
171 }
172
177 result<bool> is_healthy() {
178 if (config_.enable_circuit_breaker && circuit_breaker_) {
179 auto state = circuit_breaker_->get_state();
180 if (state == circuit_state::open) {
181 return make_success(false);
182 }
183 }
184 return make_success(true);
185 }
186
190 fault_tolerance_metrics get_metrics() const {
191 return metrics_;
192 }
193
197 const std::string& get_name() const {
198 return name_;
199 }
200
201private:
202 void initialize() {
203 if (config_.enable_circuit_breaker) {
204 circuit_breaker_ = std::make_unique<circuit_breaker<T>>(name_ + "_cb", config_.circuit_config);
205 }
206 if (config_.enable_retry) {
207 retry_executor_ = std::make_unique<retry_executor<T>>(name_ + "_retry", config_.retry_cfg);
208 }
209 }
210
211 template<typename Func>
212 result<T> execute_internal(Func&& func) {
213 if (config_.circuit_breaker_first) {
214 return execute_circuit_breaker_first(std::forward<Func>(func));
215 }
216 return execute_retry_first(std::forward<Func>(func));
217 }
218
219 template<typename Func>
220 result<T> execute_circuit_breaker_first(Func&& func) {
221 if (config_.enable_circuit_breaker && circuit_breaker_) {
222 if (config_.enable_retry && retry_executor_) {
223 return circuit_breaker_->execute([this, &func]() {
224 return retry_executor_->execute(func);
225 });
226 }
227 return circuit_breaker_->execute(std::forward<Func>(func));
228 }
229 if (config_.enable_retry && retry_executor_) {
230 return retry_executor_->execute(std::forward<Func>(func));
231 }
232 return func();
233 }
234
235 template<typename Func>
236 result<T> execute_retry_first(Func&& func) {
237 if (config_.enable_retry && retry_executor_) {
238 if (config_.enable_circuit_breaker && circuit_breaker_) {
239 return retry_executor_->execute([this, &func]() {
240 return circuit_breaker_->execute(func);
241 });
242 }
243 return retry_executor_->execute(std::forward<Func>(func));
244 }
245 if (config_.enable_circuit_breaker && circuit_breaker_) {
246 return circuit_breaker_->execute(std::forward<Func>(func));
247 }
248 return func();
249 }
250
251 std::string name_;
252 fault_tolerance_config config_;
253 std::unique_ptr<circuit_breaker<T>> circuit_breaker_;
254 std::unique_ptr<retry_executor<T>> retry_executor_;
255 mutable fault_tolerance_metrics metrics_;
256};

261class circuit_breaker_registry {
262public:
263 template<typename T>
264 void register_circuit_breaker(const std::string& name, std::shared_ptr<circuit_breaker<T>> breaker) {
265 std::lock_guard<std::mutex> lock(mutex_);
266 registry_[name] = std::move(breaker);
267 }
268
269 template<typename T>
270 std::shared_ptr<circuit_breaker<T>> get_circuit_breaker(const std::string& name) {
271 std::lock_guard<std::mutex> lock(mutex_);
272 auto it = registry_.find(name);
273 if (it != registry_.end()) {
274 return std::any_cast<std::shared_ptr<circuit_breaker<T>>>(it->second);
275 }
276 return nullptr;
277 }
278
279 void remove_circuit_breaker(const std::string& name) {
280 std::lock_guard<std::mutex> lock(mutex_);
281 registry_.erase(name);
282 }
283
284 std::vector<std::string> get_all_names() const {
285 std::lock_guard<std::mutex> lock(mutex_);
286 std::vector<std::string> names;
287 names.reserve(registry_.size());
288 for (const auto& pair : registry_) {
289 names.push_back(pair.first);
290 }
291 return names;
292 }
293
294 void clear() {
295 std::lock_guard<std::mutex> lock(mutex_);
296 registry_.clear();
297 }
298
299private:
300 mutable std::mutex mutex_;
301 std::unordered_map<std::string, std::any> registry_;
302};
