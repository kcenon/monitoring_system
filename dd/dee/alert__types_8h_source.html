125struct alert_labels {
126 std::unordered_map<std::string, std::string> labels;
127
128 alert_labels() = default;
129
130 explicit alert_labels(std::unordered_map<std::string, std::string> lbl)
131 : labels(std::move(lbl)) {}
132
138 void set(const std::string& key, const std::string& value) {
139 labels[key] = value;
140 }
141
147 std::string get(const std::string& key) const {
148 auto it = labels.find(key);
149 return it != labels.end() ? it->second : "";
150 }
151
157 bool has(const std::string& key) const {
158 return labels.find(key) != labels.end();
159 }
160
165 std::string fingerprint() const {
166 std::string result;
167 std::vector<std::pair<std::string, std::string>> sorted_labels(
168 labels.begin(), labels.end());
169 std::sort(sorted_labels.begin(), sorted_labels.end());
170 for (const auto& [key, value] : sorted_labels) {
171 result += key + "=" + value + ",";
172 }
173 return result;
174 }
175
176 bool operator==(const alert_labels& other) const {
177 return labels == other.labels;
178 }
179};

210struct alert {
211 std::string name;
212 alert_labels labels;
213 alert_annotations annotations;
214 alert_severity severity = alert_severity::warning;
215 alert_state state = alert_state::inactive;
216 double value = 0.0;
217
218 std::chrono::steady_clock::time_point created_at;
219 std::chrono::steady_clock::time_point updated_at;
220 std::optional<std::chrono::steady_clock::time_point> started_at;
221 std::optional<std::chrono::steady_clock::time_point> resolved_at;
222
223 uint64_t id = 0;
224 std::string rule_name;
225 std::string group_key;
226
230 alert()
231 : created_at(std::chrono::steady_clock::now())
232 , updated_at(created_at)
233 , id(generate_id()) {}
234
240 alert(std::string alert_name, alert_labels lbl)
241 : name(std::move(alert_name))
242 , labels(std::move(lbl))
243 , created_at(std::chrono::steady_clock::now())
244 , updated_at(created_at)
245 , id(generate_id()) {}
246
251 std::string fingerprint() const {
252 return name + "{" + labels.fingerprint() + "}";
253 }
254
259 bool is_active() const {
260 return state == alert_state::pending || state == alert_state::firing;
261 }
262
267 std::chrono::steady_clock::duration state_duration() const {
268 return std::chrono::steady_clock::now() - updated_at;
269 }
270
275 std::chrono::steady_clock::duration firing_duration() const {
276 if (state == alert_state::firing && started_at.has_value()) {
277 return std::chrono::steady_clock::now() - *started_at;
278 }
279 return std::chrono::steady_clock::duration::zero();
280 }
281
287 bool transition_to(alert_state new_state) {
288 if (!is_valid_transition(state, new_state)) {
289 return false;
290 }
291
292 auto now = std::chrono::steady_clock::now();
293 state = new_state;
294 updated_at = now;
295
296 if (new_state == alert_state::firing && !started_at.has_value()) {
297 started_at = now;
298 } else if (new_state == alert_state::resolved) {
299 resolved_at = now;
300 }
301
302 return true;
303 }
304
305private:
306 static uint64_t generate_id() {
307 static std::atomic<uint64_t> counter{0};
308 return counter.fetch_add(1, std::memory_order_relaxed);
309 }
310
314 static bool is_valid_transition(alert_state from, alert_state to) {
315 // All transitions are valid from suppressed (when silence expires)
316 if (from == alert_state::suppressed) {
317 return true;
318 }
319
320 // Transition to suppressed is always valid
321 if (to == alert_state::suppressed) {
322 return true;
323 }
324
325 switch (from) {
326 case alert_state::inactive:
327 return to == alert_state::pending;
328 case alert_state::pending:
329 return to == alert_state::firing || to == alert_state::inactive;
330 case alert_state::firing:
331 return to == alert_state::resolved;
332 case alert_state::resolved:
333 return to == alert_state::pending || to == alert_state::inactive;
334 default:
335 return false;
336 }
337 }
338};

393 alert_severity max_sev = alert_severity::info;

410struct alert_silence {
411 uint64_t id = 0;
412 std::string comment;
413 std::string created_by;
414 alert_labels matchers;
415 std::chrono::steady_clock::time_point starts_at;
416 std::chrono::steady_clock::time_point ends_at;
417
418 alert_silence()
419 : id(generate_id())
420 , starts_at(std::chrono::steady_clock::now())
421 , ends_at(starts_at + std::chrono::hours(1)) {}
422
427 bool is_active() const {
428 auto now = std::chrono::steady_clock::now();
429 return now >= starts_at && now < ends_at;
430 }
431
437 bool matches(const alert& a) const {
438 if (!is_active()) {
439 return false;
440 }
441
442 // All matcher labels must be present in alert labels
443 for (const auto& [key, value] : matchers.labels) {
444 if (a.labels.get(key) != value) {
445 return false;
446 }
447 }
448 return true;
449 }
450
451private:
452 static uint64_t generate_id() {
453 static std::atomic<uint64_t> counter{0};
454 return counter.fetch_add(1, std::memory_order_relaxed);
455 }
456};
