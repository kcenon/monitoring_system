437TEST_F(PerformanceMonitoringTest, ConcurrentRecording) {
438 const int num_threads = 10;
439 const int samples_per_thread = 100;
440
441 std::vector<std::thread> threads;
442
443 for (int t = 0; t < num_threads; ++t) {
444 threads.emplace_back([this, t, samples_per_thread]() {

446 profiler.record_sample(

451 }
452 });
453 }
