268
269 auto start_result = sys_monitor.start_monitoring(std::chrono::milliseconds(100));
270 ASSERT_TRUE(start_result.is_ok());
271
272 // Poll for samples instead of sleeping for fixed duration
273 auto deadline = std::chrono::steady_clock::now() + std::chrono::seconds(5);
274 std::vector<system_metrics> history;
275 while (std::chrono::steady_clock::now() < deadline) {
276 history = sys_monitor.get_history(std::chrono::seconds(1));
277 if (history.size() >= 2) {
278 break;
279 }
280 std::this_thread::yield();
281 }
282
283 EXPECT_GE(history.size(), 2); // Should have at least 2 samples (relaxed for CI)
284
285 // Check that timestamps are increasing
286 for (size_t i = 1; i < history.size(); ++i) {
287 EXPECT_GT(history[i].timestamp, history[i - 1].timestamp);
288 }
289
290 auto stop_result = sys_monitor.stop_monitoring();
291 ASSERT_TRUE(stop_result.is_ok());
292}
