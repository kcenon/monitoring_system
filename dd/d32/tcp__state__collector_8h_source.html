83inline std::string tcp_state_to_string(tcp_state state) {

123 uint64_t get_count(tcp_state state) const {
124 switch (state) {
125 case tcp_state::ESTABLISHED: return established;
126 case tcp_state::SYN_SENT: return syn_sent;
127 case tcp_state::SYN_RECV: return syn_recv;
128 case tcp_state::FIN_WAIT1: return fin_wait1;
129 case tcp_state::FIN_WAIT2: return fin_wait2;
130 case tcp_state::TIME_WAIT: return time_wait;
131 case tcp_state::CLOSE: return close;
132 case tcp_state::CLOSE_WAIT: return close_wait;
133 case tcp_state::LAST_ACK: return last_ack;
134 case tcp_state::LISTEN: return listen;
135 case tcp_state::CLOSING: return closing;
136 default: return unknown;
137 }
138 }
139
144 void increment(tcp_state state) {
145 switch (state) {
146 case tcp_state::ESTABLISHED: ++established; break;
147 case tcp_state::SYN_SENT: ++syn_sent; break;
148 case tcp_state::SYN_RECV: ++syn_recv; break;
149 case tcp_state::FIN_WAIT1: ++fin_wait1; break;
150 case tcp_state::FIN_WAIT2: ++fin_wait2; break;
151 case tcp_state::TIME_WAIT: ++time_wait; break;
152 case tcp_state::CLOSE: ++close; break;
153 case tcp_state::CLOSE_WAIT: ++close_wait; break;
154 case tcp_state::LAST_ACK: ++last_ack; break;
155 case tcp_state::LISTEN: ++listen; break;
156 case tcp_state::CLOSING: ++closing; break;
157 default: ++unknown; break;
158 }
159 }
160
165 uint64_t total() const {
166 return established + syn_sent + syn_recv + fin_wait1 + fin_wait2 +
167 time_wait + close + close_wait + last_ack + listen + closing + unknown;
168 }
169};
