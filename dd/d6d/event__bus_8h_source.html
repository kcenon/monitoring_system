36struct event_envelope {
37 std::type_index type;
38 std::any payload;
39 event_priority priority;
40 std::chrono::steady_clock::time_point timestamp;
41 uint64_t id;
42
43 event_envelope(std::type_index t, std::any p, event_priority pr)
44 : type(t), payload(std::move(p)), priority(pr),
45 timestamp(std::chrono::steady_clock::now()),
46 id(generate_id()) {}
47
48 bool operator<(const event_envelope& other) const {
49 // Higher priority events come first
50 if (priority != other.priority) {
51 return static_cast<int>(priority) < static_cast<int>(other.priority);
52 }
53 // For same priority, older events come first
54 return timestamp > other.timestamp;
55 }
56
57private:
58 static uint64_t generate_id() {
59 static std::atomic<uint64_t> counter{0};
60 return counter.fetch_add(1, std::memory_order_relaxed);
61 }
62};

119 // Start the event bus
120 result_void start() override {
121 std::lock_guard<std::mutex> lock(bus_mutex_);
122
123 if (is_running_) {
124 return result_void::error(monitoring_error_code::already_started,
125 "Event bus is already running");
126 }
127
128 stop_requested_ = false;
129 is_running_ = true;
130
131 // Start worker threads
132 for (size_t i = 0; i < config_.worker_thread_count; ++i) {
133 workers_.emplace_back(&event_bus::process_events_worker, this);
134 }
135
136 return result_void::success();
137 }
138
139 // Stop the event bus
140 result_void stop() override {
141 {
142 std::lock_guard<std::mutex> lock(bus_mutex_);
143 if (!is_running_) {
144 return result_void::success();
145 }
146
147 stop_requested_ = true;
148 is_running_ = false;
149 }
150
151 // Wake up all worker threads
152 queue_cv_.notify_all();
153
154 // Wait for all workers to finish
155 for (auto& worker : workers_) {
156 if (worker.joinable()) {
157 worker.join();
158 }
159 }
160 workers_.clear();
161
162 // Process any remaining events
163 process_all_pending();
164
165 return result_void::success();
166 }
167
168 // Check if event bus is active
169 bool is_active() const override {
170 std::lock_guard<std::mutex> lock(bus_mutex_);
171 return is_running_;
172 }
173
174 // Get pending event count
175 size_t get_pending_event_count() const override {
176 std::lock_guard<std::mutex> lock(queue_mutex_);
177 return event_queue_.size();
178 }

186 // Unsubscribe from events
187 result_void unsubscribe_event(const subscription_token& token) override {
188 std::lock_guard<std::mutex> lock(handlers_mutex_);
189
190 auto it = handlers_.find(token.get_event_type());
191 if (it != handlers_.end()) {
192 auto& handler_list = it->second;
193 handler_list.erase(
194 std::remove_if(handler_list.begin(), handler_list.end(),
195 [&token](const event_handler_wrapper& wrapper) {
196 return wrapper.id == token.get_handler_id();
197 }),
198 handler_list.end()
199 );
200
201 if (handler_list.empty()) {
202 handlers_.erase(it);
203 }
204 }
205
206 return result_void::success();
207 }

238protected:
239 // Publish event implementation
240 result_void publish_event_impl(std::type_index event_type,
241 std::any event) override {
242 // Check for back pressure
243 if (config_.enable_back_pressure) {
244 std::lock_guard<std::mutex> lock(queue_mutex_);
245 if (event_queue_.size() >= config_.max_queue_size) {
246 total_events_dropped_.fetch_add(1);
247 return result_void::error(monitoring_error_code::resource_exhausted,
248 "Event queue is full");
249 }
250
251 if (event_queue_.size() >= config_.back_pressure_threshold) {
252 // Apply back pressure - could implement adaptive strategies here
253 std::this_thread::sleep_for(std::chrono::milliseconds(1));
254 }
255 }
256
257 // Queue the event
258 {
259 std::lock_guard<std::mutex> lock(queue_mutex_);
260 event_queue_.emplace(event_type, std::move(event), event_priority::normal);
261 total_events_published_.fetch_add(1);
262 }
263
264 queue_cv_.notify_one();
265 return result_void::success();

268 // Subscribe event implementation
269 result<subscription_token> subscribe_event_impl(
270 std::type_index event_type,
271 std::function<void(const std::any&)> handler,
272 uint64_t handler_id,
273 event_priority priority) override {
274
275 std::lock_guard<std::mutex> lock(handlers_mutex_);
276
277 handlers_[event_type].emplace_back(std::move(handler), priority, handler_id);
278
279 // Sort handlers by priority
280 std::sort(handlers_[event_type].begin(), handlers_[event_type].end(),
281 [](const event_handler_wrapper& a, const event_handler_wrapper& b) {
282 return static_cast<int>(a.priority) > static_cast<int>(b.priority);
283 });
284
285 return subscription_token(event_type, handler_id);
286 }

303 // Process events worker thread
304 void process_events_worker() {
305 while (!stop_requested_) {
306 std::unique_lock<std::mutex> lock(queue_mutex_);
307
308 // Wait for events or stop signal
309 queue_cv_.wait_for(lock, config_.processing_interval,
310 [this] { return !event_queue_.empty() || stop_requested_; });
311
312 if (stop_requested_ && event_queue_.empty()) {
313 break;
314 }
315
316 // Process a batch of events
317 std::vector<event_envelope> batch;
318 size_t batch_size = std::min(size_t(10), event_queue_.size());
319
320 for (size_t i = 0; i < batch_size && !event_queue_.empty(); ++i) {
321 batch.push_back(std::move(const_cast<event_envelope&>(event_queue_.top())));
322 event_queue_.pop();
323 }
324
325 lock.unlock();
326
327 // Process the batch
328 for (auto& envelope : batch) {
329 dispatch_event(envelope);
330 total_events_processed_.fetch_add(1);
331 }
332 }
333 }
334
335 // Process all pending events
336 void process_all_pending() {
337 std::lock_guard<std::mutex> lock(queue_mutex_);
338
339 while (!event_queue_.empty()) {
340 auto envelope = std::move(const_cast<event_envelope&>(event_queue_.top()));
341 event_queue_.pop();
342 dispatch_event(envelope);
343 total_events_processed_.fetch_add(1);
344 }
345 }

347 // Dispatch event to handlers
348 void dispatch_event(const event_envelope& envelope) {
349 std::lock_guard<std::mutex> lock(handlers_mutex_);
350
351 auto it = handlers_.find(envelope.type);
352 if (it != handlers_.end()) {
353 for (const auto& wrapper : it->second) {
354 try {
355 wrapper.handler(envelope.payload);
356 } catch (const std::exception& e) {
357 // Log error but continue processing
358 // In production, this would be logged properly
359 }
360 }
361 }
362 }
