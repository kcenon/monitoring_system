68struct event_envelope {
69 std::type_index type;
70 std::any payload;
71 event_priority priority;
72 std::chrono::steady_clock::time_point timestamp;
73 uint64_t id;
74
75 event_envelope(std::type_index t, std::any p, event_priority pr)
76 : type(t), payload(std::move(p)), priority(pr),
77 timestamp(std::chrono::steady_clock::now()),
78 id(generate_id()) {}
79
80 bool operator<(const event_envelope& other) const {
81 // Higher priority events come first
82 if (priority != other.priority) {
83 return static_cast<int>(priority) < static_cast<int>(other.priority);
84 }
85 // For same priority, older events come first
86 return timestamp > other.timestamp;
87 }
88
89private:
90 static uint64_t generate_id() {
91 static std::atomic<uint64_t> counter{0};
92 return counter.fetch_add(1, std::memory_order_relaxed);
93 }
94};

200 // Start the event bus
201 result_void start() override {
202 std::lock_guard<std::mutex> lock(bus_mutex_);
203
204 if (is_running_) {
205 return make_void_error(monitoring_error_code::already_started,
206 "Event bus is already running");
207 }
208
209 stop_requested_ = false;
210 is_running_ = true;
211
212 // Start worker threads
213 for (size_t i = 0; i < config_.worker_thread_count; ++i) {
214 workers_.emplace_back(&event_bus::process_events_worker, this);
215 }
216
217 return make_void_success();
218 }
219
220 // Stop the event bus
221 result_void stop() override {
222 {
223 std::lock_guard<std::mutex> lock(bus_mutex_);
224 if (!is_running_) {
225 return make_void_success();
226 }
227
228 stop_requested_ = true;
229 is_running_ = false;
230 }
231
232 // Wake up all worker threads
233 queue_cv_.notify_all();
234
235 // Wait for all workers to finish
236 for (auto& worker : workers_) {
237 if (worker.joinable()) {
238 worker.join();
239 }
240 }
241 workers_.clear();
242
243 // Process any remaining events
244 process_all_pending();
245
246 return make_void_success();
247 }
248
249 // Check if event bus is active
250 bool is_active() const override {
251 std::lock_guard<std::mutex> lock(bus_mutex_);
252 return is_running_;
253 }
254
255 // Get pending event count
256 size_t get_pending_event_count() const override {
257 std::lock_guard<std::mutex> lock(queue_mutex_);
258 return event_queue_.size();
259 }
260
261 // Process all pending events synchronously
262 result_void process_pending_events() override {
263 process_all_pending();
264 return make_void_success();
265 }
266
267 // Unsubscribe from events
268 result_void unsubscribe_event(const subscription_token& token) override {
269 std::lock_guard<std::mutex> lock(handlers_mutex_);
270
271 auto it = handlers_.find(token.get_event_type());
272 if (it != handlers_.end()) {
273 auto& handler_list = it->second;
274 handler_list.erase(
275 std::remove_if(handler_list.begin(), handler_list.end(),
276 [&token](const event_handler_wrapper& wrapper) {
277 return wrapper.id == token.get_handler_id();
278 }),
279 handler_list.end()
280 );
281
282 if (handler_list.empty()) {
283 handlers_.erase(it);
284 }
285 }
286
287 return make_void_success();
288 }

290 // Get statistics
291 struct stats {
292 uint64_t total_published;
293 uint64_t total_processed;
294 uint64_t total_dropped;
295 size_t current_queue_size;
296 size_t subscriber_count;
297 bool is_back_pressure_active;
298 };
299
300 stats get_stats() const {
301 // Use scoped_lock to acquire both mutexes atomically with consistent ordering
302 // This prevents deadlock by acquiring locks in a deterministic order
303 // Lock ordering: queue_mutex before handlers_mutex (alphabetical for consistency)
304 std::scoped_lock lock(queue_mutex_, handlers_mutex_);
305
306 size_t queue_size = event_queue_.size();
307 bool back_pressure_active = queue_size >= config_.back_pressure_threshold;
308
309 size_t total_subscribers = 0;
310 for (const auto& [type, handlers] : handlers_) {
311 total_subscribers += handlers.size();
312 }
313
314 return {
315 total_events_published_.load(),
316 total_events_processed_.load(),
317 total_events_dropped_.load(),
318 queue_size,
319 total_subscribers,
320 back_pressure_active
321 };
322 }

324protected:
325 // Publish event implementation
326 result_void publish_event_impl(std::type_index event_type,
327 std::any event) override {
328 bool should_sleep = false;
329
330 {
331 std::lock_guard<std::mutex> lock(queue_mutex_);
332
333 if (config_.enable_back_pressure) {
334 const auto current_size = event_queue_.size();
335 if (current_size >= config_.max_queue_size) {
336 total_events_dropped_.fetch_add(1);
337 return make_void_error(
338 monitoring_error_code::resource_exhausted,
339 "Event queue is full");
340 }
341 should_sleep = current_size >= config_.back_pressure_threshold;
342 }
343
344 event_queue_.emplace(event_type, std::move(event), event_priority::normal);
345 total_events_published_.fetch_add(1);
346 }
347
348 if (should_sleep) {
349 std::this_thread::sleep_for(std::chrono::milliseconds(1));
350 }
351
352 queue_cv_.notify_one();
353 return make_void_success();
354 }
355
356 // Subscribe event implementation
357 result<subscription_token> subscribe_event_impl(
358 std::type_index event_type,
359 std::function<void(const std::any&)> handler,
360 uint64_t handler_id,
361 event_priority priority) override {
362
363 std::lock_guard<std::mutex> lock(handlers_mutex_);
364
365 handlers_[event_type].emplace_back(std::move(handler), priority, handler_id);
366
367 // Sort handlers by priority
368 std::sort(handlers_[event_type].begin(), handlers_[event_type].end(),
369 [](const event_handler_wrapper& a, const event_handler_wrapper& b) {
370 return static_cast<int>(a.priority) > static_cast<int>(b.priority);
371 });
372
373 return subscription_token(event_type, handler_id);
374 }
375
376 // Clear subscriptions implementation
377 result_void clear_subscriptions_impl(std::type_index event_type) override {
378 std::lock_guard<std::mutex> lock(handlers_mutex_);
379 handlers_.erase(event_type);
380 return make_void_success();
381 }
382
383 // Get subscriber count implementation
384 size_t get_subscriber_count_impl(std::type_index event_type) const override {
385 std::lock_guard<std::mutex> lock(handlers_mutex_);
386 auto it = handlers_.find(event_type);
387 return it != handlers_.end() ? it->second.size() : 0;
388 }
389
390private:
391 // Process events worker thread
392 void process_events_worker() {
393 while (!stop_requested_) {
394 std::unique_lock<std::mutex> lock(queue_mutex_);
395
396 // Wait for events or stop signal
397 queue_cv_.wait_for(lock, config_.processing_interval,
398 [this] { return !event_queue_.empty() || stop_requested_; });
399
400 if (stop_requested_ && event_queue_.empty()) {
401 break;
402 }
403
404 // Process a batch of events
405 std::vector<event_envelope> batch;
406 size_t batch_size = std::min(size_t(10), event_queue_.size());
407
408 for (size_t i = 0; i < batch_size && !event_queue_.empty(); ++i) {
409 // Copy element then pop to avoid const_cast UB
410 batch.push_back(event_queue_.top());
411 event_queue_.pop();
412 }
413
414 lock.unlock();
415
416 // Process the batch
417 for (auto& envelope : batch) {
418 dispatch_event(envelope);
419 total_events_processed_.fetch_add(1);
420 }
421 }
422 }
423
424 // Process all pending events
425 void process_all_pending() {
426 std::vector<event_envelope> pending;
427 {
428 std::lock_guard<std::mutex> lock(queue_mutex_);
429 pending.reserve(event_queue_.size());
430 while (!event_queue_.empty()) {
431 pending.push_back(event_queue_.top());
432 event_queue_.pop();
433 }
434 }
435
436 for (auto& envelope : pending) {
437 dispatch_event(envelope);
438 total_events_processed_.fetch_add(1);
439 }
440 }
441
442 // Dispatch event to handlers
443 void dispatch_event(const event_envelope& envelope) {
444 std::lock_guard<std::mutex> lock(handlers_mutex_);
445
446 auto it = handlers_.find(envelope.type);
447 if (it != handlers_.end()) {
448 for (const auto& wrapper : it->second) {
449 try {
450 wrapper.handler(envelope.payload);
451 } catch (const std::exception& e) {
452 (void)e; // Suppress unused variable warning
453 // Log error but continue processing
454 // In production, this would be logged properly
455 }
456 }
457 }
458 }
