103 std::string render(const alert& a) const {
104 std::string result = template_str_;
105
106 // Build variable map
107 std::unordered_map<std::string, std::string> vars;
108
109 // Alert properties
110 vars["name"] = a.name;
111 vars["state"] = alert_state_to_string(a.state);
112 vars["severity"] = alert_severity_to_string(a.severity);
113 vars["value"] = format_value(a.value);
114 vars["fingerprint"] = a.fingerprint();
115 vars["rule_name"] = a.rule_name;
116 vars["group_key"] = a.group_key;
117
118 // Labels
119 for (const auto& [key, value] : a.labels.labels) {
120 vars["labels." + key] = value;
121 }
122
123 // Annotations
124 vars["annotations.summary"] = a.annotations.summary;
125 vars["annotations.description"] = a.annotations.description;
126 if (a.annotations.runbook_url) {
127 vars["annotations.runbook_url"] = *a.annotations.runbook_url;
128 }
129 for (const auto& [key, value] : a.annotations.custom) {
130 vars["annotations." + key] = value;
131 }
132
133 // Custom variables
134 for (const auto& [key, value] : custom_vars_) {
135 vars[key] = value;
136 }
137
138 // Substitute variables
139 result = substitute_variables(result, vars);
140
141 return result;
142 }
143
147 const std::string& template_string() const { return template_str_; }

153 result_void validate() const {
154 // Check for unclosed variable references
155 std::regex pattern(R"(\$\{[^}]*$)");
156 if (std::regex_search(template_str_, pattern)) {
157 return make_void_error(monitoring_error_code::validation_failed,
158 "Unclosed variable reference in template");
159 }
160 return make_void_success();
161 }
162
163private:
164 static std::string format_value(double value) {
165 std::ostringstream oss;
166 oss << std::fixed << std::setprecision(2) << value;
167 return oss.str();
168 }
169
170 static std::string substitute_variables(
171 const std::string& input,
172 const std::unordered_map<std::string, std::string>& vars) {
173 std::string result = input;
174 std::regex pattern(R"(\$\{([^}]+)\})");
175
176 std::string::const_iterator start = result.cbegin();
177 std::smatch match;
178 std::string output;
179
180 while (std::regex_search(start, result.cend(), match, pattern)) {
181 output += std::string(start, match[0].first);
182
183 std::string var_name = match[1].str();
184 auto it = vars.find(var_name);
185 if (it != vars.end()) {
186 output += it->second;
187 } else {
188 output += match[0].str(); // Keep original if not found
189 }
190
191 start = match[0].second;
192 }
193 output += std::string(start, result.cend());
194
195 return output;
196 }
197
198 std::string template_str_;
199 std::unordered_map<std::string, std::string> custom_vars_;

246class rule_builder {
247public:
253 static result<std::shared_ptr<alert_rule>> build(const rule_definition& def) {
254 // Validate required fields
255 if (def.name.empty()) {
256 return make_error<std::shared_ptr<alert_rule>>(
257 monitoring_error_code::invalid_argument,
258 "Rule name is required");
259 }
260 if (def.metric_name.empty()) {
261 return make_error<std::shared_ptr<alert_rule>>(
262 monitoring_error_code::invalid_argument,
263 "Metric name is required");
264 }
265
266 auto rule = std::make_shared<alert_rule>(def.name);
267
268 // Set group
269 if (!def.group.empty()) {
270 rule->set_group(def.group);
271 }
272
273 // Set metric name
274 rule->set_metric_name(def.metric_name);
275
276 // Set severity
277 auto severity = parse_severity(def.severity);
278 if (!severity.is_ok()) {
279 return make_error<std::shared_ptr<alert_rule>>(
280 monitoring_error_code::invalid_argument,
281 severity.error().message);
282 }
283 rule->set_severity(severity.value());
284
285 // Set enabled
286 rule->set_enabled(def.enabled);
287
288 // Build trigger
289 auto trigger = build_trigger(def.trigger);
290 if (!trigger.is_ok()) {
291 return make_error<std::shared_ptr<alert_rule>>(
292 monitoring_error_code::invalid_argument,
293 trigger.error().message);
294 }
295 rule->set_trigger(trigger.value());
296
297 // Set timing
298 rule->set_evaluation_interval(
299 std::chrono::seconds(def.evaluation_interval_seconds));
300 rule->set_for_duration(
301 std::chrono::seconds(def.for_duration_seconds));
302 rule->set_repeat_interval(
303 std::chrono::seconds(def.repeat_interval_seconds));
304
305 // Set labels
306 for (const auto& [key, value] : def.labels) {
307 rule->add_label(key, value);
308 }
309
310 // Set annotations
311 if (!def.summary.empty()) {
312 rule->set_summary(def.summary);
313 }
314 if (!def.description.empty()) {
315 rule->set_description(def.description);
316 }
317 if (!def.runbook_url.empty()) {
318 rule->set_runbook_url(def.runbook_url);
319 }
320
321 return make_success(std::move(rule));
322 }
323
324private:
325 static result<alert_severity> parse_severity(const std::string& str) {
326 if (str.empty() || str == "warning") {
327 return make_success(alert_severity::warning);
328 }
329 if (str == "info") {
330 return make_success(alert_severity::info);
331 }
332 if (str == "critical") {
333 return make_success(alert_severity::critical);
334 }
335 if (str == "emergency") {
336 return make_success(alert_severity::emergency);
337 }
338 return make_error<alert_severity>(
339 monitoring_error_code::invalid_argument,
340 "Unknown severity: " + str);
341 }
342
343 static result<std::shared_ptr<alert_trigger>> build_trigger(
344 const rule_definition::trigger_config& cfg) {
345 if (cfg.type.empty() || cfg.type == "threshold") {
346 auto op = parse_operator(cfg.operator_str);
347 if (!op.is_ok()) {
348 return make_error<std::shared_ptr<alert_trigger>>(
349 monitoring_error_code::invalid_argument,
350 op.error().message);
351 }
352 return make_success(std::shared_ptr<alert_trigger>(
353 std::make_shared<threshold_trigger>(cfg.threshold, op.value())));
354 }
355
356 if (cfg.type == "rate") {
357 return make_success(std::shared_ptr<alert_trigger>(
358 std::make_shared<rate_of_change_trigger>(
359 cfg.rate_threshold,
360 std::chrono::seconds(cfg.window_seconds))));
361 }
362
363 if (cfg.type == "anomaly") {
364 return make_success(std::shared_ptr<alert_trigger>(
365 std::make_shared<anomaly_trigger>(
366 cfg.sensitivity,
367 static_cast<size_t>(cfg.window_seconds))));
368 }
369
370 if (cfg.type == "absent") {
371 return make_success(std::shared_ptr<alert_trigger>(
372 std::make_shared<absent_trigger>(
373 std::chrono::seconds(cfg.absent_seconds))));
374 }
375
376 return make_error<std::shared_ptr<alert_trigger>>(
377 monitoring_error_code::invalid_argument,
378 "Unknown trigger type: " + cfg.type);
379 }
380
381 static result<comparison_operator> parse_operator(const std::string& str) {
382 if (str.empty() || str == ">") {
383 return make_success(comparison_operator::greater_than);
384 }
385 if (str == ">=") {
386 return make_success(comparison_operator::greater_or_equal);
387 }
388 if (str == "<") {
389 return make_success(comparison_operator::less_than);
390 }
391 if (str == "<=") {
392 return make_success(comparison_operator::less_or_equal);
393 }
394 if (str == "==" || str == "=") {
395 return make_success(comparison_operator::equal);
396 }
397 if (str == "!=" || str == "<>") {
398 return make_success(comparison_operator::not_equal);
399 }
400 return make_error<comparison_operator>(
401 monitoring_error_code::invalid_argument,
402 "Unknown operator: " + str);
403 }

413class rule_registry {
414public:
415 using rule_change_callback = std::function<void(
416 const std::string& rule_name,
417 const std::shared_ptr<alert_rule>& rule,

425 result_void register_rule(std::shared_ptr<alert_rule> rule) {
426 if (!rule) {
427 return make_void_error(monitoring_error_code::invalid_argument,
428 "Rule cannot be null");
429 }
430
431 std::lock_guard<std::mutex> lock(mutex_);
432
433 std::string name = rule->name();
434 rules_[name] = rule;
435
436 // Notify listeners
437 for (const auto& callback : change_callbacks_) {
438 callback(name, rule, false);
439 }
440
441 return make_void_success();
442 }
443
449 result_void unregister_rule(const std::string& name) {
450 std::lock_guard<std::mutex> lock(mutex_);
451
452 auto it = rules_.find(name);
453 if (it == rules_.end()) {
454 return make_void_error(monitoring_error_code::not_found,
455 "Rule not found: " + name);
456 }
457
458 auto rule = it->second;
459 rules_.erase(it);
460
461 // Notify listeners
462 for (const auto& callback : change_callbacks_) {
463 callback(name, rule, true);
464 }
465
466 return make_void_success();
467 }
468
472 std::shared_ptr<alert_rule> get_rule(const std::string& name) const {
473 std::lock_guard<std::mutex> lock(mutex_);
474 auto it = rules_.find(name);
475 return it != rules_.end() ? it->second : nullptr;
476 }
477
481 std::vector<std::shared_ptr<alert_rule>> get_all_rules() const {
482 std::lock_guard<std::mutex> lock(mutex_);
483 std::vector<std::shared_ptr<alert_rule>> result;
484 result.reserve(rules_.size());
485 for (const auto& [name, rule] : rules_) {
486 result.push_back(rule);
487 }
488 return result;
489 }
490
494 std::vector<std::shared_ptr<alert_rule>> get_rules_by_group(
495 const std::string& group) const {
496 std::lock_guard<std::mutex> lock(mutex_);
497 std::vector<std::shared_ptr<alert_rule>> result;
498 for (const auto& [name, rule] : rules_) {
499 if (rule->group() == group) {
500 result.push_back(rule);
501 }
502 }
503 return result;
504 }
505
509 size_t rule_count() const {
510 std::lock_guard<std::mutex> lock(mutex_);
511 return rules_.size();
512 }
513
517 void on_rule_change(rule_change_callback callback) {
518 std::lock_guard<std::mutex> lock(mutex_);
519 change_callbacks_.push_back(std::move(callback));
520 }
521
527 result<size_t> load_definitions(const std::vector<rule_definition>& definitions) {
528 size_t loaded = 0;
529 std::vector<std::string> errors;
530
531 for (const auto& def : definitions) {
532 auto rule_result = rule_builder::build(def);
533 if (rule_result.is_ok()) {
534 auto reg_result = register_rule(rule_result.value());
535 if (reg_result.is_ok()) {
536 ++loaded;
537 } else {
538 errors.push_back(def.name + ": " + reg_result.error().message);
539 }
540 } else {
541 errors.push_back(def.name + ": " + rule_result.error().message);
542 }
543 }
544
545 if (!errors.empty() && loaded == 0) {
546 return make_error<size_t>(
547 monitoring_error_code::configuration_parse_error,
548 "Failed to load any rules: " + errors.front());
549 }
550
551 return make_success(loaded);
552 }
553
557 void clear() {
558 std::lock_guard<std::mutex> lock(mutex_);
559
560 for (const auto& [name, rule] : rules_) {
561 for (const auto& callback : change_callbacks_) {
562 callback(name, rule, true);
563 }
564 }
565
566 rules_.clear();
567 }
568
569private:
570 mutable std::mutex mutex_;
571 std::unordered_map<std::string, std::shared_ptr<alert_rule>> rules_;
572 std::vector<rule_change_callback> change_callbacks_;
573};
