83inline std::string security_event_type_to_string(security_event_type type) {

116struct security_event_counts {

132 uint64_t get_count(security_event_type type) const {
133 switch (type) {
134 case security_event_type::login_success: return login_success;
135 case security_event_type::login_failure: return login_failure;
136 case security_event_type::logout: return logout;
137 case security_event_type::sudo_usage: return sudo_usage;
138 case security_event_type::permission_change: return permission_change;
139 case security_event_type::account_created: return account_created;
140 case security_event_type::account_deleted: return account_deleted;
141 case security_event_type::account_modified: return account_modified;
142 default: return unknown;
143 }
144 }
145
150 void increment(security_event_type type) {
151 switch (type) {
152 case security_event_type::login_success: ++login_success; break;
153 case security_event_type::login_failure: ++login_failure; break;
154 case security_event_type::logout: ++logout; break;
155 case security_event_type::sudo_usage: ++sudo_usage; break;
156 case security_event_type::permission_change: ++permission_change; break;
157 case security_event_type::account_created: ++account_created; break;
158 case security_event_type::account_deleted: ++account_deleted; break;
159 case security_event_type::account_modified: ++account_modified; break;
160 default: ++unknown; break;
161 }
162 }
163
168 uint64_t total() const {
169 return login_success + login_failure + logout + sudo_usage +
170 permission_change + account_created + account_deleted +
171 account_modified + unknown;
172 }
173};
