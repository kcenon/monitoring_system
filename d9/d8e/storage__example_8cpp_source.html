39using namespace kcenon::monitoring;

41int main() {
42 std::cout << "=== Time Series Storage Example ===" << std::endl;
43
44 // Configure storage
45 storage_config config;
46 config.data_directory = "./tsdb_test_data";
47 config.compression = compression_algorithm::lz4;
48 config.memtable_size_mb = 16;
49
50 // Create time series engine
51 auto engine = std::make_unique<timeseries_engine>(config);
52
53 // Write some test metrics
54 std::cout << "\nWriting test metrics..." << std::endl;
55 auto now = std::chrono::steady_clock::now();
56
57 for (int i = 0; i < 10; ++i) {
58 double value = 100.0 + i * 10;
59 auto timestamp = now - std::chrono::seconds(60 * (10 - i));
60
61 engine->write("cpu.usage", value, timestamp, {{"host", "server1"}});
62 engine->write("memory.usage", value * 0.8, timestamp, {{"host", "server1"}});
63
64 std::cout << " Written metric at t-" << (10 - i) << " minutes: " << value << std::endl;
65 }
66
67 // Query metrics
68 std::cout << "\nQuerying CPU usage for last hour..." << std::endl;
69 auto results = engine->query("cpu.usage",
70 now - std::chrono::hours(1),
71 now,
72 {{"host", "server1"}});
73
74 for (const auto& series : results) {
75 std::cout << "Series: " << series.metric_name << std::endl;
76 std::cout << " Points: " << series.points.size() << std::endl;
77 std::cout << " Min: " << series.min_value << std::endl;
78 std::cout << " Max: " << series.max_value << std::endl;
79 std::cout << " Avg: " << series.average() << std::endl;
80 }
81
82 // Test database with partitioning
83 std::cout << "\n=== Metric Database Example ===" << std::endl;
84
85 database_config db_config;
86 db_config.data_directory = "./metrics_test_db";
87 db_config.partition_strategy = partition_strategy::by_metric_name;
88
89 auto database = std::make_unique<metric_database>(db_config);
90
91 // Write batch of metrics
92 std::vector<metric> batch;
93 for (int i = 0; i < 5; ++i) {
94 metric m;
95 m.name = "test.metric";
96 m.value = metric_value{50.0 + i};
97 m.timestamp = std::chrono::system_clock::now();
98 m.tags = {{"env", "test"}, {"instance", std::to_string(i)}};
99 batch.push_back(m);
100 }
101
102 size_t written = database->write_batch(batch);
103 std::cout << "Written " << written << " metrics to database" << std::endl;
104
105 // Query with aggregation
106 std::cout << "\n=== Query Engine Example ===" << std::endl;
107
108 auto query_engine = std::make_unique<metric_query_engine>(database.get());
109
110 // Simple query example (would need parser implementation for full functionality)
111 std::string query_str = "SELECT cpu.usage WHERE host='server1' FROM -1h";
112 std::cout << "Query: " << query_str << std::endl;
113
114 // Get database stats
115 auto stats = database->get_stats();
116 std::cout << "\nDatabase Statistics:" << std::endl;
117 std::cout << " Total metrics: " << stats.total_metrics << std::endl;
118 std::cout << " Total points: " << stats.total_points << std::endl;
119 std::cout << " Total partitions: " << stats.total_partitions << std::endl;
120
121 std::cout << "\nStorage example completed successfully!" << std::endl;
122 return 0;
123}
