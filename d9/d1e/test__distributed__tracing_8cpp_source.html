386// Mock exporter for testing exporter integration
387class MockTraceExporter : public trace_exporter_interface {
388public:
389 std::vector<trace_span> exported_spans;
390 std::atomic<std::size_t> export_count{0};
391 std::atomic<std::size_t> flush_count{0};
392 std::atomic<std::size_t> shutdown_count{0};
393 bool should_fail = false;
394
395 kcenon::common::VoidResult export_spans(const std::vector<trace_span>& spans) override {
396 if (should_fail) {
397 return kcenon::common::VoidResult::err(error_info(monitoring_error_code::operation_failed,
398 "Mock export failure", "test").to_common_error());
399 }
400 for (const auto& span : spans) {
401 exported_spans.push_back(span);
402 }
403 export_count++;
404 return kcenon::common::ok();
405 }
406
407 kcenon::common::VoidResult flush() override {
408 flush_count++;
409 return kcenon::common::ok();
410 }
411
412 kcenon::common::VoidResult shutdown() override {
413 shutdown_count++;
414 return kcenon::common::ok();
415 }
416
417 std::unordered_map<std::string, std::size_t> get_stats() const override {
418 return {
419 {"exported_spans", exported_spans.size()},
420 {"export_count", export_count.load()},
421 {"flush_count", flush_count.load()},
422 {"shutdown_count", shutdown_count.load()}
423 };
424 }
425};
