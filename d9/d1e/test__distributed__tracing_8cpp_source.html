148TEST_F(DistributedTracingTest, InjectExtractContext) {
149 auto span_result = tracer.start_span("test_operation");
150 ASSERT_TRUE(span_result.is_ok());
151 auto span = span_result.value();
152
153 span->baggage["test_key"] = "test_value";
154
155 // Inject into carrier (simulating HTTP headers)
156 std::unordered_map<std::string, std::string> headers;
157 auto context = tracer.extract_context(*span);
158 tracer.inject_context(context, headers);
159
160 EXPECT_TRUE(headers.contains("traceparent"));
161 EXPECT_TRUE(headers.contains("baggage-test_key"));
162
163 // Verify traceparent header format (00-traceid-spanid-flags)
164 auto traceparent = headers["traceparent"];
165 EXPECT_EQ(traceparent.substr(0, 3), "00-");
166 EXPECT_FALSE(traceparent.empty());
167
168 // Extract from carrier
169 auto extracted_result = tracer.extract_context_from_carrier(headers);
170 ASSERT_TRUE(extracted_result.is_ok());
171 auto extracted = extracted_result.value();
172
173 // Verify baggage is preserved through inject/extract cycle
174 EXPECT_EQ(extracted.baggage["test_key"], "test_value");
175
176 // Verify trace_id and span_id are extracted (may differ from original due to W3C format)
177 EXPECT_FALSE(extracted.trace_id.empty());
178 EXPECT_FALSE(extracted.span_id.empty());
179}
