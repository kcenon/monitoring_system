45// Test health check implementations
46class test_health_check : public health_check {
47private:
48 std::string name_;
49 health_check_type type_;
50 std::atomic<health_status> status_;
51 std::string message_;
52 mutable std::mutex mutex_;
53
54public:
55 test_health_check(
56 const std::string& name,
57 health_check_type type,
58 health_status status = health_status::healthy,
59 const std::string& message = "OK"
60 ) : name_(name), type_(type), status_(status), message_(message) {}
61
62 std::string get_name() const override { return name_; }
63 health_check_type get_type() const override { return type_; }
64
65 health_check_result check() override {
66 std::lock_guard<std::mutex> lock(mutex_);
67 health_check_result result;
68 result.status = status_.load();
69 result.message = message_;
70 result.timestamp = std::chrono::system_clock::now();
71 result.check_duration = std::chrono::milliseconds(10);
72 return result;
73 }
74
75 void set_status(health_status status) { status_.store(status); }
76 void set_message(const std::string& msg) {
77 std::lock_guard<std::mutex> lock(mutex_);
78 message_ = msg;
79 }
80};

624 }
