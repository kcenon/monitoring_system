91 void add_value(double value) {
92 std::unique_lock<std::shared_mutex> lock(mutex_);
93 count_++;
94 sum_ += value;
95
96 double delta = value - mean_;
97 mean_ += delta / static_cast<double>(count_);
98 double delta2 = value - mean_;
99 m2_ += delta * delta2;
100
101 if (count_ == 1) {
102 min_value_ = value;
103 max_value_ = value;
104 } else {
105 min_value_ = std::min(min_value_, value);
106 max_value_ = std::max(max_value_, value);
107 }
108 }

254private:
255 void init_markers() {
256 // Initialize marker positions
257 n_[0] = 1;
258 n_[1] = 2;
259 n_[2] = 3;
260 n_[3] = 4;
261 n_[4] = 5;
262
263 // Initialize desired marker positions
264 n_prime_[0] = 1;
265 n_prime_[1] = 1 + 2 * p_;
266 n_prime_[2] = 1 + 4 * p_;
267 n_prime_[3] = 3 + 2 * p_;
268 n_prime_[4] = 5;
269
270 // Increments
271 dn_[0] = 0;
272 dn_[1] = p_ / 2;
273 dn_[2] = p_;
274 dn_[3] = (1 + p_) / 2;
275 dn_[4] = 1;
276 }

278public:
279
284 void add_observation(double x) {
285 std::unique_lock<std::shared_mutex> lock(mutex_);
286 count_++;
287
288 if (count_ <= 5) {
289 q_[count_ - 1] = x;
290 if (count_ == 5) {
291 std::sort(q_, q_ + 5);
292 }
293 return;
294 }
295
296 // Find cell k
297 int k;
298 if (x < q_[0]) {
299 q_[0] = x;
300 k = 0;
301 } else if (x >= q_[4]) {
302 q_[4] = x;
303 k = 3;
304 } else {
305 for (k = 0; k < 4; ++k) {
306 if (x < q_[k + 1]) {
307 break;
308 }
309 }
310 }
311
312 // Increment positions
313 for (int i = k + 1; i < 5; ++i) {
314 n_[i]++;
315 }
316
317 // Update desired positions
318 for (int i = 0; i < 5; ++i) {
319 n_prime_[i] += dn_[i];
320 }
321
322 // Adjust marker heights
323 for (int i = 1; i < 4; ++i) {
324 double d = n_prime_[i] - n_[i];
325 if ((d >= 1 && n_[i + 1] - n_[i] > 1) ||
326 (d <= -1 && n_[i - 1] - n_[i] < -1)) {
327 int sign = (d >= 0) ? 1 : -1;
328 double q_new = parabolic(i, sign);
329
330 if (q_[i - 1] < q_new && q_new < q_[i + 1]) {
331 q_[i] = q_new;
332 } else {
333 q_[i] = linear(i, sign);
334 }
335 n_[i] += sign;
336 }
337 }
338 }
339
344 double get_quantile() const {
345 std::shared_lock<std::shared_mutex> lock(mutex_);
346 if (count_ < 5) {
347 if (count_ == 0) {
348 return 0.0;
349 }
350 // For small samples, use simple interpolation
351 std::vector<double> sorted(q_, q_ + count_);
352 std::sort(sorted.begin(), sorted.end());
353 size_t idx = static_cast<size_t>(p_ * (count_ - 1));
354 return sorted[idx];
355 }
356 return q_[2]; // Middle marker for p-quantile
357 }
358
362 size_t count() const {
363 std::shared_lock<std::shared_mutex> lock(mutex_);
364 return count_;
365 }
366
370 void reset() {
371 std::unique_lock<std::shared_mutex> lock(mutex_);
372 count_ = 0;
373 n_[0] = 1; n_[1] = 2; n_[2] = 3; n_[3] = 4; n_[4] = 5;
374 n_prime_[0] = 1;
375 n_prime_[1] = 1 + 2 * p_;
376 n_prime_[2] = 1 + 4 * p_;
377 n_prime_[3] = 3 + 2 * p_;
378 n_prime_[4] = 5;
379 }
380
381private:
382 double parabolic(int i, int sign) const {
383 double qi = q_[i];
384 double qim1 = q_[i - 1];
385 double qip1 = q_[i + 1];
386 int ni = n_[i];
387 int nim1 = n_[i - 1];
388 int nip1 = n_[i + 1];
389
390 double term1 = static_cast<double>(sign) / (nip1 - nim1);
391 double term2 = (ni - nim1 + sign) * (qip1 - qi) / (nip1 - ni);
392 double term3 = (nip1 - ni - sign) * (qi - qim1) / (ni - nim1);
393
394 return qi + term1 * (term2 + term3);
395 }
396
397 double linear(int i, int sign) const {
398 int idx = (sign < 0) ? i - 1 : i + 1;
399 return q_[i] + static_cast<double>(sign) * (q_[idx] - q_[i]) /
400 (n_[idx] - n_[i]);
401 }
402
403 mutable std::shared_mutex mutex_;
404 double p_;
405 size_t count_ = 0;
406 double q_[5] = {0};
407 int n_[5] = {0};
408 double n_prime_[5] = {0};
409 double dn_[5] = {0};
410};

420template<typename T>
421class moving_window_aggregator {
422public:
423 using time_point = std::chrono::system_clock::time_point;
424 using duration = std::chrono::system_clock::duration;
425
431 moving_window_aggregator(std::chrono::milliseconds window_duration,
432 size_t max_size)
433 : window_duration_(window_duration)
434 , max_size_(max_size) {}
435
441 void add_value(const T& value, time_point timestamp) {
442 std::unique_lock<std::shared_mutex> lock(mutex_);
443
444 // Remove expired entries
445 expire_old_entries(timestamp);
446
447 // Add new entry
448 if (entries_.size() >= max_size_) {
449 entries_.pop_front();
450 }
451 entries_.push_back({value, timestamp});
452 }
453
457 size_t size() const {
458 std::shared_lock<std::shared_mutex> lock(mutex_);
459 return entries_.size();
460 }
461
465 bool empty() const {
466 std::shared_lock<std::shared_mutex> lock(mutex_);
467 return entries_.empty();
468 }
469
473 std::vector<T> get_values() const {
474 std::shared_lock<std::shared_mutex> lock(mutex_);
475 std::vector<T> result;
476 result.reserve(entries_.size());
477 for (const auto& entry : entries_) {
478 result.push_back(entry.value);
479 }
480 return result;
481 }
482
486 void clear() {
487 std::unique_lock<std::shared_mutex> lock(mutex_);
488 entries_.clear();
489 }
490
491private:
492 struct entry {
493 T value;
494 time_point timestamp;
495 };
496
497 void expire_old_entries(time_point current) {
498 auto cutoff = current - window_duration_;
499 while (!entries_.empty() && entries_.front().timestamp < cutoff) {
500 entries_.pop_front();
501 }
502 }
503
504 mutable std::shared_mutex mutex_;
505 std::chrono::milliseconds window_duration_;
506 size_t max_size_;
507 std::deque<entry> entries_;
508};

548class stream_aggregator {
549public:
553 stream_aggregator() : stream_aggregator(stream_aggregator_config{}) {}
554
559 explicit stream_aggregator(const stream_aggregator_config& config)
560 : config_(config) {
561 // Initialize percentile estimators
562 for (double p : config_.percentiles_to_track) {
563 percentile_estimators_.emplace(p, quantile_estimator(p));
564 }
565 }
566
572 common::VoidResult add_observation(double value) {
573 std::unique_lock<std::shared_mutex> lock(mutex_);
574
575 // Check for outlier
576 if (config_.enable_outlier_detection && stats_.count() > 10) {
577 double z_score = std::abs(value - stats_.mean()) /
578 (stats_.stddev() + 1e-10);
579 if (z_score > config_.outlier_threshold) {
580 outlier_count_++;
581 outliers_.push_back(value);
582 if (outliers_.size() > 100) {
583 outliers_.erase(outliers_.begin());
584 }
585 }
586 }
587
588 stats_.add_value(value);
589
590 // Update percentile estimators
591 for (auto& [p, estimator] : percentile_estimators_) {
592 estimator.add_observation(value);
593 }
594
595 return common::ok();
596 }
597
601 streaming_statistics get_statistics() const {
602 std::shared_lock<std::shared_mutex> lock(mutex_);
603 auto stats = stats_.get_statistics();
604 stats.outlier_count = outlier_count_;
605 stats.outliers = outliers_;
606
607 // Add percentiles
608 for (const auto& [p, estimator] : percentile_estimators_) {
609 stats.percentiles[p] = estimator.get_quantile();
610 }
611
612 return stats;
613 }
614
620 std::optional<double> get_percentile(double p) const {
621 std::shared_lock<std::shared_mutex> lock(mutex_);
622 auto it = percentile_estimators_.find(p);
623 if (it != percentile_estimators_.end()) {
624 return it->second.get_quantile();
625 }
626 return std::nullopt;
627 }
628
632 size_t count() const {
633 return stats_.count();
634 }
635
639 double mean() const {
640 return stats_.mean();
641 }
642
646 double variance() const {
647 return stats_.variance();
648 }
649
653 double stddev() const {
654 return stats_.stddev();
655 }
656
660 void reset() {
661 std::unique_lock<std::shared_mutex> lock(mutex_);
662 stats_.reset();
663 outlier_count_ = 0;
664 outliers_.clear();
665 for (auto& [p, estimator] : percentile_estimators_) {
666 estimator.reset();
667 }
668 }
669
670private:
671 mutable std::shared_mutex mutex_;
672 stream_aggregator_config config_;
673 online_statistics stats_;
674 std::map<double, quantile_estimator> percentile_estimators_;
675 size_t outlier_count_ = 0;
676 std::vector<double> outliers_;
677};

685inline double pearson_correlation(const std::vector<double>& x,
686 const std::vector<double>& y) {
687 if (x.size() != y.size() || x.size() < 2) {
688 return 0.0;
689 }
690
691 size_t n = x.size();
692 double sum_x = 0, sum_y = 0, sum_xy = 0;
693 double sum_x2 = 0, sum_y2 = 0;
694
695 for (size_t i = 0; i < n; ++i) {
696 sum_x += x[i];
697 sum_y += y[i];
698 sum_xy += x[i] * y[i];
699 sum_x2 += x[i] * x[i];
700 sum_y2 += y[i] * y[i];
701 }
702
703 double num = n * sum_xy - sum_x * sum_y;
704 double den = std::sqrt((n * sum_x2 - sum_x * sum_x) *
705 (n * sum_y2 - sum_y * sum_y));
706
707 if (den < 1e-10) {
708 return 0.0;
709 }
710
711 return num / den;
712}
