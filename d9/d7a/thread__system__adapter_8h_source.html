65 // Returns true when thread_system headers are available and a monitorable
66 // provider can be discovered at runtime (best‑effort).
67 bool is_thread_system_available() const {

75 return static_cast<bool>(monitorable);

81# endif

84#endif

87 // One‑shot collection. When thread_system is not available, returns empty.
88 result<std::vector<metric>> collect_metrics() {
89 std::vector<metric> out;
90
91#if MONITORING_THREAD_SYSTEM_AVAILABLE && defined(MONITORING_HAS_COMMON_INTERFACES)
92 try {
93 auto& container = kcenon::thread::service_container::global();
94 auto monitorable = container.resolve<kcenon::common::interfaces::IMonitorable>();
95 if (monitorable) {
96 // Convert minimal subset of thread_system metrics into adapter metrics
97 // thread_system v3.0 uses common::interfaces::IMonitorable
98 auto monitoring_result = monitorable->get_monitoring_data();
99 if (!monitoring_result.is_err()) {
100 const auto& snap = monitoring_result.value();
101 // Map metrics from common_system snapshot
102 for (const auto& m : snap.metrics) {
103 metric adapted{m.name, m.value, {}};
104 out.emplace_back(std::move(adapted));
105 }
106 }
107 }
108 } catch (...) {
109 // Fall back to empty on any failure
110 }
111#endif
112
113 return make_success(std::move(out));
114 }

129 // Start periodic collection (best‑effort). When publish_events is true,
130 // collected metrics are emitted via metric_collection_event.
131 result_void start_collection(const collection_config& cfg) {
132 if (running_.exchange(true)) {
133 return make_void_success(); // already running
134 }
135
136 if (!bus_) {
137 running_ = false;
138 return make_result_void(monitoring_error_code::operation_failed, "event_bus not set");
139 }
140
141 worker_ = std::thread([this, cfg]() {
142 while (running_.load()) {
143 auto res = collect_metrics();
144 if (res.is_ok() && cfg.publish_events && !res.value().empty()) {
145 bus_->publish_event(metric_collection_event("thread_system_adapter", res.value()));
146 }
147 std::this_thread::sleep_for(cfg.interval);
148 }
149 });
150
151 return make_void_success();
152 }
153
154 result_void stop_collection() {
155 if (!running_.exchange(false)) {
156 return make_void_success();
157 }
158 if (worker_.joinable()) {
159 worker_.join();
160 }
161 return make_void_success();
162 }
163
164 ~thread_system_adapter() { (void)stop_collection(); }
165
166private:
167 std::shared_ptr<event_bus> bus_;
168 std::atomic<bool> running_{false};
169 std::thread worker_;
170};
