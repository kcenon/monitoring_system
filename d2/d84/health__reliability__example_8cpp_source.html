56// Simulate a database connection
57class DatabaseConnection {
58private:
59 std::atomic<bool> is_healthy_{true};
60 std::atomic<int> query_count_{0};
61 std::mt19937 rng_{std::random_device{}()};
62
63public:
64 void set_healthy(bool healthy) {
65 is_healthy_ = healthy;
66 }
67
68 result<std::string> execute_query(const std::string& query) {
69 query_count_++;
70
71 // Simulate latency
72 std::this_thread::sleep_for(10ms);
73
74 // Simulate failures
75 if (!is_healthy_) {
76 return make_error<std::string>(
77 monitoring_error_code::service_unavailable,
78 "Database connection lost"
79 );
80 }
81
82 // Random transient failures (10% chance)
83 std::uniform_int_distribution<> dist(1, 10);
84 if (dist(rng_) == 1) {
85 return make_error<std::string>(
86 monitoring_error_code::operation_timeout,
87 "Query timeout"
88 );
89 }
90
91 return make_success<std::string>("Query result for: " + query);
92 }
93
94 int get_query_count() const { return query_count_; }
95};

97// Simulate an external API
98class ExternalApiClient {
99private:
100 std::atomic<int> failure_count_{0};
101 std::atomic<int> call_count_{0};
102
103public:
104 result<std::string> call_api(const std::string& endpoint) {
105 call_count_++;
106
107 // Simulate increasing failures
108 if (failure_count_ > 5) {
109 // API is down
110 return make_error<std::string>(
111 monitoring_error_code::service_unavailable,
112 "Service unavailable"
113 );
114 }
115
116 // Simulate intermittent failures
117 if (call_count_ % 3 == 0) {
118 failure_count_++;
119 return make_error<std::string>(
120 monitoring_error_code::operation_failed,
121 "Internal server error"
122 );
123 }
124
125 failure_count_ = 0; // Reset on success
126 return make_success<std::string>("API response from: " + endpoint);
127 }
128
129 void reset() {
130 failure_count_ = 0;
131 call_count_ = 0;
132 }
133
134 int get_call_count() const { return call_count_; }
135};

137// Demonstrate health monitoring
138void demonstrate_health_monitoring() {
139 std::cout << "\n=== Health Monitoring Demo ===" << std::endl;
140
141 // Create health monitor
142 health_monitor_config config;
143 config.check_interval = 2s;
144 config.cache_duration = 1s;
145
146 health_monitor monitor(config);
147
148 // Create database connection for health checks
149 auto database = std::make_shared<DatabaseConnection>();
150
151 // Register liveness check
152 monitor.register_check("database_liveness",
153 std::make_shared<functional_health_check>(
154 "database_liveness",
155 health_check_type::liveness,
156 [database]() -> health_check_result {
157 // Simple ping check
158 auto result = database->execute_query("SELECT 1");
159 if (result.is_ok()) {
160 return health_check_result::healthy("Database is alive");
161 } else {
162 return health_check_result::unhealthy(
163 "Database unreachable: " + result.error().message
164 );
165 }
166 },
167 500ms, // timeout
168 true // critical
169 )
170 );
171
172 // Register readiness check
173 monitor.register_check("database_readiness",
174 std::make_shared<functional_health_check>(
175 "database_readiness",
176 health_check_type::readiness,
177 [database]() -> health_check_result {
178 // Check if database can handle queries
179 auto result = database->execute_query("SELECT COUNT(*) FROM users");
180 if (result.is_ok()) {
181 int query_count = database->get_query_count();
182 if (query_count > 100) {
183 return health_check_result::degraded(
184 "High query count: " + std::to_string(query_count)
185 );
186 }
187 return health_check_result::healthy("Database ready");
188 } else {
189 return health_check_result::unhealthy(
190 "Database not ready: " + result.error().message
191 );
192 }
193 },
194 1000ms, // timeout
195 false // non-critical
196 )
197 );
198
199 // Register startup check
200 monitor.register_check("system_startup",
201 std::make_shared<functional_health_check>(
202 "system_startup",
203 health_check_type::startup,
204 []() -> health_check_result {
205 // Check system initialization
206 static bool initialized = false;
207 if (!initialized) {
208 std::this_thread::sleep_for(100ms); // Simulate initialization
209 initialized = true;
210 }
211 return health_check_result::healthy("System initialized");
212 }
213 )
214 );
215
216 // Start health monitoring
217 monitor.start();
218
219 std::cout << "Health monitoring started" << std::endl;
220
221 // Perform health checks
222 std::cout << "\n1. Initial health check:" << std::endl;
223 auto all_checks = monitor.check_all();
224 for (const auto& [name, result] : all_checks) {
225 std::cout << " " << name << ": "
226 << (result.status == health_status::healthy ? "HEALTHY" :
227 result.status == health_status::degraded ? "DEGRADED" : "UNHEALTHY")
228 << " - " << result.message << std::endl;
229 }
230
231 // Get overall status
232 auto overall = monitor.get_overall_status();
233 std::cout << " Overall status: "
234 << (overall == health_status::healthy ? "HEALTHY" :
235 overall == health_status::degraded ? "DEGRADED" : "UNHEALTHY")
236 << std::endl;
237
238 // Simulate database failure
239 std::cout << "\n2. Simulating database failure..." << std::endl;
240 database->set_healthy(false);
241 std::this_thread::sleep_for(1s);
242
243 all_checks = monitor.check_all();
244 for (const auto& [name, result] : all_checks) {
245 if (name.find("database") != std::string::npos) {
246 std::cout << " " << name << ": "
247 << (result.status == health_status::healthy ? "HEALTHY" : "UNHEALTHY")
248 << " - " << result.message << std::endl;
249 }
250 }
251
252 // Register recovery handler
253 monitor.register_recovery_handler("database_liveness",
254 [database]() -> bool {
255 std::cout << " Attempting database recovery..." << std::endl;
256 database->set_healthy(true);
257 return true;
258 }
259 );
260
261 // Recover database
262 std::cout << "\n3. Triggering recovery..." << std::endl;
263 monitor.refresh();
264 std::this_thread::sleep_for(2s);
265
266 all_checks = monitor.check_all();
267 std::cout << " Database status after recovery: "
268 << (all_checks["database_liveness"].status == health_status::healthy ?
269 "HEALTHY" : "UNHEALTHY") << std::endl;
270
271 // Get health report
272 std::cout << "\n4. Health Report:" << std::endl;
273 std::cout << monitor.get_health_report() << std::endl;
274
275 monitor.stop();
276}

278// Demonstrate circuit breaker
279void demonstrate_circuit_breaker() {
280 std::cout << "\n=== Circuit Breaker Demo ===" << std::endl;
281
282 // Create external API client
283 auto api_client = std::make_shared<ExternalApiClient>();
284
285 // Configure circuit breaker
286 circuit_breaker_config cb_config;
287 cb_config.failure_threshold = 3;
288 // cb_config.failure_ratio = 0.5; // Not available in current API
289 cb_config.timeout = 100ms;
290 cb_config.reset_timeout = 2s;
291 cb_config.success_threshold = 2;
292
293 circuit_breaker<std::string> breaker("api_breaker", cb_config);
294
295 std::cout << "Circuit breaker configured:" << std::endl;
296 std::cout << " Failure threshold: " << cb_config.failure_threshold << std::endl;
297 std::cout << " Reset timeout: 2s" << std::endl;
298
299 // Define the operation
300 auto api_operation = [api_client]() -> result<std::string> {
301 return api_client->call_api("/users");
302 };
303
304 // Define fallback
305 auto fallback = []() -> result<std::string> {
306 return make_success<std::string>("Cached response (fallback)");
307 };
308
309 // Make calls through circuit breaker
310 std::cout << "\n1. Making API calls through circuit breaker:" << std::endl;
311
312 for (int i = 1; i <= 10; ++i) {
313 auto result = breaker.execute(api_operation, fallback);
314
315 std::cout << " Call " << i << ": ";
316 if (result.is_ok()) {
317 std::cout << "SUCCESS - " << result.value() << std::endl;
318 } else {
319 std::cout << "FAILED - " << result.error().message << std::endl;
320 }
321
322 // Check circuit state
323 auto state = breaker.get_state();
324 if (state == circuit_state::open) {
325 std::cout << " [Circuit OPEN - using fallback]" << std::endl;
326 } else if (state == circuit_state::half_open) {
327 std::cout << " [Circuit HALF-OPEN - testing]" << std::endl;
328 }
329
330 std::this_thread::sleep_for(300ms);
331 }
332
333 // Get circuit breaker metrics
334 auto metrics = breaker.get_metrics();
335 std::cout << "\n2. Circuit Breaker Metrics:" << std::endl;
336 std::cout << " Total calls: " << metrics.total_calls << std::endl;
337 std::cout << " Successful calls: " << metrics.successful_calls << std::endl;
338 std::cout << " Failed calls: " << metrics.failed_calls << std::endl;
339 std::cout << " Rejected calls: " << metrics.rejected_calls << std::endl;
340 std::cout << " State transitions: " << metrics.state_transitions << std::endl;
341
342 // Wait for circuit to reset
343 std::cout << "\n3. Waiting for circuit reset..." << std::endl;
344 api_client->reset(); // Reset API client
345 std::this_thread::sleep_for(3s);
346
347 // Try again after reset
348 std::cout << "\n4. Trying after reset:" << std::endl;
349 for (int i = 1; i <= 3; ++i) {
350 auto result = breaker.execute(api_operation, fallback);
351 std::cout << " Call " << i << ": ";
352 if (result.is_ok()) {
353 std::cout << "SUCCESS" << std::endl;
354 } else {
355 std::cout << "FAILED" << std::endl;
356 }
357 }
358}

360// Demonstrate retry policy (simplified)
361void demonstrate_retry_policy() {
362 std::cout << "\n=== Retry Policy Demo ===" << std::endl;
363
364 // Configure retry policy
365 retry_config config;
366 config.max_attempts = 3;
367 config.strategy = retry_strategy::exponential_backoff;
368 config.initial_delay = 100ms;
369 config.max_delay = 2s;
370 config.backoff_multiplier = 2.0;
371
372 std::cout << "Retry policy configured:" << std::endl;
373 std::cout << " Max attempts: " << config.max_attempts << std::endl;
374 std::cout << " Strategy: exponential backoff" << std::endl;
375 std::cout << " Initial delay: 100ms" << std::endl;
376
377 // Simulate manual retry logic (since retry_policy class not available)
378 std::cout << "\n1. Executing flaky operation with manual retry:" << std::endl;
379
380 std::atomic<int> attempt_count{0};
381 auto flaky_operation = [&attempt_count]() -> result<std::string> {
382 attempt_count++;
383 std::cout << " Attempt " << attempt_count << "..." << std::endl;
384
385 // Fail first 2 attempts
386 if (attempt_count <= 2) {
387 return make_error<std::string>(
388 monitoring_error_code::operation_timeout,
389 "Operation timed out"
390 );
391 }
392
393 return make_success<std::string>("Operation succeeded!");
394 };
395
396 result<std::string> final_result = make_error<std::string>(
397 monitoring_error_code::operation_failed, "Initialization pending");
398 for (int i = 0; i < static_cast<int>(config.max_attempts); ++i) {
399 final_result = flaky_operation();
400 if (final_result.is_ok()) {
401 break;
402 }
403
404 // Wait before retry
405 if (i < static_cast<int>(config.max_attempts) - 1) {
406 auto delay = config.initial_delay * static_cast<int>(std::pow(config.backoff_multiplier, i));
407 std::this_thread::sleep_for(delay);
408 }
409 }
410
411 if (final_result.is_ok()) {
412 std::cout << " Final result: SUCCESS - " << final_result.value() << std::endl;
413 } else {
414 std::cout << " Final result: FAILED - " << final_result.error().message << std::endl;
415 }
416
417 std::cout << " Total attempts: " << attempt_count << std::endl;
418}

420// Demonstrate error boundaries
421void demonstrate_error_boundaries() {
422 std::cout << "\n=== Error Boundaries Demo ===" << std::endl;
423
424 // Configure error boundary
425 error_boundary_config config;
426 config.error_threshold = 5; // Use correct field name
427 config.error_window = 60s;
428 config.enable_fallback_logging = true; // Use correct field name
429
430 error_boundary<std::string> boundary("critical_section", config); // Specify template type
431
432 // Set error handler
433 boundary.set_error_handler([](const error_info& error, degradation_level level) {
434 std::cout << " Error handler called: " << error.message
435 << " (degradation level: " << static_cast<int>(level) << ")" << std::endl;
436 });
437
438 std::cout << "Error boundary configured:" << std::endl;
439 std::cout << " Max errors: " << config.error_threshold << std::endl;
440 std::cout << " Error window: 60s" << std::endl;
441
442 // Execute operations within boundary
443 std::cout << "\n1. Executing operations within error boundary:" << std::endl;
444
445 for (int i = 1; i <= 7; ++i) {
446 auto result = boundary.execute([i]() -> ::result<std::string> {
447 std::cout << " Operation " << i << ": ";
448
449 // Simulate failures on odd numbers
450 if (i % 2 == 1) {
451 std::cout << "FAILED" << std::endl;
452 return make_error<std::string>(
453 monitoring_error_code::operation_failed,
454 "Operation " + std::to_string(i) + " failed"
455 );
456 }
457
458 std::cout << "SUCCESS" << std::endl;
459 return make_success<std::string>("Result " + std::to_string(i));
460 });
461
462 if (result.is_err() && result.error().code == static_cast<int>(monitoring_error_code::circuit_breaker_open)) {
463 std::cout << " [Error boundary triggered - too many errors]" << std::endl;
464 break;
465 }
466 }
467
468 // Get statistics
469 auto stats = boundary.get_metrics();
470 std::cout << "\n2. Error Boundary Statistics:" << std::endl;
471 std::cout << " Total operations: " << stats.total_operations << std::endl;
472 std::cout << " Failed operations: " << stats.failed_operations << std::endl;
473 std::cout << " Success rate: "
474 << (stats.total_operations > 0 ?
475 100.0 * (stats.total_operations - stats.failed_operations) / stats.total_operations : 0)
476 << "%" << std::endl;
477}
