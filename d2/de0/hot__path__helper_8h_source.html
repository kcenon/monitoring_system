93template <typename Map, typename Key, typename CreateFn>
94auto get_or_create(Map& map, std::shared_mutex& mutex, const Key& key,
95 CreateFn create_fn) ->
96 typename std::remove_reference<decltype(*map.begin()->second)>::type* {
97 // Fast path: try with shared lock (read-only, allows concurrent readers)
98 {
99 std::shared_lock<std::shared_mutex> read_lock(mutex);
100 auto it = map.find(key);
101 if (it != map.end()) {
102 return it->second.get();
103 }
104 }
105
106 // Slow path: upgrade to write lock and double-check
107 {
108 std::unique_lock<std::shared_mutex> write_lock(mutex);
109 // Double-check after acquiring write lock
110 // Another thread may have created the entry while we were waiting
111 auto& ptr = map[key];
112 if (!ptr) {
113 ptr = create_fn();
114 }
115 return ptr.get();
116 }
117}

159template <typename Map, typename Key, typename CreateFn, typename InitFn>
160auto get_or_create_with_init(Map& map, std::shared_mutex& mutex, const Key& key,
161 CreateFn create_fn, InitFn init_fn) ->
162 typename std::remove_reference<decltype(*map.begin()->second)>::type* {
163 // Fast path: try with shared lock (read-only, allows concurrent readers)
164 {
165 std::shared_lock<std::shared_mutex> read_lock(mutex);
166 auto it = map.find(key);
167 if (it != map.end()) {
168 return it->second.get();
169 }
170 }
171
172 // Slow path: upgrade to write lock and double-check
173 {
174 std::unique_lock<std::shared_mutex> write_lock(mutex);
175 // Double-check after acquiring write lock
176 // Another thread may have created the entry while we were waiting
177 auto& ptr = map[key];
178 if (!ptr) {
179 ptr = create_fn();
180 // Initialize the newly created entry while still under write lock
181 init_fn(*ptr);
182 }
183 return ptr.get();
184 }
185}

228template <typename Map, typename Key, typename CreateFn, typename UpdateFn>
229auto get_or_create_and_update(Map& map, std::shared_mutex& mutex, const Key& key,
230 CreateFn create_fn, UpdateFn update_fn)
231 -> decltype(update_fn(*map.begin()->second)) {
232 // Get or create the entry
233 auto* ptr = get_or_create(map, mutex, key, std::move(create_fn));
234
235 // Apply the update function outside of map locks
236 // The value's own synchronization handles thread safety
237 return update_fn(*ptr);
238}
