154 template<typename Func, typename Fallback>
155 common::Result<T> execute(Func&& func, Fallback&& fallback) {
156 metrics_.total_calls++;
157
158 auto current_state = check_state();
159
160 if (current_state == circuit_state::open) {
161 metrics_.rejected_calls++;
162 return fallback();
163 }
164
165 auto op_result = func();
166 if (op_result.is_ok()) {
167 on_success();
168 return op_result;
169 } else {
170 on_failure();
171 return fallback();
172 }

235 void reset() {
236 std::lock_guard<std::mutex> lock(state_mutex_);
237 transition_to(circuit_state::closed);
238 failure_count_.store(0, std::memory_order_release);
239 consecutive_successes_.store(0, std::memory_order_release);
240 }

242private:
246 circuit_state check_state() {
247 auto current = state_.load(std::memory_order_acquire);
248
249 if (current == circuit_state::open) {
250 auto now = clock::now();
251 auto opened_at = last_failure_time_.load(std::memory_order_acquire);
252
253 if (now - opened_at >= config_.reset_timeout) {
254 std::lock_guard<std::mutex> lock(state_mutex_);
255 if (state_.load(std::memory_order_acquire) == circuit_state::open) {
256 transition_to(circuit_state::half_open);
257 consecutive_successes_.store(0, std::memory_order_release);
258 return circuit_state::half_open;
259 }
260 }
261 }
262
263 return state_.load(std::memory_order_acquire);
264 }
265
269 void on_success() {
270 metrics_.successful_calls++;
271
272 auto current = state_.load(std::memory_order_acquire);
273
274 if (current == circuit_state::half_open) {
275 auto successes = consecutive_successes_.fetch_add(1, std::memory_order_acq_rel) + 1;
276
277 if (successes >= config_.success_threshold) {
278 std::lock_guard<std::mutex> lock(state_mutex_);
279 if (state_.load(std::memory_order_acquire) == circuit_state::half_open) {
280 transition_to(circuit_state::closed);
281 failure_count_.store(0, std::memory_order_release);
282 consecutive_successes_.store(0, std::memory_order_release);
283 }
284 }
285 } else if (current == circuit_state::closed) {
286 failure_count_.store(0, std::memory_order_release);
287 }
288 }
289
293 void on_failure() {
294 metrics_.failed_calls++;
295 last_failure_time_.store(clock::now(), std::memory_order_release);
296
297 auto current = state_.load(std::memory_order_acquire);
298
299 if (current == circuit_state::half_open) {
300 std::lock_guard<std::mutex> lock(state_mutex_);
301 if (state_.load(std::memory_order_acquire) == circuit_state::half_open) {
302 transition_to(circuit_state::open);
303 consecutive_successes_.store(0, std::memory_order_release);
304 }
305 } else if (current == circuit_state::closed) {
306 auto failures = failure_count_.fetch_add(1, std::memory_order_acq_rel) + 1;
307
308 if (failures >= config_.failure_threshold) {
309 std::lock_guard<std::mutex> lock(state_mutex_);
310 if (state_.load(std::memory_order_acquire) == circuit_state::closed &&
311 failure_count_.load(std::memory_order_acquire) >= config_.failure_threshold) {
312 transition_to(circuit_state::open);
313 }
314 }
315 }
316 }
317
321 void transition_to(circuit_state new_state) {
322 auto old_state = state_.load(std::memory_order_acquire);
323 if (old_state != new_state) {
324 state_.store(new_state, std::memory_order_release);
325 metrics_.state_transitions++;
326 }
327 }
