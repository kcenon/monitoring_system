126 template<typename Func, typename Fallback>
127 auto execute(Func&& func, Fallback&& fallback) -> decltype(func()) {
128 metrics_.total_calls++;
129
130 // Check and potentially transition state
131 auto current_state = check_state();
132
133 if (current_state == circuit_state::open) {
134 metrics_.rejected_calls++;
135 return fallback();
136 }
137
138 try {
139 auto result = func();
140 on_success();
141 return result;
142 } catch (...) {
143 on_failure();
144 return fallback();
145 }
146 }

209 void reset() {
210 std::lock_guard<std::mutex> lock(state_mutex_);
211 transition_to(circuit_state::closed);
212 failure_count_.store(0, std::memory_order_release);
213 consecutive_successes_.store(0, std::memory_order_release);
214 }
215
216private:
220 circuit_state check_state() {
221 auto current = state_.load(std::memory_order_acquire);
222
223 if (current == circuit_state::open) {
224 auto now = clock::now();
225 auto opened_at = last_failure_time_.load(std::memory_order_acquire);
226
227 if (now - opened_at >= config_.reset_timeout) {
228 std::lock_guard<std::mutex> lock(state_mutex_);
229 // Double-check after acquiring lock
230 if (state_.load(std::memory_order_acquire) == circuit_state::open) {
231 transition_to(circuit_state::half_open);
232 consecutive_successes_.store(0, std::memory_order_release);
233 return circuit_state::half_open;
234 }
235 }
236 }
237
238 return state_.load(std::memory_order_acquire);
239 }
240
244 void on_success() {
245 metrics_.successful_calls++;
246
247 auto current = state_.load(std::memory_order_acquire);
248
249 if (current == circuit_state::half_open) {
250 auto successes = consecutive_successes_.fetch_add(1, std::memory_order_acq_rel) + 1;
251
252 if (successes >= config_.success_threshold) {
253 std::lock_guard<std::mutex> lock(state_mutex_);
254 if (state_.load(std::memory_order_acquire) == circuit_state::half_open) {
255 transition_to(circuit_state::closed);
256 failure_count_.store(0, std::memory_order_release);
257 consecutive_successes_.store(0, std::memory_order_release);
258 }
259 }
260 } else if (current == circuit_state::closed) {
261 // Reset failure count on success in closed state
262 failure_count_.store(0, std::memory_order_release);
263 }
264 }
265
269 void on_failure() {
270 metrics_.failed_calls++;
271 last_failure_time_.store(clock::now(), std::memory_order_release);
272
273 auto current = state_.load(std::memory_order_acquire);
274
275 if (current == circuit_state::half_open) {
276 // Any failure in half_open immediately opens the circuit
277 std::lock_guard<std::mutex> lock(state_mutex_);
278 if (state_.load(std::memory_order_acquire) == circuit_state::half_open) {
279 transition_to(circuit_state::open);
280 consecutive_successes_.store(0, std::memory_order_release);
281 }
282 } else if (current == circuit_state::closed) {
283 auto failures = failure_count_.fetch_add(1, std::memory_order_acq_rel) + 1;
284
285 if (failures >= config_.failure_threshold) {
286 std::lock_guard<std::mutex> lock(state_mutex_);
287 if (state_.load(std::memory_order_acquire) == circuit_state::closed &&
288 failure_count_.load(std::memory_order_acquire) >= config_.failure_threshold) {
289 transition_to(circuit_state::open);
290 }
291 }
292 }
293 }
294
298 void transition_to(circuit_state new_state) {
299 auto old_state = state_.load(std::memory_order_acquire);
300 if (old_state != new_state) {
301 state_.store(new_state, std::memory_order_release);
302 metrics_.state_transitions++;
303 }
304 }
305
306 std::string name_;
307 config config_;
308
309 std::atomic<size_t> failure_count_{0};
310 std::atomic<size_t> consecutive_successes_{0};
311 std::atomic<circuit_state> state_{circuit_state::closed};
312 std::atomic<clock::time_point> last_failure_time_{clock::now()};
313
314 mutable std::mutex state_mutex_;
315 mutable circuit_breaker_metrics metrics_;
316};
