54 size_t failure_threshold = 5;

68 config.success_threshold = success_threshold;

83 circuit_breaker_metrics() = default;

91 double get_success_rate() const {
92 auto total = total_calls.load();
93 if (total == 0) {
94 return 1.0;
95 }
96 return static_cast<double>(successful_calls.load()) / static_cast<double>(total);

155 template<typename Func, typename Fallback>
156 common::Result<T> execute(Func&& func, Fallback&& fallback) {
157 metrics_.total_calls++;
158
159 if (!impl_->allow_request()) {
160 metrics_.rejected_calls++;
161 return fallback();
162 }
163
164 auto op_result = func();
165 if (op_result.is_ok()) {
166 impl_->record_success();
167 metrics_.successful_calls++;
168 return op_result;
169 } else {
170 impl_->record_failure();
171 metrics_.failed_calls++;
172 return fallback();
173 }
174 }
