70// Custom trigger implementation: Periodic trigger (fires every N evaluations)
71class periodic_trigger : public alert_trigger {
72public:
73 explicit periodic_trigger(size_t period)
74 : period_(period), count_(0) {}
75
76 bool evaluate(double /*value*/) const override {
77 ++count_;
78 if (count_ >= period_) {
79 count_ = 0;
80 return true;
81 }
82 return false;
83 }
84
85 std::string type_name() const override {
86 return "periodic";
87 }
88
89 std::string description() const override {
90 return "fires every " + std::to_string(period_) + " evaluations";
91 }
92
93 void reset() {
94 count_ = 0;
95 }
96
97private:
98 size_t period_;
99 mutable size_t count_;
100};

102// Custom trigger: Moving average based trigger
103class moving_average_trigger : public alert_trigger {
104public:
105 moving_average_trigger(size_t window_size, double threshold)
106 : window_size_(window_size)
107 , threshold_(threshold) {}
108
109 bool evaluate(double value) const override {
110 values_.push_back(value);
111 if (values_.size() > window_size_) {
112 values_.erase(values_.begin());
113 }
114
115 if (values_.size() < window_size_) {
116 return false; // Not enough data yet
117 }
118
119 double sum = 0.0;
120 for (double v : values_) {
121 sum += v;
122 }
123 double avg = sum / static_cast<double>(values_.size());
124
125 return avg > threshold_;
126 }
127
128 std::string type_name() const override {
129 return "moving_average";
130 }
131
132 std::string description() const override {
133 return "MA(" + std::to_string(window_size_) + ") > " + std::to_string(threshold_);
134 }
135
136 double current_average() const {
137 if (values_.empty()) return 0.0;
138 double sum = 0.0;
139 for (double v : values_) {
140 sum += v;
141 }
142 return sum / static_cast<double>(values_.size());
143 }
144
145 void reset() {
146 values_.clear();
147 }
148
149private:
150 size_t window_size_;
151 double threshold_;
152 mutable std::vector<double> values_;
153};
