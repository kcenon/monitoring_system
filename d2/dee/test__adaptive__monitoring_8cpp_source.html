148TEST_F(AdaptiveMonitoringTest, AdaptationStatistics) {
149 auto mock = std::make_shared<mock_collector>("test_collector");
150 adaptive_collector collector(mock);
151
152 // Simulate load changes
153 system_metrics low_load;
154 low_load.cpu_usage_percent = 30.0;
155 low_load.memory_usage_percent = 40.0;
156
157 system_metrics high_load;
158 high_load.cpu_usage_percent = 85.0;
159 high_load.memory_usage_percent = 70.0;
160
161 // Adapt to low load
162 collector.adapt(low_load);
163 auto stats = collector.get_stats();
164 EXPECT_EQ(stats.current_load_level, load_level::low);
165
166 // Adapt to high load
167 collector.adapt(high_load);
168 stats = collector.get_stats();
169 // After smoothing with default factor 0.7, the effective CPU would be:
170 // 0.7 * 85 + 0.3 * 30 = 59.5 + 9 = 68.5 (high, not critical)
171 // But memory is 70%, which should elevate it
172 EXPECT_GE(static_cast<int>(stats.current_load_level), static_cast<int>(load_level::high));
173 EXPECT_GT(stats.total_adaptations, 0);
174 EXPECT_GT(stats.downscale_count, 0);
175}

411TEST_F(AdaptiveMonitoringTest, ConcurrentCollectorAccess) {
412 const int num_threads = 10;
413 const int collectors_per_thread = 5;
414
415 std::vector<std::thread> threads;
416
417 for (int t = 0; t < num_threads; ++t) {
418 threads.emplace_back([this, t, collectors_per_thread]() {
419 for (int c = 0; c < collectors_per_thread; ++c) {
420 std::string name = "collector_" + std::to_string(t) + "_" + std::to_string(c);
421 auto mock = std::make_shared<mock_collector>(name);
422
423 monitor.register_collector(name, mock);
424
425 // Random operations
426 if (c % 2 == 0) {
427 monitor.set_collector_priority(name, t * 10 + c);
428 }
429
430 if (c % 3 == 0) {
431 monitor.get_collector_stats(name);
432 }
433 }
434 });
435 }
436
437 for (auto& thread : threads) {
438 thread.join();
439 }
440
441 auto all_stats = monitor.get_all_stats();
442 EXPECT_EQ(all_stats.size(), num_threads * collectors_per_thread);
