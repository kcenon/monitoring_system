57enum class comparison_operator {
58 greater_than,
59 greater_or_equal,
60 less_than,
61 less_or_equal,
62 equal,
63 not_equal
64};
65
69constexpr const char* comparison_operator_to_string(comparison_operator op) noexcept {
70 switch (op) {
71 case comparison_operator::greater_than: return ">";
72 case comparison_operator::greater_or_equal: return ">=";
73 case comparison_operator::less_than: return "<";
74 case comparison_operator::less_or_equal: return "<=";
75 case comparison_operator::equal: return "==";
76 case comparison_operator::not_equal: return "!=";
77 default: return "?";
78 }
79}

101class threshold_trigger : public alert_trigger {
102public:
109 explicit threshold_trigger(double threshold,
110 comparison_operator op = comparison_operator::greater_than,
111 double epsilon = 1e-9)
112 : threshold_(threshold)
113 , operator_(op)
114 , epsilon_(epsilon) {}
115
116 bool evaluate(double value) const override {
117 switch (operator_) {
118 case comparison_operator::greater_than:
119 return value > threshold_;
120 case comparison_operator::greater_or_equal:
121 return value >= threshold_ - epsilon_;
122 case comparison_operator::less_than:
123 return value < threshold_;
124 case comparison_operator::less_or_equal:
125 return value <= threshold_ + epsilon_;
126 case comparison_operator::equal:
127 return std::abs(value - threshold_) <= epsilon_;
128 case comparison_operator::not_equal:
129 return std::abs(value - threshold_) > epsilon_;
130 default:
131 return false;
132 }
133 }
134
135 std::string type_name() const override {
136 return "threshold";
137 }
138
139 std::string description() const override {
140 return "value " + std::string(comparison_operator_to_string(operator_)) +
141 " " + std::to_string(threshold_);
142 }
143
147 double threshold() const { return threshold_; }
148
152 comparison_operator op() const { return operator_; }
153

266class rate_of_change_trigger : public alert_trigger {
267public:
272 enum class rate_direction {
273 increasing,
274 decreasing,
275 either
276 };
277
285 rate_of_change_trigger(double rate_threshold,
286 std::chrono::milliseconds window,
287 rate_direction direction = rate_direction::either,
288 size_t min_samples = 2)
289 : rate_threshold_(rate_threshold)
290 , window_(window)
291 , direction_(direction)
292 , min_samples_(min_samples) {}
293
294 bool evaluate(double value) const override {
295 auto now = std::chrono::steady_clock::now();
296
297 std::lock_guard<std::mutex> lock(mutex_);
298
299 // Add new sample
300 samples_.push_back({value, now});
301
302 // Remove old samples outside window
303 auto cutoff = now - window_;
304 while (!samples_.empty() && samples_.front().timestamp < cutoff) {
305 samples_.pop_front();
306 }
307
308 // Need minimum samples to calculate rate
309 if (samples_.size() < min_samples_) {
310 return false;
311 }
312
313 // Calculate rate of change
314 double rate = calculate_rate();
315
316 switch (direction_) {
317 case rate_direction::increasing:
318 return rate > rate_threshold_;
319 case rate_direction::decreasing:
320 return rate < -rate_threshold_;
321 case rate_direction::either:
322 return std::abs(rate) > rate_threshold_;
323 default:
324 return false;
325 }
326 }
327
328 std::string type_name() const override {
329 return "rate_of_change";
330 }
331
332 std::string description() const override {
333 std::string dir_str;
334 switch (direction_) {
335 case rate_direction::increasing: dir_str = "increase"; break;
336 case rate_direction::decreasing: dir_str = "decrease"; break;
337 case rate_direction::either: dir_str = "change"; break;
338 }
339 return dir_str + " rate > " + std::to_string(rate_threshold_) +
340 " per " + std::to_string(window_.count()) + "ms";
341 }
342
346 void reset() {
347 std::lock_guard<std::mutex> lock(mutex_);
348 samples_.clear();
349 }
350
351private:
352 struct sample {
353 double value;
354 std::chrono::steady_clock::time_point timestamp;
355 };
356
357 double calculate_rate() const {
358 if (samples_.size() < 2) {
359 return 0.0;
360 }
361
362 // Use linear regression for smoother rate calculation
363 double sum_x = 0.0, sum_y = 0.0, sum_xy = 0.0, sum_xx = 0.0;
364 auto base_time = samples_.front().timestamp;
365
366 for (const auto& s : samples_) {
367 double x = std::chrono::duration<double, std::milli>(
368 s.timestamp - base_time).count();
369 double y = s.value;
370 sum_x += x;
371 sum_y += y;
372 sum_xy += x * y;
373 sum_xx += x * x;
374 }
375
376 double n = static_cast<double>(samples_.size());
377 double denominator = n * sum_xx - sum_x * sum_x;
378
379 if (std::abs(denominator) < 1e-10) {
380 return 0.0;
381 }
382
383 // Slope is rate of change per millisecond
384 double slope = (n * sum_xy - sum_x * sum_y) / denominator;
385
386 // Convert to rate per window
387 return slope * static_cast<double>(window_.count());
388 }
389
390 double rate_threshold_;
391 std::chrono::milliseconds window_;
392 rate_direction direction_;
393 size_t min_samples_;
394
395 mutable std::mutex mutex_;
396 mutable std::deque<sample> samples_;
397};

417class anomaly_trigger : public alert_trigger {
418public:
425 explicit anomaly_trigger(double sensitivity = 3.0,
426 size_t window_size = 100,
427 size_t min_samples = 10)
428 : sensitivity_(sensitivity)
429 , window_size_(window_size)
430 , min_samples_(min_samples) {}
431
432 bool evaluate(double value) const override {
433 std::lock_guard<std::mutex> lock(mutex_);
434
435 // Add to history
436 if (history_.size() >= window_size_) {
437 history_.pop_front();
438 }
439 history_.push_back(value);
440
441 // Need minimum samples for statistics
442 if (history_.size() < min_samples_) {
443 return false;
444 }
445
446 // Calculate statistics
447 double mean_val = mean();
448 double stddev = standard_deviation(mean_val);
449
450 // Avoid division by zero or very small stddev
451 if (stddev < 1e-10) {
452 return false;
453 }
454
455 // Calculate z-score
456 double z_score = std::abs(value - mean_val) / stddev;
457
458 return z_score > sensitivity_;
459 }
460
461 std::string type_name() const override {
462 return "anomaly";
463 }
464
465 std::string description() const override {
466 return "value > " + std::to_string(sensitivity_) + " std devs from mean";
467 }
468
472 double current_mean() const {
473 std::lock_guard<std::mutex> lock(mutex_);
474 return mean();
475 }
476
480 double current_stddev() const {
481 std::lock_guard<std::mutex> lock(mutex_);
482 return standard_deviation(mean());
483 }
484
488 void reset() {
489 std::lock_guard<std::mutex> lock(mutex_);
490 history_.clear();
491 }
492
493private:
494 double mean() const {
495 if (history_.empty()) {
496 return 0.0;
497 }
498 double sum = std::accumulate(history_.begin(), history_.end(), 0.0);
499 return sum / static_cast<double>(history_.size());
500 }
501
502 double standard_deviation(double mean_val) const {
503 if (history_.size() < 2) {
504 return 0.0;
505 }
506 double sq_sum = 0.0;
507 for (double val : history_) {
508 double diff = val - mean_val;
509 sq_sum += diff * diff;
510 }
511 return std::sqrt(sq_sum / static_cast<double>(history_.size() - 1));
512 }
513
514 double sensitivity_;
515 size_t window_size_;
516 size_t min_samples_;
517
518 mutable std::mutex mutex_;
519 mutable std::deque<double> history_;
520};

553class composite_trigger : public alert_trigger {
554public:
560 composite_trigger(composite_operation op,
561 std::vector<std::shared_ptr<alert_trigger>> triggers)
562 : operation_(op)
563 , triggers_(std::move(triggers)) {}
564
568 bool evaluate(double value) const override {
569 std::vector<double> values(triggers_.size(), value);
570 return evaluate_multi(values);
571 }
572
578 bool evaluate_multi(const std::vector<double>& values) const {
579 if (triggers_.empty()) {
580 return false;
581 }
582
583 std::vector<bool> results;
584 results.reserve(triggers_.size());
585
586 for (size_t i = 0; i < triggers_.size(); ++i) {
587 double val = (i < values.size()) ? values[i] : values.back();
588 results.push_back(triggers_[i]->evaluate(val));
589 }
590
591 switch (operation_) {
592 case composite_operation::AND:
593 return std::all_of(results.begin(), results.end(),
594 [](bool b) { return b; });
595 case composite_operation::OR:
596 return std::any_of(results.begin(), results.end(),
597 [](bool b) { return b; });
598 case composite_operation::XOR: {
599 size_t count = std::count(results.begin(), results.end(), true);
600 return count == 1;
601 }
602 case composite_operation::NOT:
603 return !results.front();
604 default:
605 return false;
606 }
607 }
608
609 std::string type_name() const override {
610 return "composite";
611 }
612
613 std::string description() const override {
614 std::string op_str;
615 switch (operation_) {
616 case composite_operation::AND: op_str = " AND "; break;
617 case composite_operation::OR: op_str = " OR "; break;
618 case composite_operation::XOR: op_str = " XOR "; break;
619 case composite_operation::NOT: return "NOT (" + triggers_.front()->description() + ")";
620 }
621
622 std::string result = "(";
623 for (size_t i = 0; i < triggers_.size(); ++i) {
624 if (i > 0) {
625 result += op_str;
626 }
627 result += triggers_[i]->description();
628 }
629 result += ")";
630 return result;
631 }
632
636 const std::vector<std::shared_ptr<alert_trigger>>& triggers() const {
637 return triggers_;
638 }
639
640 // Factory methods
644 static std::shared_ptr<composite_trigger> all_of(
645 std::vector<std::shared_ptr<alert_trigger>> triggers) {
646 return std::make_shared<composite_trigger>(composite_operation::AND,
647 std::move(triggers));
648 }
649
653 static std::shared_ptr<composite_trigger> any_of(
654 std::vector<std::shared_ptr<alert_trigger>> triggers) {
655 return std::make_shared<composite_trigger>(composite_operation::OR,
656 std::move(triggers));
657 }
658
662 static std::shared_ptr<composite_trigger> invert(
663 std::shared_ptr<alert_trigger> trigger) {
664 return std::make_shared<composite_trigger>(composite_operation::NOT,
665 std::vector<std::shared_ptr<alert_trigger>>{std::move(trigger)});
666 }
667
668private:
669 composite_operation operation_;
670 std::vector<std::shared_ptr<alert_trigger>> triggers_;
671};

680class absent_trigger : public alert_trigger {
681public:
686 explicit absent_trigger(std::chrono::milliseconds absent_duration)
687 : absent_duration_(absent_duration) {}
688
689 bool evaluate(double /*value*/) const override {
690 auto now = std::chrono::steady_clock::now();
691
692 std::lock_guard<std::mutex> lock(mutex_);
693
694 auto previous = last_seen_;
695 last_seen_ = now;
696
697 // First evaluation - not absent yet
698 if (previous == std::chrono::steady_clock::time_point{}) {
699 return false;
700 }
701
702 // Check if the gap since previous value exceeds threshold
703 return (now - previous) > absent_duration_;
704 }
705
706 std::string type_name() const override {
707 return "absent";
708 }
709
710 std::string description() const override {
711 return "no data for " + std::to_string(absent_duration_.count()) + "ms";
712 }
713
717 void reset() {
718 std::lock_guard<std::mutex> lock(mutex_);
719 last_seen_ = std::chrono::steady_clock::time_point{};
720 }
721
722private:
723 std::chrono::milliseconds absent_duration_;
724 mutable std::mutex mutex_;
725 mutable std::chrono::steady_clock::time_point last_seen_{};
726};

735class delta_trigger : public alert_trigger {
736public:
742 explicit delta_trigger(double delta_threshold, bool absolute = true)
743 : delta_threshold_(delta_threshold)
744 , absolute_(absolute) {}
745
746 bool evaluate(double value) const override {
747 std::lock_guard<std::mutex> lock(mutex_);
748
749 if (!has_previous_) {
750 previous_value_ = value;
751 has_previous_ = true;
752 return false;
753 }
754
755 double delta = value - previous_value_;
756 previous_value_ = value;
757
758 if (absolute_) {
759 return std::abs(delta) > delta_threshold_;
760 }
761 return delta > delta_threshold_;
762 }
763
764 std::string type_name() const override {
765 return "delta";
766 }
767
768 std::string description() const override {
769 if (absolute_) {
770 return "|delta| > " + std::to_string(delta_threshold_);
771 }
772 return "delta > " + std::to_string(delta_threshold_);
773 }
774
778 void reset() {
779 std::lock_guard<std::mutex> lock(mutex_);
780 has_previous_ = false;
781 }
782
783private:
784 double delta_threshold_;
785 bool absolute_;
786 mutable std::mutex mutex_;
787 mutable double previous_value_ = 0.0;
788 mutable bool has_previous_ = false;
789};
