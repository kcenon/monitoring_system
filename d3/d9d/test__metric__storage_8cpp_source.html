57TEST_F(MetricStorageTest, RingBufferBasicOperations) {
58 ring_buffer_config config;
59 config.capacity = 8; // Small capacity for testing
60 config.overwrite_old = false;
61
62 ring_buffer<int> buffer(config);
63
64 EXPECT_EQ(buffer.capacity(), 8);
65 EXPECT_TRUE(buffer.empty());
66 EXPECT_FALSE(buffer.full());
67 EXPECT_EQ(buffer.size(), 0);
68
69 // Write elements
70 for (int i = 0; i < 7; ++i) {
71 auto result = buffer.write(std::move(i));
72 EXPECT_TRUE(result) << "Failed to write " << i;
73 }
74
75 EXPECT_EQ(buffer.size(), 7);
76 EXPECT_FALSE(buffer.empty());
77 EXPECT_TRUE(buffer.full()); // 7 elements in 8-capacity buffer means full
78
79 // Try to write when full
80 auto result = buffer.write(std::move(999));
81 EXPECT_FALSE(result); // Should fail as overwrite_old is false
82
83 // Read elements
84 int value;
85 for (int i = 0; i < 7; ++i) {
86 auto read_result = buffer.read(value);
87 EXPECT_TRUE(read_result);
88 EXPECT_EQ(value, i);
89 }
90
91 EXPECT_TRUE(buffer.empty());
92}

354TEST_F(MetricStorageTest, MetricStorageThreadSafety) {
355 metric_storage storage;
356
357 const int num_threads = 4;
358 const int metrics_per_thread = 100;
359 std::vector<std::thread> threads;
360
361 // Launch multiple threads writing metrics
362 for (int t = 0; t < num_threads; ++t) {
363 threads.emplace_back([&storage, t, metrics_per_thread]() {
364 std::random_device rd;
365 std::mt19937 gen(rd());
366 std::uniform_real_distribution<> dis(0.0, 100.0);
367
368 for (int i = 0; i < metrics_per_thread; ++i) {
369 std::string metric_name = "thread_" + std::to_string(t) + "_metric_" + std::to_string(i);
370 double value = dis(gen);
371 storage.store_metric(metric_name, value);
372
373 // Small delay to increase chance of contention
374 std::this_thread::sleep_for(std::chrono::microseconds(1));
375 }
376 });
377 }
378
379 // Wait for all threads to complete
380 for (auto& thread : threads) {
381 thread.join();
382 }
383
384 storage.flush();
385
386 // Verify we stored a reasonable number of metrics
387 const auto& stats = storage.get_stats();
388 EXPECT_GT(stats.total_metrics_stored.load(), num_threads * metrics_per_thread * 0.8);
389
390 auto names = storage.get_metric_names();
391 EXPECT_GT(names.size(), 0);
392}
