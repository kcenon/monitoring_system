54class monitor_factory : public IMonitorProvider {
55private:
56 static std::shared_ptr<monitor_factory> instance_;
57 static std::mutex instance_mutex_;
58
59 std::shared_ptr<IMonitor> default_monitor_;
60 std::unordered_map<std::string, std::shared_ptr<IMonitor>> named_monitors_;
61 std::shared_ptr<ILogger> shared_logger_;
62 mutable std::mutex factory_mutex_;
63
64 monitor_factory() = default;
65
66public:
67 static std::shared_ptr<monitor_factory> instance() {
68 std::lock_guard<std::mutex> lock(instance_mutex_);
69 if (!instance_) {
70 instance_ = std::shared_ptr<monitor_factory>(new monitor_factory());
71 }
72 return instance_;
73 }
74
75 void set_shared_logger(std::shared_ptr<ILogger> logger) {
76 std::lock_guard<std::mutex> lock(factory_mutex_);
77 shared_logger_ = logger;
78
79 // Note: Logger injection not available in performance_monitor yet
80 // Future enhancement: Add ILogger support to performance_monitor
81 }
82
83 // IMonitorProvider implementation
84 std::shared_ptr<IMonitor> get_monitor() override {
85 std::lock_guard<std::mutex> lock(factory_mutex_);
86
87 if (!default_monitor_) {
88 default_monitor_ = std::make_shared<performance_monitor>();
89 }
90
91 return default_monitor_;
92 }
93
94 std::shared_ptr<IMonitor> create_monitor(const std::string& name) override {
95 std::lock_guard<std::mutex> lock(factory_mutex_);
96
97 auto it = named_monitors_.find(name);
98 if (it != named_monitors_.end()) {
99 return it->second;
100 }
101
102 auto monitor = std::make_shared<performance_monitor>();
103 named_monitors_[name] = monitor;
104 return monitor;
105 }
106
107 std::vector<std::string> list_monitors() const {
108 std::lock_guard<std::mutex> lock(factory_mutex_);
109 std::vector<std::string> names;
110 names.reserve(named_monitors_.size());
111
112 for (const auto& [name, _] : named_monitors_) {
113 names.push_back(name);
114 }
115
116 return names;
117 }
118
119 size_t monitor_count() const {
120 std::lock_guard<std::mutex> lock(factory_mutex_);
121 return named_monitors_.size() + (default_monitor_ ? 1 : 0);
122 }
123
124 void reset() {
125 std::lock_guard<std::mutex> lock(factory_mutex_);
126 default_monitor_.reset();
127 named_monitors_.clear();
128 }
129};

137class example_logger : public ILogger {
138private:
139 std::string prefix_;
140 std::atomic<size_t> count_{0};
141 log_level min_level_ = log_level::trace;
142
143public:
144 explicit example_logger(const std::string& prefix = "LOG")
145 : prefix_(prefix) {}
146
147 common::VoidResult log(log_level level, const std::string& message) override {
148 std::cout << "[" << prefix_ << "] [" << to_string(level) << "] "
149 << message << std::endl;
150 count_++;
151 return common::ok();
152 }
153
154 common::VoidResult log(log_level level, const std::string& message,
155 const std::string&, int, const std::string&) override {
156 return log(level, message);
157 }
158
159 common::VoidResult log(const log_entry& entry) override {
160 return log(entry.level, entry.message);
161 }
162
163 bool is_enabled(log_level level) const override {
164 return static_cast<int>(level) >= static_cast<int>(min_level_);
165 }
166
167 common::VoidResult set_level(log_level level) override {
168 min_level_ = level;
169 return common::ok();
170 }
171
172 log_level get_level() const override {
173 return min_level_;
174 }
175
176 common::VoidResult flush() override {
177 std::cout << std::flush;
178 return common::ok();
179 }
180
181 size_t count() const { return count_.load(); }
182};

325class aggregating_monitor : public IMonitor {
326private:
327 std::vector<std::shared_ptr<IMonitor>> monitors_;
328 std::mutex mutex_;
329
330public:
331 void add_monitor(std::shared_ptr<IMonitor> monitor) {
332 std::lock_guard<std::mutex> lock(mutex_);
333 monitors_.push_back(monitor);
334 }
335
336 common::VoidResult record_metric(const std::string& name, double value) override {
337 std::lock_guard<std::mutex> lock(mutex_);
338 for (auto& monitor : monitors_) {
339 monitor->record_metric(name, value);
340 }
341 return common::ok();
342 }
343
344 common::VoidResult record_metric(
345 const std::string& name,
346 double value,
347 const std::unordered_map<std::string, std::string>& tags) override
348 {
349 std::lock_guard<std::mutex> lock(mutex_);
350 for (auto& monitor : monitors_) {
351 monitor->record_metric(name, value, tags);
352 }
353 return common::ok();
354 }
355
356 common::Result<metrics_snapshot> get_metrics() override {
357 std::lock_guard<std::mutex> lock(mutex_);
358
359 metrics_snapshot combined;
360 combined.source_id = "aggregating_monitor";
361 combined.capture_time = std::chrono::system_clock::now();
362
363 for (auto& monitor : monitors_) {
364 auto result = monitor->get_metrics();
365 if (result) {
366 for (const auto& metric : result.value().metrics) {
367 combined.metrics.push_back(metric);
368 }
369 }
370 }
371
372 return common::ok(std::move(combined));
373 }
374
375 common::Result<health_check_result> check_health() override {
376 health_check_result result;
377 result.timestamp = std::chrono::system_clock::now();
378 result.status = health_status::healthy;
379 result.message = "Aggregating monitor";
380
381 return common::ok(std::move(result));
382 }
383
384 common::VoidResult reset() override {
385 std::lock_guard<std::mutex> lock(mutex_);
386 for (auto& monitor : monitors_) {
387 monitor->reset();
388 }
389 return common::ok();
390 }
391
392 size_t monitor_count() const {
393 std::lock_guard<std::mutex> lock(mutex_);
394 return monitors_.size();
395 }
396};
