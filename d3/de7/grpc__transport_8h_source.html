135class stub_grpc_transport : public grpc_transport {
136private:
137 std::string host_;
138 uint16_t port_{0};
139 bool connected_{false};
140 bool simulate_success_{true};
141 std::function<grpc_response(const grpc_request&)> response_handler_;
142 mutable std::atomic<std::size_t> requests_sent_{0};
143 mutable std::atomic<std::size_t> bytes_sent_{0};
144 mutable std::atomic<std::size_t> send_failures_{0};
145
146public:
147 stub_grpc_transport() = default;
148
152 void set_simulate_success(bool success) {
153 simulate_success_ = success;
154 }
155
159 void set_response_handler(std::function<grpc_response(const grpc_request&)> handler) {
160 response_handler_ = std::move(handler);
161 }
162
163 result_void connect(const std::string& host, uint16_t port) override {
164 if (!simulate_success_) {
165 return result_void::err(error_info(
166 monitoring_error_code::network_error,
167 "Simulated connection failure",
168 "stub_grpc_transport"
169 ).to_common_error());
170 }
171 host_ = host;
172 port_ = port;
173 connected_ = true;
174 return common::ok();
175 }
176
177 result<grpc_response> send(const grpc_request& request) override {
178 if (!connected_) {
179 send_failures_.fetch_add(1, std::memory_order_relaxed);
180 return make_error<grpc_response>(
181 monitoring_error_code::network_error,
182 "Not connected");
183 }
184
185 if (response_handler_) {
186 auto response = response_handler_(request);
187 requests_sent_.fetch_add(1, std::memory_order_relaxed);
188 bytes_sent_.fetch_add(request.body.size(), std::memory_order_relaxed);
189 return make_success(response);
190 }
191
192 if (!simulate_success_) {
193 send_failures_.fetch_add(1, std::memory_order_relaxed);
194 return make_error<grpc_response>(
195 monitoring_error_code::network_error,
196 "Simulated send failure");
197 }
198
199 grpc_response response;
200 response.status_code = 0; // OK in gRPC
201 response.status_message = "OK";
202 response.elapsed = std::chrono::milliseconds(10);
203
204 requests_sent_.fetch_add(1, std::memory_order_relaxed);
205 bytes_sent_.fetch_add(request.body.size(), std::memory_order_relaxed);
206
207 return make_success(response);
208 }
209
210 bool is_connected() const override {
211 return connected_;
212 }
213
214 void disconnect() override {
215 connected_ = false;
216 host_.clear();
217 port_ = 0;
218 }
219
220 bool is_available() const override {
221 return true;
222 }
223
224 std::string name() const override {
225 return "stub";
226 }
227
228 grpc_statistics get_statistics() const override {
229 return {
230 requests_sent_.load(std::memory_order_relaxed),
231 bytes_sent_.load(std::memory_order_relaxed),
232 send_failures_.load(std::memory_order_relaxed)
233 };
234 }
235
236 void reset_statistics() override {
237 requests_sent_.store(0, std::memory_order_relaxed);
238 bytes_sent_.store(0, std::memory_order_relaxed);
239 send_failures_.store(0, std::memory_order_relaxed);
240 }
241
242 // Test helpers
243 std::string get_host() const { return host_; }
244 uint16_t get_port() const { return port_; }
245};
